REPLACE:country_is_abolishing_bakufu = {
    AND = {
        has_law = law_type:law_bakufu
        OR = {
            is_enacting_law = law_type:law_no_separate_hog
            is_enacting_law = law_type:law_appointed_on_advice
            is_enacting_law = law_type:law_appointed_by_hos
        }
    }
}

REPLACE:japanese_imperial_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		is_ruler_of_own_country = yes
		owner ?= {
			NOR = {
				has_government_type = gov_domain_alliance
				has_government_type = gov_domain_frontier
			}
		}
     	home_country ?= {
			OR = {
     			has_law = law_type:law_monarchy
     			has_law = law_type:law_social_monarchy
			}
     	}
		OR = {
			culture = {
				has_discrimination_trait = language_japanese
			}
			c:JAP ?= home_country
		}
	}
}

REPLACE:character_is_emperor_of_japan = {
	c:JAP ?= owner
    is_ruler_of_own_country = yes
}
