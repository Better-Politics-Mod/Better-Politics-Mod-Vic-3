bpm_update_ig_front_support = {
	scope = interest_group


	effect = {
		save_scope_as = selected_ig
		clear_variable_map = bpm_front_attraction
		ordered_in_global_list = {
			variable = bpm_existing_front_names
			order_by = bpm_front_attraction_dispatch_overall
			scope:selected_ig = {
				add_to_variable_map = {
					name = bpm_front_attraction
					key = prev
					value = prev.bpm_front_attraction_dispatch_overall
				}
			}
		}
		clear_saved_scope = selected_ig
	}
}

bpm_action_visible_sgui = {
	scope = political_movement

	saved_scopes = {
		polmov
	}

	visible = {
		this ?= scope:polmov
	}
}

bpm_setup_movement_panel = {
	scope = political_movement

	effect = {
		save_scope_as = movement
		clear_variable_list = bpm_ig_view
		owner = {
			every_supportive_interest_group = {
				prev = {
					add_to_variable_list = {
						name = bpm_ig_view
						target = prev
					}
				}
			}
		}
	}
}

bpm_cell_is_growing = {
	scope = state

	saved_scopes = {
		movement
	}

	is_shown = {
		owner = { save_temporary_scope_as = country }
		bpm_cell_growth_chance > bpm_cell_shrinking_chance
	}
	effect = {
		save_scope_as = country
		bpm_increase_or_shrink_tt = yes
	}
}

bpm_front_has_ongoing_action = {
	scope = political_movement

	is_shown = {
		save_temporary_scope_as = movement
		owner = {
			any_in_list = {
				variable = bpm_active_coalition_actions
				var:bpm_initiator ?= scope:movement
			}
		}
	}
}
