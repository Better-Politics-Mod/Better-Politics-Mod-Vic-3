namespace = bpm_election_events

bpm_election_events.1 = {
    type = country_event
	placement = root
	title = bpm_election_events.1.t
	desc = bpm_election_events.1.d
	flavor = bpm_election_events.1.f

	dlc = dlc_bpm
	event_image = { video = "gfx/event_pictures/europenorthamerica_political_extremism.bk2" }

	#left_icon = scope:bpm_new_ruler_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

    duration = 4

    trigger = {
		# triggered by new ruler election
	}

	immediate = {
		if = {
			limit = {
				has_law_or_variant = law_type:law_answers_to_legislature
			}
			random_scope_character = {
				limit = {
					has_role = bpm_character_role_hog
				}
				save_scope_as = bpm_new_ruler_scope
			}
		}
		else = {
			ruler = {
				save_scope_as = bpm_new_ruler_scope
			}
		}
	}

    option = { # strategy: greedy ruler
		name = bpm_election_events.1.a
		set_variable = { 
			name = bpm_cabappt_strategy
			value = flag:bpm_greedy_ruler
		}
		custom_tooltip = bpm_cabinet_strategy_greedy_tt
	}

	option = { # strategy: keep the status quo
		name = bpm_election_events.1.b
		remove_variable = bpm_cabappt_strategy
		custom_tooltip = bpm_cabinet_strategy_statusquo_tt
	}

	option = { # strategy: prominent ideologues
		name = bpm_election_events.1.c
		trigger = {
			scope:bpm_new_ruler_scope.interest = {
				custom_tooltip = {
					text = "bpm_not_anti_establishment_tt"
					bpm_ig_is_radical_opposition = no
				}
				custom_tooltip = {
					text = "bpm_not_extremist_tt"
					bpm_ig_is_very_radical_opposition = no
				}
			}
		}
		set_variable = { 
			name = bpm_cabappt_strategy
			value = flag:bpm_pick_prominent_idg
		}
		custom_tooltip = bpm_cabinet_strategy_promidg_tt
	}

	option = { # strategy: prominent institutionalists
		name = bpm_election_events.1.e
		trigger = {
			scope:bpm_new_ruler_scope.interest = {
				custom_tooltip = {
					text = "bpm_not_anti_establishment_tt"
					bpm_ig_is_radical_opposition = no
				}
				custom_tooltip = {
					text = "bpm_not_extremist_tt"
					bpm_ig_is_very_radical_opposition = no
				}
			}
		}
		set_variable = {
			name = bpm_cabappt_strategy
			value = flag:bpm_pick_prominent_ing
		}
		custom_tooltip = bpm_cabinet_strategy_proming_tt
	}

}
