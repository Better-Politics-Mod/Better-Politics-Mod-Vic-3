bpm_movement_is_significant_faction = {
	OR = {
		AND = {
			scope:polmov = { is_political_movement_type = movement_liberal }
			OR = { 
				is_interest_group_type = ig_intelligentsia
				is_interest_group_type = ig_industrialists
			}
		}
		AND = {
			scope:polmov = { 
				OR = {
					is_political_movement_type = movement_royalist_absolutist
					is_political_movement_type = movement_legitimist
					is_political_movement_type = movement_carlist
					is_political_movement_type = movement_miguelist
				}
			}
			is_interest_group_type = ig_landowners
		}
	}
}
