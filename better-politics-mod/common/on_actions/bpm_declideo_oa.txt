on_game_started = {
    on_actions = {
        bpm_on_game_started_declideo
    }
}

on_law_activated = {
    on_actions = {
        bpm_on_law_activated_declideo
    }
}

bpm_on_game_started_declideo = {
    effect = {
        bpm_handle_previous_law_saving_to_varlist = yes
    }
}

bpm_on_law_activated_declideo = {
    effect = {
        if = {
            limit = {
                has_game_started = yes
            }
            bpm_handle_previous_law_saving = yes
            bpm_handle_law_entrenchment = yes
        }
    }
}
