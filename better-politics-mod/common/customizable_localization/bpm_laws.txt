	type = country

		trigger = {
			has_law = law_type:law_presidential_republic
		}
		localization_key = bpm_repeatable_politics_topple_name_presidential
	}

	text = {
		trigger = {
			NOT = {
				has_law = law_type:law_presidential_republic

		}

	}

	text = {
		trigger = {
			always = not
		}

		fallback = yes 
		localization_key = bpm_repeatable_politics_topple_name_base
	}
