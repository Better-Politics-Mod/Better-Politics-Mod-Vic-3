namespace = bpm_send_notification

bpm_send_notification.1 = { 
	type = country_event
	placement = root
	hidden = yes

	immediate = {
		if = {
			limit = {
				country_rank >= rank_value:major_power
			}
			save_scope_as = revolution_country
			every_country = {
				post_notification = bpm_significant_new_republic
			}
		}
	}
}
