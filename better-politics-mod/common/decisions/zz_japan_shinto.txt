REPLACE:shinto_decision = {
	is_shown = {
		exists = c:JAP
		c:JAP = THIS
		NOT = { country_has_state_religion = rel:shinto }
		NOT = { has_dlc_feature = ep2_content }
		has_variable = japan_emperor_restored
	}

	possible = {
		ig:ig_devout = {
			bpm_ig_is_marginal = no
		}
		has_law = law_type:law_monarchy
		NOT = { has_law_or_variant = law_type:law_state_religion }
	}

	when_taken = {
		set_state_religion = rel:shinto
		state_religion_switch_effect = yes
		ig:ig_devout = {
			bpm_remove_all_cit_ideologies = yes
			bpm_remove_all_dop_ideologies = yes
			add_ideology = ideology_dop_autocrat
			add_ideology = ideology_cit_ethnonationalist
			bpm_reorder_ideologies = yes
		}
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:shinto
				value = 0.4
			}
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = { is_capital = yes }
			}
			convert_population = {
				target = rel:shinto
				value = 0.2
			}
		}
	}

	ai_chance = {
		value = 10
	}
}
