					}
				}
				multiply = 0.5
			}

			if = {
				limit = {
					interest_group ?= {
						ig_approval <= -5
					}
				}
				multiply = 1.25
			}
			else_if = {
				limit = {
					interest_group ?= {
						ig_approval <= -10
					}
				}
				multiply = 1.5
			}

			if = { # If we are Conservative dont immediately hang any Liberal politician and so on

					scope:actor ?= {
						has_strategy = ai_strategy_conservative_agenda
					}
					OR = {

					}
				}

			}
			else_if = {
				limit = {
					scope:actor ?= {
						has_strategy = ai_strategy_progressive_agenda
					}
					OR = {

						bpm_leader_is_progressive_liberal = yes
					}
				}

			}
