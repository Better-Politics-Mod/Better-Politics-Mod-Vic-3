on_monthly_pulse_character = {
	on_actions = {
		bpm_prominerf
	}
}

on_game_started = {
	on_actions = {
		bpm_prominerfall
	}
}

bpm_prominerf = {
	effect = {
		if = {
			limit = {
				interest_group = {
					bpm_interest_group_is_activated = no
				}
			}
			add_modifier = { name = bpm_character_with_irrelevant_ig }
		}
		else_if = {
			limit = {
				has_modifier = bpm_character_with_irrelevant_ig
			}
			remove_modifier = bpm_character_with_irrelevant_ig
		}
	}
}

bpm_prominerfall = {
	effect = {
		every_country = {
			every_scope_character = {
				if = {
					limit = {
						interest_group = {
							bpm_interest_group_is_activated = no
						}
					}
					add_modifier = { name = bpm_character_with_irrelevant_ig }
				}
				else_if = {
					limit = {
						has_modifier = bpm_character_with_irrelevant_ig
					}
					remove_modifier = bpm_character_with_irrelevant_ig
				}
			}
		}

	}
}

bpm_ig_shoving = {
	effect = {
		if = {
			limit = {
				interest_group = {
					bpm_interest_group_is_activated = no
				}
			}
			# reactionaries
			if = {
				limit = {
					interest_group = {
						is_interest_group_type = ig_reactionaries
						is_interest_group_type = ig_conservatives
					}
				}
				set_interest_group = owner.ig:ig_landowners
			}
			if = {
				limit = {
					interest_group = {
						is_interest_group_type = ig_socialists
						is_interest_group_type = ig_revolutionist_socialists
						is_interest_group_type = ig_reformist_socialists
						is_interest_group_type = ig_fascists
					}
				}
				set_interest_group = owner.ig:ig_trade_unions
			}
			if = {
				limit = {
					interest_group = {
						is_interest_group_type = ig_market_liberals
					}
				}
				set_interest_group = owner.ig:ig_industrialists
			}
			if = {
				limit = {
					interest_group = {
						is_interest_group_type = ig_national_liberals
						is_interest_group_type = ig_liberals
					}
				}
				set_interest_group = owner.ig:ig_petty_bourgeoisie
			}
			if = {
				limit = {
					interest_group = {
						is_interest_group_type = ig_anarchists
						is_interest_group_type = ig_agrarian_populists
					}
				}
				set_interest_group = owner.ig:ig_rural_folk
			}
		}

	}
}
