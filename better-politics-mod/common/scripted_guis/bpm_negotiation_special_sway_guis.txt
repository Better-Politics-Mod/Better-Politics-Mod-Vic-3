bpm_pm_sway_gui = {
	scope = interest_group
	is_shown = {
		owner = { has_modifier = bpm_in_negotiation }
		owner = { bpm_country_has_parliamentary_electoral_system = yes }
		NOT = { owner = { any_interest_group = { has_variable = bpm_pm_promised_ig }} } # only one promise at a time
		NOT = { owner.var:bpm_headofgov ?= leader }
		is_in_government = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		set_variable = {
			name = bpm_pm_promised_ig
			days = 360
		}
		add_modifier = {
			name = bpm_favored_pm
			months = normal_modifier_time
			is_decaying = yes
		}
		leader = {
            bpm_appoint_as_hog = yes
            set_variable = bpm_rightfully_elected_headofgov
		}
	}
}

bpm_military_concessions_sway_gui = {
	scope = interest_group
	is_shown = {
		owner = { has_modifier = bpm_in_negotiation }
		#owner = { bpm_country_has_parliamentary_electoral_system = yes }
		NOT = { owner = { has_journal_entry = je_bpm_concession_military } } # only one promise at a time
		#NOT = { owner.var:bpm_headofgov ?= leader }
		is_interest_group_type = ig_armed_forces
		#is_in_government = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		owner = { add_journal_entry = {
			type = je_bpm_concession_military
		} }
	}
}
