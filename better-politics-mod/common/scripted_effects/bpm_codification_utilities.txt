bpm_enact_law = {
	bpm_codify_law = {
		law = $law$
		level = $level$
	}
	activate_law = $law$
	set_variable = bpm_codification_law_passed_by_enactment_consumable
}

bpm_activate_law = {
	save_scope_as = activating_law
	owner = {
		bpm_codify_law = {
			law = scope:activating_law.type
			level = $level$
		}
		activate_law = scope:activating_law.type
		set_variable = bpm_codification_law_passed_by_enactment_consumable
	}
}

bpm_codify_law = {
	save_scope_as = associated_country

	if = {
		limit = {
			flag:$level$ = flag:traditional
			$law$ = {
				bpm_codification_can_be_traditional = yes
			}
		}
		custom_tooltip = bpm_codification_set_traditional_tooltip
		$law$ = {
			bpm_codification_set_traditional = yes
		}
	}
	else_if = {
		limit = {
			flag:$level$ = flag:decree
			$law$ = {
				bpm_codification_can_be_decree = yes
			}
		}
		custom_tooltip = bpm_codification_set_decree_tooltip
		$law$ = {
			bpm_codification_set_decree = yes
		}
	}
	else_if = {
		limit = {
			flag:$level$ = flag:legal
			$law$ = {
				bpm_codification_can_be_legal = yes
			}
		}
		custom_tooltip = bpm_codification_set_legal_tooltip
		$law$ = {
			bpm_codification_set_legal = yes
		}
	}
	else_if = {
		limit = {
			flag:$level$ = flag:constitutional
			$law$ = {
				bpm_codification_can_be_constitutional = yes
			}
		}
		$law$ = {
			custom_tooltip = bpm_codification_set_constitutional_tooltip
			bpm_codification_set_constitutional = yes
		}
	}

}
