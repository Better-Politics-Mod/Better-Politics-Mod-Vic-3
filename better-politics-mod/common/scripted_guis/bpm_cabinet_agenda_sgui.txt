bpm_cabinet_agenda_set_slot_one = {
	scope = country
	saved_scopes = {
		slot
	}
	effect = {
		je:je_bpm_cabinet_agendas = {
			set_variable = {
				name = bpm_cabinet_agenda_slot_one
				value = scope:slot
			}
			set_variable = {
				name = bpm_cabinet_agenda_slot_one_progress
				value = 0
			}
		}
	}
}

bpm_cabinet_agenda_slot_one_sgui = {
    scope = country
	# IS_ACTIVE
    is_shown = {
		je:je_bpm_cabinet_agendas = {
			# There is a slot
			has_variable = bpm_cabinet_agenda_slot_one
			# The slot is not empty
			NOT = {
				var:bpm_cabinet_agenda_slot_one = flag:bpm_no_cabinet_agenda
			}
		}
    }
	# IS_OPEN
    is_valid = {
		je:je_bpm_cabinet_agendas = {
			# There is a slot
			has_variable = bpm_cabinet_agenda_slot_one
			# The slot is empty
			var:bpm_cabinet_agenda_slot_one = flag:bpm_no_cabinet_agenda
		}
    }
}
