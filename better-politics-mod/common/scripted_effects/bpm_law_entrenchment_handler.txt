bpm_handle_law_entrenchment = {
    save_scope_as = law
    owner = {
        if = {
            limit = {
                has_law_or_variant = law_type:law_hereditary_bureaucrats
            }
            ig:ig_landowners ?= { save_scope_as = selected_ig }
        }
        else = {
            ig:ig_petty_bourgeoisie ?= { save_scope_as = selected_ig }
        }
        if = {
            limit = {
                scope:law.type = {
                    is_same_law_group_as = law_type:law_monarchy
                }
            }
            scope:law = {
                add_amendment = {
                    type = bpm_amendment_newly_enacted
                    sponsor = scope:selected_ig
                    timeout = 60
                }
            }
        }
        else = {
            scope:law = {
                add_amendment = {
                    type = bpm_amendment_newly_enacted
                    sponsor = scope:selected_ig
                    timeout = 30
                }
            }
        }
        bpm_update_ideology_all_igs = yes
    }
}

bpm_handle_previous_law_saving = {    
    save_scope_as = law
    owner = {
        if = {
            limit = {
                NOT = {
                    has_variable_list = bpm_country_laws_list
                }
            }
            bpm_handle_previous_law_saving_to_varlist = yes
        }

        random_in_list = {
            variable = bpm_country_laws_list
            limit = { this ?= scope:law.type }
            save_scope_as = prev_law_type
        }
        if = {
            limit = {
                NOT = {
                    scope:law = {
                        has_amendment = bpm_amendment_newly_enacted
                    }
                }
            }
            update_variable_map_value = { 
                name = bpm_previous_laws 
                key = scope:law.type 
                value = scope:prev_law_type 
            }
        }
        else = {
            update_variable_map_value = { 
                name = bpm_previous_laws 
                key = scope:law.type 
                value = variable_map(bpm_previous_laws|scope:prev_law_type) 
            }
        }

    }

}

bpm_handle_previous_law_saving_to_varlist = {
    clear_variable_list = bpm_country_laws_list
    every_active_law = {
        owner = {
            add_to_variable_list = {
                name = bpm_country_laws_list
                target = prev.type
            }
        }
    }
}

update_variable_map_value = {
    if = {
        limit = {
            is_key_in_variable_map = {
                name = $name$
                target = $key$
            }
        }
        remove_from_variable_map = {
            name = $name$
            key = $key$
        }
    }
    add_to_variable_map = {
        name = $name$
        key = $key$
        value = $value$
    }
}
