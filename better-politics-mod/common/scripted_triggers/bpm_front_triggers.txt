bpm_trigger_any_front = {
	OR = {
		$trigger$ = { front = urban }
		$trigger$ = { front = popular }
		$trigger$ = { front = liberal }
		$trigger$ = { front = republican }
		$trigger$ = { front = labor }
		$trigger$ = { front = socialist }
		$trigger$ = { front = reactionary }
		$trigger$ = { front = fascist }
		$trigger$ = { front = nationalist }
		$trigger$ = { front = peasant }
	}
}


bpm_any_state_has_rev = {
	any_scope_state = {
		bpm_state_has_rev = yes
	}
}

bpm_state_has_rev = {
	OR = {
		has_modifier = bpm_revcell_$front$_lv1
		has_modifier = bpm_revcell_$front$_lv2
		has_modifier = bpm_revcell_$front$_lv3
	}
}

bpm_state_front_is_growing = {
	bpm_state_repression <= bpm_front_$front$_progressrate
	NOT = {
		turmoil < 0.25
	}
}
