bpm_setup_expected_cabpos = {
	every_interest_group = {
		remove_variable = bpm_expected_cabpos
		remove_variable = bpm_actual_cabpos
		remove_modifier = bpm_cabinet_positions_expected_modifier
		remove_modifier = bpm_cabinet_positions_jealousy_modifier
	}
	every_interest_group = {
		limit = {
			bpm_expected_cabpos_base > 0
		}
		set_variable = {
			name = bpm_expected_cabpos
			value = bpm_expected_cabpos_base
		}
	}
	every_interest_group = {
		set_variable = {
			name = bpm_actual_cabpos
			value = bpm_number_of_positions_ig
		}
	}

	every_interest_group = {
		save_scope_value_as = {
			name = ig_inq
			value = bpm_jealousy_cabpos
		}
		root = {
			set_variable = {
				name = bpm_jealousy_cabpos
				value = scope:ig_inq
			}
		}
		add_modifier = {
			name = bpm_cabinet_positions_jealousy_modifier
			multiplier = var:bpm_jealousy_cabpos
		}
		save_scope_value_as = {
			name = ig_inq
			value = bpm_expectation_cabpos
		}
		root = {
			set_variable = {
				name = bpm_jealousy_cabpos
				value = scope:ig_inq
			}
		}
		add_modifier = {
			name = bpm_cabinet_positions_expected_modifier
			multiplier = var:bpm_jealousy_cabpos
		}
	}
}
