REPLACE:je_meiji_imperial_marriage = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_meiji_restoration

	immediate = {
        random_scope_character = {
            limit = {
                has_role = bpm_character_role_hog
            }
            save_scope_as = imperial_marriage_ruler
        }
		random_scope_character = {
			limit = {
				has_variable = japan_imperial_princess
			}
			save_scope_as = imperial_princess_scope
		}
		ordered_scope_treaty = {
			limit = {
				binds = ROOT
				any_scope_article = {
					OR = {
						has_type = trade_privilege
						has_type = foreign_investment_rights
						has_type = law_commitment
					}
				}
			}
			order_by = {
				value = "ROOT.ai_treaty_value(scope:source_country)"
				multiply = -1
			}
			save_scope_as = relevant_treaty_scope
		}
		if = {
			limit = {
				exists = scope:relevant_treaty_scope
			}
			scope:journal_entry ?= {
				set_variable = japan_break_treaty_var
			}
		}
	}

	complete = {
		trigger_if = {
			limit = {
				scope:journal_entry ?= {
					has_variable = japan_break_treaty_var
				}
			}
			trigger_if = {
				limit = {
					exists = scope:relevant_treaty_scope
				}
				custom_tooltip = {
					text = has_broken_relevant_treaty_tt
					NOT = {
						exists = scope:relevant_treaty_scope
					}
				}
			}
			trigger_else = {
				custom_tooltip = {
					text = im_broken_applicable_treaty_tt
					NOT = {
						exists = scope:relevant_treaty_scope
					}
				}
			}
		}
		has_law_or_variant = law_type:law_closed_borders
		NOT = {
			has_law_or_variant = law_type:law_free_trade
		}
		NOT = {
			any_state_in_home_islands = {
				is_treaty_port = yes
			}
		}
	}

	on_complete = {
		custom_tooltip = {
			text = kobu_gattai_progress_tt
			set_variable = japan_completed_imperial_marriage
		}
	}

	fail = {
		OR = {
			custom_tooltip = {
				text = ruler_does_not_have_imperial_marriage_tt
                any_scope_character = {
                    has_role = bpm_character_role_hog
                    NOT = {
		        		has_variable = japan_imperial_marriage
		        	}
                }
			}
			custom_tooltip = {
				text = je_has_timed_out_tt
				NOT = {
					has_journal_entry = je_meiji_imperial_marriage
				}
			}
		}
	}

	on_fail = {
		ruler = {
			remove_variable = japan_imperial_marriage
		}
		custom_tooltip = {
			text = kobu_gattai_no_longer_viable_tt
			remove_variable = kobu_gattai
			set_variable = kogi_yoron
		}
		custom_tooltip = kogi_yoron_forced_tt
	}

	on_timeout = {
		custom_tooltip = je_will_fail_tt
		ruler = {
			remove_variable = japan_imperial_marriage
		}
		remove_variable = kobu_gattai
		set_variable = kogi_yoron
	}

	invalid = {
		NOR = {
			has_law_or_variant = law_type:law_monarchy
			has_law = law_type:law_bakufu
		}
	}

	timeout = 1825

	weight = 10000
	should_be_pinned_by_default_uninvolved_or_context = yes
	can_revolution_inherit = yes
}

	custom_on_failure_header = je_meiji_restoration_on_shogunate_victory_header
	on_fail = {
		if = {
			limit = {
				has_variable = kobu_gattai
			}
			trigger_event = {
				id = ep2_meiji.8 popup = yes
			}
		}
		else_if = {
			limit = {
				has_variable = kogi_yoron
			}
			trigger_event = {
				id = ep2_meiji.9 popup = yes
			}
			set_variable = japan_emperor_restored
		}
		else = {
			custom_tooltip = je_meiji_shogunate_results_unknown_tt
		}
		set_variable = japan_restoration_complete
	}

	event_outcome_failed_effect_desc = {
        effect = {
			if = {
				limit = {
					has_variable = kobu_gattai
				}
				custom_tooltip_no_bullet = ep2_meiji.8.a
				add_modifier = {
					name = modifier_court_and_shogunate
					days = very_long_modifier_time
					is_decaying = yes
				}
				random_scope_character = {
					limit = {
						has_role = character_role_emperor_of_japan
					}
					add_trait = literary
					add_modifier = {
						name = modifier_veiled_by_awe
						days = stupidly_long_modifier_time
					}
				}
			}
        }
    }

    event_outcome_failed_effect_desc = {
        effect = {
			if = {
				limit = {
					has_variable = kogi_yoron
				}
				if = {
					limit = {
						NOT = {
							any_political_movement = {
								is_political_movement_type = movement_modernizer
							}
						}
					}
					create_political_movement = { type = movement_modernizer }
				}
				japan_display_add_meiji_reforms = yes
			}
        }
    }

    event_outcome_failed_effect_desc = {
        effect = {
			if = {
				limit = {
					has_variable = kogi_yoron
				}
				custom_tooltip_no_bullet = ep2_meiji.9.a
				add_modifier = {
					name = modifier_continuity_and_enlightenment
					days = very_long_modifier_time
					is_decaying = yes
				}
			}
        }
    }

    event_outcome_failed_effect_desc = {
        effect = {
			if = {
				limit = {
					has_variable = kogi_yoron
				}
				custom_tooltip_no_bullet = ep2_meiji.9.b
				add_modifier = {
					name = modifier_taikun_monarchy
					days = very_long_modifier_time
					is_decaying = yes
				}
				every_scope_character = {
					limit = {
						has_variable = house_tokugawa
					}
					add_modifier = {
						name = modifier_taikun_monarchy_character
						days = stupidly_long_modifier_time
					}
				}
			}
        }
    }

	invalid = {
		NOT = {
			country_has_monarchy_law = yes
		}
	}

	on_invalid = {
		set_variable = japan_restoration_complete
		if = {
			limit = {
				has_variable = emperor_var
			}
			remove_variable = emperor_var
		}
	}

	weight = 10000
	should_be_pinned_by_default_uninvolved_or_context = yes
	can_revolution_inherit = yes
}
