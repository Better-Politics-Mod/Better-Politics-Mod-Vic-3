bpm_revolutionary_zeal_value = {
	if = {
		limit = {
			scope:level ?= flag:lv1
		}
		value = {
			value = scope:faction.var:bpm_revolutionary_zeal
			multiply = 4
			max = 100
		}
	}
	else_if = {
		limit = {
			scope:level ?= flag:lv2
		}
		value = {
			value = scope:faction.var:bpm_revolutionary_zeal
			subtract = 25
			multiply = 4
			max = 100
		}
	}
	else_if = {
		limit = {
			scope:level ?= flag:lv3
		}
		value = {
			value = scope:faction.var:bpm_revolutionary_zeal
			subtract = 50
			multiply = 4
			max = 100
		}
	}
	else_if = {
		limit = {
			scope:level ?= flag:lv4
		}
		value = {
			value = scope:faction.var:bpm_revolutionary_zeal
			subtract = 75
			multiply = 4
			max = 100
		}
	}
	divide = 100
}

bpm_revolutionary_zeal_change_value = {
	if = {
		limit = {
			scope:faction.var:bpm_faction_type ?= flag:bpm_faction_the_republic_of_virtue
		}
		value = {
			value = -1
			desc = "BPM_BASE_VALUE"
		}
	}
}

bpm_overall_revolutionary_zeal = {
	every_in_list = {
		variable = bpm_revolution_victory_igs
		add = {
			value = ig_clout
			value = { every_in_list = {
				variable = bpm_postrev_factions
				add = var:bpm_revolutionary_zeal
			} }
			divide = {
				every_in_list = {
					variable = bpm_postrev_factions
					add = 1
				}
				min = 1
			}
		}
	}
	divide = { every_in_list = {
		variable = bpm_revolution_victory_igs
		add = 1
	} min = 1 }
}
