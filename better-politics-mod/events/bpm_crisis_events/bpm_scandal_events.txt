bpm_powerful_underworld_events.2 = { ### Powerful Underworld; Train Robbery
	type = country_event
	placement = scope:bpm_crisis_pu_targeted_state

	title = bpm_powerful_underworld_events.2.t
	desc = {
		triggered_desc = {
			desc = bpm_powerful_underworld_events.2.desc_citizens_target
			trigger = {
				has_variable = bpm_crisis_pu_citizens_targetted
			}
		}
		triggered_desc = {
			desc = bpm_powerful_underworld_events.2.desc_both_target
			trigger = {
				has_variable = bpm_crisis_pu_both_targetted
			}
		}
		triggered_desc = {
			desc = bpm_powerful_underworld_events.2.desc_cargo_target
			trigger = {
				has_variable = bpm_crisis_pu_cargo_targetted
			}
		}
		triggered_desc = {
			desc = bpm_powerful_underworld_events.2.desc_extremism_target
			trigger = {
				has_variable = bpm_crisis_pu_valuables_extremism_targeted
			}
		}
		triggered_desc = {
			desc = bpm_powerful_underworld_events.2.desc_civilians_info
			trigger = {
				always = yes
			}
		}
	}
	flavor = bpm_powerful_underworld_events.2.f

	event_image = {
		video = "unspecific_trains"
	}

	gui_window = event_window_superevent_modern_full

	on_created_soundeffect = "event:/SFX/UI/Alerts/Toasts/country_revolution"

	icon = "gfx/interface/icons/invention_icons/manufacturies.dds"

	duration = 1

	cooldown = {
		
	}

	dlc = dlc_bpm

	trigger = {
		any_scope_state = {
			has_building = building_railway
			has_variable = bpm_crisis_pu_affected_state
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				has_variable = bpm_crisis_pu_affected_state
			}
			save_scope_as = bpm_crisis_pu_targeted_state
		}
		random_list = {
			1 = {
				modifier = {
					add = bpm_underworld_industrial_wealthy_buildings
					divide = bpm_underworld_armament_and_bureaucracy
					divide = 2
				}
				set_variable = bpm_crisis_pu_citizens_targetted
			}	
			1 = {
				modifier = {
					add = bpm_underworld_industrial_wealthy_buildings
					divide = bpm_underworld_armament_and_bureaucracy
					divide = 1.5
				}
				set_variable = bpm_crisis_pu_both_targetted
			}
			1 = {
				modifier = {
					add = bpm_underworld_industrial_wealthy_buildings
					divide = bpm_underworld_armament_and_bureaucracy
					divide = 4
					if = {
						limit = {
							any_interest_group = {
								bpm_ig_is_socialist = yes
							}
						}
						every_interest_group = {
							limit = {
								bpm_ig_is_socialist = yes
							}
							add = ig_clout
							multiply = 2.5
						}
					}
					add = scope:bpm_crisis_pu_targeted_state.radical_fraction
				}
				set_variable = bpm_crisis_pu_valuables_extremism_targeted
			}	
			2 = {
				modifier = {
					if = {
						limit = {
							has_technology_researched = dynamite
						}
						add = 2
					}
				}
				set_variable = bpm_crisis_pu_cargo_targetted
			}	
		}
	}

	after = {
		remove_variable = bpm_crisis_pu_citizens_targetted
		remove_variable = bpm_crisis_pu_cargo_targetted
		remove_variable = bpm_crisis_pu_valuables_extremism_targeted
		remove_variable = bpm_crisis_pu_both_targetted
	}

	option = { 
		name = bpm_powerful_underworld_events.2.a

		scope:bpm_crisis_pu_targeted_state = {
			add_modifier = {
				name = bpm_crisis_pu_limited_travels
				years = 2
				is_decaying = yes
			}
		}
	}

	option = { 
		name = bpm_powerful_underworld_events.2.b

		scope:bpm_crisis_pu_targeted_state = {
			add_modifier = {
				name = bpm_crisis_pu_worried_no_invest
				is_decaying = yes
				years = 2
			}
		}
		ig:ig_industrialists = {
			add_modifier = {
				name = bpm_crisis_pu_money_cannot_flow
				is_decaying = yes
				years = 2
			}
		}

		if = {
			limit = {
				NOT = {
					has_technology_researched = steel_railway_cars
				}
			}
			add_technology_progress = {
				progress = 2000
				technology = steel_railway_cars
			}
		}
	}

	option = { 
		name = bpm_powerful_underworld_events.2.c

		scope:bpm_crisis_pu_targeted_state = {
			add_radicals_in_state = {
				value = 0.1
				strata = upper
			}
			add_radicals_in_state = {
				value = 0.05
				strata = middle
			}
			add_modifier = {
				name = bpm_crisis_pu_worried_no_invest
				is_decaying = yes
				years = 2
			}
		}

		bpm_remove_political_stability_small = yes
		custom_tooltip = {
			text = bpm_crisis_pu_10_percent_less
			set_variable = {
				name = bpm_crisis_pu_10_percent_less_var
				months = 6
			}
		}
	}

	option = { 
		name = bpm_powerful_underworld_events.2.d

		scope:bpm_crisis_pu_targeted_state = {
			add_radicals_in_state = {
				value = 0.075
				strata = middle
			}
			add_modifier = {
				name = bpm_crisis_pu_defence_against_criminals
				is_decaying = yes
				years = 2
			}
		}

		custom_tooltip = {
			text = bpm_crisis_pu_business_pool_events
			
		}

		custom_tooltip = {
			text = bpm_crisis_pu_city_fights_tt
			
		}
	}
}
