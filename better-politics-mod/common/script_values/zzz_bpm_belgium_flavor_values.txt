bpm_liberal_drift_away = {
	value = 0

	add = {
		desc = "BPM_VALUE_FROM_GOVERNMENT_IS_NOT_LIB_HOMGENOUS"
		if = {
			limit = {
				bpm_belgium_government_is_liberal_homogenous = no
				bpm_belgium_government_is_empty = no
			}
			add = 0.2
		}
		else_if = {
			limit = {
				bpm_belgium_government_is_liberal_homogenous = yes
				bpm_belgium_government_is_empty = no
			}
			add = 0.1
		}
	}

	add = {
		desc = "BPM_VALUE_FROM_LIB_APPROVAL_BELOW_0"
		if = {
			limit = {
				bpm_belgium_government_is_liberal_homogenous = no
				bpm_belgium_government_is_empty = no
				ig:ig_liberals ?= {
					ig_approval < 0
				}
			}
			subtract = {
				value = ROOT.ig:ig_liberals.ig_clout
				multiply = ROOT.ig:ig_liberals.ig_approval
				divide = 7.77
			}
		}
	}

	subtract = {
		desc = "BPM_VALUE_FROM_UNIONISM_STILL_NEEDED"
		if = {
			limit = {
				NAND = {
					has_global_variable = london_treaty_has_concluded_global_var
					has_global_variable = london_treaty_signed_bel_global_var
					has_global_variable = london_treaty_signed_net_global_var
				}
			}
			add = {
				value = ROOT.ig:ig_liberals.ig_clout
				multiply = 1.33
			}
		}
	}
}

bpm_france_orleanist_igs = {
	value = 0

	every_interest_group = {
		limit = {
			OR = {
				leader = {
					has_ideology = ideology:ideology_orleanist
				}
				AND = {
					is_interest_group_type = ig_liberals
					NOR = {
						leader = {
							OR = {
								has_ideology = ideology:ideology_radical
								has_ideology = ideology:ideology_republican
							}
						}
						has_ideology = ideology:ideology_gov_liberal_republican
						has_ideology = ideology:ideology_gov_radical_republican
					}
				}	
			}
		}
		add = ig_clout
	}
}
