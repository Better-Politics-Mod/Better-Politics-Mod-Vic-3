# Tenpo - Hardliner: Liberals Marginalized
tenpo_checkbox_liberals_marginalized = {
	type = country
	text = {
		trigger = {
            any_interest_group = {
                bpm_ig_is_liberal = yes
                bpm_ig_is_marginal = no
            }
		}
		localization_key = checkbox_unticked
	}
	text = {
		trigger = {
			NOT = {
                any_interest_group = {
                    bpm_ig_is_liberal = yes
                    bpm_ig_is_marginal = no
                }
            }
		}
		localization_key = checkbox_ticked
	}
}

# Tenpo - Reformer: Liberals In Government
tenpo_checkbox_government = {
	type = country
	text = {
		trigger = {
			NOT = {
                any_interest_group = {
                    bpm_ig_is_liberal = yes
                    is_in_government = yes
                }
            }
		}
		localization_key = checkbox_unticked
	}
	text = {
		trigger = {
			any_interest_group = {
                bpm_ig_is_liberal = yes
                is_in_government = yes
            }
		}
		localization_key = checkbox_ticked
	}
}
