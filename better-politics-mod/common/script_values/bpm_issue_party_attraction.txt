#ISSUEREG

bpm_issue_attraction = {
    if = {
        limit = {
            this = flag:bpm_issue_sol
        }
        value = bpm_issue_attraction_issue_sol
    }
    else_if = {
        limit = {
            this = flag:bpm_issue_anti_war
        }
        value = bpm_issue_attraction_issue_anti_war
    }
    else_if = {
        limit = {
            this = flag:bpm_issue_anti_liberal
        }
        value = bpm_issue_attraction_issue_anti_liberal
    }
    else_if = {
        limit = {
            this = flag:bpm_issue_trade_deficit
        }
        value = bpm_issue_attraction_issue_trade_deficit
    }
    multiply = bpm_issue_interest
}

bpm_issue_attraction_issue_sol = {
    value = 0
    scope:checking_party = {
        every_member = {
            limit = {
                OR = {
                    bpm_ig_is_paternalistic = yes
                    bpm_ig_is_socialist = yes
                }
            }
            add = 1
        }
    }
}

bpm_issue_attraction_issue_anti_war = {
    value = 0
    scope:checking_party = {
        every_member = {
            limit = {
                OR = {
                    is_interest_group_type = ig_market_liberals
                    is_interest_group_type = ig_rural_folk
                    is_interest_group_type = ig_intelligentsia
                    is_interest_group_type = ig_liberals
                    bpm_ig_is_socialist = yes
                }
            }
            add = 1
        }
    }
    if = {
        limit = {
            scope:country = {
                is_at_war = no
            }
        }
        add = -999
    }
}

bpm_issue_attraction_issue_anti_liberal = {
    value = 0
    scope:checking_party = {
        every_member = {
            limit = {
                OR = {
                    is_interest_group_type = ig_reactionaries
                    bpm_ig_is_monarchist = yes
                }
            }
            add = 1
        }
    }
    if = {
        limit = {
            scope:checking_party = scope:country.py:liberal_party
        }
        add = -999
    }
}

bpm_issue_attraction_issue_trade_deficit = {
    value = 0
    scope:checking_party = {
        every_member = {
            limit = {
                law_stance = {
                    law = law_type:law_free_trade
                    value < neutral
                }
            }
            add = 1
        }
    }
    if = {
        limit = {
            scope:checking_party = scope:country.py:free_trade_party
        }
        add = -999
    }
}
