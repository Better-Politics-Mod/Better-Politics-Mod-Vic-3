ideology_moderate_valid_trigger = {
	OR = {
		bpm_ig_is_institutional = yes
		NOR = {
			bpm_ig_is_socialist = yes
			bpm_ig_is_traditionalist = yes
			bpm_ig_is_fascist = yes
		}
		is_interest_group_type = ig_christian_democrats
	}

	NAND = {
		is_interest_group_type = ig_intelligentsia
		modifier:interest_group_bpm_intelligentsia_force_leader_non_moderate = yes # Forced to have opinions
	}
}

REPLACE:ideology_traditionalist_valid_trigger = {
	bpm_interest_group_is_activated = yes
	bpm_ig_is_liberal = no
	bpm_ig_is_proletarian = no
	NOR = {
		has_ideology = ideology:ideology_rel_radical_secularist
		has_ideology = ideology:ideology_rel_secularist
		has_ideology = ideology:ideology_rel_atheist
		has_ideology = ideology:ideology_rel_pragmatic_moralist
	}
	trigger_if = {
		limit = {
			owner = {
				c:BIC ?= owner
			}
		}			
		NOR = {
			is_interest_group_type = ig_trade_unions
			is_interest_group_type = ig_rural_folk
			is_interest_group_type = ig_petty_bourgeoisie
			is_interest_group_type = ig_landowners
			is_interest_group_type = ig_devout
			is_interest_group_type = ig_intelligentsia
		}
	}
	trigger_if = {
		limit = {
			owner = {
				NOR = {
					has_law = law_type:law_state_atheism
					country_has_state_religion = rel:atheist
				}
			}
		}
		leader = { religion ?= owner.religion }
	}

}
