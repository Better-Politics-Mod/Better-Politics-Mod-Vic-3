# The Outlaw Prince
ethiopia.1 = {
	type = country_event
	placement = ROOT

	title = ethiopia.1.t
	desc = ethiopia.1.d
	flavor = ethiopia.1.f

	event_image = {
		video = "africa_soldiers_breaking_protest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
		NOT = { has_global_variable = tewodros_var }
		# country_has_primary_culture = cu:amhara
		has_law = law_type:law_monarchy
		this ?= c:BGM
	}

	immediate = {
		set_global_variable = tewodros_var
	}

	option = {
		name = ethiopia.1.a
		default_option = yes
		create_character = {
			first_name = Tewodros
			last_name = Solomon
			historical = yes
			ruler = yes
			noble = yes
			age = 45
			dna = dna_emperor_tewodros_02
			interest_group = ig_armed_forces
			culture = cu:amhara
			religion = rel:oriental_orthodox
			ideology = ideology_abolitionist
			traits = {
				romantic
				expert_offensive_planner
			}
			on_created = {
				add_modifier = {
					name = tewodros_modifier
				}
			}
		}
	}
	option = {
		name = ethiopia.1.b
		# remove_global_variable = tewodros_var
		add_radicals = {
			value = small_radicals
		}
	}
}
