je_march_to_radicalism = {
	icon = "gfx/interface/icons/event_icons/phrygian.dds"

	group = bpm_je_group_global

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_march_to_radicalism_awaiting_spark_status
				trigger = {
					global_var:bpm_je_march_to_radicalism_progress >= 100
				}
			}
			triggered_desc = {
				desc = je_march_to_radicalism_status
				trigger = {
					always = yes
				}
			}
		}
	}


	is_shown_when_inactive = {
		hidden_trigger = {
			has_global_variable = bpm_je_march_to_radicalism_progress
			NOT = { has_global_variable = bpm_peoples_springtime_happened }
			NOT = { has_global_variable = bpm_peoples_springtime_happening }
		}
	}

	should_be_involved = {
		bpm_country_is_in_europe = yes
	}

	#change this for the compatch
	possible = {
		hidden_trigger = {
			has_global_variable = bpm_je_march_to_radicalism_progress
			NOT = { has_global_variable = bpm_peoples_springtime_happened }
			NOT = { has_global_variable = bpm_peoples_springtime_happening }
		}
	}

	should_show_when_not_involved = {
		bpm_country_is_in_europe = yes
		#OR = {
		#	#is_in_geographic_region = geographic_region_europe
		#	#country_has_interest_marker_in_europe = yes
        #    #top_overlord ?= {
        #    #    capital = {
        #    #        is_in_geographic_region = geographic_region_europe
        #    #    }
        #    #}
		#}
	}

	invalid = {
		hidden_trigger = {
			OR = {
				has_variable = bpm_march_to_radicalism_complete_local_var
				has_global_variable = bpm_peoples_springtime_happened
			}
		} 
		# The people have already sprung
	}

	immediate = {
		
	}

	complete = {
		AND = {
			custom_tooltip = {
				text = "bpm_je_progress_complete"
				has_global_variable = bpm_je_march_to_radicalism_progress
				global_var:bpm_je_march_to_radicalism_progress >= 120
			}
			custom_tooltip = {
				text = "bpm_peoples_springtime_spark_happened"
				has_global_variable = bpm_peoples_springtime_happening
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			scope:journal_entry = {
				every_scope_je_involved = {
					if = {
						limit = {
							bpm_country_is_in_europe = yes
							NOT = { this = c:RUS }
						}
						hidden_effect = {
							remove_modifier = bpm_pre_peoples_springtime
						}
						add_modifier = {
							name = bpm_pre_peoples_springtime
							multiplier = {
								value = global_var:bpm_je_march_to_radicalism_progress
								divide = 100
								max = 1
							}
							months = 24
						}
						if = {
							limit = {
								global_var:bpm_je_march_to_radicalism_progress >= 100
								bpm_country_laws_ok_voting = no
								NOR = {
									this = c:GBR
								}
								NOT = { has_global_variable = bpm_march_to_radicalism_complete_var }
							}
							add_radicals = {
								value = small_radicals
							}
						}
					}
				}
			}
			if = {
				NOT = { has_global_variable = bpm_peoples_springtime_happening }
				NOT = { has_global_variable = bpm_peoples_springtime_happened }
            	limit = {
               		any_country = {
                    	bpm_country_is_in_europe = yes
						has_revolution = yes
                    }
                }
				set_global_variable = peoples_springtime_happening
            }   
		}
	}

	progressbar = yes
	current_value = {
		value = global_var:bpm_je_march_to_radicalism_progress
		max = 120
	}

	goal_add_value = {
		add = 120
		max = 120
	}

	on_complete_all_involved = {
		set_global_variable = {
			name = bpm_march_to_radicalism_complete_var
		}
		set_variable = {
			name = bpm_march_to_radicalism_complete_local_var
		}
		custom_tooltip = "bpm_activate_je_springtimes"
	}

	on_complete = {
		hidden_effect = {
			add_contextless_journal_entry = je_springtime_of_the_peoples
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes
}

REPLACE_OR_CREATE:je_springtime_of_the_peoples = {
	icon = "gfx/interface/icons/event_icons/48je.dds"

	group = bpm_je_group_global

	should_be_involved = {
		bpm_country_is_in_europe = yes
	}

	immediate = {
		add_contextless_journal_entry = je_frankfurt_parliament
	}

	immediate_all_involved = {
		trigger_event = { id = 1848.4 }
		if = {
			limit = {
				NOT = {
					any_political_movement = {
						OR = {
							is_political_movement_type = movement_radical
							is_political_movement_type = movement_german_national
							is_political_movement_type = movement_italian_national
						}
					}
				}
			}
			if = {
				limit = {
					OR = {
						country_has_primary_culture = cu:north_german
						country_has_primary_culture = cu:south_german
					}
				}
				create_political_movement = { type = movement_german_national }
			}
			else_if = {
				limit = {
					OR = {
						country_has_primary_culture = cu:north_italian
						country_has_primary_culture = cu:south_italian
					}
				}
				create_political_movement = { type = movement_italian_national }
			}
			else = {
				create_political_movement = { type = movement_radical }
			}
		}
		if = {
			limit = {
				exists = c:PRU
				this = c:PRU
			}
			create_character = {
				culture = cu:north_german
				female = no
				ideology = ideology_radical
				interest_group = ig_radicals
				is_agitator = yes 
				traits = {
					inspirational_orator
					ambitious
					firebrand
				}
			}
		}
		start_national_awakening = {
			culture = cu:north_italian
			months = 48
			state_region = s:STATE_VENETIA
		}
		add_modifier = {
			name = bpm_peoples_springtime_modifier
			months = 26
		}
		every_interest_group = {
			limit = {
				bpm_ig_is_ideological = yes
			}
			change_variable = {
				name = bpm_ig_organization
				add = 20
			}
		}
	}

	on_timeout_all_involved = {
		set_variable = completed_peoples_springtime_je
		remove_global_variable = bpm_peoples_springtime_happening
		set_global_variable = bpm_peoples_springtime_happened
		if = {
			limit = { NOT = { has_global_variable = recently_completed_peoples_springtime_je } }
			set_global_variable = {
				name = recently_completed_peoples_springtime_je
				months = 24
			}
		}
		custom_tooltip = je_springtime_of_the_peoples_tt_3
		hidden_effect = {
			remove_modifier = bpm_pre_peoples_springtime
			remove_modifier = bpm_peoples_springtime_modifier
			every_interest_group = {
				remove_modifier = bpm_1848_stubborn_ig
			}
			if = {
				limit = { is_player = no }
				add_loyalists = {
					value = 0.5
					strata = upper
				}
				add_loyalists = {
					value = 0.3
					strata = middle
				}
				add_loyalists = {
					value = 0.2
					strata = lower
				}
			}
		}
		custom_tooltip = je_springtime_of_the_peoples_tt_4
	}

	on_weekly_pulse = {
		events = {
			peoples_springtime.100
		}
	}

	on_monthly_pulse = {
		events = {
			1848.5
			bpm_radlib_utils.5
			bpm_radlib_utils.6
		}
		random_events = {
			100 = 0
			10 = peoples_springtime.3
			10 = peoples_springtime.4
			10 = peoples_springtime.5
		}
		effect = {
			scope:journal_entry = {
				every_scope_je_involved = {
					bpm_setup_country_stubbornness = yes
				}
			}
		}
	}

	invalid = {
		has_variable = completed_peoples_springtime_je
	}

	weight = 10000
	should_be_pinned_by_default = yes

	transferable = no
	can_revolution_inherit = yes

	timeout = 730
}

je_frankfurt_parliament = {
	icon = "gfx/interface/icons/event_icons/germany.dds"

	group = je_group_global_international_situations

	should_be_involved = {
		country_is_in_german_confederation = yes
	}

	should_show_when_not_involved = {
		OR = {
			country_is_in_german_confederation = yes
            top_overlord ?= {
                country_is_in_german_confederation = yes
            }
		}
	}

	possible = {
		any_country = {
			has_global_variable = bpm_march_to_radicalism_complete_var
		}
	}

	should_be_pinned_by_default = yes
	weight = 10000


	is_shown_when_inactive = {
		NOT = {
			has_global_variable = grunderzeit_happened_global_var
		}
	}

	modifiers_while_active = {
		
	}

	immediate = {
		if = {
			limit = {
				any_country = {
					is_player = yes
					country_is_in_german_confederation = yes
				}
			}
			every_country = {
				limit = {
					is_player = yes
					country_is_in_german_confederation = yes
				}
				set_variable = bpm_frankfurt_host_tag
			}
			
		}
		else = {
			c:PRU = {
				set_variable = bpm_frankfurt_host_tag
			}
		}
		set_global_variable = {
			name = bpm_frankfurt_progress
			value = 0
		}
	}

	on_weekly_pulse = {
		effect = {
			change_global_variable = {
				name = bpm_frankfurt_progress
				add = 10
			}
		}
		events = {
			bpm_frankfurt_parliament_events.1
			bpm_frankfurt_parliament_events.2
			bpm_frankfurt_parliament_events.3
			bpm_frankfurt_parliament_events.4
			bpm_frankfurt_parliament_events.5
			bpm_frankfurt_parliament_events.6
			bpm_frankfurt_parliament_events.7
		}
	}


	immediate_all_involved = {

	}

	current_value = {
        value = global_var:bpm_frankfurt_progress
		max = 100
    }

	goal_add_value = {
		add = 100
		max = 100
	}

    progressbar = yes


}
