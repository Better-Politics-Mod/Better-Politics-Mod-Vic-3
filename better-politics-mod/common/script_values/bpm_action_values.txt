bpm_revolt_progress_value_polmov = {
	save_temporary_scope_as = movement
	owner = {
		value = bpm_revolt_progress_value
		divide = 100
	}
}

bpm_number_of_states_with_cells = {
	save_temporary_scope_as = movement
	owner = {
		value = 0
		every_scope_state = {
			limit = {
				bpm_has_revcell = yes
			}
			add = 1
		}
	}
}

bpm_revolt_progress_value = {
	value = 0
	every_scope_state = {
		limit = {
			bpm_has_revcell = yes
		}
		value = {
			value = 0
			if = {
				limit = {
					bpm_has_revcell_level = { lvl = 1 }
				}
				add = 1
			}
			else_if = {
				limit = {
					bpm_has_revcell_level = { lvl = 2 }
				}
				add = 3
			}
			else_if = {
				limit = {
					bpm_has_revcell_level = { lvl = 3 }
				}
				add = 10
			}
			multiply = bpm_population_fraction
		}
	}
}

bpm_population_fraction = {
	value = state_population
	divide = owner.total_population
}
