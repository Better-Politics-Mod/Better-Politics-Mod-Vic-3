bpm_land_focus_effect = {
	bpm_land_focus_effect_economy = yes
}

bpm_land_focus_effect_economy = {
	if = {
		limit = {
			has_variable = bpm_land_focus_result_economy
		}
		bpm_land_focus_effect_economy_success = yes
	}
	else = {
		bpm_land_focus_effect_economy_failure = yes
	}
}

# LAND FOCUS

## Success

bpm_land_focus_effect_economy_success = {
	if = {
		limit = {
			has_variable = bpm_minister_of_economy
		}
		var:bpm_minister_of_economy = {
			switch = {
				trigger = is_interest_group_type
				ig_petty_bourgeoisie = {
					owner = {
						bpm_land_focus_effect_economy_success_bureaucrats = yes
					}
				}
			}
		}
	}
}

bpm_land_focus_effect_economy_success_bureaucrats = {
	if = {
		limit = {
			has_law = law_type:law_council_republic
		}
		bpm_add_law_enactment_boost_specified = { law = collectivized_agriculture }
	}
	else_if = {
		limit = {
			has_law = law_type:law_serfdom
		}
		bpm_add_law_enactment_boost_specified = { law = tenant_farmers }
	}
	else_if = {
		limit = {
			has_law = law_type:law_tenant_farmers
		}
		bpm_add_law_enactment_boost_specified = { law = commercialized_agriculture }
	}
	else = {
		add_modifier = {
			name = bpm_ca_economic_bureaucrat_success_no_aris
			days = 9125
			is_decaying = yes
		}
	}
	add_modifier = {
		name = bpm_ca_economic_bureaucrat_success_bureaustr
		days = 9125
		is_decaying = yes
	}
	bpm_auth_add_exhaustion_normal = yes
}

## Failure

bpm_land_focus_effect_economy_failure = {
	if = {
		limit = {
			has_variable = bpm_minister_of_economy
		}
		var:bpm_minister_of_economy = {
			switch = {
				trigger = is_interest_group_type
				ig_petty_bourgeoisie = {
					owner = {
						bpm_land_focus_effect_economy_failure_bureaucrats = yes
					}
				}
			}
		}
	}
	bpm_auth_add_exhaustion_large = yes
}

bpm_land_focus_effect_economy_failure_bureaucrats = {
	add_modifier = {
		name = bpm_ca_economic_bureaucrat_failure_yesaris
		days = 9125
		is_decaying = yes
	}
}
