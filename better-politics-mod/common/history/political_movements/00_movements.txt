	every_country = {
		limit = { NOT = { is_country_type = decentralized } }
		# if = {
		# 	limit = {
		# 		bpm_country_is_pre_modern = yes
		# 	}
		# 	create_political_movement = { type = movement_modernizer }
		# }
		bpm_create_necessary_movements = yes
		if = {
			limit = {
				has_law = law_type:law_monarchy
				country_has_voting_franchise = no
				NOT = { c:FRA ?= this } # No generic royalist movement for France
				NOT = { c:POR ?= this } # No absolutists for Portugal
				NOR = {
					c:SPA ?= this # No absolutists for the Spains
					c:SPC ?= this
				}
			}
			create_political_movement = { type = movement_royalist_absolutist }
		}
		else_if = {
			limit = {
				has_law = law_type:law_monarchy
				OR = {
					country_has_voting_franchise = yes
					c:POR ?= this
				}
				NOT = { c:FRA ?= this } # No generic royalist movement for France
			}
			create_political_movement = { type = movement_royalist_constitutional }
		}
		if = {
			limit = {
				NOT = { has_law = law_type:law_slavery_banned }
			}
			if = {
				limit = {
					NOT = {
						has_law_or_variant = law_type:law_colonial_slavery
					}
				}
				create_political_movement = { type = movement_pro_slavery }
			}
			
			if = {
				limit = { has_technology_researched = empiricism }
				create_political_movement = { type = movement_anti_slavery }
			}
		}		
		if = {
			limit = {
				c:BIC ?= this
			}
			create_political_movement = { type = movement_religious_minority religion = rel:hindu }
			create_political_movement = { type = movement_religious_minority religion = rel:sunni }
			create_political_movement = { type = movement_cultural_minority culture = cu:bengali }
		}
		else_if = { # France gets Legitimists instead, Spain gets Carlists, Portugal gets Miguelists
			limit = {
				NOR = {
					c:FRA ?= this
					c:SPA ?= this
					c:SPC ?= this
					c:POR ?= this
				}
			}
			create_political_movement = { type = movement_religious_majority }
		}
		# if = {
		# 	limit = {
		# 		NOR = { # Will not spawn if you're an un-unified German or Italian state
		# 			AND = {
		# 				OR = {
		# 					country_has_primary_culture = cu:north_german
		# 					country_has_primary_culture = cu:south_german
		# 				}
		# 				NOR = {
		# 					c:GER ?= this
		# 					c:AUS ?= this
		# 					c:TRS ?= this
		# 					c:KUK ?= this
		# 					c:HRE ?= this
		# 				}
		# 			}
		# 			AND = {
		# 				OR = {
		# 					country_has_primary_culture = cu:north_italian
		# 					country_has_primary_culture = cu:south_italian
		# 				}
		# 				NOR = {
		# 					c:ITA ?= this
		# 					c:AUS ?= this
		# 					c:KUK ?= this
		# 					c:HRE ?= this
		# 				}
		# 			}
		# 		}
		# 	}
		# 	create_political_movement = { type = movement_cultural_majority }
		# }
		
		if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:north_german
					country_has_primary_culture = cu:south_german
				}
			}
			create_political_movement = { type = movement_german_national }
		}
		else_if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:north_italian
					country_has_primary_culture = cu:south_italian
				}
			}
			create_political_movement = { type = movement_italian_national }
		}
		else = {
			create_political_movement = { type = movement_cultural_majority }
		}
	}	
	c:GBR ?= {
		create_political_movement = { type = movement_labor }
		create_political_movement = { type = movement_cultural_minority culture = cu:irish }
	}
	# brewing rebellions of 1837
	c:ONT ?= {
		create_political_movement = { type = movement_radical }
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_radical
			}
			add_modifier = {
				name = can_reformer_movement_fury
				days = very_long_modifier_time
			}
		}
	}
	c:QUE ?= {
		create_political_movement = { type = movement_radical }
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_radical
			}
			add_modifier = {
				name = can_reformer_movement_fury
				days = very_long_modifier_time
			}
		}
	}

	}

		create_political_movement = { type = movement_legitimist }
		create_political_movement = { type = movement_bonapartist }

	}

		create_political_movement = { type = movement_carlist }
		create_political_movement = { type = movement_liberal }

		random_political_movement = {
			limit = {
				is_political_movement_type = movement_liberal
			}
			set_suppression = yes

			limit = {
				is_political_movement_type = movement_miguelist

		create_political_movement = { type = movement_cultural_minority culture = cu:czech }
		create_political_movement = { type = movement_cultural_minority culture = cu:north_italian }
		create_political_movement = { type = movement_liberal }
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_liberal
			}
			set_suppression = yes
		}

			remove_modifier = initial_movement_enthusiasm
		}
	}
