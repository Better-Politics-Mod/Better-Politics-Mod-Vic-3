bpm_expected_cabpos_base = {
	if = {
		limit = {
			owner = {
				OR = {
					bpm_country_has_parliamentary_electoral_system = yes
					bpm_country_has_executive_electoral_system = yes
				}
			}
			is_in_government = yes
		}
		add = owner.bpm_total_num_positions
		multiply = {
			value = ig_clout
			divide = {
				owner = {
					value = 0
					every_interest_group = {
						limit = {
							is_in_government = yes
						}
						add = ig_clout
					}
					min = 0.001
				}
			}
		}
		floor = yes
	}
	else_if = {
		limit = {
			owner = {
				OR = {
					bpm_country_has_parliamentary_oligarchic_system = yes
				}
			}
			bpm_ig_is_state_institutional = yes
		}
		add = owner.bpm_total_num_positions
		multiply = {
			value = ig_clout
			divide = {
				owner = {
					value = 0
					every_interest_group = {
						limit = {
							bpm_ig_is_state_institutional = yes
						}
						add = ig_clout
					}
					min = 0.001
				}
			}
		}
	}
}

bpm_expected_cabpos = {
	if = {
		limit = {
			has_variable = bpm_expected_cabpos
		}
		value = var:bpm_expected_cabpos
	}
	else = {
		value = bpm_expected_cabpos_base
	}
}

bpm_actual_cabpos = {
	if = {
		limit = {
			has_variable = bpm_actual_cabpos
		}
		value = var:bpm_actual_cabpos
	}
	else = {
		value = 0
	}
}

bpm_diff_from_expectation = {
	value = bpm_actual_cabpos
	subtract = bpm_expected_cabpos
}

bpm_jealousy_cabpos = {
	value = 0
	save_temporary_scope_as = iginq
	owner = { every_interest_group = {
		limit = {
			NOT = {
				THIS = scope:iginq
			}
		}
		if = {
			limit = {
				bpm_diff_from_expectation > 0	
			}
			add = bpm_diff_from_expectation
		}
	} }
	multiply = -2
}

bpm_expectation_cabpos = {
	value = 0
	add = {	
		value = bpm_diff_from_expectation
		max = 0
	}
	multiply = 3
}
