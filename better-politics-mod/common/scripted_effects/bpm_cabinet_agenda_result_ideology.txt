bpm_ideology_effect_education_devout = {
	if = {
		limit = {
			has_law = law_type:law_religious_schools
		}
		add_modifier = {
			name = bpm_ideology_education_devout_success
		}
		
		random_list = {
			5 = {
				random_scope_pop = {
					limit = {
						NOT = {
							religion = {
								has_discrimination_trait = state_religion
							}
						}
					}
					religion = {
						save_scope_as = majoritarian_religion
					}
				}
				change_pop_religion = {
					target = scope:majoritarian_religion
					value = 0.05
				}
			}
			95 = {}
		}
	}
}
