### Springtime ###
je_vormaerz = {
	icon = "gfx/interface/icons/notification_icons/secession.dds"

	group = je_group_historical_content

    modifiers_while_active = {
        vormaerz_calm_before_the_storm_modifier
    }

    is_shown_when_inactive = {
        OR = {
            country_has_primary_culture = cu:north_german
            country_has_primary_culture = cu:south_german
        }
        NOR = {
            this ?= c:LUX
            this ?= c:TRS
        }
    }

    possible = {
        OR = {
            country_has_primary_culture = cu:north_german
            country_has_primary_culture = cu:south_german
        }
        NOR = {
            this ?= c:LUX
            this ?= c:TRS
        }
    }

	invalid = {
		hidden_trigger = {
			OR = {
				has_variable = bpm_march_to_radicalism_complete_local_var
				has_global_variable = bpm_peoples_springtime_happened
			}
		} 
		# The people have already sprung
	}
	
	complete = {
		custom_tooltip = {
			text = "bpm_peoples_springtime_spark_happened"
            has_journal_entry = je_springtime_of_the_peoples
            #has_global_variable = peoples_springtime_happening
		}
	}

    on_complete = {
        c:PRU = { trigger_event = { id = robert_blum.1 popup = yes } } 
    }

	should_be_pinned_by_default = yes

	on_monthly_pulse = {
		events = {
           vormaerz.1 # Code Civil
           vormaerz.2 # Nationalism
           vormaerz.3 # Sports Organisations
           vormaerz.4 # Poverty and famine
           vormaerz.5 # Communist Manifesto
		}
    
       effect = {
           if = {
                limit = {
                    has_technology_researched = mass_communication
                    has_technology_researched = international_relations
                    NOT = { has_technology_researched = nationalism }
                }
                add_technology_progress = { progress = 250 technology = nationalism }
            }
            if = {
               limit = {
                    has_technology_researched = nationalism
                    NOT = { has_technology_researched = organized_sports }
                }
                add_technology_progress = { progress = 250 technology = organized_sports }
            }
        }
    }

	weight = 100
}
