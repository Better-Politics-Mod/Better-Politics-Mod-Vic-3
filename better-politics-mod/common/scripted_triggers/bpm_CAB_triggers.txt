# -- Cabinet Register Institutions (Antagonism) --

bpm_is_antagonistic = {
    trigger_if = {
        limit = {
            NOT = { has_variable = bpm_cabinet_minister }
        }
        always = no
    }
    trigger_else_if = {
        limit = {
            owner = { has_institution = institution_colonial_affairs }
            var:bpm_cabinet_minister.type ?= owner.institution:institution_colonial_affairs.type
        }
        bpm_is_antagonistic_colonial_affairs = yes
    }
    trigger_else_if = {
        limit = {
            owner = { has_institution = institution_social_security }
            var:bpm_cabinet_minister.type ?= owner.institution:institution_social_security.type
        }
        bpm_is_antagonistic_social_security = yes
    }
    trigger_else = {
        always = no
    }
}

bpm_is_antagonistic_hypothetical = {
    trigger_if = {
        limit = {
            NOT = { owner = { has_variable = bpm_cabinet_char_select} }
        }
        always = no
    }
    trigger_if = {
        limit = {
            owner = { has_institution = institution_colonial_affairs }
            owner.var:bpm_cabinet_char_select.type ?= owner.institution:institution_colonial_affairs.type
        }
        bpm_is_antagonistic_colonial_affairs = yes
    }
    trigger_else_if = {
        limit = {
            owner = { has_institution = institution_social_security }
            owner.var:bpm_cabinet_char_select.type ?= owner.institution:institution_social_security.type
        }
        bpm_is_antagonistic_social_security = yes
    }
    trigger_else = {
        always = no
    }
}

bpm_is_antagonistic_schools = {
    always = no
}

bpm_is_antagonistic_colonial_affairs = {
    interest_group = {
        # Interest groups that are antagonistic to the colonial affairs institution
        OR = {
            is_interest_group_type = ig_agrarian_populists
            is_interest_group_type = ig_anarchists
            is_interest_group_type = ig_revolutionist_socialists
            is_interest_group_type = ig_socialists
            is_interest_group_type = ig_rural_folk
            is_interest_group_type = ig_trade_unions
        }
    }
}

bpm_is_antagonistic_social_security = {
    interest_group = {
        # Interest groups that are antagonistic to the social security institution
        OR = {
            is_interest_group_type = ig_fascists
            is_interest_group_type = ig_reactionaries
        }
    }
}
