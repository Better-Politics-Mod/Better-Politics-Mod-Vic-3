namespace = bpm_marlib_split

bpm_marlib_split.1 = {
	type = country_event
	placement = root

	event_image = { video = "gfx/event_pictures/europenorthamerica_capitalists_meeting.bk2" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	title = bpm_marlib_split.1.t
	desc = bpm_marlib_split.1.d
	flavor = bpm_marlib_split.1.f
	duration = 3

	immediate = {
		set_variable = local_marlib_split_var
		bpm_initialize_marlibs = yes
	}

	option = {
		name = bpm_marlib_split.1.a
		default_option = yes
		hidden_effect = {
			ig:ig_market_liberals = { leader = { kill_character = yes } }
		}
	}
}

bpm_marlib_split.2 = { #Brazil Market Liberals
	type = country_event
	placement = root

	event_image = { video = "gfx/event_pictures/europenorthamerica_capitalists_meeting.bk2" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	title = bpm_marlib_split.1.t
	desc = bpm_marlib_split.1.d
	flavor = bpm_marlib_split.1.f
	duration = 3

	immediate = {
		set_variable = local_marlib_split_var

		ig:ig_market_liberals = {
			remove_ideology = ideology_nonexistent
			if = {
				limit = { ROOT = { bpm_country_is_republic = yes } }
				add_ideology = ideology_gov_liberal_republican
			}
			else = { add_ideology = ideology_gov_moderate_monarchist }
			add_ideology = ideology_dop_moderate_democrat
			add_ideology = ideology_sta_progressive_state
			add_ideology = ideology_cit_liberal_nationalist
			add_ideology = ideology_rel_moralist
			add_ideology = ideology_mil_patriotic
			add_ideology = ideology_eco_laissez_faire
			add_ideology = ideology_wel_market_welfare
			add_ideology = ideology_fem_benevolent_sexism
		}

		ig:ig_liberals = {
			set_interest_group_name = ig_moderate_liberals
		}
	}

	option = {
		name = bpm_marlib_split.1.a
		default_option = yes
		hidden_effect = {
			ig:ig_market_liberals = { leader = { kill_character = yes } }
		}
	}
}
