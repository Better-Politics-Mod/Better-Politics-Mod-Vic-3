bpm_auth_force_law_through_gui = {
	scope = country

	is_valid = {
		enacting_any_law = yes
		enactment_phase < 2
		bpm_auth_authority_and_recent_exhaustion >= 500
	}

	effect = {
		bpm_auth_add_exhaustion_huge = yes
		add_enactment_phase = yes
	}
}

bpm_auth_state_capacity_level = {
	scope = country
	saved_scopes = {
		level
	}

	is_shown = {
		trigger_if = {
			limit = {
				scope:level = flag:lv1
			}
			bpm_low_state_capacity = yes
		}
		trigger_if = {
			limit = {
				scope:level = flag:lv2
			}
			bpm_medium_state_capacity = yes
		}
		trigger_if = {
			limit = {
				scope:level = flag:lv3
			}
			bpm_high_state_capacity = yes
		}
	}
}
