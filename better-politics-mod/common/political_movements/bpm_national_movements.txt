movement_hindu_nationalist = {
	category = movement_category_religious
	
	ideology = ideology_traditionalist_minoritarian
	character_ideologies = {
		ideology_fascist
		ideology_theocrat
	}		
	
	creation_trigger = {
		OR = {
			AND = {
				c:BIC ?= this
				has_technology_researched = pan-nationalism
				any_scope_state = {
					state_is_homeland_of_indian_cultures = yes
					any_scope_pop = {
						religion = rel:hindu
					}
				}
			}
			AND = {
				c:MUG ?= this
				has_technology_researched = pan-nationalism
				any_scope_state = {
					state_is_homeland_of_indian_cultures = yes
					any_scope_pop = {
						religion = rel:hindu
					}
					count >= 5
				}
			}
		}
	}
	creation_weight = {
		value = 100
	}		
	on_created = {}
	
	religion_selection_trigger = {
		this = rel:hindu
	}
	religion_selection_weight = {
		value = 100
	}	
	
	character_support_trigger = {
		culture = {
			has_discrimination_trait = south_asian_heritage
		}
		religion = rel:hindu
	}
	character_support_weight = {
		value = 100
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_traditionalist_minoritarian
					has_ideology = ideology:ideology_fascist
					has_ideology = ideology:ideology_theocrat
				}
			}
			multiply = {
				value = 5
			}
		}
	}	
	
	pop_support_trigger = {
		culture = {
			has_discrimination_trait = south_asian_heritage
		}
		religion = rel:hindu
	}

	pop_support_factors = {
		movement_support_hindu_religion
		movement_support_low_literacy
		movement_support_academics
		movement_support_clergymen
		movement_support_bureaucrats
		movement_support_clerks
		movement_support_corporatism_tech
		movement_support_je_communal_divides
	}
	
	pop_support_weight = {
		if = {
			limit = {
				religion = rel:hindu
			}
			add = {
				value = 9
				desc = "POP_HINDU"
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 12
					desc = "POP_ACADEMICS"
				}
			}
			if = {
				limit = {
					is_pop_type = clergymen
				}
				add = {
					value = 25
					desc = "POP_CLERGYMEN"
				}
			}
			if = {
				limit = {
					is_pop_type = bureaucrats
				}
				add = {
					value = 9
					desc = "POP_BUREAUCRATS"
				}
			}		
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 12
					desc = "POP_CLERKS"
				}
			}
			if = {
				limit = {
					owner ?= {
						has_modifier = modifier_hightened_intercommunal_tension
					}
				}
				add = {
					value = 25
					desc = "from_communal_divides_je"
				}
			}
			if = {
				limit = {
					owner ?= {
						has_variable = religious_nationalism_decrease_var
					}
				}
				add = {
					value = -75
					desc = "from_communal_divides_completed_je"
				}
			}
			add = { # Weakened by increasing literacy rates
				desc = "POP_LITERACY"
				this = {
					value = literacy_rate
					multiply = -2
				}
			}
			if = {
				limit = {
					owner ?= {
						has_technology_researched = corporatism
					}
				}
				multiply = {
					value = 1.5
					desc = "CORPORATISM_TECH"
				}
			}
		}
	}
	revolution = {
		possible = {
		    always = no		political_movement_support >= 0.05
		}	
	
		weight = {
			value = 100
		}
	
		state_weight = {
			value = 1
			if = {
			   limit = { 
					NOT = {
						state_is_homeland_of_indian_cultures = yes
					}
				}
				multiply = 0
			}

			if = {
				limit= { bpm_auth_state_can_join_nonideological_civil_war = no }
				multiply = 0
			}
		}
		forced_tags = {
			BHT = {
				trigger = {
					always = yes
				}
				weight = 1000
			}
		}
	}
	
	secession = {
		possible = {
			scope:clout <= 0
		}
	
		weight = {
			value = 1000		
		}	
		
		state_weight = {
			value = 1
			if = {
			   limit = { 
					NOT = {
						state_is_homeland_of_indian_cultures = yes
					}
				}
				multiply = 0
			}

			if = {
				limit= { bpm_auth_state_can_join_nonideological_civil_war = no }
				multiply = 0
			}
		}
		forced_tags = {
			BHT = {
				trigger = {
					always = yes
				}
				weight = 1000
			}
		}		
	}
	
	additional_radicalism_factors = {
		if = {
			limit = {
				owner = {
					is_subject = yes
				}
			}
			add = {
				desc = "INDIA_SUBJECT"
				if = {
					limit = {
						owner = {
							OR = {
								is_subject_type = subject_type_colony
								is_subject_type = subject_type_chartered_company
							}
						}
					}
					value = 0.15
				}
				if = {
					limit = {
						owner = {
							is_subject_type = subject_type_dominion
						}
					}
					value = 0.05
				}
			}
		}	
		if = {
			limit = {
				owner = {
					is_subject = yes
				}
			}
			add = {
				desc = "LIBERTY_DESIRE"
				value = owner.liberty_desire
				divide = 500
			}
		}	
	}
}
