bpm_movement_factions_sg = {
    scope = interest_group
    saved_scopes = {
        polmov
        result
    }
    effect = {
        bpm_movement_setter = yes
        save_scope_as = selected_ig
        if = {
            limit = {
                scope:polmov = { any_influenced_interest_group = { type = scope:selected_ig.type } }
            }
            change_variable = {
                name = bpm_strongly_influenced_clout
                add = scope:result
            }
        }
        change_variable = {
            name = bpm_influenced_clout
            add = scope:result
        }
    }
}


bpm_ig_is_institutional_sg = {
    scope = interest_group
    is_shown = {
    }
    is_valid = {
        bpm_ig_is_institutional = yes
    }
    effect = {
    }
}

bpm_movement_is_significant_faction_sg = {
    scope = interest_group
    saved_scopes = {
        polmov
    }
    is_shown = {
        OR = {
            AND = {
                scope:polmov = { is_political_movement_type = movement_liberal }
                is_interest_group_type = ig_intelligentsia
            }
            AND = {
                scope:polmov = { is_political_movement_type = movement_royalist_absolutist }
                is_interest_group_type = ig_landowners
            }
        }
    }
}
