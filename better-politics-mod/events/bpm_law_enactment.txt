namespace = bpm_law_veto

bpm_law_veto.1 = {
	type = country_event
	placement = root
	title = bpm_law_veto.1.t
	desc = bpm_law_veto.1.d
	flavor = bpm_law_veto.1.f

	category = enactment

	gui_window = event_window_1char_tabloid
	left_icon = scope:executive

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"


	duration = 10

	dlc = dlc_bpm

	trigger = {
		# NOT = { ruler = { has_trait = trait_child } }
	}

	cancellation_trigger = {
		enacting_any_law = no
	}
	
	immediate = {
		ruler = {
			save_scope_as = executive
		}
		add_modifier = {
			name = bpm_block_law_cancel
			months = 10 # Matching event duration in case it gets cancelled before resolving
		}
	}

	option = {
		name = bpm_law_veto.1.a
		default_option = yes
		trigger = {
			bpm_law_enactment_ruler_approval > 0
		}
		add_enactment_phase = 1
		hidden_effect = { remove_modifier = bpm_block_law_cancel }
	}
	option = { 
		name = bpm_law_veto.1.b
		trigger = {
			enactment_setback_count < 2
			ruler = {
				NOR = {
					var:bpm_ig_enacting_law_change_steps > 0
					AND = {
						var:bpm_ig_enacting_law_change_steps = 0
						interest_group.var:bpm_ig_enacting_law_change_steps > 0
					}
				}
			}
		}
		add_enactment_setback = 1
		hidden_effect = { remove_modifier = bpm_block_law_cancel }
	}
	option = {
		name = bpm_law_veto.1.c
		trigger = {
			ruler = {
				NOR = {
					var:bpm_ig_enacting_law_change_steps > 0
					AND = {
						var:bpm_ig_enacting_law_change_steps = 0
						interest_group.var:bpm_ig_enacting_law_change_steps > 0
					}
				}
			}
		}
		every_interest_group = {
			limit = { bpm_ig_law_support_total_value > 0 }
			owner = {
				add_radicals = {
					value = very_large_radicals
					interest_group = prev
				}
			}
		}
		ruler = {
			hidden_effect = {
				remove_modifier = corrupted_by_power
			}
			add_modifier = {
				name = corrupted_by_power
				months = 48
				is_decaying = yes
			}
		}
		trigger_event = { id = unable_to_enact.2 popup = yes }
		hidden_effect = { remove_modifier = bpm_block_law_cancel }
	}
}

# The [HoG]'s desk
bpm_law_veto.2 = {
	type = country_event
	placement = root
	title = bpm_law_veto.2.t
	desc = bpm_law_veto.2.d
	flavor = bpm_law_veto.2.f

	category = enactment

	gui_window = event_window_1char_tabloid
	left_icon = scope:executive

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 5

	dlc = dlc_bpm

	trigger = {
		# NOT = { ruler = { has_trait = trait_child } }
	}

	immediate = {
		var:bpm_headofgov = {
			save_scope_as = executive
		}
		add_modifier = {
			name = bpm_block_law_cancel
			months = 5 # Matching event duration in case it gets cancelled before resolving
		}
	}

	# Send to the [Ruler]'s desk directly
	option = {
		name = bpm_law_veto.2.a
		default_option = yes
		trigger = {
			bpm_law_enactment_hog_approval > 0
			bpm_cabinet_current_law_support_percentage > 0.2
		}
		if = {
			limit = {
				NOT = { bpm_law_enactment_ruler_must_accept = yes }
			}
			trigger_event = { id = bpm_law_veto.1 popup = yes }
		}
		else = {
			add_enactment_phase = 1
		}
		hidden_effect = { remove_modifier = bpm_block_law_cancel }
	}
	# Override the Cabinet
	option = {
		name = bpm_law_veto.2.b
		default_option = yes
		trigger = {
			bpm_law_enactment_hog_approval > 0
			NOT = { bpm_cabinet_current_law_support_percentage > 0.2 }
		}
		if = {
			limit = {
				NOT = { bpm_law_enactment_ruler_must_accept = yes }
			}
			trigger_event = { id = bpm_law_veto.1 popup = yes }
		}
		else = {
			add_enactment_phase = 1
		}
		hidden_effect = { remove_modifier = bpm_block_law_cancel }
	}
	# The Cabinet forces the [HoG]'s hand
	option = {
		name = bpm_law_veto.2.c
		default_option = yes
		trigger = {
			bpm_law_enactment_hog_approval < 0
			bpm_cabinet_current_law_support_percentage > 0.75
			bpm_cabinet_number_of_members >= 4
		}
		if = {
			limit = {
				bpm_law_enactment_ruler_must_accept = yes
			}
			add_enactment_phase = 1
		}
		hidden_effect = { remove_modifier = bpm_block_law_cancel }
	}
	# Send back to the legislature
	option = { 
		name = bpm_law_veto.2.e
		add_enactment_setback = 1
		hidden_effect = { remove_modifier = bpm_block_law_cancel }
	}
}
