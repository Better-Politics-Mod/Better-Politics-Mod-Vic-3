je_bpm_july_monarchy = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	group = je_group_historical_content

	immediate = {
		set_variable = {
			name = bpm_july_monarchy_prog_var
			value = 0
		}
	}

	is_progressing = {
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.10
			}
		}
		government_legitimacy >= 50
	}
	
	progress_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_fast
				trigger = {
					scope:journal_entry = {
						is_progressing = yes
					}
					government_legitimacy >= 90
				}
			}
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_normal
				trigger = {
					scope:journal_entry = {
						is_progressing = yes
					}
					government_legitimacy >= 75
				}
			}
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_slow
				trigger = {
					scope:journal_entry = {
						is_progressing = yes
					}
				}
			}
			triggered_desc = {
				desc = je_bpm_july_monarchy_progressing_civil_war
				trigger = {
					any_civil_war = {
						civil_war_progress > 0.10
					}
				}
			}
			desc = je_bpm_july_monarchy_progressing_low_legitimacy
		}
	}
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.10
			}
		}
		OR = {
			has_law = law_type:law_census_voting
			has_law = law_type:law_universal_suffrage
		}
	}

	on_complete = {
		set_variable = {
			name = one_true_french_dynasty
		}
		trigger_event = { id = bpm_france.2 popup = yes }
	}

	fail = {
		OR = {
			NOT = { has_law = law_type:law_monarchy }
			capital = {
				turmoil >= 0.25
			}
			OR = {
				AND = {
					has_journal_entry = je_springtime_of_the_peoples
					radical_fraction = {
						value >= 0.25
					}
				}
				any_civil_war = {
					civil_war_progress > 0.75
				}
			}
		}
	}

	on_fail = {
		remove_variable = bpm_july_monarchy_prog_var
		trigger_event = { id = bpm_france.1 popup = yes }
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					scope:journal_entry = {
						is_progressing = yes
					}
				}
				if = {
					limit = {
						government_legitimacy >= 90
					}
					change_variable = {
						name = bpm_july_monarchy_prog_var
						add = 2
					}
				}
				else_if = {
					limit = {
						government_legitimacy >= 75
					}
					change_variable = {
						name = bpm_july_monarchy_prog_var
						add = 1
					}
				}
				else = {
					change_variable = {
						name = bpm_july_monarchy_prog_var
						add = 0.5
					}
				}
			}
		}
	}

	current_value = {
		value = root.var:bpm_july_monarchy_prog_var
	}

	goal_add_value = {
		value = 100
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}
