bpm_front_urban_multiplier = {
	value = 1
	if = {
		limit = {
			owner ?={
				OR = {
					AND = {
						has_variable_list = bpm_front_urban_members
						exists = scope:selected_ig
						is_target_in_variable_list = {
							name = bpm_front_urban_members
							target = scope:selected_ig
						}
					}
					AND = {
						has_variable_list = bpm_front_urban_movs
						exists = scope:selected_mov
						is_target_in_variable_list = {
							name = bpm_front_urban_movs
							target = scope:selected_mov
						}
					}
				}
			}
		}
		add = 1
	}
}
