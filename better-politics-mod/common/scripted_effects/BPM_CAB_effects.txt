

bpm_fix_cabinet_pool = {
    every_in_list = {
        variable = bpm_CAB_char_select_pool
        if = {
            limit = { 
                is_character_alive = yes 
                #exists = interest_group
            } 
            owner = { add_to_variable_list = { name = bpm_CAB_temp_pool target = PREV } }
       }
    }
    clear_variable_list = bpm_CAB_char_select_pool
    every_in_list = {
        variable = bpm_CAB_temp_pool
        if = {
            limit = { 
                is_character_alive = yes 
                #exists = interest_group
            } 
            owner = { add_to_variable_list = { name = bpm_CAB_char_select_pool target = PREV } }
       }
    }
}

bpm_clear_cabinet_pool = {
    every_in_list = {
        variable = bpm_CAB_char_select_pool
        if = {
            limit = { 
                OR = {
                    NOR = {
                        has_variable = bpm_cabinet_minister
                        has_variable = bpm_cabinet_important_character
                    }
                    is_character_alive = no
                }
                is_in_void = yes
            } 
            free_character_from_void = yes  
       }
    }
    clear_variable_list = bpm_CAB_temp_pool
    every_in_list = {
        variable = bpm_CAB_char_select_pool
        if = {
            limit = { 
                is_character_alive = yes 
                OR = {
                    has_variable = bpm_cabinet_minister
                    has_variable = bpm_cabinet_important_character
                }
            } 
            owner = { add_to_variable_list = { name = bpm_CAB_temp_pool target = PREV } }
       }
    }
    clear_variable_list = bpm_CAB_char_select_pool
    every_in_list = {
        variable = bpm_CAB_temp_pool
        if = {
            limit = { 
                is_character_alive = yes 
            } 
            owner = { add_to_variable_list = { name = bpm_CAB_char_select_pool target = PREV } }
       }
    }
}

bpm_pick_cabinet_minister = {
    every_interest_group = {
        limit = {
            bpm_interest_group_is_activated = yes
            bpm_ig_is_marginal = no
        }
        if = {
            limit = {
                NOT = {
                    has_variable = bpm_cabinet_interest_group_luck_factor
                }
            }
            set_variable = {
                name = bpm_cabinet_interest_group_luck_factor
                value = 0
            }
        }
        change_variable = {
            name = bpm_cabinet_interest_group_luck_factor
            add = { 0 10 }
        }
    }
    ordered_interest_group = {
        limit = {
            bpm_interest_group_is_activated = yes
            bpm_ig_is_marginal = no
            is_in_government = yes
        }
        min = 0
        max = 1

        order_by = bpm_cabinet_character_pool_ig_weight
        
        save_scope_as = ranig
    }

    scope:ranig ?= {
        remove_variable = bpm_cabinet_interest_group_luck_factor
    }

    if = {
        limit = {
            NOT = { has_variable_list = bpm_CAB_char_select_pool }
            scope:ranig.leader ?= {
                NOT = {
                    has_variable = bpm_cabinet_minister
                }
            }
        }
        scope:ranig.leader ?= {
            set_variable = bpm_send_to_void_if_not_anchored
        }
        add_to_variable_list = {
            name = bpm_CAB_char_select_pool
            target = scope:ranig.leader
        }
    }
    else_if = {
        limit = {
            exists = scope:ranig
        }
        if = {
            limit = {
                NOT = { is_target_in_variable_list = { name = bpm_CAB_char_select_pool target = scope:ranig.leader } }
                scope:ranig.leader ?= {
                    NOT = {
                        has_variable = bpm_cabinet_minister
                    }
                }
            }
            scope:ranig.leader ?= {
                set_variable = bpm_send_to_void_if_not_anchored
            }
            add_to_variable_list = {
                name = bpm_CAB_char_select_pool
                target = scope:ranig.leader
            }
        }
        else = {
            create_character = {
                interest_group = scope:ranig
                on_created = {
                    place_character_in_void = 10000
                }
                trait_generation = { }
                save_scope_as = ranchar
            }
            scope:ranchar ?= {
                bpm_generate_character_traits = yes
            }
            add_to_variable_list = {
                name = bpm_CAB_char_select_pool
                target = scope:ranchar
            }
        }
    }
}

bpm_clear_hog = {
    remove_variable = bpm_headofgov
    every_scope_character = {
        limit = {
            has_role = bpm_character_role_hog
        }
        remove_character_role = bpm_character_role_hog
    }
}

bpm_set_headofgov_sticky = {
    set_variable = bpm_headofgov_sticky
    owner = {
        bpm_clear_hog = yes
        set_variable = {
            name = bpm_headofgov
            value = prev
        }
    }
    add_character_role = bpm_character_role_hog
}

bpm_remove_headofgov_sticky = {
    remove_variable = bpm_headofgov_sticky
    owner = {
        bpm_choose_hog = yes
    }
}

bpm_choose_hog_parlelec = {
    random_interest_group = {
        limit = {
            bpm_is_strongest_ig_in_government = yes
            bpm_interest_group_is_activated = yes
        }
        save_scope_as = ranig
    }
    if = {
        limit = {
            scope:ranig.leader ?= ruler
        }
        if = {
            # If the current HoG already is the leader of the strongest IG in government & isn't the ruler, we don't need to do anything
            limit = {
                NOT = {
                    AND = {
                        var:bpm_headofgov ?= {
                            interest_group = scope:ranig
                        } 
                        NOT = { var:bpm_headofgov ?= ruler }
                    }
                }
            }
            if = {
                limit = {
                    scope:ranig = {
                        has_variable = bpm_attached_hog_non_ruler
                    }
                    NOT = { scope:ranig.var:bpm_attached_hog_non_ruler ?= scope:ranig.leader }
                    NOT = { scope:ranig.var:bpm_attached_hog_non_ruler ?= ruler }
                }
                bpm_clear_hog = yes
                scope:ranig.var:bpm_attached_hog_non_ruler = {
                    bpm_make_headofgov = yes
                }
            }
            else = {
                create_character = {
                    interest_group = scope:ranig
                    on_created = {
                        place_character_in_void = 10000
                    }
                    save_scope_as = ranchar
                }
                bpm_clear_hog = yes
                scope:ranchar = { bpm_make_headofgov = yes }
                scope:ranig = {
                    set_variable = {
                        name = bpm_attached_hog_non_ruler
                        value = scope:ranchar
                    }
                }
            }
        }
    }
    else = {
        bpm_clear_hog = yes
        scope:ranig.leader = {
            bpm_make_headofgov = yes
        }
    }
}

bpm_make_headofgov = {
    owner = {
        set_variable = {
            name = bpm_headofgov
            value = prev
        }
        every_scope_character = {
            limit = {
                has_role = bpm_character_role_hog
            }
            remove_character_role = bpm_character_role_hog
        }
    }
    add_character_role = bpm_character_role_hog
}
bpm_choose_hog = {
    if = {
        limit = {
            has_variable = bpm_hog_heir
            has_law = law_type:law_advisory_body
            has_law_or_variant = law_type:law_answers_to_legislature
            OR = {
                NOT = { exists = var:bpm_headofgov }
                var:bpm_headofgov = {
                    is_character_alive = no
                }
            }
        }
        var:bpm_hog_heir ?= {
            # handle scripted heirs
            bpm_set_headofgov_sticky = yes
            if = {
                limit = {
                    owner = {
                        has_law = law_type:law_bakufu
                    }
                }
                set_variable = shogun_var
            }
            remove_character_role = bpm_character_role_protege
        }
        post_notification = bpm_dictatorial_hog_top
        remove_variable = bpm_hog_heir
        bpm_do_hog_transition = yes
    }
    else_if = {
        limit = {
            bpm_is_hoshog = yes
        }
        bpm_clear_hog = yes
        set_variable = {
            name = bpm_headofgov
            value = ruler
        }
    }
    else = {
        bpm_choose_hog_others = yes
    }
    if = {
		limit = {
            has_law_or_variant = law_type:law_answers_to_legislature
            has_law = law_type:law_advisory_body
			var:bpm_headofgov ?= {
                is_adult = no
                NOT = {has_template = JAP_iemochi_tokugawa} # has his own event
            }
		}
		owner = {
			trigger_event = {
				id = hog_regency.1
				popup = yes
			}
		}
	}
    if = {
        limit = {
            c:JAP ?= this
        }
        var:bpm_headofgov ?= {
            character_japan_ruler_succession_chain_effect = yes
        }
    }
}

bpm_choose_hog_others = {
    if = {
        limit = {
            NOT = {
                var:bpm_headofgov ?= {
                    has_variable = bpm_headofgov_sticky
                }
            }
            NOT = {
                owner = {
                    var:bpm_headofgov ?= {
                        has_variable = bpm_headofgov_sticky
                    }
                }
            }
        }
        if = {
            limit = {
                modifier:country_bpm_hog_top_parlelec = yes
                has_law = law_type:law_advisory_body
                NOT = { has_variable = bpm_hog_heir }
            }
            var:bpm_headofgov ?= {
                set_variable = bpm_headofgov_sticky
            }
            bpm_do_hog_transition = yes
        }
        else_if = {
            limit = {
                modifier:country_bpm_hog_top_parlelec = yes
            }
            bpm_choose_hog_parlelec = yes
        }
        else_if = {
            limit = {
                NAND = {
                    has_variable = bpm_headofgov
		            exists = var:bpm_headofgov
                }
            }
            bpm_clear_hog = yes
            set_variable = {
                name = bpm_headofgov
                value = ruler
            }
        }
    }

    bpm_reload_institution_modifiers_hog = yes
    bpm_reload_interim_penalty = yes
}

bpm_do_hog_transition = {
    if = {
        limit = {
            var:bpm_hog_heir ?= {
                NOT = { is_character_alive = yes}
            }
        }
        remove_variable = bpm_hog_heir
    }
    
    if = {
        limit = {
            NOT = {
                has_variable = bpm_hog_heir
            }
            has_law_or_variant = law_type:law_answers_to_legislature
            has_law = law_type:law_advisory_body
        }
        bpm_create_hog_heir_hook = yes
    }
    else_if = {
        limit = {
            has_variable = bpm_hog_heir
            NAND = {
                has_law_or_variant = law_type:law_answers_to_legislature
                has_law = law_type:law_advisory_body
            }
        }
        remove_variable = bpm_hog_heir
        every_scope_character = {
			limit = {has_role = bpm_character_role_protege}
			remove_character_role = bpm_character_role_protege
		}
    }
}

bpm_fill_cabinet_pool = {
    if = {
        limit = {
            # only the player has a cabinet pool
            # the AI should use bpm_newcab_char_pick
            is_player = yes
        }
        bpm_pick_cabinet_minister = yes
        while = {
            limit  = {
                trigger_if = {
                    limit = {
                        modifier:country_bpm_cabinet_variety = yes
                    }
                    variable_list_size = {
                        name = bpm_CAB_char_select_pool
                        value <= 15
                    }
                }
                trigger_else = {
                    variable_list_size = {
                        name = bpm_CAB_char_select_pool
                        value <= 9
                    }
                }
            }
            bpm_pick_cabinet_minister = yes
            bpm_fix_cabinet_pool = yes
        }
    }

}

bpm_fire_cabinet_minister = {
    custom_tooltip = {
        text = bpm_remove_from_cabinet_effect_tt
        bpm_remove_character_from_cabinet = yes
    }

    hidden_effect ={ owner = { bpm_update_government_and_opposition = yes } }

    if = {
        limit = {
            popularity > 0
        }
        owner = { add_radicals = {
            value = {
                value = prev.popularity
                divide = 100
                multiply = 0.5
            }
            interest_group = prev.interest_group
        } }
    }
    else = {
        owner = { add_loyalists = {
            value = {
                value = prev.popularity
                divide = 100
                multiply = -0.05
            }
        } }
    }

    interest_group = {
        add_modifier = {
            name = bpm_recently_removed_from_cabinet
            days = short_modifier_time
            is_decaying = yes
        }
    }
}

    if = {
        limit = {
            bpm_ruler_is_acting_as_interim_head_of_government = yes

                has_modifier = bpm_interim_government_synergy_penalty
            }

    }
    else_if = {

            NOT = { bpm_ruler_is_acting_as_interim_head_of_government = yes }

        }
        remove_modifier = bpm_interim_government_synergy_penalty
    }

bpm_on_new_hog_dictatorship = {
    if = {
        limit = {
            has_law = law_type:law_advisory_body
            has_law_or_variant = law_type:law_answers_to_legislature
            has_game_started = yes
        }
        if = {
            limit = {
                has_law = law_type:law_bakufu
            }
            bpm_create_hog_heir_hook = yes
            var:bpm_hog_heir ?= {
                bpm_set_headofgov_sticky = yes
                if = {
                    limit = {
                        owner = {
                            has_law = law_type:law_bakufu
                        }
                    }
                    set_variable = shogun_var
                }
                remove_character_role = bpm_character_role_protege
            }
            remove_variable = bpm_hog_heir
            bpm_do_hog_transition = yes
        }
        else_if = {
            limit = {
                OR = {
                    NOT = {has_variable = bpm_headofgov}
                    var:bpm_headofgov ?= ruler
                }
            }
            ordered_interest_group = {
                limit = {is_in_government = yes}
                order_by = ig_clout
                position = 0
                ordered_scope_character = {
                    order_by = prominence
                    position = 0
                    bpm_appoint_as_hog = yes
                }
            }
            bpm_do_hog_transition = yes
        }
    }
}

bpm_choose_regent = {
    if = {
        limit = {
            OR = {
                has_variable = bpm_regency_init
                AND = {
                    bpm_has_gov_regency = yes
                    OR = {
                        NOT = { exists = var:bpm_regent }
                        var:bpm_regent ?= {
                            is_character_alive = no
                        }
                    }
                }
            }
        }
        if = {
            limit = {
                any_scope_character = {
                    interest_group = {
                        is_in_government = yes
                    }
                    bpm_char_can_be_regent = yes
                }
            }
            ordered_interest_group = {
                limit = {
                        is_in_government = yes

                            bpm_char_can_be_regent = yes
                        }
                    }
                order_by = ig_clout
                position = 0

                    limit = {
                        bpm_char_can_be_regent = yes
                    }
                    order_by = prominence
                    position = 0
                    bpm_set_character_as_regent = yes
                }
            }
        }
        else = {
            ordered_interest_group = {
                limit = {
                    is_in_government = yes
                }
                order_by = ig_clout
                position = 0
                save_temporary_scope_as = biggest_ig

	            }
            }
        }
    }
    if = {
        limit = {has_variable = bpm_regency_init}
        remove_variable = bpm_regency_init
    }
}
