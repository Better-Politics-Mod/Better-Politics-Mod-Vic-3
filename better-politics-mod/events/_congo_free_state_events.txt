	type = country_event
	placement = scope:afs_capital_scope

			}
		}
		create_diplomatic_pact = {
			country = c:AFS
			type = personal_union
		}
	}

	option = { #Let it happen

		default_option = yes
		set_variable = free_state_made
 		scope:free_state = {
