	}

	sponsor_modifier = {
		interest_group_pol_str_mult = 1

		bpm_france_artisanal_lowest_level_pm = yes

		country_production_tech_spread_mult = -0.25

		building_group_bg_extraction_self_investment_chance_add = 0.9
		building_group_bg_infrastructure_self_investment_chance_add = 0.9
	}

	on_activate = {
		hidden_effect = {
			if = {
				limit = { game_date < 1836.1.2 }
				add_modifier = {
					name = bpm_urban_collectivization_workaround
					multiplier = 0.9
					months = 1
				}
			}
		}
		bpm_france_make_igs_support_artisanal_econ = yes
	}

	on_deactivate = {
		bpm_france_make_igs_not_support_artisanal_econ = yes
	}

	possible = {
		c:FRA ?= this
	}

	would_sponsor = {
		is_interest_group_type = ig_intelligentsia
	}

	ai_will_revoke = {
		always = yes
	}

	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}
}
