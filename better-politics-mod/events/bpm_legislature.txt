namespace = bpm_legislature
bpm_legislature.1 = {
	#run state elections
	type = state_event
	placement = root
	hidden = yes

	trigger = {
		has_variable = next_election
		total_months = var:next_election
	}
	immediate = {
		bpm_state_do_election = yes
		if = {
			limit = {
				owner = {
					has_variable = months_between_elections
				}
			 }
			change_variable = {
				name = next_election
				add = owner.var:months_between_elections
			}
		}
		else = {
			remove_variable = next_election
		}

		owner = {
			calc_legislators = yes
		}
	}
}
