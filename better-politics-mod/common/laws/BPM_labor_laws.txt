# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

INJECT:law_no_workers_rights = {
	on_activate = {
		if = {
			limit = {
				NOT = { has_law = law_type:law_unregulated_work_time }
			}
			activate_law = law_type:law_unregulated_work_time
		}
	}
}

INJECT:law_regulatory_bodies = {
	ai_enact_weight_modifier = { #Petitions
		# Basically that's one thing almost all communists can agree on
		if = {
			limit = { 
				has_law = law_type:law_council_republic
				NOT = { has_technology_researched = human_rights } # They'd rather have full protections in that case
			}
			add = 750
		}

		if = {
			limit = {
				has_journal_entry = je_bpm_unions_des_oppositions
			}
			add = 25
		}

		if = {
			limit = {
				has_variable = bpm_belgium_modest_reform
			}
			add = 100
		}
	}
}

INJECT:law_worker_protections = {
	ai_enact_weight_modifier = { #Petitions
		value = 0

		# Basically that's one thing almost all communists can agree on
		if = {
			limit = { 
				has_law = law_type:law_council_republic
			}
			add = 750
		}
	}
}
