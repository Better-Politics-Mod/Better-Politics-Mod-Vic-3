bpm_ideology_visibility_gui = {
	scope = country
	saved_scopes = {
		ideology
	}

	is_shown = {
		AND = {
			NOT = { is_target_in_global_variable_list = { name = bpm_vanilla_ideologies target = scope:ideology } }
			NOT = { is_target_in_global_variable_list = { name = bpm_extra_ideologies target = scope:ideology } }
		}
	}
}

	}
