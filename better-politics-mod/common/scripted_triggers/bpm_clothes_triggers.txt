REPLACE:american_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		OR = {
			home_country = {
				exists = c:USA
				c:USA = this
				is_subject = no
			}
			home_country = {
				exists = c:FSA
				c:FSA = this
				is_subject = no
			}
			home_country = {
				exists = c:CSA
				c:CSA = this
				is_subject = no
			}
		}
	}
}

REPLACE:regency_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = owner
		has_role = bpm_character_role_regent
	}
}

REPLACE:monarchy_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = owner
		NOT = { culture = cu:griqua } # Griqua captains wear plainer clothing
		OR = {
			AND = {
				has_role = bpm_character_role_regent
				is_female = yes # Female Regents wear more stylish clothes than male ones, as female regents trend towards being royals themselves
			}
			is_ruler_of_own_country = yes
	     	is_heir_of_own_country = yes
		}
     	owner = {
			OR = {
     			has_law = law_type:law_monarchy
     			has_law = law_type:law_social_monarchy
			}
     	}
	}
}

REPLACE:japanese_shogun_clothes_trigger = {
	exists = scope:character
	scope:character = {
		has_role = bpm_character_role_hog
		owner ?= {
			has_law = law_type:law_bakufu
		}
     	home_country ?= {
			OR = {
     			has_law = law_type:law_monarchy
     			has_law = law_type:law_social_monarchy
			}
     	}
		OR = {
			culture = {
				has_discrimination_trait = language_japanese
			}
			c:JAP ?= home_country
		}
	}
}
