bpm_misc.1 = { # A Word of Warning
	type = country_event
	placement = root

	event_image = { video = "gfx/event_pictures/europenorthamerica_political_extremism.bk2" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/political_extremism"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	title = bpm_misc.1.t
	desc = bpm_misc.1.d
	flavor = bpm_misc.1.f
	duration = 2

	trigger = {
		is_player = yes
		NOT = { has_global_variable = bpm_been_warned_var }
	}

	immediate = {
		set_global_variable = bpm_been_warned_var
	}

	option = {
		name = bpm_misc.1.a
		default_option = yes
	}
}

bpm_misc.2 = {
	type = country_event
	placement = root
	hidden = yes

	trigger = {
		country_has_voting_franchise = no
		NOT = {
			any_interest_group = { has_party = yes }
		}
		OR = {
			any_interest_group = {
				has_ideology = ideology:ideology_ig_united_socialist
				ig_clout > 0.05
			}
			any_interest_group = {
				has_ideology = ideology:ideology_republican
				ig_clout > 0.10
			}
		}
	}

	immediate = {
		scripted_effect_parties_emergence = yes
	}
}
