bpm_value_from_scandals = {
	value = 0
	if = {
		limit = {
			is_in_government = yes
			owner = {
				has_journal_entry = je_bpm_crises_government_scandal
				je:je_bpm_crises_government_scandal.var:bpm_government_scandal_progress >= 50
			}
		}
		value = je:je_bpm_crises_government_scandal.var:bpm_government_scandal_progress
		divide = -10
		subtract = 4
		round = yes
	}
}

bpm_number_of_crises = {
	value = 0
	add = bpm_number_of_crises_dissent
	add = bpm_number_of_crises_crises
	if = {
		limit = {
			is_at_war = yes
		}
		add = 1
	}
}
