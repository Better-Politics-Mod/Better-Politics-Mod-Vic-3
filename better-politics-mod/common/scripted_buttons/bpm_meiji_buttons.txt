		has_journal_entry = je_meiji_restoration
	}

	possible = {
		OR = {
			custom_tooltip = {
				text = sufficient_pressure_from_sonno_joi_tt

					is_political_movement_type = movement_meiji_restorationist
					political_movement_radicalism > define:NPolitics|RADICALISM_PROTESTING_THRESHOLD
				}
			}
			custom_tooltip = {
				text = hog_is_modernizer_or_mitogaku_tt
				any_scope_character = {
					has_role = bpm_character_role_hog
					OR = {
						has_ideology = ideology:ideology_modernizer_leader
						has_ideology = ideology:ideology_mitogaku
					}
				}
			}
			AND = {
				custom_tooltip = {
					text = kobu_gattai_trigger_tt
					has_variable = kobu_gattai
				}
				custom_tooltip = {
					text = shogun_imperial_marriage_tt
					has_variable = japan_completed_imperial_marriage
				}
			}
		}
		NOT = {
			bpm_has_gov_regency = yes
		}
		has_law = law_type:law_bakufu
		custom_tooltip = {
			text = shogunate_has_a_vision_tt
			OR = {
				has_variable = kobu_gattai
				has_variable = kogi_yoron
			}
		}
		is_not_in_civil_war = yes
		custom_tooltip = {
			text = not_already_used_this_interaction_tt
			NOT = {
				has_variable = je_meiji_restoration_proclaim_imperial_restoration_var
			}
		}
	}

	selected = {
		has_variable = je_meiji_restoration_proclaim_imperial_restoration_var
	}

	effect = {
		if = {
			limit = {
				has_variable = kobu_gattai
			}
			custom_tooltip = meiji_restoration_restore_emperor_1_tt
		}
		else_if = {
			limit = {
				has_variable = kogi_yoron
			}
			custom_tooltip = meiji_restoration_restore_emperor_2_tt
		}
		trigger_event = {
			id = ep2_meiji.5
		}
		custom_tooltip = {
			text = meiji_restoration_restore_emperor_warning_tt
		}
		set_variable = je_meiji_restoration_proclaim_imperial_restoration_var
	}

	ai_chance = {
		value = 1
		if = {
			limit = {
				any_political_movement = {
					is_political_movement_type = movement_meiji_restorationist
					political_movement_radicalism > define:NPolitics|RADICALISM_PROTESTING_THRESHOLD
				}
			}
			add = 5
		}
		else_if = {
			limit = {
				any_political_movement = {
					is_political_movement_type = movement_meiji_restorationist
					political_movement_radicalism > define:NPolitics|RADICALISM_MILITANT_THRESHOLD
				}
			}
			add = 25
		}
		add = { # -15 to 5
			value = government_legitimacy
			subtract = 75
			divide = 5
		}
		if = {
			limit = {
				has_variable = kogi_yoron
				ruler ?= {
					OR = {
						has_ideology = ideology:ideology_modernizer_leader
						has_ideology = ideology:ideology_mitogaku
					}
				}
			}
			add = 10
		}
		if = {
			limit = {
				has_variable = kobu_gattai
				NOT = {
					has_variable = japan_completed_imperial_marriage
				}
			}
			add = -6
		}
	}
}

				}
