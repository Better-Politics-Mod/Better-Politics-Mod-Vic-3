bpm_fix_front = {
	clear_variable_list = bpm_front_$name$_temp
	every_in_list = {
	variable = bpm_front_$name$
	owner = { add_to_variable_list = { 
	name =  bpm_front_$name$_temp
	target = prev }
	} }
	clear_variable_list = bpm_front_$name$
	every_in_list = {
	variable = bpm_front_$name$_temp
	owner = { add_to_variable_list = { 
	name =  bpm_front_$name$
	target = prev }
	} }
	clear_variable_list = bpm_front_$name$_temp
}
bpm_choose_front_ig_subp = {
	ig:ig_$ig$ = {
		save_scope_as = selected_ig
	}
	if = {
		limit = {
			scope:selected_ig ?= {
				bpm_interest_group_is_activated = yes
				# Tiny unorganized IGs won't join revolutions
				OR = {
					ig_clout > 0.005
					bpm_ig_organization_val > 0.35
					AND = {
						ig_clout > 0.01
						bpm_ig_organization_val > 0.2
					}
				}
			}

		}
		ordered_in_global_list = {
			variable = bpm_existing_front_names
			position = 0
			order_by = bpm_front_attraction_dispatch_$ig$
			if = { 
				limit = {
					bpm_front_attraction_dispatch_$ig$ > 0
				}
				save_scope_as = selected_front_name
				save_scope_as = sel_POINT_$ig$
			}
		}
		ig:ig_$ig$ = {
			save_scope_as = selected_ig
		}
		if = {
			limit = {
				exists = scope:selected_front_name
				exists = scope:sel_POINT_$ig$
			}
			bpm_add_ig_to_front = yes
		}
		else = {
			bpm_remove_ig_from_all_fronts = yes
		}

	}
}

bpm_choose_front_mov_subp = {
	random_political_movement = {
		limit = {
			is_political_movement_type = movement_$mov$
		}
		save_scope_as = selected_mov
	}
	if = {
		limit = {
			any_political_movement = {
				is_political_movement_type = movement_$mov$
				# Movements will only join revolutions over 50% radicalism and some support
				political_movement_radicalism > 0.5
				political_movement_support > 0.05
				count > 0
			}
		}
		ordered_in_global_list = {
			variable = bpm_existing_front_names
			position = 0
			order_by = bpm_front_attraction_dispatch_$mov$
			if = { 
				limit = {
					bpm_front_attraction_dispatch_$mov$ > 0
				}
				save_scope_as = selected_front_name
				save_scope_as = sel_POINT_$mov$
			}
		}
		if = {
			limit = {
				exists = scope:selected_front_name
				exists = scope:sel_POINT_$mov$
			}
			bpm_add_mov_to_front = yes
		}
		else = {
			bpm_remove_mov_from_all_fronts = yes
		}
	}
}

bpm_front_create_revolutionary_tag_subp = {
	scope:journal_entry = {
		var:bpm_front_currently_planning = {
			save_scope_as = selected_front_name
		}
	}
	ordered_scope_state = {
		limit = {
			is_capital = no
		}
		position = 0
		order_by = bpm_state_front_weight
		save_scope_as = new_capital_state
	}
	create_country = {
		tag = $tag$
		origin = this
		state = scope:new_capital_state
		on_created = {
			set_variable = bpm_country_mark_rev
		}
	}
	random_country = {
		limit = {
			has_variable = bpm_country_mark_rev
		}
		remove_variable = bpm_country_mark_rev
		save_scope_as = new_country
	}
	ordered_scope_state = {
		limit = {
			is_capital = no
			#bpm_state_front_weight > 25
		}
		min = 1
		max = 4
		order_by = bpm_state_front_weight
		set_state_owner = scope:new_country
	}
	ordered_scope_state = {
		limit = {
			is_capital = no
			bpm_state_front_weight > 25
		}
		order_by = bpm_state_front_weight
		set_state_owner = scope:new_country
	}
	create_diplomatic_play = {  
		name = war_between_states
		target_country = scope:new_country
		type = dp_revolution
		annex_as_civil_war = yes
	}

bpm_increase_cell_level_spillover = {
	if = {
		limit = {
			has_modifier = bpm_revcell_$front$_lv1
		}
		bpm_remove_revcell = { front = $front$ lvl = lv1 }
		bpm_add_revcell = { front = $front$ lvl = lv2 }
	}
	else_if = {
		limit = {
			has_modifier = bpm_revcell_$front$_lv2
		}
		bpm_remove_revcell = { front = $front$ lvl = lv2 }
		bpm_add_revcell = { front = $front$ lvl = lv3 }
	}
	else_if = {
		limit = {
			NOR = {
				has_modifier = bpm_revcell_$front$_lv1
				has_modifier = bpm_revcell_$front$_lv2
				has_modifier = bpm_revcell_$front$_lv3
			}
		}
		bpm_add_revcell = { front = $front$ lvl = lv1 }
	}
	save_scope_as = spillover_state
	post_notification = bpm_revolution_spillover
}

bpm_increase_cell_level = {
	set_local_variable = {
		name = saved_front
		value = flag:bpm_front_$front$_name
	}
	save_scope_as = starting_state
	if = {
		limit = {
			has_modifier = bpm_revcell_$front$_lv1
		}
		bpm_remove_revcell = { front = $front$ lvl = lv1 }
		bpm_add_revcell = { front = $front$ lvl = lv2 }
		post_notification = bpm_revolution_grows
	}
	else_if = {
		limit = {
			has_modifier = bpm_revcell_$front$_lv2
		}
		bpm_remove_revcell = { front = $front$ lvl = lv2 }
		bpm_add_revcell = { front = $front$ lvl = lv3 }
		post_notification = bpm_revolution_grows
	}
	else_if = {
		limit = {
			has_modifier = bpm_revcell_$front$_lv3
			any_neighbouring_state = {
				turmoil > 0.25
				NOT = {
					has_modifier = bpm_revcell_$front$_lv3
				}
			}
		}
		random_neighbouring_state = {
			limit = {
				turmoil > 0.25
				NOT = {
					has_modifier = bpm_revcell_$front$_lv3
				}
			}
			bpm_increase_cell_level_spillover = {front = $front$}
		}
	}
	else_if = {
		limit = {
			has_modifier = bpm_revcell_$front$_lv3
			is_coastal = yes
			market = {
				any_scope_country = {
					any_scope_state = {
						turmoil > 0.25
						NOT = {
							has_modifier = bpm_revcell_$front$_lv3
						}
					}
				}
			}
		}
		market = {
			random_scope_country = {
				random_scope_state = {
					limit = {
						turmoil > 0.25
						NOT = {
							has_modifier = bpm_revcell_$front$_lv3
						}
					}
					bpm_increase_cell_level_spillover = {front = $front$}
				}
			}
		}
	}
	else_if = {
		limit = {
			NOR = {
				has_modifier = bpm_revcell_$front$_lv1
				has_modifier = bpm_revcell_$front$_lv2
				has_modifier = bpm_revcell_$front$_lv3
			}
		}
		bpm_add_revcell = { front = $front$ lvl = lv1 }
		post_notification = bpm_revolution_appears
	}
}

bpm_increase_or_shrink_cell_level = {
	scope:bpm_front_$front$ = {
		save_scope_as = selected_front_name
	}
	if = {
		limit = {
			NOT = {
				has_variable = bpm_revcell_$front$_progress
			}
		}
		set_variable = {
			name = bpm_revcell_$front$_progress
			value = 3
		}
	}
	else_if = {
		limit = {
			# below 25% turmoil it will go away
			OR = {
				bpm_state_repression > bpm_front_$front$_progressrate
				turmoil < 0.25
			}
		}
		change_variable = {
			name = bpm_revcell_$front$_progress
			subtract = 1
		}
		#bpm_reduce_cell_level = { front = $front$ }
		set_local_variable = bpm_shrinking
	}
	else_if = {
		limit = {
			bpm_state_repression <= bpm_front_$front$_progressrate
		}
		change_variable = {
			name = bpm_revcell_$front$_progress
			add = 1
		}
		#bpm_increase_cell_level = { front = $front$ }
		set_local_variable = bpm_growing
	}
	if = {
		limit = {
			var:bpm_revcell_$front$_progress <= 0
		}
		set_variable = {
			name = bpm_revcell_$front$_progress
			value = 3
		}
		bpm_reduce_cell_level = { front = $front$ }
	}
	else_if = {
		limit = {
			var:bpm_revcell_$front$_progress >= 6
		}
		set_variable = {
			name = bpm_revcell_$front$_progress
			value = 3
		}
		bpm_increase_cell_level = { front = $front$ }
	}
}


bpm_shrink_or_grow_in_states = {
	if = {
		limit = {
			any_scope_state = {
				has_variable = bpm_revcell_$front$
			}
		}
		scope:selected_je = {
			set_variable = bpm_je_is_urgent
		}
		random_scope_state = {
			limit = {
				has_variable = bpm_revcell_$front$
			}
			bpm_increase_or_shrink_cell_level = { front = $front$ }
		}
	}
	else = {
		set_variable = {
			name = bpm_front_$front$_organization
			value = 0
		}
		scope:selected_je = {
			remove_variable = bpm_je_is_urgent
		}
		random_scope_state = {
			limit = {
				turmoil >= 0.25
				NOT = {
					has_variable = bpm_revcell_$front$_recently_subsided
				}
				NOT = {
					owner = {
						has_variable = bpm_revcell_$front$_recently_subsided
					}
				}
			}
			scope:selected_je = {
				set_variable = bpm_je_is_urgent
			}
			bpm_increase_cell_level = { front = $front$ }
		}
	}
}

		value = {
			value = 0
			every_scope_state = {
				limit = {
					has_variable = bpm_revcell_$front$
				}
				add = {
					value = modifier:state_bpm_revcell_progress_$front$_popscaled_add
					multiply = {

						divide = owner.total_population
					}
				}
			}
		}
	}

	if = {
		limit = {
			NOT = {
				has_variable = bpm_front_$front$_organization
			}
		}
		set_variable = {
			name = bpm_front_$front$_organization
			value = local_var:new_org
		}
	}
	else = {
		change_variable = {
			name = bpm_front_$front$_organization
			add = local_var:new_org
		}
	}
