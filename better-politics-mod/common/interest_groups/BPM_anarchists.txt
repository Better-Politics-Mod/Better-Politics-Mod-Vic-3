ig_anarchists = {
	color = hsv{ 0 0 0 }
	texture = "gfx/interface/icons/ig_icons/anarchists.dds"
	layer = "revolution_dynamic_trade_unions"
	index = 9

	ideologies = {
		ideology_nonexistent
	}
	
	enable = {
		has_global_variable = global_anarchists_split_var
	}	
	
	on_enable = {
		ig:ig_anarchists = {
			set_ig_trait = ig_trait:ig_trait_left_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_left_bastion_of_freedom
			set_ig_trait = ig_trait:ig_trait_left_systematic_opposition
		}
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
		}
		owner = {
			has_variable = local_anarchists_split_var
		}
	}
	
	pop_weight = {
		value = 15

		add = {
			desc = "UNEMPLOYED"
			if = {
				limit = { 
					is_employed = no
				}
				value = 20
			}
		}		
	
		add = {
			desc = "POP_URBAN_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
					}
				}
				value = 50
			}
		}

		add = {
			desc = "POP_EDUCATED_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						AND = {
							is_pop_type = laborers
							literacy_rate >= 0.50
						}
						is_pop_type = machinists
					}
				}
				value = 50
			}
		}	

		add = {
			desc = "POP_LEADERS"
			if = {
				limit = { 
					OR = {
						is_pop_type = academics
						is_pop_type = engineers
					}
				}
				value = 40
			}
		}

		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
					owner = {
						radical_fraction = {
							value >= 0.2
							pop_type = laborers
						}
					}
				}
				value = 10
			}
		}
		add = {
			desc = "POPS_ANGER_SYMPATHY"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
						is_pop_type = soldiers
						is_pop_type = farmers
						is_pop_type = academics
					}
					owner = {
						radical_fraction = {
							value >= 0.5
							pop_type = laborers
						}
					}
				}
				value = 100
			}
		}

		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 50
			}
		}	
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01
					add = 1
					max = 2.0
					min = 0.5
				}
			}
		}	

		multiply = {
			desc = "NO_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_marginal = yes
						}
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "ESTABLISHED_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_powerful = yes
						}
					}
				}
				value = 2.0
			}		
		}		

		multiply = {
			desc = "AT_WAR"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						is_at_war = yes
					}
				}
				value = 1.8
			}		
		}	
		multiply = {
			desc = "BEFORE_FIRST_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = first_intl_split_complete_var } }
				value = 0.6
			}
		}
		multiply = {
			desc = "BEFORE_SECOND_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = second_intl_split_complete_var } }
				value = 0.7
			}
		}
		multiply = {
			desc = "POLITICAL_AGITATION"	
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = political_agitation } }
				value = 1.3
			}		
		}
		multiply = {
			desc = "NO_ANARCHISM"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = anarchism }
					}
				}
				value = 0.4
			}		
		}	

		multiply = {
			desc = "NO_LABOR_MOVEMENT"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = labor_movement }
					}
				}
				value = 0.5
			}		
		}		

		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = egalitarianism }
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "REFORMISTS_RADICALS_SUPPRESSED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							NOT = { has_ideology = ideology:ideology_anarchist }
							OR = {
								has_ideology = ideology:ideology_proletarian
								has_ideology = ideology:ideology_republican
								has_ideology = ideology:ideology_egalitarian
								leader = {
									OR = {
										has_ideology = ideology:ideology_radical
										has_ideology = ideology:ideology_abolitionist
										has_ideology = ideology:ideology_feminist
									}
								}
							}
							is_being_suppressed = yes
						}
					}
				}
				value = 1.2
			}		
		}
		multiply = {
			desc = "CONSERVATIVES_BOLSTERED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							OR = {
								has_ideology = ideology:ideology_paternalistic
								has_ideology = ideology:ideology_reactionary
								has_ideology = ideology:ideology_pro_slavery
								has_ideology = ideology:ideology_stratocratic
								has_ideology = ideology:ideology_laissez_faire
							}
							is_being_bolstered = yes
						}
					}
				}
				value = 1.1
			}		
		}	

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}	
		multiply = ig_attraction_from_radicals_and_loyalists
	}
	
	monarch_weight = {
		value = 0.0
	}	

	agitator_weight = {
		value = 1.0
		multiply = bpm_base_agitator_weight
	}

	commander_weight = {
		value = 1.0
		multiply = bpm_commander_assignment_from_laws_factor
	}	
	
	noble_chance = {
		value = 0.0
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}

}
