
bpm_repeal_amendment_confirmation_gui = {
    scope = country
    saved_scopes = {
        at
    }

	confirm_title = bpm_repeal_amendment_title
	confirm_text = bpm_repeal_amendment_text
	
	is_valid = {
		NOT = {
			custom_tooltip = { 
				text = "BPM_AMENDMENT_CAN_BE_REPEALED"
				is_target_in_global_variable_list = {
					name = bpm_unrepealable_amendment_types
					target = scope:at.type
				}
			}
		}
		enacting_any_law = no
	}
    effect = {
        set_variable = {
            name = bpm_repealing_amendment
            value = scope:at
        }
		set_variable = {
			name = bpm_repealing_amendment_law
			value = scope:at.attached_law
		}
        show_as_tooltip = {
			start_enactment = law_type:law_repeatable_politics_repeal_amendment_other
		}
		hidden_effect = {
			start_enactment = law_type:law_repeatable_politics_repeal_amendment
			add_enactment_phase = 2
		}
    }
}
