bpm_repeatable_politics_topple_name = {
	type = country
	random_valid = no

	text = {
		trigger = {
			has_law = law_type:law_presidential_republic
		}
		localization_key = bpm_repeatable_politics_topple_name_presidential
	}

	text = {
		trigger = {
			NOT = {
				has_law = law_type:law_presidential_republic
			}
			country_has_voting_franchise = yes
		}
		localization_key = bpm_repeatable_politics_topple_name_parliamentary # Council Republic, Constitutional Monarchy, Parliamentary Republic, etc.
	}

	text = {
		trigger = {
			always = not
		}

		fallback = yes 
		localization_key = bpm_repeatable_politics_topple_name_base
	}
}
