british_raj_decision = {
	is_shown = {
		exists = c:GBR
		exists = c:BIC
		c:GBR = ROOT
		c:BIC = {
			is_subject_of = ROOT
			is_player = no
		}
		NOT = { has_variable = british_raj_var }
	}

	possible = {
		hidden_trigger = {
			c:BIC ?= {
				is_player = no
			}
		}
		is_at_war = no
		OR = {
			institution_investment_level = {
				institution = institution_colonial_affairs
				value >= 4
			}
			AND = {
				is_ai = yes
				has_technology_researched = quinine
			}
		}
	}

	when_taken = {
		trigger_event = { id = raj.1 days = 0 popup = yes }
	}

	ai_chance = {
		value = 1000		
	}
}
