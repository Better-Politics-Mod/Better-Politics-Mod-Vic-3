bpm_reload_institution_modifiers_economy_noncr = {
    institution:institution_economy ?= {
        bpm_remove_institution_modifiers_economy = yes
    }
    if = {
        limit = {
            var:bpm_is_institution_economy ?= {
                OR = {
                    is_character_alive = no
                    NOT = { exists = interest_group } 
                }
            }
        }
        remove_variable = bpm_is_institution_economy
    }
    if = {
        limit = {
            has_variable = bpm_is_institution_economy
        }
        var:bpm_is_institution_economy.interest_group = {
           bpm_reload_modifier_inst_singlet = {
                IG1 = agrarian_populists
                INST = economy
                MOD = bpm_economy_agpops_modifier
            }
           bpm_reload_modifier_inst_doublet = {
                IG1 = conservatives
                IG2 = reactionaries
                INST = economy
                MOD = bpm_economy_cons_modifier
            }
           bpm_reload_modifier_inst_singlet = {
                IG1 = fascists
                INST = economy
                MOD = bpm_economy_fash_modifier
            }
           bpm_reload_modifier_inst_doublet = {
                IG1 = liberals
                IG2 = national_liberals
                INST = economy
                MOD = bpm_economy_libs_modifier
            }
           bpm_reload_modifier_inst_singlet = {
                IG1 = market_liberals
                INST = economy
                MOD = bpm_economy_marlibs_modifier
            }
           bpm_reload_modifier_inst_singlet = {
                IG1 = radicals
                INST = economy
                MOD = bpm_economy_radicals_modifier
            }
           bpm_reload_modifier_inst_singlet = {
                IG1 = reformist_socialists
                INST = economy
                MOD = bpm_economy_reformists_modifier
            }
           bpm_reload_modifier_inst_doublet = {
                IG1 = revolutionist_socialists
                IG2 = socialists
                INST = economy
                MOD = bpm_economy_commies_modifier
            }
           bpm_reload_modifier_inst_singlet = {
                IG1 = armed_forces
                INST = economy
                MOD = bpm_economy_military_modifier
            }
           bpm_reload_modifier_inst_singlet = {
                IG1 = devout

                MOD = bpm_economy_neutraling_modifier
            }
           bpm_reload_modifier_inst_singlet = {
                IG1 = landowners

bpm_is_institution_economy_modifier = {
    text = {
        trigger = {
            has_law = law_type:law_council_republic
        }
        localization_key = bpm_economy_council_modifier_desc
    }
    text = {
        trigger = {
            always = no
        }
        fallback = yes
        localization_key = bpm_economy_noncr_modifier_desc
    }
}
