bpm_update_consciousness = {
	bpm_update_consciousness_labor = yes
	bpm_update_consciousness_liberal = yes
	bpm_cache_consciousness_labor = yes
	bpm_cache_consciousness_liberal = yes
}

bpm_update_consciousness_labor = {
	set_variable = {
		name = bpm_consciousness_labor_change_cache
		value = bpm_consciousness_labor_change
	}
	set_variable = {
		name = bpm_consciousness_labor_target_cache
		value = bpm_consciousness_labor_target
	}
	if = {
		limit = {
			NOT = {
				has_variable = bpm_consciousness_labor
			}
		}
		set_variable = {
			name = bpm_consciousness_labor
			value = var:bpm_consciousness_labor_target_cache
		}
	}
	change_variable = {
		name = bpm_consciousness_labor
		add = var:bpm_consciousness_labor_change_cache
	}
	clamp_variable = {
		name = bpm_consciousness_labor
		min = 0
		max = 100
	}


}
