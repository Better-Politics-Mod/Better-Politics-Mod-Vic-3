je_bpm_auth_tracker = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = { NOT = { is_country_type = decentralized } }
	possible = { always = yes }

    immediate = {
		
    }

    on_weekly_pulse = {
        effect = {
			every_scope_state = {
				bpm_auth_handle_potential_collapses = yes
				bpm_auth_remove_partisans_if_accepted = yes
				bpm_auth_handle_suppression_and_order = yes
				if = {
					limit = { has_variable = bpm_auth_setup_check_for_partisans }
					remove_variable = bpm_auth_setup_check_for_partisans
					if = {
						limit = { bpm_auth_will_spawn_partisans = yes }
						bpm_auth_spawn_partisans = yes
					}
				}
			}
			bpm_auth_update_state_exhaustion = yes
			bpm_auth_handle_authority_from_extra_sources = yes
			bpm_auth_handle_authority_shortage = yes
			bpm_auth_handle_activism_from_collapses_and_partisans = yes
        }
    }

	weight = -100

	should_be_pinned_by_default = no
}
