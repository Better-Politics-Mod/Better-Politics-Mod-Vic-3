bpm_CAB_toggle_char_select = {
	scope = institution
	is_shown = {
	}
	is_valid = {
	}
	effect = {
        owner = { 
            set_variable = {
                name = bpm_cabinet_char_select
                value = prev
            }
            bpm_fix_cabinet_pool = yes
        }
	}
}

bpm_CAB_toggle_char_select_off = {
    scope = institution
    is_shown = {
    }
    is_valid = {
    }
    effect = {
        remove_variable = bpm_cabinet_char_select
    }
}
