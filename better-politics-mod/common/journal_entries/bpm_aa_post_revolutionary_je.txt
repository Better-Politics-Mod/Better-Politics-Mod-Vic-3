je_bpm_revolutionary_damage = {
	icon = "gfx/interface/icons/notification_icons/revolution.dds"

	group = je_group_internal_affairs

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_bpm_revolutionary_damage_desc
			}
		}
	}

	should_be_pinned_by_default = yes

	can_deactivate = yes

	modifiers_while_active = {
		# add modifiers
	}

	is_shown_when_inactive = {
		# this should always be yes
		always = yes
	}

	possible = {
		OR = {
			bpm_rev_exists = yes
			has_variable = bpm_marker_revolutionary_government
		}
	}

	complete = {
		hidden_trigger = {
			always = no
		}
		# tooltip generation
		custom_tooltip = my_custom_end_tt
	}
	
	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					bpm_rev_exists = yes
				}
				while = {
					count = 5
					random_scope_state = {
						limit = {
							devastation < 40
						}
						state_region = {
							random_list = {
								10 = {}
								5 = {
									add_devastation = 1
								}
								3 = {
									add_devastation = 2
								}
								1 = {
									add_devastation = 3
								}
							}
						}
					}
				}
			}
		}
	}
	on_monthly_pulse = {
		effect = {
			je:je_bpm_revolutionary_damage = {
				remove_modifier = bpm_revolutionary_damage_bureaucracy_mod
				add_modifier = {
					name = bpm_revolutionary_damage_bureaucracy_mod
					multiplier = "je:je_bpm_revolutionary_damage.scripted_bar_progress(bpm_revolutionary_bureaucratic_damage_progress)"
				}
				remove_modifier = bpm_revolutionary_damage_military_mod
				add_modifier = {
					name = bpm_revolutionary_damage_military_mod
					multiplier = "je:je_bpm_revolutionary_damage.scripted_bar_progress(bpm_revolutionary_military_damage_progress)"
				}
				remove_modifier = bpm_revolutionary_damage_economy_mod
				add_modifier = {
					name = bpm_revolutionary_damage_economy_mod
					multiplier = "je:je_bpm_revolutionary_damage.scripted_bar_progress(bpm_revolutionary_economic_damage_progress)"
				}
			}
		}
	}
	
	scripted_progress_bar = bpm_revolutionary_bureaucratic_damage_progress
	scripted_progress_bar = bpm_revolutionary_military_damage_progress
	scripted_progress_bar = bpm_revolutionary_economic_damage_progress

	should_be_pinned_by_default = no
}
