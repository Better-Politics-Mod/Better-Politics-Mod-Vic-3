bpm_add_law_to_draft = {
	scope = law
	is_shown = {
	}
	is_valid = {
	}
	effect = {
		save_scope_as = bpm_selected_law
		owner = { 
			add_to_variable_list = {
				name = bpm_negotiation_targets
				target = scope:bpm_selected_law.type
			}
			bpm_cache_support_for_target_laws = yes
		}
	}
}

bpm_remove_law_from_draft = {
	scope = country
	saved_scopes = {
		law_type
	}
	effect = {
		remove_list_variable = {
			name = bpm_negotiation_targets
			target = scope:law_type
		}
		bpm_cache_support_for_target_laws = yes
	}
}

bpm_no_law_allowed = {
	scope = law
	is_shown = {
		save_temporary_scope_as = bpm_selected_law
		owner = { 
			NOT = { any_in_list = {
				variable = bpm_negotiation_targets
				is_same_law_group_as = scope:bpm_selected_law.type
			} }
		}
	}
}

bpm_begin_constitutional_drafting = {
	scope = country
	effect = {
		bpm_setup_constitutional_drafting = yes
	}
	is_valid = {
		NOT = {
			has_modifier = bpm_in_negotiation
		}
		NOT = {
			has_variable = bpm_negotiation_banner_visible
		}
		OR = {
			custom_tooltip = {
				text = bpm_tooltip_political_stability_too_high
				bpm_political_stability <= 40
			}
			trigger_if = {
				limit = {
					bpm_country_is_in_europe = yes
				}
				has_journal_entry = je_springtime_of_the_peoples
			}
		}
	}
}

bpm_negotiation_const_complete = {
	scope = country
	effect = {
		bpm_approve_draft_constitution = yes
	}
}

bpm_negotiation_const_end = {
	scope = country
	effect = {
		bpm_end_constitutional_drafting = yes
	}
}
