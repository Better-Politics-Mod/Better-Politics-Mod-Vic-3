bpm_consolidation_of_the_state_values = {
	value = 0

	add = {
		desc = "BPM_VALUE_FROM_CLOUT_AND_APPROVAL_OF_BUREAUCRATS"
		if = {
			limit = {
				exists = ig:ig_petty_bourgeoisie
				ig:ig_petty_bourgeoisie = {
					is_in_government = yes
				}
			}
			add = {
				value = ROOT.ig:ig_petty_bourgeoisie.ig_clout
				multiply = ROOT.ig:ig_petty_bourgeoisie.ig_approval
				divide = 2.5
			}
		}
	}
	
	add = {
		desc = "BPM_VALUE_FROM_CLOUT_AND_APPROVAL_OF_PROFESSIONALS"
		if = {
			limit = {
				exists = ig:ig_intelligentsia
				ig:ig_intelligentsia = {
					is_in_government = yes
				}
			}
			add = {
				value = ROOT.ig:ig_intelligentsia.ig_clout
				multiply = ROOT.ig:ig_intelligentsia.ig_approval
				divide = 2.5
			}
		}
	}
}

bpm_left_unionist_interest_groups_clout = {
	value = 0

	add = {
		every_interest_group = {
			limit = {
				OR = {
					is_interest_group_type = ig_armed_forces # Military Interests
					is_interest_group_type = ig_intelligentsia # Provincial Liberals
					is_interest_group_type = ig_liberals # Left Unionists
					is_interest_group_type = ig_national_liberals # Conservative Unionists
				}
				NOT = {
					has_ideology = ideology:ideology_nonexistent
				}
			}
			add = ig_clout
		}
	}
}
