			}
			owner = {

			}
		}
		add = {

		}

	}

		limit = {
			OR = {
				bpm_movement_is_socialist = yes

				bpm_country_is_pre_constitutional = yes

			desc = "Opposition to absolute monarchy"

		}

			owner = {

			value = 0.2

			desc = "Unsatisfied with Unfair Voting System"

		}
	}

		limit = {
			owner = { 
				bpm_country_laws_are_liberal_democratic = no

		}

			desc = "Country is not fully Liberal Democratic"
			value = 0.1
		}
	}
	
	if = {
		limit = {
			is_political_movement_type = movement_communist
			has_global_variable = first_intl_chain_complete_var

			desc = "Communists after first international"

	}

		limit = {
			is_political_movement_type = movement_communist
			has_global_variable = second_intl_chain_complete_var
			owner = { NOT = { has_law_or_variant = law_type:law_council_republic } }
		}
		add = {
			desc = "Communists after second international"
			value = 0.2 
		}
	}

		limit = {
			bpm_movement_is_modernizer = yes
			owner = { country_has_laws_in_need_of_modernization = yes }
		}
		add = {
			desc = "Supporting modernization"
			value = 0.25
		}
	}
	else_if = {
		limit = {
			OR = {
				bpm_movement_is_liberal_nationalist = yes
				bpm_movement_is_laissez_faire = yes
				bpm_movement_is_modernizer = yes
			}
			owner = { 
				bpm_country_laws_have_traditional_economics = yes
			}
		}
		add = {
			desc = "Opposition to Traditional economy"
			add = 0.2
		}
	}
	else_if = {
		limit = {
			bpm_movement_is_laissez_faire = yes
		}

			limit = { owner = { bpm_country_laws_are_very_capitalist = no } }
			add = {
				desc = "Laws are not capitalist enought" 
				value = 0.2
			}
		}
		if = {
			limit = { owner = { bpm_country_laws_have_max_welfare = yes } }
			add = {
				desc = "Opposition to welfare state"
				value = 0.2
			}
		}
		else_if = {
			limit = { owner = { bpm_country_laws_has_market_welfare = no } }
			add = {
				desc = "Support for market welfare"
				value = 0.1 
			}
		}
	}
	else_if = {
		limit = {
			bpm_movement_is_abolitionist = yes
			owner = { bpm_country_has_unfree_labor = yes }

		add = {
			desc = "Opposition to slavery"

		}

					has_law_or_variant = law_type:law_legacy_slavery

			}
		}

			desc = "Wants to uphold slavery"

			}

		limit = {
			bpm_movement_is_liberal_bourgeois = yes

			}
		}
		add = {
			desc = "Opposition to Socialist Economy"
			value = 0.2
		}
	}

		limit = {
			OR = {
				bpm_movement_is_reactionary_monarchist = yes
				is_political_movement_type = movement_fascist
				is_political_movement_type = movement_reactionary
			}

				bpm_country_laws_has_socialist_economics = yes
				bpm_country_laws_are_liberal_democratic = yes

			desc = "Movement is anti democractic"
			value = 0.2

			desc = "Opposition to monarchy"
			value = 0.2
		}
	}

			bpm_movement_is_reactionary_monarchist = yes

				}
			}
		}

			desc = "Laws are not nationalist"
			value = 0.2
		}
	}

			OR = {
				is_political_movement_type = movement_religious_minority
