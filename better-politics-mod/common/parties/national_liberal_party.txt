
national_liberal_party = {
	color = hsv{ 0.50 0.9 0.8 }
	name = party_natlib_ger

	icon = {
		default = "/gfx/interface/icons/political_parties/liberal.dds"
	}
	unlocking_technologies = {
		nationalism
	}
	# THIS IS RUN ONLY ONCE AT GAME START
	valid_for_country = {
		OR = {
            country_has_primary_culture = cu:north_german
            country_has_primary_culture = cu:north_german 
        }
        NOR = {
            c:LUX ?= this
            c:TRS ?= this
        }
	}

	available_for_interest_group = {
		bpm_interest_group_is_activated = yes
        OR = {
            is_interest_group_type = ig_national_liberals
            is_interest_group_type = ig_market_liberals
            is_interest_group_type = ig_industrialists
            is_interest_group_type = ig_rural_folk
        }
	}
    
	join_weight = {
		value = {
			value = 0
			desc = "base_value"
		}
        if = {
            limit = {
                OR = {
                    AND = {
                        exists = c:PRU
                        owner = { this ?= c:PRU }
                    }
                    AND = {
                        exists = c:NGF
                        owner = { this ?= c:NGF }
                    }
                    AND = {
                        exists = c:GER
                        owner = { this ?= c:GER }
                    }
                }
            }
		    if = {
                limit = {
                    OR = {
                        is_interest_group_type = ig_national_liberals
                        is_interest_group_type = ig_rural_folk
                    }
                }
                add = {
                    value = 100
                    desc = "bpm_is_national_liberal"
                }
            }
		    if = {
                limit = {
                    OR = {
                        is_interest_group_type = ig_market_liberals
                        is_interest_group_type = ig_industrialists
                    } 
                }
                if = {
                    limit = {
                        owner = { 
                            has_variable = bpm_schutzzoll_party_split 
                            has_law = law_type:law_monarchy
                        }
                    }
                    add = {
                        value = -100
                        desc = "bpm_schutzzoll_party_split_desc"
                    }
                }
                else = {
                    add = {
                        value = 100
                        desc = "bpm_natlib_business_alliance_desc"
                    }
                }
            }
        }
	}
}
