gg_korea.4 = { # Donghak Petition
	type = country_event
	placement = ROOT

	title = {
		first_valid = {
			triggered_desc = {
				desc = gg_korea.4.ta
				trigger = {
					scope:petition_state = {
						state_region = s:STATE_YANGHO
					}
				}
			}
			triggered_desc = {
				desc = gg_korea.4.tb
				trigger = {
					scope:petition_state = {
						NOT = {
							state_region = s:STATE_YANGHO
						}
					}
				}
			}
		}
	}
	desc = gg_korea.4.d
	flavor = gg_korea.4.f

	dlc = dlc010

	event_image = {
		video = "ep1_donghak_rebellion"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
		NOT = {
			has_variable = donghak_petition_event_var
		}
		radical_fraction = {
			value >= 0.2
		}
		ig:ig_agrarian_populists ?= {
			is_marginal = no
			is_in_government = no
		}
	}

	immediate = {
		ig:ig_agrarian_populists ?= {
			save_scope_as = donghak_ig
		}
		if = {
			limit = {
				any_scope_state = {
					state_region = s:STATE_YANGHO
					radical_fraction = {
						value >= 0.1
					}
				}
			}
			random_scope_state = {
				limit = {
					state_region = s:STATE_YANGHO
					radical_fraction = {
						value >= 0.1
					}
				}
				save_scope_as = petition_state
			}
		}
		else = {
			random_scope_state = {
				limit = {
					radical_fraction = {
						value >= 0.1
					}
				}
				save_scope_as = petition_state
			}
		}
		set_variable = donghak_petition_event_var
	}

	option = {
		name = gg_korea.4.a
		default_option = yes
		scope:donghak_ig = {
			add_modifier = {
				name = modifier_donghak_petition_rejected
				months = normal_modifier_time
			}
		}
		scope:petition_state = {
			add_radicals_in_state = {
				value = medium_radicals
				pop_type = peasants
			}
		}
	}

	option = {
		name = gg_korea.4.b
		add_journal_entry = { type = je_gyojo_shinwon }
	}
}

	placement = ROOT

	title = korea_monarchy.3.t
	desc = korea_monarchy.3.d
	flavor = korea_monarchy.3.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:myeongseong_scope

	dlc = dlc010

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
		NOT = {
			has_variable = myeongseong_event_var
		}
		has_dlc_feature = ep1_content
		c:KOR ?= this
		game_date >= 1873.1.1
		ruler = {
			has_template = KOR_yi_myongbok
		}
	}

	immediate = {
		ruler = {
			save_scope_as = gojong_scope
		}
		create_character = {
			template = KOR_empress_myeongseong
			on_created = {
				set_variable = myeongseong_var
				place_character_in_void = 6
			}
		}
		random_character_in_void = {
			limit = {
				has_variable = myeongseong_var
			}
			save_scope_as = myeongseong_scope
		}
		set_variable = myeongseong_event_var
	}

	option = {
		name = korea_monarchy.3.a
		default_option = yes
		show_as_tooltip = {
			create_character = {
				template = KOR_empress_myeongseong
			}
		}
		hidden_effect = {
			random_character_in_void = {
				limit = {
					has_variable = myeongseong_var
				}
				free_character_from_void = yes
				set_as_interest_group_leader = yes
			}
		}
		if = {
			limit = {
				ig:ig_industrialists = {
					is_in_government = no
				}
			}
			ig:ig_industrialists = {
				add_ruling_interest_group = yes
			}
		}
	}

	option = {
		name = korea_monarchy.3.b
		add_modifier = {
			name = modifier_empress_myeongseong
			months = very_long_modifier_time
		}
		hidden_effect = {
			random_character_in_void = {
				limit = {
					has_variable = myeongseong_var
				}
				kill_character = {
					hidden = yes
				}
			}
		}
	}
}
