ig_liberals = {
	color = hsv{ 0.16 1 1 }
	texture = "gfx/interface/icons/ig_icons/liberal.dds"
	
	ideologies = {
		ideology_liberal
		ideology_meritocratic
		ideology_laissez_faire
		ideology_individualist
	}
	
	enable = {
		always = yes
	}
	
	traits = {
		ig_trait_left_systematic_corruption
		ig_trait_left_bastion_of_freedom
		ig_trait_left_systematic_opposition
	}	
	
	on_enable = {}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
	}
	
	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 200
			}
		}	
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 50
			}
		}	

		add = {
			desc = "MIDDLE_CLASS"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					strata = middle
				}
				value = 100
			}
		}

		add = {
			desc = "URBAN_MIDDLE_CLASS"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					strata = middle
				}
				value = 100
			}
		}

		add = {
			desc = "URBAN_UPPER_CLASS"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					strata = rich
				}
				value = 50
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}	

		multiply = {
			desc = "UNEDUCATED"
			value = 1.0

			if = {
				limit = {
					literacy_rate <= 0.5
				}
				value = 0.5
			}
		}
		multiply = {
			desc = "VERY_UNEDUCATED"
			value = 1.0

			if = {
				limit = {
					literacy_rate <= 0.25
				}
				value = 0.5
			}
		}

		multiply = {
			desc = "NO_DEMOCRACY"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = democracy }
					}
				}
				value = 0.5
			}		
		}
		multiply = {
			desc = "NO_RATIONALISM"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = rationalism }
					}
				}
				value = 0.0
			}		
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}

				value = 0.1
			}
		}

		if = {
			limit = {

					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
