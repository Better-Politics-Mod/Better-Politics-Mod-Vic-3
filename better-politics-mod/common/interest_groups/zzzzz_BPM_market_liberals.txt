ig_market_liberals = {
	color = hsv{ 0.45 0.7 0.9 }
	texture = "gfx/interface/icons/ig_icons/market_liberals.dds"
	layer = "revolution_dynamic_industrialists"
	index = 12
	
	ideologies = {
		ideology_nonexistent
	}
	
	character_ideologies = {
		ideology_center_leader
		ideology_republican_leader
		ideology_liberal_leader
		ideology_radical
		ideology_market_liberal
		ideology_land_reformer
		ideology_reformer
		ideology_positivist
		ideology_utilitarian_leader
	}

	enable = {
		always = yes
	}
	
	on_enable = {
		bpm_update_name_market_liberals = yes
	}
	on_disable = {}

	priority_cultures = { #Country scope
		rule = {
			trigger = {
				c:BIC ?= this
				any_primary_culture = { 
					cu:british = this
				}
			}
			cultures = {
				british scottish irish
			}
		}
	}

	on_character_ig_membership = {}
	
	pop_potential = {
		custom_tooltip = {
			text = BPM_VALID_POP_FOR_IG
			owner ?= {
				NOT = {
					is_country_type = decentralized
				}
				ig:ig_market_liberals ?= {
					NOT = {
						has_ideology = ideology:ideology_nonexistent
					}
					bpm_interest_group_is_activated = yes
				}
			}
		}
	}
	
	pop_weight = {
		value = 40

		if = {
			limit = {
				owner = {
					ig:ig_market_liberals ?= {
						OR = {
							has_ideology = ideology:ideology_nonexistent
							bpm_interest_group_is_activated = no
						}
					}	
				}
			}
			value = -999
		}

		if = {
			limit = { is_pop_type = peasants }
			add = {
				desc = "POP_EXCLUSIONS"
				value = -50
			}
		}
		else_if = {
			limit = { is_pop_type = aristocrats }
			add = {
				desc = "POP_ARISTOCRATS"
				value = 60
			}
		}
		else_if = {
			limit = {
				strata = middle
				NOT = { is_pop_type = clergymen }
			}
			add = {
				desc = "MIDDLE_CLASS"
				value = 20
			}
		}

		if = {
			limit = { 
				OR = {
					is_pop_type = capitalists 
					is_pop_type = shopkeepers
				}
			}
			add = {
				desc = "POP_CAPITALISTS"
				value = var:bpm_consciousness_marlib_igs_capitalists_shopkeepers
			}
		}
		else_if = {
			limit = {
				OR = {
					is_pop_type = academics
					is_pop_type = engineers
				}
			}
			add = {
				desc = "POP_ACADEMICS_ENGINEERS"
				value = var:bpm_consciousness_marlib_igs_academics_engineers
			}
		}
		else = {
			add = {
				desc = "LIBERAL_CONSCIOUSNESS"
				value = var:bpm_consciousness_marlib_igs_others
			}
		}


		# URBAN EMPLOYMENT
		if = {
			limit = {
				bpm_pop_employment_is_rural = no
			}
			add = {
				value = 30
				desc = "URBAN"
			}
			if = {
				limit = {
					strata = middle
					NOT = { is_pop_type = clergymen }
				}
				add = {
					value = 50
					desc = "URBAN_MIDDLE_CLASS"
				}
			}
			else_if = {
				limit = { strata = upper }
				add = {
					value = 30
					desc = "URBAN_UPPER_CLASS"
				}
			}
		}

		if = {
			limit = {
				owner = {
					c:BIC ?= this
				}
				culture = {
					has_discrimination_trait_group = heritage_group_european
				}
				OR = {
					strata = upper
					strata = middle
				}
				NOT = {
					is_pop_type = officers
				}
			}
			add = {
				desc = "POP_INDIA_EUROPEAN_ADMINISTRATORS"
				value = 1000 #250
			}
		}
		
		if = {
			limit = {
				owner = {
					c:BIC ?= this
				}
			}
			if = {
				limit = {
					NOT = {
						culture = {
							has_discrimination_trait_group = heritage_group_european
						}
					}
					OR = {
						is_pop_type = bureaucrats
						is_pop_type = clerks
					}
				}
				add = {
					desc = "POP_INDIA_NATIVE_ADMINISTRATORS"
					value = 100 #25
				}
			}
			else_if = {
				limit = {
					NOT = {
						culture = {
							has_discrimination_trait_group = heritage_group_european
						}
					}
					NOR = {
						is_pop_type = bureaucrats
						is_pop_type = clerks
					}
				}
				multiply = {
					desc = "POP_INDIA_ADMINISTRATION_PRIVILEGES"
					value = 0.5 #0.05
				}
			}
		}

		if = {
			limit = { owner = { bpm_performance_significant_country = yes } }
			add = {
				desc = "SLAVERY_DETRACTORS"
				if = {
					limit = {
						owner = { NOT = { has_law_or_variant = law_type:law_slavery_banned } }
						OR = {
							pop_acceptance <= acceptance_status_3
							state = { is_slave_state = no }
							AND = {
								state = { is_incorporated = no }
								owner = { has_law = law_type:law_colonial_slavery }
							}
						}
					}
					value = 50
				}
			}
			multiply = {
				desc = "LEADER_POPULARITY"
				scope:interest_group = {
					leader = {
						value = popularity
						multiply = 0.005
						add = 1
						max = 1.5
						min = 0.67
					}
				}
			}
			multiply = {
				desc = "TRADITIONAL_MONARCHY"	
				value = 1.0
				
				if = {
					limit = { 
						owner = {
							has_law_or_variant = law_type:law_monarchy
							OR = {
								has_law_or_variant = law_type:law_oligarchy
								has_law_or_variant = law_type:law_autocracy
							}
						}
					}
					value = 0.75
				}		
			}
			multiply = bpm_ig_agitators_popularity
			multiply = bpm_ig_ruler_popularity
			multiply = { desc = bpm_ig_attraction_from_radicals_and_loyalists_tt value = ig_attraction_from_radicals_and_loyalists }
		}
		multiply = {
			desc = "UNEDUCATED"
			value = literacy_rate
			multiply = 1.5
			max = 1.0
		}
		multiply = ig_attraction_to_ruler_ideology_from_stability
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
	}
	
	monarch_weight = {
		owner.ig:ig_market_liberals = {
			value = bpm_ig_monarch_weight
		}
	}
	
	agitator_weight = {
		owner.ig:ig_market_liberals = {
			value = bpm_base_agitator_weight
		}
	}
	
	commander_weight = {
		owner.ig:ig_market_liberals = {
			value = bpm_commander_weight
		}
	}

	executive_weight = {
		value = 1.0
		if ={
			limit = {
				owner.ig:ig_market_liberals = { bpm_interest_group_is_activated = no }
			}
			value = 0.0
		}
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}

	female_executive_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
