bpm_codification_decree_support = {
    if = {
        limit = {
            is_target_in_variable_list = {
                name = bpm_codification_legal_laws
                target = var:bpm_decree_cooldown_current_law
            }
            NOT = { has_law = law_type:law_advisory_body }
        }
        add = {
            value = -3
            desc = "bpm_codification_decree_current_legal_codification"
        }
    }
    else_if = {
        limit = {
            is_target_in_variable_list = {
                name = bpm_codification_constitutional_laws
                target = var:bpm_decree_cooldown_current_law 
            }
            NOT = { has_law = law_type:law_advisory_body }
        }
        add = {
            value = -6
            desc = "bpm_codification_decree_current_constitutional_codification"
        }
    }
    if = {
        limit = {
            has_law = law_type:law_advisory_body
        }
        add = {
            value = 3
            desc = "bpm_codification_decree_support_from_advisory_body"
        }
    }
    if = {
        limit = {
            has_law = law_type:law_hereditary_bureaucrats
        }
        if = {
            limit = {
                ig:ig_landowners = {
                    law_stance = {
                        law = owner.var:bpm_decree_cooldown
                        value = approve
                    }
                }
            }
            add = {
                value = 1
                desc = "bpm_codification_decree_support_from_aristocrats"
            }
        }
        else_if = {
            limit = {
                ig:ig_landowners = {
                    law_stance = {
                        law = owner.var:bpm_decree_cooldown
                        value = strongly_approve
                    }
                }
            }
            add = {
                value = 2
                desc = "bpm_codification_decree_support_from_aristocrats"
            }
        }
    }
    else = {
        if = {
            limit = {
                ig:ig_petty_bourgeoisie = {
                    law_stance = {
                        law = owner.var:bpm_decree_cooldown
                        value = approve
                    }
                }
            }
            add = {
                value = 1
                desc = "bpm_codification_decree_support_from_bureaucrats"
            }
        }
        else_if = {
            limit = {
                ig:ig_petty_bourgeoisie = {
                    law_stance = {
                        law = owner.var:bpm_decree_cooldown
                        value = strongly_approve
                    }
                }
            }
            add = {
                value = 2
                desc = "bpm_codification_decree_support_from_bureaucrats"
            }
        }
    }

    if = {
        limit = {
            ig:ig_armed_forces = {
                law_stance = {
                    law = owner.var:bpm_decree_cooldown
                    value = approve
                }
            }
        }
        add = {
            value = 1
            desc = "bpm_codification_decree_support_from_armed_forces"
        }
    }
    else_if = {
        limit = {
            ig:ig_armed_forces = {
                law_stance = {
                    law = owner.var:bpm_decree_cooldown
                    value = strongly_approve
                }
            }
        }
        add = {
            value = 2
            desc = "bpm_codification_decree_support_from_armed_forces"
        }
    }

    if = {
        limit = {
            has_law = law_type:law_appointed_bureaucrats
            ig:ig_petty_bourgeoisie = { 
                ig_approval >= 0
            }
        }
        add = {
            value = 1
            desc = "bpm_codification_decree_support_from_loyal_bureaucrats"
        }
    }
    else_if = {
        limit = {
            has_law = law_type:law_elected_bureaucrats
            ig:ig_petty_bourgeoisie = { 
                ig_approval >= happy
            }
        }
        add = {
            value = 1
            desc = "bpm_codification_decree_support_from_loyal_bureaucrats"
        }
    }

    if = {
        limit = {
            has_law = law_type:law_no_separate_hog
        }
        add = {
            value = 1
            desc = "bpm_codification_decree_support_from_strong_executive"
        }
    }

    if = {
        limit = {
            has_law = law_type:law_unitary
        }
        add = {
            value = 2
            desc = "bpm_codification_decree_support_from_centralization"
        }
    }
    else_if = {
        limit = {
            NOT = { has_law = law_type:law_devolved } 
        }
        add = {
            value = -2
            desc = "bpm_codification_decree_support_from_centralization"
        }
    }

    if = {
        limit = {
            OR = {
                has_law = law_type:law_censorship
                has_law = law_type:law_outlawed_dissent
            }
        }
        add = {
            value = 1
            desc = "bpm_codification_decree_support_from_censorship"
        }
    }
    if = {
        limit = {
            has_law = law_type:law_supreme_legislature
        }
        add = {
            value = -4
            desc = "bpm_codification_decree_support_from_supreme_legislature"
        }
    }
    min = 0
}
