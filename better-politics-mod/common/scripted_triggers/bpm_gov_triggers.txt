# TWO PARTY SYSTEM

bpm_country_has_two_party_system = {
	has_law = law_type:law_presidential_republic
}

bpm_country_has_historical_two_party_system = {
	OR = {
		AND = {
			has_law = law_type:law_presidential_republic
			exists = c:USA
			this = c:USA
		}
		AND = {
			has_law = law_type:law_monarchy
			exists = c:GBR
			this = c:GBR
		}
	}
}
