je_bpm_belgium_current_government_type = {
	type = country
	random_valid = no

	### This is the shittiest code to have ever been made

	### None

	text = {
		trigger = {
			bpm_overall_unionist_ministers = 0
		}
		localization_key = je_bpm_none_in_government
	}

	### Homogenous government

	text = {
		trigger = {
			bpm_left_unionist_ig_government = 0
		}
		localization_key = je_bpm_belgium_government_homogenous_catholic
	}

	text = {
		trigger = {
			bpm_right_unionist_ig_government = 0
		}
		localization_key = je_bpm_belgium_government_homogenous_liberal
	}

	### Mixed

	text = {
		trigger = {
			OR = {
				bpm_right_unionist_ig_government = bpm_left_unionist_ig_government
				AND = {
					bpm_minimum_for_dominance_government >= bpm_left_unionist_ig_government
					bpm_minimum_for_dominance_government >= bpm_right_unionist_ig_government
				}
			}
		}
		localization_key = je_bpm_belgium_government_unionist_mixed
	}

	### Dominant Left/Right

	text = {
		trigger = {
			bpm_right_unionist_ig_government < bpm_left_unionist_ig_government
		}
		localization_key = je_bpm_belgium_government_unionist_left
	}

	text = {
		trigger = {
			bpm_right_unionist_ig_government > bpm_left_unionist_ig_government
		}
		localization_key = je_bpm_belgium_government_unionist_right
	}
}
