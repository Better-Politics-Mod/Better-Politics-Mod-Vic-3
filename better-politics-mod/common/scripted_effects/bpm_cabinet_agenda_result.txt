bpm_calrbinet_agenda_result = {
	switch = {
		trigger = var:bpm_calrbinet_agenda_$slot$
		flag:bpm_calrbinet_agenda_land_focus = {
			owner = { bpm_land_focus_decide_success = yes }
			trigger_event = {
				id = bpm_calrbinet_agenda_events.2
				popup = yes
			}
		}
	}
}


# LAND FOCUS

bpm_land_focus_decide_success = {
	# We have three ministerial axes
	random_list = {
		1 = { 
			set_variable = {
				name = bpm_land_focus_result_economy
			}
		}
		1 = {
			remove_variable = bpm_land_focus_result_economy
		}
	}
	random_list = {
		1 = { 
			set_variable = {
				name = bpm_land_focus_result_police
			}
		}
		1 = {
			remove_variable = bpm_land_focus_result_police
		}
	}
	random_list = {
		1 = { 
			set_variable = {
				name = bpm_land_focus_result_education
			}
		}
		1 = {
			remove_variable = bpm_land_focus_result_education
		}
	}
}

bpm_land_focus_effect = {
	bpm_land_focus_effect_economy = yes
}

bpm_land_focus_effect_economy = {
	if = {
		limit = {
			has_variable = bpm_land_focus_result_economy
		}
		bpm_land_focus_effect_economy_success = yes
	}
	else = {
		bpm_land_focus_effect_economy_failure = yes
	}
}


## Success

bpm_land_focus_effect_economy_success = {
	if = {
		limit = {
			has_variable = bpm_minister_of_economy
		}
		var:bpm_minister_of_economy = {
			switch = {
				trigger = is_interest_group_type
				ig_petty_bourgeoisie = {
					owner = {
						bpm_land_focus_effect_economy_success_bureaucrats = yes
					}
				}
			}
		}
	}
}

bpm_land_focus_effect_economy_success_bureaucrats = {
	if = {
		limit = {
			has_law = law_type:law_council_republic
		}
		bpm_add_law_enactment_boost_specified = { law = collectivized_agriculture }
	}
	else_if = {
		limit = {
			has_law = law_type:law_serfdom
		}
		bpm_add_law_enactment_boost_specified = { law = tenant_farmers }
	}
	else_if = {
		limit = {
			has_law = law_type:law_tenant_farmers
		}
		bpm_add_law_enactment_boost_specified = { law = commercialized_agriculture }
	}
	else = {
		remove_modifier = bpm_calr_economic_bureaucrat_success_no_aris
		add_modifier = {
			name = bpm_calr_economic_bureaucrat_success_no_aris
			days = 9125
			is_decaying = yes
		}
	}
	add_modifier = {
		name = bpm_calr_economic_bureaucrat_success_bureaustr
		days = 9125
		is_decaying = yes
	}
	bpm_auth_add_exhaustion_normal = yes
}

		name = bpm_calr_economic_rural_folk_success_rurafostr
		days = 9125
		is_decaying = yes
	}
	add_modifier = {
		name = bpm_calr_economic_rural_folk_success_qol
	}
	add_loyalists = {
		value = 0.05
		strata = lower
	}
	add_loyalists = {
		value = 0.1
		pop_type = peasants
	}
}

## Failure

bpm_land_focus_effect_economy_failure = {
	if = {
		limit = {
			has_variable = bpm_minister_of_economy
		}
		var:bpm_minister_of_economy = {
			switch = {
				trigger = is_interest_group_type
				ig_petty_bourgeoisie = {
					owner = {
						bpm_land_focus_effect_economy_failure_bureaucrats = yes
					}
				}
			}
		}
	}
	bpm_auth_add_exhaustion_large = yes
}
