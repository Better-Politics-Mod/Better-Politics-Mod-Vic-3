je_bpm_winds_of_change {
    should_be_pinned_by_default = yes
    group = je_group_tutorial

	icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"

    can_deactivate = yes
    is_shown_when_inactive = {
        is_player = yes
		NOT = { has_variable = bpm_tutorial_1_done }
	}

    possible = {
        custom_tooltip = {
            text = bpm_tutorial_stability
            var:bpm_political_stability > 0.8
        }
        
	}

    complete = {
        custom_tooltip = {
            text = bpm_tutorial_stability
            var:bpm_political_stability <= 0.66
        }
		is_at_war = no
	}

    on_complete = {
		set_variable = bpm_tutorial_1_done
        add_loyalists = {
				value = 0.1
				pop_type = bureaucrats
		}
		trigger_event = { id = bpm_tutorial.1 }
	}

}
