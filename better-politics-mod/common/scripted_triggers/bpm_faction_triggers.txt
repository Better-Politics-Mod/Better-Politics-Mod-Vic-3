bpm_movement_is_significant_faction = {
	OR = {
		AND = {
			scope:polmov = { is_political_movement_type = movement_liberal }
			OR = { 
				is_interest_group_type = ig_intelligentsia
				is_interest_group_type = ig_industrialists
			}
		}
		AND = {
			scope:polmov = { is_political_movement_type = movement_royalist_absolutist }
			is_interest_group_type = ig_landowners
		}
	}
}

bpm_faction_not_expelled = {
	owner = {
		any_political_movement = {
			is_political_movement_type = $MOVEMENT$
			NOT = {
				any_supporting_character = {
					modifier:character_bpm_character_faction_expelled_from_ig = yes
					interest_group = root
				}
			}
		}
	}
}
