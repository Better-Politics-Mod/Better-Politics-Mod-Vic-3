bpm_ig_is_radical_opposition_status_quo_dop = {
	owner = {
		NAND = {
			has_law_or_variant = law_type:law_separated_powers
			OR = { 
				has_law_or_variant = law_type:law_anti_strike_laws
				has_law_or_variant = law_type:law_right_to_associate
			}
			trigger_if = {
				limit = {
					ig:ig_christian_democrats = { has_ideology = ideology:ideology_dop_divine_democrat }
				}
				OR = {
					has_law_or_variant = law_type:law_census_voting
					has_law_or_variant = law_type:law_wealth_voting
				}
			}
			trigger_else_if = {
				limit = { 
					ig:ig_christian_democrats = { has_ideology = ideology:ideology_dop_conservative_democrat }
				}
				OR = {
					has_law_or_variant = law_type:law_census_voting
					has_law_or_variant = law_type:law_universal_suffrage
				}
			}
			trigger_else_if = {
				limit = {
					has_ideology = ideology:ideology_dop_christian_democrat
				}
				has_law_or_variant = law_type:law_universal_suffrage
			}
			trigger_else = {
				always = yes
			}
		}
	}
}

	custom_description = {
		text = bpm_ig_is_radical_opposition_tooltip
		bpm_ig_is_ideological = yes
		OR = {
			# Monarchy: Republicans are Radical Opposition
			# Exception: Conservatives are not radical opposition in monarchies
			bpm_ig_is_radical_opposition_republican_in_monarchy = yes
	
			# Non-Democracy: Democrats/Anarchists are Radical Opposition
			# Exception: Conservatives are not radical opposition in non-democracies
			bpm_ig_is_radical_opposition_democrat_in_undemocratic = yes
	
			# Democracy: Authoritarians are Radical Opposition 
			bpm_ig_is_radical_opposition_authoritarian_in_democracy = yes
	
			# Illegal Labor Unions: Labor radicals are Radical Opposition
			bpm_ig_is_radical_opposition_unionist_in_illegal_unions = yes

			# Non-council republic: Socialists are Radical Opposition
			bpm_ig_is_radical_opposition_non_cr_socialist = yes
	
			# Fascists: Fascists are always Radical Opposition, tbh
			bpm_ig_is_radical_opposition_fascist = yes
			
			# Anarchists: Anarchists are always Radical Opposition as well
			bpm_ig_is_radical_opposition_anarchist = yes
	
			# Council Republic: Non-Socialist, Non-institutional IGs are Radical Opposition
			bpm_ig_is_radical_opposition_cr_non_socialist = yes

			# Christian Democrats: Christians Democrats do not like the status quo

			bpm_ig_is_radical_opposition_status_quo = yes

			# French Monarchists oppose rival dynaties
			bpm_ig_is_radical_opposition_orleanists = yes
			bpm_ig_is_radical_opposition_legitimists = yes
			bpm_ig_is_radical_opposition_bonapartists = yes

			# Miguelists and Carlists want their line on throne
			bpm_ig_is_radical_opposition_carlists = yes
			bpm_ig_is_radical_opposition_miguelists = yes

			# State Atheism: Moralist igs do not like state atheis
			bpm_ig_is_radical_oppostion_moralists_in_state_atheism = yes

			# Catholics dont like being discriminated
			bpm_ig_is_radical_opposition_discriminated_catholics = yes

			# Ideological igs under orgdem who don't support it
			bpm_ig_is_radical_opposition_organic_democracy = yes
