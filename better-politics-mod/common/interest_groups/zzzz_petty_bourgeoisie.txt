		}

			limit = { is_pop_type = clerks }

				desc = "POP_CLERKS"
				value = 100

		}
		else_if = {
			limit = { is_pop_type = bureaucrats }
			add = {
				desc = "POP_BUREAUCRATS"
				value = 250
			}
		}
		else_if = {
			limit = { is_pop_type = capitalists }
			add = {
				desc = "POP_CAPITALISTS"

			limit = {

				desc = "POP_ARISTOCRATS_CON"

			}
		}

		if = {

			limit = {
