bpm_ig_reset_ideologies = {
    bpm_remove_all_eco_ideologies = yes
    bpm_remove_all_gov_ideologies = yes
    bpm_remove_all_dop_ideologies = yes
    bpm_remove_all_rel_ideologies = yes
    bpm_remove_all_mil_ideologies = yes
    bpm_remove_all_fem_ideologies = yes
    bpm_remove_all_cit_ideologies = yes
    bpm_remove_all_wel_ideologies = yes
    bpm_remove_all_sta_ideologies = yes
    bpm_remove_all_exe_ideologies = yes
    bpm_remove_all_ban_ideologies = yes
}

bpm_ig_set_ideology_no_re = {
    bpm_remove_all_$type$_ideologies = yes
    add_ideology = ideology_$type$_$name$
}

#=====================================
# Armed Forces
#=====================================

bpm_update_ideology_armed_forces = {
    ig:ig_armed_forces ?= {
        bpm_ig_reset_ideologies = yes

        ## DEFAULT
        bpm_ig_set_ideology_no_re = {
            type = mil
            name = institutional_militarist
        }
        bpm_ig_set_ideology_no_re = {
            type = sta
            name = institutional_state
        }
        bpm_ig_set_ideology_no_re = {
            type = dop
            name = oligarch
        }
    }

    # CONDITIONAL

    if = {
        limit = {
            has_variable = red_army_var
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = mass_army
            }
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = party_socialist
            }
        }
    }

    if = {
        limit = {
			active_law:lawgroup_economic_system = {
				has_amendment = amendment_type:bpm_amendment_artisanal_economy
			}
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = eco
                name = artisan
            }
        }
    }


    ig:ig_armed_forces ?= { bpm_reorder_ideologies = yes }
}
