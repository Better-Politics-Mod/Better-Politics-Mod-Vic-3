        bpm_ig_reset_ideologies = yes
        bpm_trait_gov_$gov$ = yes
        bpm_trait_exe_$exe$ = yes
        bpm_trait_dop_$dop$ = yes
        bpm_trait_sta_$sta$ = yes
        bpm_trait_eco_$eco$ = yes
        bpm_trait_wel_$wel$ = yes
        bpm_trait_cit_$cit$ = yes
        bpm_trait_rel_$rel$ = yes
        bpm_trait_mil_$mil$ = yes
        bpm_trait_fem_$fem$ = yes
        bpm_trait_ban_$ban$ = yes
        bpm_trait_vanilla_$vanilla$ = yes
    }
}

bpm_apply_traits_ig_deactivatable = {
    ig:ig_$ig$ = {
        if = {
            limit = {
                bpm_interest_group_is_activated = yes
            }
            bpm_ig_reset_ideologies = yes
            bpm_trait_gov_$gov$ = yes
            bpm_trait_exe_$exe$ = yes
            bpm_trait_dop_$dop$ = yes
            bpm_trait_sta_$sta$ = yes
            bpm_trait_eco_$eco$ = yes
            bpm_trait_wel_$wel$ = yes
            bpm_trait_cit_$cit$ = yes
            bpm_trait_rel_$rel$ = yes
            bpm_trait_mil_$mil$ = yes
            bpm_trait_fem_$fem$ = yes
            bpm_trait_ban_$ban$ = yes
            bpm_trait_vanilla_$vanilla$ = yes
        }
        else = {
            bpm_ig_reset_ideologies = yes
            bpm_trait_has_vanilla_ideology = { ideology = ideology_nonexistent }   
        }
    }
}

bpm_check_deactivation = {
    if = {
        limit = {
            bpm_interest_group_is_activated = no
        }
        bpm_ig_reset_ideologies = yes
        bpm_trait_has_vanilla_ideology = { ideology = ideology_nonexistent }    
    }
}
