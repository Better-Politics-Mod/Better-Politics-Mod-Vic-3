ig_petty_bourgeoisie = {
	color = hsv{ 0 0 0.5 }
	texture = "gfx/interface/icons/ig_icons/petty_bourgeoisie.dds"
	layer = "revolution_dynamic_petty_bourgeoisie"
	index = 5

	ideologies = {
		ideology_plutocratic
	}
	
	enable = {
		always = yes
	}	
	
	on_enable = {
		ig:ig_petty_bourgeoisie = {
			set_ig_trait = ig_trait:ig_trait_urban_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_urban_well_oiled_machine
			set_ig_trait = ig_trait:ig_trait_urban_systematic_opposition
		}
	}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {	
	}	
	
	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_ACADEMICS"
			if = {
				limit = { 
					is_pop_type = academics
				}
				value = 50
			}
		}	
		
		add = {
			desc = "POP_CLERKS"
			if = {
				limit = { 
					is_pop_type = clerks
				}
				value = 100
			}
		}		
		
		add = {
			desc = "POP_BUREAUCRATS"
			if = {
				limit = { 
					is_pop_type = bureaucrats
				}
				value = 250
			}
		}
		
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 50
			}
		}				
		
		add = {
			desc = "POP_OFFICERS"
			if = {
				limit = { 
					is_pop_type = officers
				}
				value = 20
			}
		}		

		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = { 
					is_pop_type = aristocrats
				}
				value = 50
			}
		}		
				
		add = {
			desc = "NONPRIMARY_CULTURE"
			if = {
				limit = {
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					pop_has_primary_culture = no
				}
				value = 100
			}
		}	

		add = {
			desc = "POLITICAL_MACHINES"
			if = {
				limit = {
					strata = middle
					owner = {
						OR = {
							has_law = law_type:law_landed_voting
							has_law = law_type:law_wealth_voting
							has_law = law_type:law_census_voting
							has_law = law_type:law_appointed_bureaucrats
						}
					}
				}
				value = 100
			}
		}	
		add = {
			desc = "POLITICAL_MACHINES"
			if = {
				limit = {
					strata = poor
					owner = {
						OR = {
							has_law = law_type:law_census_voting
							has_law = law_type:law_elected_bureaucrats
						}
					}
				}
				value = 50
			}
		}	

		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}
	
	monarch_weight = {
		value = 0.25
	}
	
	commander_weight = {
		value = 1.0
	}	
	
	noble_chance = {
		value = 0.5
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.001

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
