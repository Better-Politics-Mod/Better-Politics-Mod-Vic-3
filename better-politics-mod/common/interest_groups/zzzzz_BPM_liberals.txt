ig_liberals = {
	color = hsv{ 0.16 1 1 }
	texture = "gfx/interface/icons/ig_icons/liberals.dds"
	layer = "revolution_dynamic_industrialists"
	index = 12
	
	ideologies = {
		ideology_nonexistent
		
		# Vanilla
		ideology_liberal
		ideology_laissez_faire
		ideology_individualist
	}
	
	character_ideologies = {
		ideology_center_leader
		ideology_republican_leader
		ideology_liberal_leader
		ideology_radical
		ideology_market_liberal
		ideology_protectionist
		ideology_land_reformer
		ideology_reformer
	}

	enable = {
		always = yes
	}
	
	on_enable = {
		bpm_update_name_liberals = yes
	}
	on_disable = {}

	priority_cultures = { #Country scope
		rule = {
			trigger = {
				c:BIC ?= this
				any_primary_culture = { 
					cu:british = this
				}
			}
			cultures = {
				assamese avadhi baluchi bengali bihari gujarati kannada malayalam marathi oriya panjabi pathan rajput sindi tamil telegu hindustani bundeli pahari gondi bageli chhattisgarhi deccani
			}
		}
	}

	on_character_ig_membership = {}
	
	pop_potential = {
		custom_tooltip = {
			text = BPM_VALID_POP_FOR_IG
			owner ?= {
				has_technology_researched = rationalism
				ig:ig_liberals ?= {
					NOT = { has_ideology = ideology:ideology_nonexistent }
					bpm_interest_group_is_activated = yes
				}
			}
		}
	}
	
	pop_weight = {
		value = 40

		if = {
			limit = { is_pop_type = peasants }
			if = {
				limit = {
					owner = {
						this = c:SWI
					}
				}
				add = {
					desc = "POP_INCLUSIONS"
					value = 60
				}	
			}
			else = {
				add = {
					desc = "POP_EXCLUSIONS"
					value = -50
				}
			}
		}
		else_if = {
			limit = { is_pop_type = aristocrats }
			add = {
				desc = "POP_ARISTOCRATS"
				value = 60
			}
			if = {
				limit = { 
					owner = {
						has_law_or_variant = law_type:law_monarchy
						OR = {
							has_law_or_variant = law_type:law_landed_voting
							has_law_or_variant = law_type:law_autocracy
						}
					}
				}
				add = {
					desc = "ARISTOCRAT_REFORMERS"
					value = 30
				}
			}
			else_if = {
				if = {
					limit = { 
						owner = {
							has_law_or_variant = law_type:law_monarchy
							has_law_or_variant = law_type:law_wealth_voting
						}
					}
					add = {
						desc = "ARISTOCRAT_REFORMERS"
						value = 20
					}
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
				NOT = { is_pop_type = clergymen }
			}
			add = {
				desc = "MIDDLE_CLASS"
				value = 30
			}
		}

		if = {
			limit = { OR = {
				is_pop_type = capitalists 
				is_pop_type = shopkeepers
			} }
			add = {
				desc = "POP_CAPITALISTS"
				value = var:bpm_consciousness_liberal_igs_capitalists_shopkeepers
			}
		}
		else_if = {
			limit = {
				OR = {
					is_pop_type = academics
					is_pop_type = engineers
				}
			}
			add = {
				desc = "POP_ACADEMICS"
				value = var:bpm_consciousness_liberal_igs_academics_engineers
			}
		}
		else = {
			add = {
				desc = "LABOR_CONSCIOUSNESS"
				value = var:bpm_consciousness_liberal_igs_others
			}
		}

		# URBAN EMPLOYMENT
		if = {
			limit = {
				bpm_pop_employment_is_rural = no
			}
			add = {
				value = 30
				desc = "URBAN"
			}
			if = {
				limit = {
					strata = middle
					NOT = { is_pop_type = clergymen }
				}
				add = {
					value = 100
					desc = "URBAN_MIDDLE_CLASS"
				}
			}
			else_if = {
				limit = { strata = upper }
				add = {
					value = 50
					desc = "URBAN_UPPER_CLASS"
				}
			}
		}

		if = {
			limit = { 
				AND = {
					owner = { bpm_performance_significant_country = yes }
					NOT = { owner = { has_variable = bpm_2nd_party_system } }
				}
			}

			add = {
				desc = "SLAVERY_DETRACTORS"
				if = {
					limit = {
						owner = { NOT = { has_law_or_variant = law_type:law_slavery_banned } }
						OR = {
							pop_acceptance <= acceptance_status_3
							state = { is_slave_state = no }
							AND = {
								state = { is_incorporated = no }
								owner = { has_law = law_type:law_colonial_slavery }
							}
						}
					}
					value = 100
				}
			}

			multiply = {
				desc = "LEADER_POPULARITY"
				scope:interest_group = {
					leader = {
						value = popularity
						multiply = 0.005
						add = 1
						max = 1.5
						min = 0.67
					}
				}
			}
			multiply = bpm_ig_agitators_popularity
			multiply = bpm_ig_ruler_popularity
			multiply = { desc = bpm_ig_attraction_from_radicals_and_loyalists_tt value = ig_attraction_from_radicals_and_loyalists }
		}
		multiply = {
			desc = "UNEDUCATED"
			value = literacy_rate
			multiply = 1.5
			max = 1.0
		}
		if = {
			limit = { game_date < 1855.1.1 }
			if = {
				limit = {
					owner = {
						NOT = { has_variable = completed_peoples_springtime_je }
						has_global_variable = bpm_je_march_to_radicalism_progress
						is_in_geographic_region = geographic_region_europe
						country_has_voting_franchise = no
					}
				}
				multiply = {
					desc = "PEOPLES_SPRINGTIME_BUILDUP"
					value = global_var:bpm_je_march_to_radicalism_progress
					multiply = 0.033 # 0.05
				}
			}
		}
		multiply = ig_attraction_to_ruler_ideology_from_stability
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
        scope:interest_group = { multiply = { value =  bpm_ig_revolutionary_government_attraction desc = "bpm_ig_revolutionary_government_attraction" } }
	}
	
	monarch_weight = {
		owner.ig:ig_liberals = {
			value = bpm_ig_monarch_weight
		}
	}
	
	agitator_weight = {
		owner.ig:ig_liberals = {
			value = bpm_base_agitator_weight
		}
	}
	
	commander_weight = {
		owner.ig:ig_liberals = {
			value = bpm_commander_weight
		}
	}	

	executive_weight = {
		value = 0.5
		if ={
			limit = {
				owner.ig:ig_liberals = { bpm_interest_group_is_activated = no }
			}
			value = 0.0
		}
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}

	female_executive_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
