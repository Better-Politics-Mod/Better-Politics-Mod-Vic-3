namespace = bpm_provisional_government_events

bpm_provisional_government_events.1 = {
	type = country_event
	placement = root
	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	title = bpm_provisional_government_events.1.t
	desc = bpm_provisional_government_events.1.d
	flavor = bpm_provisional_government_events.1.f

	duration = 3

	trigger = {
		#Triggered by on_action
	}

	immediate = {
		remove_variable = bpm_can_new_constitution
	}

	option = {
		name = bpm_provisional_government_events.1.a
		default_option = yes
        add_modifier = {
            name = bpm_new_constitution
            months = 12
        }
		add_loyalists = {
			value = 0.2
		}
		add_radicals = {
			value = -0.1
		}
	}
}

bpm_provisional_government_events.2 = {
	type = country_event
	placement = root
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = bpm_provisional_government_events.2.t
	desc = bpm_provisional_government_events.2.d
	flavor = bpm_provisional_government_events.2.f

	duration = 3

	trigger = {
		#Triggered by JE
	}

	immediate = {
		any_interest_group = {
			if = {
				limit = {
					has_variable = bpm_promised_ig
				}
				ordered_preferred_law = {
					min = 0
					max = 2
					check_range_bounds = no
					save_scope_as = government_petition_law
				}
				set_variable = { name = petitioning_ig days = 100 }
			}
        }
	}

	cancellation_trigger = {
		NOR = {
			exists = scope:government_petition_ig
			exists = scope:government_petition_law
		}
	}

	option = {
		name = bpm_provisional_government_events.2.a
		default_option = yes
        if = {
            limit = {
                exists = scope:government_petition_law
            }
            set_variable = { name = desired_law_var value = scope:government_petition_law }
        }
		custom_tooltip = government_petition_tt
		add_journal_entry = {
			type = je_government_petition
		}
	}
	

}
