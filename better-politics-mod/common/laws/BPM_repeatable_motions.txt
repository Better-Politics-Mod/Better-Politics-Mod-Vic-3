	group = lawgroup_repeatable_motions

		bpm_decorative_repeatable_effect_no_permanent = yes
		bpm_decorative_repeatable_effect_temporary = yes
		bpm_decorative_repeatable_stronger_from_home_affairs = yes
	}

	on_enact = {
		custom_tooltip = bpm_law_effect_tt_repeatable_politics_military_purge # temp
		hidden_effect = {
			call_election = {
                months = 1
            }
            every_active_party = {

                    }
                }
                add_momentum = -2
            }

		}

		add_modifier = {
			name = bpm_repeatable_law_military_purge_result
			years = bpm_repeatable_law_result_duration
			is_decaying = yes
		}
		kill_population_percent = {
			percent = 0.01
			pop_type = officers
		}
		
		bpm_finalize_repeatable_law = {
			LAW = law_repeatable_no_confidence_motion
		}
	}

	can_enact = {
		bpm_repeatable_law_not_on_cooldown = {
			LAW = law_repeatable_no_confidence_motion
		}
	}
	
    # Must be a democratic parliamentary republic
    
    disallowing_laws = {
        law_autocracy
        law_oligarchy
        law_technocracy
        law_military_junta
    }
	possible_political_movements = { }
	
	unlocking_laws = { 
		law_parliamentary_republic
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
