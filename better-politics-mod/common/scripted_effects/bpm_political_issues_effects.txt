		bpm_setup_issues_varlist_subp = { issue = trade_deficit }
		bpm_setup_issues_varlist_subp = { issue = unemployment }
		bpm_setup_issues_varlist_subp = { issue = anti_socialist }
		bpm_setup_issues_varlist_subp = { issue = anti_monarchist }

# post hook
bpm_post_electoral_mandate_hook = {
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_sol
		}
		set_variable = {
			name = bpm_electoral_mandate_sol_baseline_sol
			value = average_sol
		}
	}
	set_variable = {
		name = bpm_electoral_mandate_progress
		value = 0
	}
	set_variable = {
		name = bpm_electoral_mandate_time_progress
		value = 0
	}
	bpm_clear_macros_issues = yes
}

bpm_electoral_mandate_status_track_special = {
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_sol
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = average_sol
				subtract = var:bpm_electoral_mandate_sol_baseline_sol
				multiply = 4
				floor = yes
				min = 0
				max = 4
			}
		}	
	}
}
