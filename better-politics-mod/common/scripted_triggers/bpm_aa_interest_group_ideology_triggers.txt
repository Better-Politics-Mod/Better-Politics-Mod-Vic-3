bpm_country_has_artisanal_economy = {
    active_law:lawgroup_economic_system = {
        has_amendment = amendment_type:bpm_amendment_artisanal_economy
    }
}

bpm_country_has_republican_tradition = {
    active_law:lawgroup_governance_principles = {
        has_amendment = amendment_type:bpm_amendment_republican_tradition
    }
}

bpm_country_is_new_world_republic = {
    is_in_geographic_region = geographic_region_new_world
    bpm_country_is_republic = yes
}

bpm_has_law_entrenched = {
    has_law_or_variant = law_type:$law$
    any_active_law = {
        type = {
            is_same_law_group_as = law_type:$law$
        }
        NOT = { has_amendment = amendment_type:bpm_amendment_newly_enacted }
    }
}

bpm_was_previous_law = {
    has_variable_map = bpm_previous_laws
    exists = variable_map(bpm_previous_laws|scope:current_law.type)
    "variable_map(bpm_previous_laws|scope:current_law.type)" = law_type:$law$
}

bpm_country_has_inertia_with_law = {
    OR = {
        AND = {
            any_active_law = {
                type = {
                    is_same_law_group_as = law_type:$law$
                }
                save_temporary_scope_as = current_law
            }
            scope:current_law = {
                has_amendment = amendment_type:bpm_amendment_newly_enacted
            }
            bpm_was_previous_law = {
                law = $law$
            }
        }
        bpm_has_law_entrenched = {
            law = $law$
        }
    }
}

bpm_has_republic_entrenched = {
    OR = {
        bpm_has_law_entrenched = { law = law_presidential_republic }
        bpm_has_law_entrenched = { law = law_parliamentary_republic }
        bpm_has_law_entrenched = { law = law_council_republic }
    }
}

bpm_has_inertia_with_republic = {
    OR = {
        bpm_country_has_inertia_with_law = { law = law_presidential_republic }
        bpm_country_has_inertia_with_law = { law = law_parliamentary_republic }
        bpm_country_has_inertia_with_law = { law = law_council_republic }
    }
}
