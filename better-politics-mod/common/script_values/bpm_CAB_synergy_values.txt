bpm_calculate_synergy = {
    value = 0


    every_law = {

        if = {
            limit = {
                NOR = {
                    AND = {
                        scope:BPM_CHAR_1.interest_group ?= {
                            law_stance = {
                                law = prev.type
                                value = neutral
                            }
                        }
                        scope:BPM_CHAR_1 ?= {
                            law_stance = {
                                law = prev.type
                                value = neutral
                            }
                        }
                    }
                    AND = {
                        scope:BPM_CHAR_2.interest_group ?= {
                            law_stance = {
                                law = prev.type
                                value = neutral
                            }
                        }
                        scope:BPM_CHAR_2 ?= {
                            law_stance = {
                                law = prev.type
                                value = neutral
                            }
                        }
                    }
                }
            }
            add = 4

            subtract = bpm_calculate_synergy_subprocess_absolute
        }
    


    }


    divide = BPM_MAX_SYN
    subtract = 0.7

    divide = 0.2

    multiply = scope:bpm_calculate_synergy_multiplier

    min = 0
    max = 1
    round_to = 0.001
}

bpm_calculate_synergy_subprocess_absolute = {
    value = 0

    if = {
        limit = {
            bpm_calculate_synergy_subprocess < 0
        }
        subtract = bpm_calculate_synergy_subprocess
    }
    else = {
        add = bpm_calculate_synergy_subprocess
    }
}

bpm_calculate_synergy_subprocess = {
    value = 0

    if = {
        limit = {
            scope:BPM_CHAR_2.interest_group ?= {
                trigger_if = {
                    limit = {
                        bpm_ig_enabled = yes
                    }
                    law_stance = {
                        law = prev.type
                        value = neutral
                    }
                }
                trigger_else = {
                    always = yes
                }
            }
        }
        scope:BPM_CHAR_1 ?= {
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_approve
                    }
                }
                add = 2
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = approve
                    }
                }
                add = 1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = disapprove
                    }
                }
                add = -1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_disapprove
                    }
                }
                add = -2
            }
        }
    }
    else_if = {
        limit = {
            scope:BPM_CHAR_1.interest_group ?= { bpm_ig_enabled = yes }
        }
        scope:BPM_CHAR_1.interest_group ?= {
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_approve
                    }
                }
                add = 2
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = approve
                    }
                }
                add = 1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = disapprove
                    }
                }
                add = -1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_disapprove
                    }
                }
                add = -2
            }
        }
    }


    if = {
        limit = {
            scope:BPM_CHAR_2.interest_group ?= {
                trigger_if = {
                    limit = {
                        bpm_ig_enabled = yes
                    }
                    law_stance = {
                        law = prev.type
                        value = neutral
                    }
                }
                trigger_else = {
                    always = yes
                }
            }
        }
        scope:BPM_CHAR_2 = {
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_approve
                    }
                }
                add = 2
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = approve
                    }
                }
                add = 1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = disapprove
                    }
                }
                add = -1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_disapprove
                    }
                }
                add = -2
            }
        }
    }
    else_if = {
        limit = {
            scope:BPM_CHAR_2.interest_group ?= { bpm_ig_enabled = yes }
        }
        scope:BPM_CHAR_2.interest_group ?= {
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_approve
                    }
                }
                add = -2
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = approve
                    }
                }
                add = -1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = disapprove
                    }
                }
                add = 1
            }
            if = {
                limit = {
                    law_stance = {
                        law = prev.type
                        value = strongly_disapprove
                    }
                }
                add = 2
            }
        }
    }
}
