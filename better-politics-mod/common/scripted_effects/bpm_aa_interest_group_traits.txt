bpm_trait_vanilla_paternalistic = {
    if = {
        limit = {
            owner = {
                bpm_has_inertia_with_republic = yes
            }
        }
        if = {
            limit = {has_ideology = ideology:ideology_paternalistic}
            remove_ideology = ideology_paternalistic
        }
		bpm_trait_has_vanilla_ideology = { ideology = ideology_republican_paternalistic }
    }
    else = {
        if = {
            limit = {has_ideology = ideology:ideology_republican_paternalistic}
            remove_ideology = ideology_republican_paternalistic
        }
		bpm_trait_has_vanilla_ideology = { ideology = ideology_paternalistic }
    }
}
