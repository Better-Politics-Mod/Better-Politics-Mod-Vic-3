# The Assassination of Abraham Lincoln
acw_followup_events.1 = {
	type = country_event

	left_icon = ruler

	title = acw_followup_events.1.t
	desc = acw_followup_events.1.d
	flavor = acw_followup_events.1.f


	event_image = {
		video = "southamerica_public_figure_assassination"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		exists = c:USA
		c:USA = ROOT
		NOT = { has_variable = fords_theatre }
		ruler = {
			has_variable = abraham_lincoln
		}
	}

	immediate = {
		set_variable = {
			name = fords_theatre
			value = yes
		}
		random_scope_character = {
			limit = {
				has_variable = abraham_lincoln
			}
			save_scope_as = lincoln_proxy
			interest_group = {
				save_scope_as = lincoln_proxy_ig
			}
		}
	}

	option = {
		name = acw_followup_events.1.a
        default_option = yes

		scope:lincoln_proxy = {
			kill_character = yes
		}
		scope:lincoln_proxy_ig = {
			add_modifier = {
				name = lincoln_martyred
				months = normal_modifier_time
			}
		}
	}
}

# Forty Acres and a Mule
acw_followup_events.4 = {
	type = country_event

	title = acw_followup_events.4.t
	desc = acw_followup_events.4.d
	flavor = acw_followup_events.4.f
	placement = scope:land_redistribution_state

	event_image = {
		video = "asia_dead_cattle_poor_harvest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	cooldown = { months = short_modifier_time }

	trigger = { # JE pulse
		any_scope_state = {
			has_modifier = unreconstructed_state
			state_region = { 
				is_homeland = cu:afro_american 
			}
			NOT = {
				has_modifier = usfp_land_redistribution
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = { 
				has_modifier = unreconstructed_state
				state_region = { 
					is_homeland = cu:afro_american 
				}
				NOT = {
					has_modifier = usfp_land_redistribution
				}
			}
			save_scope_as = land_redistribution_state
		}
	}

	option = {
		name = acw_followup_events.4.a
		default_option = yes
		add_modifier = {
			name = usfp_enforcing_reconstruction
			years = 1
		}
		scope:land_redistribution_state = {
			add_culture_standard_of_living_modifier = {
				culture = cu:afro_american
				months = normal_modifier_time
				multiplier = 1
			}
			add_modifier = {
				name = usfp_land_redistribution
				months = normal_modifier_time
			}
			add_loyalists_in_state = {
				culture = cu:afro_american
				value = large_radicals
			}
			add_radicals_in_state = {
				value = 0.25
				culture = cu:dixie
				strata = rich
			}
			add_radicals_in_state = {
				value = small_radicals
				culture = cu:dixie
			}
		}
		usfp_reconstruction_swing_freedmen = yes
	}


	option = {
		name = acw_followup_events.4.b

		scope:land_redistribution_state = {
			add_radicals_in_state = {
				culture = cu:afro_american
				value = 0.1
			}
			add_loyalists_in_state = {
				value = 0.25
				culture = cu:dixie
				strata = rich
			}
		}
		usfp_reconstruction_swing_south = yes
	}
}
