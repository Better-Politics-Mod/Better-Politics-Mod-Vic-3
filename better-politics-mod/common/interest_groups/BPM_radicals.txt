ig_radicals = {
	color = hsv{ 0.11 1 1 }
	texture = "gfx/interface/icons/ig_icons/radicals.dds"
	layer = "revolution_dynamic_intelligentsia"
	index = 13
	
	ideologies = {
		ideology_gov_radical_republican
		ideology_dop_radical_democrat
		ideology_sta_progressive_state
		ideology_cit_liberal_nationalist
		ideology_rel_secularist
		ideology_mil_patriotic
		ideology_eco_modernizer
		ideology_wel_public_welfare
		ideology_fem_gender_egalitarian
		ideology_ban_averse
	}
	
	enable = {
		always = yes
	}
	
	on_enable = {
		ig:ig_radicals = {
			# set_ig_trait = ig_trait:ig_trait_left_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_left_bastion_of_freedom
			set_ig_trait = ig_trait:ig_trait_left_systematic_opposition
		}
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
	}
	
	pop_weight = {
		value = 20

		add = {
			desc = "SLAVERY_DETRACTORS"
			if = {
				limit = {
					owner = { NOT = { has_law = law_type:law_slavery_banned } }
					OR = {
						pop_has_primary_culture = no
						state = { is_slave_state = no }
					}
				}
				value = 50
			}
		}

		add = {
			desc = "POP_EXCLUSIONS"
			if = {
				limit = { 
					is_pop_type = peasants
				}
				value = -10
			}
		}
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 100
			}
		}	

		add = {
			desc = "MIDDLE_CLASS"
			if = {
				limit = { 
					strata = middle
				}
				value = 50
			}
		}
		add = {
			desc = "URBAN_POP"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 40
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.005
					add = 1
					max = 1.5
					min = 0.67
				}
			}
		}	

		multiply = bpm_ig_agitators_popularity

		multiply = {
			desc = "LIBERALS_SUPPRESSED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_radicals }
							bpm_ig_is_liberal = yes
							is_being_suppressed = yes
						}
					}
				}
				value = 1.1
			}		
		}	
		multiply = {
			desc = "CONSERVATIVES_BOLSTERED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							bpm_ig_is_right_wing = yes
							is_being_bolstered = yes
						}
					}
				}
				value = 1.1
			}		
		}	

		multiply = {
			desc = "UNEDUCATED"
			value = literacy_rate
			multiply = 1.4
			max = 1.0
		}

		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = democracy }
					}
				}
				value = 0.5
			}		
		}
		multiply = {
			desc = "NO_DEMOCRACY"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = democracy }
					}
				}
				value = 0.0
			}		
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}	
		multiply = ig_attraction_from_radicals_and_loyalists
	}
	
	monarch_weight = {
		value = 0.1
	}	

	agitator_weight = {
		value = 1.0
		multiply = bpm_base_agitator_weight
	}
	
	commander_weight = {
		value = 1.0
		multiply = bpm_commander_assignment_from_laws_factor
	}		
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
