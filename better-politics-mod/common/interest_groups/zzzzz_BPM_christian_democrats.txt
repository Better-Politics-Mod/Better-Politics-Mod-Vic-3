ig_christian_democrats = {
	color = hsv{ 0.52 0.85 1.0 }
	texture = "gfx/interface/icons/ig_icons/christian_democrats.dds"	
	layer = "revolution_dynamic_rural_folk"
	index = 8

	ideologies = {
		ideology_nonexistent
	}

	character_ideologies = {
		ideology_center_leader
		ideology_protectionist
		ideology_sovereignist_leader
		ideology_liberal_leader
		#ideology_corporatist_leader
		ideology_reformer
	}
	
	enable = {
		always = yes
	}
	
	on_enable = {
		ig:ig_christian_democrats ?= {
			# set_ig_trait = ig_trait:ig_trait_left_systematic_corruption
			#set_ig_trait = ig_trait:ig_trait_left_bastion_of_freedom
			#set_ig_trait = ig_trait:ig_trait_left_systematic_opposition
		}
		if = {
			limit = {
				NOR = {
					country_has_state_religion = rel:catholic
					religion_percent_country = { 
						target = rel:catholic
						value >= 0.5
					}
				}	
			}
			#ig:ig_christian_democrats ?= {
			#	set_interest_group_name = ig_catholic_interests
			#}
		}
	}

	on_character_ig_membership = {
		if = {
			limit = {
				owner = { 
					religion_percent_country = { 
						target = rel:catholic
						value <= 1.0
					}
				}
			}
			change_character_religion = rel:catholic
		}
		else_if = {
			limit = {
				NOT = {
					owner = {
						has_law = law_type:law_state_atheism
					}
				}
			}
			change_character_religion = scope:interest_group.owner.religion
		}
		else = {
			change_character_religion = scope:interest_group.owner.country_definition.religion
		}
	}

	on_disable = {}	
	
	pop_potential = {
		has_pop_religion = catholic
		owner = {
			ig:ig_christian_democrats = {
				bpm_interest_group_is_activated = yes
			}
		}
		custom_tooltip = {
			text = BPM_VALID_POP_FOR_IG
			owner = {
				NOT = { is_country_type = decentralized }
				exists = ig:ig_christian_democrats
				ig:ig_christian_democrats ?= { NOT = { has_ideology = ideology:ideology_nonexistent } }
				ig:ig_christian_democrats = { bpm_interest_group_is_activated = yes }
			}
		}
	}
	
	pop_weight = {
		value = 50

        # Attraction from being Catholic
        if = {
            limit = { 
                has_pop_religion = catholic
				owner = {
					NOT = {
						country_has_state_religion = rel:catholic
					}
				}
            }
            add = { 
                desc = "Catholic Religion"
                value = 100
            }
        }

		# CLERGYMEN LIKE LOYALTY TO THE POPE

		else_if = {
			limit = { 
				is_pop_type = clergymen 
			}
			add = {
				desc = "POP_CLERGYMEN"
				value = 50
			}
		}

		# IDEOLOGY-BASED ATTRACTION: ECONOMY

		if = {
			limit = {
				owner = {
					ig:ig_christian_democrats = {
						has_ideology = ideology:ideology_eco_distrubitist # CSP example:
					}
				}
			}
			if = {
				limit = { strata = lower }
				add = {
					desc = "LOWER_CLASS"
					value = 25
				}
			}
			else_if = {
				limit = { strata = middle }
				add = {
					desc = "MIDDLE_CLASS"
					value = 75
				}
			}
			else_if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					desc = "POP_CAPITALISTS"
					value = 100
				}
			}
			else_if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					desc = "POP_ARISTOCRATS"
					value = 10
				}
			}
		}

		else_if = {
			limit = {
				owner = {
					ig:ig_christian_democrats = {
						has_ideology = ideology:ideology_eco_modern_interventionist # Popularists, Young Right, Dutch Christian Democrats
					}
				}
			}
			if = {
				limit = { strata = lower }
				add = {
					desc = "LOWER_CLASS"
					value = 50
				}
			}
			else_if = {
				limit = { strata = middle }
				add = {
					desc = "MIDDLE_CLASS"
					value = 100
				}
			}
			else_if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					desc = "POP_CAPITALISTS"
					value = 75
				}
			}
		}

		else_if = {
			limit = {
				owner = {
					ig:ig_christian_democrats = {
						has_ideology = ideology:ideology_eco_centralized_economy # Young Zentrumites, late Polish Chadecy
					}
				}
			}
			if = {
				limit = { strata = lower }
				add = {
					desc = "LOWER_CLASS"
					value = 75
				}
			}
			else_if = {
				limit = { strata = middle }
				add = {
					desc = "MIDDLE_CLASS"
					value = 100
				}
			}
			else_if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					desc = "POP_CAPITALISTS"
					value = -250
				}
			}
			else_if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					desc = "POP_ARISTOCRATS"
					value = -250
				}
			}
		}

		else_if = {
			limit = {
				owner = {
					ig:ig_christian_democrats = {
						has_ideology = ideology:ideology_eco_modern_agrarian # Bavarian People's Party, Lithuanian Christian Democrats
					}
				}
			}
			# Always rural due to pop potential
			add = {
				desc = "RURAL_POP"
				value = 50
			}		

			if = {
				limit = { is_pop_type = farmers }
				add = {
					desc = "POP_FARMERS"
					value = 100
				}
			}
			else_if = {
				limit = { is_pop_type = peasants }
				add = {
					desc = "POP_PEASANTS"
					value = 50
				}
			}

			else_if = {
				limit = { strata = middle }
				add = {
					desc = "MIDDLE_CLASS"
					value = 25
				}
			}
			else_if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					desc = "POP_ARISTOCRATS"
					value = 5
				}
			}
		}

		# IDEOLOGY-BASED ATTRACTION: DEMOCRACY

		if = {
			limit = {
				owner = {
					ig:ig_christian_democrats = {
						has_ideology = ideology:ideology_dop_iliberal_democrat # CSP, LKDP
					}
				}
			}
			else_if = {
				limit = { strata = lower }
				add = {
					desc = "LOWER_CLASS"
					value = 10
				}
			}
			
		}

		else_if = {
			limit = {
				owner = {
					ig:ig_christian_democrats = {
						has_ideology = ideology:ideology_dop_divine_democrat # Belgians, Dutch, general tendencies
					}
				}
			}
			else_if = {
				limit = { strata = middle }
				add = {
					desc = "MIDDLE_CLASS"
					value = 5
				}
			}
		}

		else_if = {
			limit = {
				owner = {
					ig:ig_christian_democrats = {
						has_ideology = ideology:ideology_dop_christian_democrat
					}
				}
			}
			if = {
				limit = { is_pop_type = laborers }
				add = {
					desc = "POP_LABORERS"
					value = 10
				}
			}
			else_if = {
				limit = { is_pop_type = machinists }
				add = {
					desc = "POP_MACHINISTS"
					value = 5
				}
			}
		}

        #Attraction from being discriminated (see the efforts by Zentrum under Poles etc)
        add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 20
			}
		}	

        multiply = { #stronger in non-Catholic countries (see Zentrum, Brabant/Limburg in the Netherlands)
            desc = "Catholic Minority"    
        
            value = 1.0 
            if = { 
                limit = { 
                    owner = { 
                        religion_percent_country = { 
                            target = rel:catholic
                            value <= 0.5 
                        }
                    }
                }
                value = 2.0
            }
        }
        multiply = { #has researched corporatism
            desc = "Has Corporatism"
            value = 1.0 
            if = { 
                limit = { 
                    owner = { 
                        has_technology_researched = corporatism
                    }
                }
                value = 1.25
            }
        }

		multiply = { #has researched mass parties
            desc = "Has Mass Parties"
            value = 0.65
            if = { 
                limit = { 
                    owner = { 
                        has_technology_researched = socialism
                    }
                }
                value = 1.0
            }
        }

		if = {
			limit = { owner = { bpm_performance_significant_country = yes } }
			multiply = {
				desc = "LEADER_POPULARITY"
				scope:interest_group = {
					leader = {
						value = popularity
						multiply = 0.005
						add = 1
						max = 1.5
						min = 0.67
					}
				}
			}	
			multiply = {
				desc = "STILL_OLIGARCHIC"
				value = 1.0
				if = {
					limit = {
						owner = {
							has_law = law_type:law_oligarchy
							has_law = law_type:law_monarchy
						}
					}
					value = 0.5
				}
			}
			multiply = bpm_ig_agitators_popularity
			multiply = bpm_ig_ruler_popularity
			multiply = ig_attraction_from_radicals_and_loyalists
		}
		multiply = ig_attraction_to_ruler_ideology_from_stability
	}
	
	monarch_weight = {
		value = 0.5
	}	

	agitator_weight = {
		value = 1.0
		multiply = bpm_base_agitator_weight
		if ={
			limit = {
				bpm_interest_group_is_activated = no
			}
			value = 0.0
		}
	}
	
	commander_weight = {
		value = 1.0
		multiply = bpm_commander_assignment_from_laws_factor
	}	
	
	noble_chance = {
		value = 0.2
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.5
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
