bpm_war_prep_decided_success = {
	bpm_decide_success = { type = war_prep ministry = military }
	bpm_decide_success = { type = war_prep ministry = economy }
	bpm_decide_success = { type = war_prep ministry = diplomacy }
}

	bpm_war_prep_effect_military = yes
	bpm_war_prep_effect_economy = yes

### MILITARY SUCCESS

bpm_war_prep_effect_military = {
	if = {
		limit = {
			has_variable = bpm_war_prep_result_military
		}
		bpm_war_prep_effect_military_success = yes
	}
	else = {
		bpm_war_prep_effect_military_failure = yes

bpm_war_prep_effect_economy = {
	if = {
		limit = {
			has_variable = bpm_war_prep_result_economy
		}
		bpm_war_prep_effect_economy_success = yes
	}
	else = {

	}
}

bpm_war_prep_effect_economy_success = {
	if = {
		limit = {
			has_variable = bpm_minister_of_economy
		}
		var:bpm_minister_of_economy = {
			switch = {
				trigger = is_interest_group_type
				
				ig_landowners = {
					owner = { bpm_war_prep_effect_economy_success_landowners = yes }
				}
				ig_armed_forces = {
					owner = { bpm_war_prep_effect_economy_success_armed_forces = yes }
				}
				ig_fascists = {
					owner = { bpm_war_prep_effect_economy_success_fascists = yes }
				}
				ig_putschists = {
					owner = { bpm_war_prep_effect_economy_success_fascists = yes }
				}
			}
		}
	}
	
	# Base effect for all
	set_variable = {
		name = bpm_war_prep_economy_next_war_bonus
		months = 120
	}
}

bpm_war_prep_effect_economy_success_landowners = {
	set_variable = {
		name = bpm_war_prep_economy_next_war_bonus_weak
		months = 120
	}
	remove_variable = bpm_war_prep_economy_next_war_bonus

bpm_war_prep_effect_economy_success_armed_forces = {
