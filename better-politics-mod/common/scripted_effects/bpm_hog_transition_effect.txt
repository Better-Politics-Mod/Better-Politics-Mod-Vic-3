bpm_create_hog_heir_hook = {
    if = {
        limit = {
            THIS = c:JAP
            has_law = law_type:law_bakufu
        }
		if = {
			limit = {
				NOT = {
					has_variable = bpm_JAP_iesada_tokugawa
				}
			}
			create_character = {
				template = JAP_iesada_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
				}
			}
			set_variable = bpm_JAP_iesada_tokugawa
		}
		else_if = {
			limit = {
				NOT = {
					has_variable = bpm_JAP_iemochi_tokugawa
				}
			}
			create_character = {
				template = JAP_iemochi_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
				}
			}
			set_variable = bpm_JAP_iemochi_tokugawa
		}
		else = {
			create_character = {
				#heir = yes
				#first_name = Ieyoshi
				last_name = Tokugawa
				#historical = yes
				#birth_date = 1793.6.22
				age = 23
				interest_group = ruler.interest_group
				#ideology = ideology_traditionalist
				#traits = {
				#	meticulous
				#	pious
				#}
				on_created = {
					owner = {
						set_variable = house_tokugawa
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
				}
			}
		}
    }
    else = {
        create_character = {
			#heir = yes
			#first_name = Ieyoshi
			#last_name = Tokugawa
			#historical = yes
			#birth_date = 1793.6.22
            age = 23
			interest_group = ruler.interest_group
			#ideology = ideology_traditionalist
			#traits = {
			#	meticulous
			#	pious
			#}
			on_created = {
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
			}
		}
    }
}
