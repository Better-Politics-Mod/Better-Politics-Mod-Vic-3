		country_legitimacy_govt_total_clout_add = 100
		country_legitimacy_govt_size_add = 1
		country_legitimacy_ideological_incoherence_mult = 0.2
		country_aristocrats_voting_power_add = 30
		country_officers_voting_power_add = 10
		country_clergymen_voting_power_add = 10
		country_capitalists_voting_power_add = 30
		country_shopkeepers_voting_power_add = 5
		country_farmers_voting_power_add = 5
		country_authority_add = 100
	}
	
	possible_political_movements = {
		law_oligarchy
		law_autocracy 
		law_weighted_wealth_voting
		law_wealth_voting
		law_census_voting
		law_weighted_universal_suffrage
		law_universal_suffrage
		law_anarchy
	}

	on_activate = {
		scripted_effect_parties_emergence = yes
	}
	
	pop_support = {
		value = 0

		add = {
			desc = "POP_LITERACY" 
			
			if = {
				limit = { 
					OR = {
						NOT = { strata = rich }
						wealth < 30
					}					
					owner = { 
						NOR = {
							has_law = law_type:law_weighted_wealth_voting
							has_law = law_type:law_wealth_voting
							has_law = law_type:law_census_voting
							has_law = law_type:law_weighted_universal_suffrage
							has_law = law_type:law_universal_suffrage
							has_law = law_type:law_anarchy
						}	
					}
				}
				add = {
					value = literacy_rate
					divide = 2
				}
			}
		}			
	}
}
