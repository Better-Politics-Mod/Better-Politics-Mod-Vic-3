### WELFARE SUCCESS

bpm_popular_welfare_effect_welfare = {
	if = {
		limit = {
			has_variable = bpm_popular_welfare_result_welfare
		}
		bpm_popular_welfare_effect_welfare_success = yes
	}
	else = {
		bpm_popular_welfare_effect_welfare_failure = yes
	}
}

bpm_popular_welfare_effect_welfare_success = {
	# Auto-upgrade welfare/safety institutions if upgrading
	if = {
		limit = {
			has_institution = institution_welfare
			institution:institution_welfare = {
				is_being_upgraded = yes
			}
		}
		institution:institution_welfare = {
			complete_upgrade = yes
		}
	}
	
	if = {
		limit = {
			has_institution = institution_workplace_safety
			institution:institution_workplace_safety = {
				is_being_upgraded = yes
			}
		}
		institution:institution_workplace_safety = {
			complete_upgrade = yes
		}
	}
	
	add_loyalists = {
		value = 0.05
		strata = lower

		limit = {
			has_variable = bpm_minister_of_social_security

			switch = {
				trigger = is_interest_group_type

	bpm_auth_add_exhaustion_huge = yes
	
	if = {
		limit = {
			has_institution = institution_welfare
			institution:institution_welfare = {
				is_at_level >= 2
			}
		}
		institution:institution_welfare = {
			downgrade = yes
		}
	}
	
	if = {
		limit = {
			has_institution = institution_workplace_safety
			institution:institution_workplace_safety = {
				is_at_level >= 2
			}
		}
		institution:institution_workplace_safety = {
