bpm_right_click_menu_spirit_make_popular_gui = {
	scope = political_movement
	is_shown = {
		has_game_rule = bpm_allow_spirit_mechanics
	}
	is_valid = {
		NOT = {
			has_modifier = bpm_spirit_movement_bolstered
		}
		bpm_spirit_value >= bpm_spirit_amount_large
	}
	effect = {
		add_modifier = { name = bpm_spirit_movement_bolstered is_decaying = yes years = 5 }
		bpm_spirit_spend_large = yes
	}
}
