bpm_state_capacity = {
	value = 0
    if = {
        limit = {
            has_variable = bpm_state_capacity
        }
        add = var:bpm_state_capacity
    }
}

bpm_state_capacity_target = {
	value = 0
	add = modifier:country_bpm_auth_state_capacity_target_add
}

bpm_state_capacity_trend = {
	value = 0
	add = modifier:country_bpm_auth_state_capacity_trend_add
}

bpm_state_capacity_trend_signed = {
	if = {
		limit = {
			bpm_state_capacity > bpm_state_capacity_target
		}
		value = {
			value = bpm_state_capacity_trend
			multiply = -1
		}
	}
	else = {
		value = bpm_state_capacity_trend
	}
	if = {
		limit = {
			OR = {
				bpm_state_capacity >= 100
				bpm_state_capacity = bpm_state_capacity_trend
				bpm_state_capacity <= 0
			}
		}
		value = 0
	}
}
