
bpm_set_start_cabinet_ministers = {
    if = {
        limit = {
            exists = c:JAP
            this = c:JAP
        }
        # We can't use random_scope_character
        # Because it won't look at void characters
        random_character = {
            limit = {
                has_variable = bpm_tadakuni_mizuno_var
            }
            owner = {
                set_variable = { name = bpm_mineco value = prev }
                set_variable = { name = bpm_minister_of_economy value = prev }
            }

            add_character_role = bpm_character_role_minister
        }
        random_character = {
            limit = {
                has_variable = bpm_masayoshi_hotta_var
            }
            owner = {
                set_variable = { name = bpm_minint value = prev }

                set_variable = { name = bpm_minister_of_centralization value = prev }
                set_variable = { name = bpm_minister_of_police value = prev }
                set_variable = { name = bpm_minister_of_home_affairs value = prev }
            }

            add_character_role = bpm_character_role_minister
        }
    }
    else_if = {
        limit = {
            exists = c:BRZ
            this = c:BRZ
        }
        # We can't use random_scope_character
        # Because it won't look at void characters
        random_character = {
            limit = {
                has_variable = bpm_manuel_do_nascimento_var
            }
            bpm_set_cabinet_minister = { INST = economy }
        }
        random_character = {
            limit = {
                has_variable = bpm_antonio_paulino_var
            }
            bpm_set_cabinet_minister = { INST = police }
        }
        random_character = {
            limit = {
                has_variable = bpm_saturnino_de_sousa_var
            }
            bpm_set_cabinet_minister = { INST = home_affairs }
        }
        random_character = {
            limit = {
                has_variable = bpm_antonio_paulino_var
            }
            bpm_set_cabinet_minister = { INST = centralization }
        }
        random_character = {
            limit = {
                has_variable = bpm_manuel_alves_branco_var
            }
            bpm_set_cabinet_minister = { INST = diplomacy }
        }
    }
}
