bpm_front_crossroad_takes_over = {
	bpm_front_has_supportive_general = { front = $front$ }
	bpm_front_controls_capital = { front = $front$ }
	bpm_country_has_strong_center = yes
}

bpm_front_crossroad_bloodless = {
	bpm_very_low_state_capacity = yes
}

# whether a front has a strong supportive general in the country
bpm_front_has_supportive_general = {
	any_scope_character = {
		interest_group = { owner = { is_target_in_variable_list = {
			name = bpm_front_$front$_members
			target = prev
		} } }
		commander_rank >= 2
        loyalty < define:NCharacters|LOYALTY_UNCERTAIN_THRESHOLD
	}
}

# whether the front controls the capital of the country
bpm_front_controls_capital = {
	capital = {
		has_variable = bpm_revcell_$front$
	}
}

# whether the country has a strong center
# determines if it will simply fall after
# the capital is taken or if it will put up
# a fight and potentially cause a civil war
bpm_country_has_strong_center = {
	NOR = {
		has_law = law_type:law_ad_hoc_centralization
		has_law = law_type:law_confederated
	}
}
