namespace = vormaerz

vormaerz.1 = { # Legacy of the Code Civil
	type = country_event
	placement = scope:code_civil_state
	title = vormaerz.1.t
	desc = vormaerz.1.d
	flavor = vormaerz.1.f

	duration = 3

	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
		any_scope_state = {
			has_state_trait = state_trait_rhine_river
		}
        NOT = { has_variable = code_civil_happened }
	}

	immediate = {
		set_variable = code_civil_happened
		random_scope_state = {
			limit = { 
				has_state_trait = state_trait_rhine_river 
				root ?= owner
			}
			save_scope_as = code_civil_state
		}

	}

	option = {
		name = vormaerz.1.a
		default_option = yes

		every_state = {
			limit = { 
				has_state_trait = state_trait_rhine_river 
				root ?= owner
			}
			add_modifier = {
				name = vormaerz_rheinrecht_modifier
				days = -1
			}
		}

    }
}

vormaerz.4 = { # Poverty in the country side
	type = country_event
	placement = root
	title = vormaerz.4.t
	desc = vormaerz.4.d
	flavor = vormaerz.4.f

	duration = 3

	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
		NOT = { has_variable = pauperism_happened }
		year > 1840 # Maybe this?
		any_scope_state = {
			owner ?= root
			any_scope_pop = {
				is_pop_type = peasants
				standard_of_living < 8
			}
		}

	}

	immediate = {
		set_variable = pauperism_happened
	}

	option = {
		name = vormaerz.4.a
		default_option = yes
		add_radicals = {
			value = medium_radicals
			pop_type = peasants
		}
		every_scope_state = {
			limit = {
				owner ?= root
				any_scope_pop = {
					is_pop_type = peasants
					standard_of_living < 8
				}
			}
			add_modifier = {
				name = vormaerz_pauperism_modifier
				days = -1
			}
			add_radicals_in_state = {
				value = large_radicals
				pop_type = peasants
			}
		}

    }
}
