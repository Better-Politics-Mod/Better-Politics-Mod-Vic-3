bpm_je_general_strike_crackdown = {
	name = "bpm_je_general_strike_crackdown_button"
	desc = "bpm_je_general_strike_crackdown_button_desc"

	visible = { always = yes }

	possible = {
		OR = {
			has_institution = institution:institution_home_affairs
			has_institution = institution:institution_police
		}
		NOT = { has_law = law_type:law_elected_police }
		custom_tooltip = {
			text = bpm_je_button_cooldown_week
			NOT = {
				has_variable = bpm_je_button_cooldown_week
			}
		}
	}

	effect = {
		hidden_effect = {
			set_variable = {
				name = bpm_je_button_cooldown_week
				days = 7
			}
		}
		add_radicals = {
			value = medium_radicals
			interest_group = ig:ig_trade_unions
		}
		custom_tooltip = {
			text = bpm_je_general_strike_add_strike_exhaustion_large
			change_variable = {
				name = bpm_je_general_strike_exhaustion
				add = {
					add = 10
					multiply = institution:institution_police.investment
				}
			}
		}
		custom_tooltip = {
			text = bpm_je_general_strike_remove_strike_progress_small
			change_variable = {
				name = bpm_je_general_strike_progress
				subtract = {
					add = 2
					multiply = institution:institution_police.investment
				}
			}
		}
	}
}
