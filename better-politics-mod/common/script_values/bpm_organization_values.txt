bpm_organization_trend = {
	value = 0
	if = {
		limit = {
			bpm_ig_is_institutional = yes
		}
		# Trend 1/50 of the way to resting point
		# Example if current org is 30 and resting point is 20, then trend will be -0.2
		add = {
			value = bpm_organization_ing_resting_point
			subtract = var:bpm_ig_organization
			divide = 50
			desc = "ORG_TREND_TO_RESTING_POINT_INSTITUTIONAL"
		}
	}
	else = {
		if = {
			limit = {
				bpm_ig_is_proletarian = yes
			}
			add = {
				value = owner.bpm_consciousness_labor_percent
				if = {
					limit = {
						OR = {
							is_interest_group_type = ig_revolutionist_socialists
							is_interest_group_type = ig_anarchists
						}
					}
					multiply = 2
				}
				desc = "From Labor Consciousness"
			}
		}
		else_if = {
			limit = {
				bpm_ig_is_liberal = yes
			}
			add = {
				value = owner.bpm_consciousness_liberal_percent
				if = {
					limit = {
						OR = {
							is_interest_group_type = ig_radicals
							is_interest_group_type = ig_national_liberals
						}
					}
					multiply = 2
				}
				desc = "From Liberal Consciousness"
			}
		}
	}
}

bpm_organization_ing_resting_point = {
	value = {
		# Default resting point is 20
		value = 20
		desc = "ORG_RP_DEFAULT"
	}
	if = {
		limit = {
			is_interest_group_type = ig_armed_forces
		}
		add = {
			value = 20
			desc = "ORG_RP_ARMED_FORCES"
		}
		if = {
			limit = {
				owner = {
					has_law = law_type:law_professional_officers
				}
			}
			add = {
				value = 20
				desc = "ORG_RP_PROFESSIONAL_OFFICERS"
			}
		}
	}
	if = {
		limit = {
			is_interest_group_type = ig_landowners
			owner = {
				has_law = law_type:law_hereditary_bureaucrats
			}
		}
		add = {
			value = 30
			desc = "ORG_RP_LANDOWNERS_HEREDITARY_BUREAUCRATS"
		}
	}
	if = {
		limit = {
			is_interest_group_type = ig_petty_bourgeoisie
		}
		add = {
			value = 20
			desc = "ORG_RP_BUREAUCRATS"
		}
		if = {
			limit = {
				owner = {
					has_law = law_type:law_hereditary_bureaucrats
				}
			}
			subtract = {
				value = 10
				desc = "ORG_RP_BUREAUCRATS_HEREDITARY_BUREAUCRATS"
			}
		}
		else_if = {
			limit = {
				owner = {
					has_law = law_type:law_appointed_bureaucrats
				}
			}
			add = {
				value = 30
				desc = "ORG_RP_BUREAUCRATS_APPOINTED_BUREAUCRATS"
			}
		}
	}
}
