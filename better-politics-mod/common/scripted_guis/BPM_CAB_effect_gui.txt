bpm_dismiss_character_from_stack = {
    scope = interest_group
    is_valid = {
        custom_tooltip = {
            text = "bpm_cab_select_ig_tt"
            OR = {
                var:bpm_skilled_characters > 0
                var:bpm_unskilled_characters > 0
                var:bpm_expert_characters > 0
            }
        }   
    }
    effect = {
        var:bpm_selected_character_igshow = {
            kill_character = yes
        }
        remove_variable = bpm_selected_character_igshow
        hidden_effect = {
            save_scope_as = bpm_selected_ig
            owner = {
                bpm_generate_cabinet_character = yes
            }
        }
        scope:bpm_selected_character = {
            bpm_character_skill_level_init = yes
        }
        set_variable = {
            name = bpm_selected_character_igshow
            value = scope:bpm_selected_character
        }
        owner = { 
            clear_variable_list = bpm_synergy_calculation_charset
            add_to_variable_list = {
                name = bpm_synergy_calculation_charset
                target = ruler
            }
            add_to_variable_list = {
                name = bpm_synergy_calculation_charset
                target = scope:bpm_selected_character
            }
            bpm_calculate_synergy = yes
            prev = { set_variable = {
                name = bpm_compatibility_with_ruler
                value = owner.var:bpm_temp_synergy_calculation
            } }        
        }

    }
}

bpm_CAB_toggle_char_select = {
	scope = institution
	is_shown = {
	}
	is_valid = {
	}
	effect = {
        owner = { 
            set_variable = {
                name = bpm_cabinet_char_select
                value = prev
            }
            clear_variable_list = bpm_CAB_ig_select_pool
            var:bpm_cabinet_char_select.type = {
                save_scope_as = bpm_reference_institution_type
            }
            bpm_reload_government_mandate = yes
            ordered_interest_group = {
                limit = {
                    OR = {
                        owner = { bpm_country_has_parliamentary_electoral_system = no }
                        is_in_government = yes
                    }
                    ig_clout >= 0.01
                }
                order_by = ig_clout
                min = 0
                owner = { add_to_variable_list = {
                    name = bpm_CAB_ig_select_pool
                    target = prev
                } }
                set_variable = {
                    name = bpm_cost_ig_position_var
                    value = bpm_cost_ig_position
                }

            }

            every_interest_group = {
                remove_variable = bpm_selected_character_igshow
                hidden_effect = {
                    save_scope_as = bpm_selected_ig
                    owner = {
                        bpm_generate_cabinet_character = yes
                    }
                }
                scope:bpm_selected_character = {
                    bpm_character_skill_level_init = yes
                }
                set_variable = {
                    name = bpm_selected_character_igshow
                    value = scope:bpm_selected_character
                }
            }
            bpm_reset_mandate_used = yes
            

            #bpm_fix_cabinet_pool = yes
            #bpm_fill_leader_pool = yes
            every_in_list = {
                variable = bpm_CAB_ig_select_pool
                save_scope_as = bpm_selected_ig
                hidden_effect = {
                    owner = {
                        bpm_generate_cabinet_character = yes
                    }
                }
                scope:bpm_selected_character ?= {
                    bpm_cab_character_calculate_hyp_synergy = yes
                }
                leader = { 
                    set_variable = {
                        name = bpm_cab_char_synergy
                        value = scope:bpm_selected_character.var:bpm_cab_char_synergy
                    }
                }
                owner = { 
                    clear_variable_list = bpm_synergy_calculation_charset
                    add_to_variable_list = {
                        name = bpm_synergy_calculation_charset
                        target = ruler
                    }
                    add_to_variable_list = {
                        name = bpm_synergy_calculation_charset
                        target = scope:bpm_selected_character
                    }
                    bpm_calculate_synergy = yes
                    prev = { set_variable = {
                        name = bpm_compatibility_with_ruler
                        value = owner.var:bpm_temp_synergy_calculation
                    } }        
                }
            }
            
        }
	}

bpm_CAB_toggle_pm_select = {
	scope = institution
	is_shown = {
	}
	is_valid = {
        owner = {
            OR = {
                has_law = law_type:law_appointed_by_hos
                has_law = law_type:law_appointed_on_advice
            }
        }
	}
	effect = {
        owner = { 
            remove_variable = bpm_cabinet_char_select
            set_variable = bpm_cabinet_pm_select
            clear_variable_list = bpm_CAB_pm_select_pool
            every_interest_group = {
                if = {
                    limit = {
                        owner = { has_law = law_type:law_appointed_by_hos }
                        NOT = { this = owner.ruler.interest_group }
                    }
                    owner = {
                        add_to_variable_list = {
                            name = bpm_CAB_pm_select_pool
                            target = prev.leader
                        }
                    }
                }
                else_if = {
                    limit = {
                        owner = { has_law = law_type:law_appointed_on_advice }
                        NOT = { this = owner.ruler.interest_group }
                        is_powerful = yes
                    }
                    owner = {
                        add_to_variable_list = {
                            name = bpm_CAB_pm_select_pool
                            target = prev.leader
                        }
                    }
                }
            }
            owner.ruler.interest_group = {
                save_scope_as = bpm_selected_ig
            }
            bpm_generate_cabinet_character = yes
            scope:bpm_selected_character = {
                set_variable = bpm_pm_candidate_var
            }
            add_to_variable_list = {
                name = bpm_CAB_pm_select_pool
                target = scope:bpm_selected_character
            }
            # bpm_reset_mandate_used = yes
            # bpm_reload_government_mandate = yes

            every_in_list = {
                variable = bpm_CAB_pm_select_pool
                # TODO FOR 3.0
                #bpm_cabinet_calculate_synergy_hypothetical_head_of_government = yes
            }
        }
	}
}

bpm_newcab_select_ig = {
    scope = interest_group
    is_shown = {
    }
    is_valid = {
        custom_tooltip = {
            text = "bpm_cab_select_ig_tt"
            OR = {
                var:bpm_skilled_characters > 0
                var:bpm_unskilled_characters > 0
                var:bpm_expert_characters > 0
            }
        }   
        # custom_tooltip = {
        #     text = "bpm_cab_select_ig_enough_mandate_tt"
        #     bpm_remaining_mandate >= var:bpm_cost_ig_position_var
        # }
        trigger_if = {
            limit = {
                owner = {
                    OR = {
                        bpm_country_has_executive_dictatorial_system = yes
                        bpm_country_has_executive_electoral_system = yes
                    }
                }
            }
            OR = {
                custom_tooltip = {
                    text = "bpm_compatibility_with_ruler"
                    var:bpm_compatibility_with_ruler < 0.35
                }
                var:bpm_selected_character_igshow.interest_group ?= owner.ruler.interest_group
            }

        }

    }
    effect = {
        save_scope_as = bpm_selected_ig
        hidden_effect = {
            owner = {
                bpm_generate_cabinet_character = yes
            }
        }
        if = {
            limit = {
                owner = {
                    has_variable = bpm_cabinet_pm_select
                }
            }
            scope:bpm_selected_character ?= {
                interest_group = {
                    bpm_update_mandate_used = { VALUE = bpm_cost_ig_position }
                }
                set_variable = {
                    name = bpm_mandate_usage_pos
                    value = interest_group.bpm_cost_ig_position
                }
                set_variable = {
                    name = bpm_mandate_usage_neg
                    value = var:bpm_mandate_usage_pos
                }
                change_variable = {
                    name = bpm_mandate_usage_neg
                    multiply = -1
                }
                bpm_appoint_as_hog = yes
            }
        }
        else = {
            scope:bpm_selected_character ?= {
                bpm_add_to_cabinet_interaction_shared_effect = yes
            }
        }
    }
}

bpm_toggle_inst = {
    scope = institution
    effect = {
        if = {
            limit = {
                owner = {
                    OR = {
                        AND = {
                            je:je_bpm_cabinet_agendas = { OR = {
                                var:bpm_cabinet_agenda_slot_one ?= flag:bpm_no_cabinet_agenda
                                var:bpm_cabinet_agenda_slot_two ?= flag:bpm_no_cabinet_agenda
                                var:bpm_cabinet_agenda_slot_three ?= flag:bpm_no_cabinet_agenda
                            } }
                            NOT = {
                                is_target_in_variable_list = {
                                    name = bpm_focused_institutions
                                    target = prev
                                }
                            }
                        }
                        is_target_in_variable_list = {
                            name = bpm_focused_institutions
                            target = prev
                        }
                    }
                }
            }
            owner = {
                if = {
                    limit = {
                        is_target_in_variable_list = {
                            name = bpm_focused_institutions
                            target = prev
                        }
                    }
                    remove_list_variable = {
                        name = bpm_focused_institutions
                        target = prev
                    }
                }
                else = {
                    add_to_variable_list = {
                        name = bpm_focused_institutions
                        target = prev
                    }
                }
                je:je_bpm_cabinet_agendas = {
                    save_scope_as = journal_entry
                }
                bpm_lock_slot_if_no_skill = yes 
            }
        }
    }
}
