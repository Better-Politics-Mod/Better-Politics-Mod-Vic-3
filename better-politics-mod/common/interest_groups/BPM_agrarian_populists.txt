ig_agrarian_populists = {
	color = hsv{ 0.32 0.67 0.75 }
	texture = "gfx/interface/icons/ig_icons/agrarian_populists.dds"	
	layer = "revolution_dynamic_rural_folk"
	index = 8

	ideologies = {
		ideology_dop_radical_democrat
		ideology_sta_decentralized_state
		ideology_cit_liberal_nationalist
		ideology_rel_secularist
		ideology_mil_antimilitarist
		ideology_eco_modern_agrarian
		ideology_ban_averse
	}
	
	enable = {
		always = yes
	}
	
	on_enable = {
		ig:ig_agrarian_populists = {
			# set_ig_trait = ig_trait:ig_trait_left_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_left_bastion_of_freedom
			set_ig_trait = ig_trait:ig_trait_left_systematic_opposition
		}
	}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
		}
	}
	
	pop_weight = {
		value = 20
	
		add = {
			desc = "EXCLUSIONS"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations				
					}
					NOT = {
						is_pop_type = academics
					}
				}
				value = -100
			}
		}		

		add = {
			desc = "RURAL_POP"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations				
					}
				}
				value = 50
			}
		}		

		add = {
			desc = "POP_FARMERS"
			if = {
				limit = { 
					is_pop_type = farmers
				}
				value = 100
			}
		}	

		add = {
			desc = "POP_EDUCATED_FARMERS"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations				
					}
					strata <= middle
				}
				value = literacy_rate
				multiply = 100
			}
		}	

		add = {
			desc = "POP_RURAL_ABOLITIONISTS"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations				
					}
					owner = {
						OR = {
							has_law = law_type:law_serfdom
							has_law = law_type:law_slave_trade
							has_law = law_type:law_legacy_slavery
						}
					}
				}
				value = 80
			}
		}

		add = {
			desc = "POP_RURAL_ANTICAPITALISM"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations				
					}
					scope:interest_group = {
						OR = {
							bpm_ig_is_socialist = yes
							leader = { bpm_leader_is_socialist = yes }
						}
					}
					strata <= middle
					owner = {
						has_law = law_type:law_commercialized_agriculture
					}
				}
				value = 100
			}
		}

		add = {
			desc = "POP_RURAL_WANTS_COLLECTIVIZATION"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations				
					}
					scope:interest_group = {
						OR = {
							bpm_ig_is_socialist = yes
							leader = { bpm_leader_is_socialist = yes }
						}
					}
					strata <= middle
					owner = {
						has_law = law_type:law_council_republic
						NOT = { has_law = law_type:law_collectivized_agriculture }
					}
				}
				value = 60
			}
		}

		add = {
			desc = "POP_RURAL_HAPPY_UNDER_SOCIALISM"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations				
					}
					scope:interest_group = {
						OR = {
							bpm_ig_is_socialist = yes
							leader = { bpm_leader_is_socialist = yes }
						}
					}
					strata <= middle
					pop_radical_fraction = 0
					owner = {
						has_law = law_type:law_collectivized_agriculture
					}
				}
				value = 150
			}
		}

		add = {
			desc = "POP_FREE_PEASANTS"
			if = {
				limit = { 
					is_pop_type = peasants
					owner = {
						NOR = {
							has_law = law_type:law_serfdom
							has_law = law_type:law_slave_trade
							has_law = law_type:law_legacy_slavery
						}
					}
				}
				value = -40
			}
		}

		add = {
			desc = "POP_LEADERS"
			if = {
				limit = { 
					OR = {
						is_pop_type = academics
					}
					owner = {
						has_law = law_type:law_serfdom
					}
				}
				value = 100
			}
		}

		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
					OR = {
						owner = {
							OR = {
								radical_fraction = {
									value >= 0.5
									pop_type = farmers
								}
								radical_fraction = {
									value >= 0.5
									pop_type = peasants
								}
							}
						}
					}
				}
				value = 20
			}
		}

		add = {
			desc = "POP_CLERGY"
			if = {
				limit = { 
					is_pop_type = clergymen
					owner = {
						OR = {
							radical_fraction = {
								value >= 0.25
								pop_type = farmers
							}
							radical_fraction = {
								value >= 0.25
								pop_type = peasants
							}
						}
					}
				}
				value = 50
			}
		}

		add = {
			desc = "POPS_ANGER_SYMPATHY"
			if = {
				limit = { 
					OR = {
						is_pop_type = clergymen
						is_pop_type = academics
					}
					owner = {
						OR = {
							radical_fraction = {
								value >= 0.2
								pop_type = farmers
							}
							radical_fraction = {
								value >= 0.2
								pop_type = peasants
							}
						}
					}
				}
				value = 100
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			if = {
				limit = {
					owner = {
						any_interest_group = {
							is_interest_group_type = ig_agrarian_populists
						}
					}
				}
				scope:interest_group = {
					leader = {
						value = popularity
						multiply = 0.005
						add = 1
						max = 1.5
						min = 0.67
					}
				}
			}
		}

		multiply = bpm_ig_agitators_popularity

		multiply = {
			desc = "CAN_GET_LABOR_MOVEMENT"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						can_research = labor_movement
					}
				}
				value = 2.0
			}		
		}	

		multiply = {
			desc = "NO_LABOR_MOVEMENT"	
			value = 1.0
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = labor_movement }
					}
				}
				value = 0.5
			}		
		}		
		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = egalitarianism }
					}
				}
				value = 0.0
			}		
		}
		multiply = {
			desc = "REFORMISTS_RADICALS_SUPPRESSED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_agrarian_populists }
							bpm_ig_is_radical_liberal = yes
							is_being_suppressed = yes
						}
					}
				}
				value = 1.2
			}		
		}
		multiply = {
			desc = "CONSERVATIVES_BOLSTERED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							bpm_ig_is_right_wing = yes
							is_being_bolstered = yes
						}
					}
				}
				value = 1.1
			}		
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
		multiply = ig_attraction_from_radicals_and_loyalists
	}
	
	monarch_weight = {
		value = 0.1
	}	

	agitator_weight = {
		value = 1.0
		multiply = bpm_base_agitator_weight
	}
	
	commander_weight = {
		value = 1.0
		multiply = bpm_commander_assignment_from_laws_factor
	}	
	
	noble_chance = {
		value = 0.2
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.5
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
