		movement_support_bureaucrats
		movement_support_clerks
		movement_support_peasants
	}

		add = {
			value = 3
			desc = "POP_BASE_SUPPORT"
		}

		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

			limit = {
				strata = upper
			}

				limit = {
					is_pop_type = capitalists
				} 
				add = {
					value = 12
					desc = "POP_CAPITALISTS"
				}
			}

				limit = {
					is_pop_type = aristocrats
				}
				add = {
					value = 18
					desc = "POP_ARISTOCRATS"
				}
			}
		}

			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 12
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = bureaucrats
				}
				add = {
					value = 12
					desc = "POP_BUREAUCRATS"
				}
			}
			else = {
				add = {
					value = 9
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 6
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants
				}
				add = {
					value = 0
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"

			value = literacy_rate

			multiply = literacy_rate
			add = 0.10
		}
		if = {
			limit = {
				owner ?= {
					any_political_movement = {
						is_political_movement_type = movement_orleanist
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
						save_temporary_scope_as = affected_movement
					}
				}
			}
			if = {
				limit = {
					scope:affected_movement = {
						is_being_suppressed = yes
					}
				}
				multiply = {
					value = 1
					subtract = owner.modifier:country_suppression_attraction_factor
					desc = "concept_suppressed"
				}
			}
			else = {
				multiply = {
					value = 1
					add = owner.modifier:country_bolster_attraction_factor
					desc = "concept_bolstered"
				}
			}
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}	
	
		weight = {
			value = 100
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}		
	}	

	additional_radicalism_factors = {
		if = {
			limit = {
				owner ?= {
					has_law = law_type:law_monarchy
					ruler ?= {
						NOT = {
							has_ideology = ideology:ideology_orleanist
						}
					}
				}
			}
			add = {
				value = 0.25
				desc = "NON_ORLEANIST_MONARCH"
			}
		}		
	}
}

movement_kaikoku = {
	category = movement_category_ideological
	
	ideology = ideology_eco_kaikoku_ha
	character_ideologies = {
		ideology_market_liberal
		ideology_liberal_leader
		ideology_republican_leader
		ideology_radical
		ideology_humanitarian
	} 
	
	creation_trigger = {
		exists = c:JAP
		c:JAP = this
	}

	creation_weight = {
		value = 9999999
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	disband_trigger = {
		has_variable = meiji_var
	}
	
	character_support_trigger = {
		trigger_if = {
			limit = {
				is_in_exile_pool = yes
			}
			OR = {
				has_ideology = ideology:ideology_market_liberal
				has_ideology = ideology:ideology_liberal_leader
				has_ideology = ideology:ideology_republican_leader
				has_ideology = ideology:ideology_radical
				has_ideology = ideology:ideology_humanitarian				
			}
		}
		trigger_else = {
			OR = {
				has_ideology = ideology:ideology_market_liberal
				has_ideology = ideology:ideology_liberal_leader
				has_ideology = ideology:ideology_republican_leader
				has_ideology = ideology:ideology_radical
				has_ideology = ideology:ideology_humanitarian
				interest_group ?= {
					OR = {
						has_ideology = ideology:ideology_liberal
						has_ideology = ideology:ideology_liberal_modern
					}
				}
			}
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_republican_leader
					has_ideology = ideology:ideology_liberal_leader
					has_ideology = ideology:ideology_market_liberal
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_radical
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_radical
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_humanitarian
				NOT = {
					interest_group = { is_interest_group_type = ig_trade_unions }
				}
			}
			multiply = {
				value = 5
			}
		}
		if = { # Liberal agitators just inherently love liberalism
			limit = {
				interest_group ?= {
					OR = {
						has_ideology = ideology:ideology_liberal
						has_ideology = ideology:ideology_liberal_modern
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}

	can_pressure_interest_group = {
		c:JAP ?= owner
	}

		NOT = {
			is_pop_type = slaves

		movement_support_aristocrats

							exists = ig:ig_liberals
						}
					}

				add = {

					desc = "MIDDLE_NO_ICON"
				}
			}
			if = {
				limit = {
					owner = {
						NOT = {
							exists = ig:ig_market_liberals
						}
					}
				}
				multiply = 6
			}
			if = {
				limit = {
					owner.ruler ?= {
						interest_group = { NOR = { 
							is_interest_group_type = ig_industrialists
							is_interest_group_type = ig_liberals
							is_interest_group_type = ig_market_liberals
							is_interest_group_type = ig_national_liberals
						} }
					}
				}
				multiply = 2
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			add = {
				value = 3
				desc = "LOWER_NO_ICON"
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"

			value = literacy_rate

			multiply = literacy_rate
			add = 0.10
		}
		multiply = 0.5
		multiply = { value = BPM_LIBERAL_STRENGTH desc = BPM_LIBERAL_AND_CONS } 
		if = {
			limit = {
				owner ?= {
					any_political_movement = {
						is_political_movement_type = movement_kaikoku
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
						save_temporary_scope_as = affected_movement
					}
				}
			}
			if = {
				limit = {
					scope:affected_movement = {
						is_being_suppressed = yes
					}
				}
				multiply = {
					value = 1
					subtract = owner.modifier:country_suppression_attraction_factor
					desc = "concept_suppressed"
				}
			}
			else = {
				multiply = {
					value = 1
					add = owner.modifier:country_bolster_attraction_factor
					desc = "concept_bolstered"
				}
			}
		}

	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.15
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = state_urbanization_rate	
		}
	}
	
	secession = {

			always = no
		}

			value = 1000
		}

		}

					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_schools"
				}		
			}	
			if = {
				limit = {
					bpm_country_is_pre_modern_absolutist = yes
				}
				add = {
					value = 0.1
					desc = "IS_PRE_MODERN_ABSOLUTIST"
				}
			}			
		}
	}
