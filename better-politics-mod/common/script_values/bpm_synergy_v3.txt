bpm_synergy_v3 = {
	#// Synergy disagreement categories
	#value = bpm_synergy_v3_denominator

	add = bpm_synergy_disagreement_governance # // 0 to 40
	add = bpm_synergy_disagreement_democracy # // 0 to 40
	add = bpm_synergy_disagreement_economy # // 0 to 40
	add = bpm_synergy_disagreement_nationalism # // 0 to 10
	add = bpm_synergy_disagreement_religion # // 0 to 10
	add = bpm_synergy_disagreement_security # // 0 to 10
	add = bpm_synergy_disagreement_unions # // 0 to 5
	add = bpm_synergy_disagreement_workers #// 0 to 10
	if = {
		limit = {
			NOT = {
				has_law = law_type:law_slavery_banned
			}
		}
		add = bpm_synergy_disagreement_slavery
	}
	if = {
		limit = {
			has_law = law_type:law_isolationism
		}
		add = bpm_synergy_disagreement_isolation
	}
	if = {
		limit = {
			has_law = law_type:law_hereditary_bureaucrats
		}
		add = bpm_synergy_disagreement_aristocracy
	}

	divide = bpm_synergy_v3_denominator
	max = 1
	min = 0
	#multiply = 100
}
