REPLACE:ig_armed_forces = {
	color = hsv{ 0.05 0.1 0.4 }
	texture = "gfx/interface/icons/ig_icons/armed_forces.dds"
	layer = "revolution_dynamic_armed_forces"
	index = 0
	
	ideologies = {
		ideology_dop_oligarch
		ideology_sta_institutional_state
		ideology_mil_institutional_militarist
		# Vanilla
		ideology_loyalist
		ideology_patriotic
	}

	character_ideologies = {
		ideology_moderate
		# ideology_orleanist
		# ideology_bonapartist
		ideology_jingoist_leader
		ideology_authoritarian
		# ideology_corporatist_leader
	}
	
	enable = {
		always = yes
	}
	
	on_enable = {
		bpm_update_ideology_armed_forces = yes
		bpm_update_name_armed_forces = yes

		# Put IG initialization here (in an arbitrary IG), so it happens before any on actions (hopefully)
		# This is a bit of a hack, but it's the only way to do it
		if = {
			limit = { game_date > 1836.1.1 }
			owner = {
				bpm_copy_variable_from_civil_war_origin = yes
			}
		}
	}
	on_disable = {}

	priority_cultures = { #Country scope
		rule = {
			trigger = {
				c:BIC ?= this
				any_primary_culture = { 
					cu:british = this
				}
			}
			cultures = {
				british scottish irish
			}
		}
		rule = {
			trigger = {
				c:ZAN ?= this
				any_primary_culture = {
					cu:bedouin = this
				}
			}
			cultures = {
				bedouin swahili baluchi
			}
		}
	}

	
	on_character_ig_membership = {}
	
	pop_potential = {
		pop_acceptance >= acceptance_status_4
		OR = {
			is_pop_type = soldiers
			is_pop_type = officers
			is_pop_type = aristocrats
			is_pop_type = bureaucrats
			is_pop_type = capitalists
			is_pop_type = shopkeepers
		}		
	}
	
	pop_weight = {
		value = 25

		add = {
			desc = "POP_WAR_PROFITEERS"
			if = {
				limit = { 
					OR = {
						pop_employment_building_group = bg_military_industry
					}
				}
				value = 125
			}
		}
		
		if = {
			limit = { is_pop_type = soldiers }
			add = {
				desc = "POP_SOLDIERS"
				value = 50
			}
		}
		else_if = {
			limit = { is_pop_type = officers }
			add = {
				desc = "POP_OFFICERS"
				value = 300
			}
		}
		else_if = {
			limit = { is_pop_type = aristocrats }
			add = {
				desc = "POP_ARISTOCRATS"
				value = 100
			}
		}
		else_if = {
			limit = { is_pop_type = bureaucrats }
			add = {
				desc = "POP_BUREAUCRATS"
				value = 50
			}
		}

		multiply = ig_attraction_to_interests_from_stability
		multiply = {
			desc = "IN_WAR"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						is_at_war = yes
					}
				}
				value = 2
			}		
		}		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01
					add = 1
					max = 2.0
					min = 0.75
				}
			}
		}	
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
	}
	
	monarch_weight = {
		owner.ig:ig_armed_forces = {
			value = bpm_ig_monarch_weight
		}
	}

	agitator_weight = {
		owner.ig:ig_armed_forces = {
			value = bpm_base_agitator_weight
		}
	}
	
	commander_weight = {
		owner.ig:ig_armed_forces = {
			value = bpm_commander_weight
		}
	}

	executive_weight = {
		value = 0.1
	}
	
	noble_chance = {
		value = 0.5
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}

	female_executive_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.75
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
