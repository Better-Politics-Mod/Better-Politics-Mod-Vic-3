bpm_cabinet_agenda_result = {
	switch = {
		trigger = var:bpm_cabinet_agenda_$slot$
		flag:bpm_cabinet_agenda_land_focus = {
			owner = { 
				bpm_land_focus_decide_success = yes
				trigger_event = {
					id = bpm_cabinet_agenda_events.2
					popup = yes
				}
			}

		}
	}
}

bpm_decide_success = {
	random_list = {
		1 = {
			set_variable = bpm_$type$_result_$ministry$
		}
		1 = {
			remove_variable = bpm_$type$_result_$ministry$
		}
	}
}
