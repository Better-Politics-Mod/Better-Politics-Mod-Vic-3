bpm_dissent_rev_small_attack_bt = {
	name = "bpm_dissent_rev_small_attack"
	desc = "bpm_dissent_rev_small_attack_desc"

	visible = { 
		#
	}

	ai_chance = {
		base = 100
	}

	possible = {
		NOT = {
			has_variable = bpm_atk_cld
		}
	}

	effect = {
		scope:journal_entry = {
			set_variable = {
				name = bpm_left_alone_counter
				value = 0
			}
			custom_tooltip = {
				text = "bpm_dissent_rev_small_attack_tt"
				change_variable = {
					name = bpm_revolution_resource_amount
					subtract = owner.bpm_revolutionary_damage_normal
				}
			}
			clamp_variable = {
				name = bpm_revolution_resource_amount
				min = 0
				max = var:bpm_revolution_resource_amount_max
			}
		}
		set_variable = {
			name = bpm_atk_cld
			value = yes
			days = 15
		}
		bpm_auth_add_exhaustion_cost = yes
	}
}
