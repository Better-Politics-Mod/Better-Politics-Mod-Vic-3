bpm_update_sup_ig = {
    remove_variable = bpm_sup_ig_$rank$
    ordered_activated_interest_group = {
        position = $rank$
        order_by = ig_clout
        check_range_bounds = no
        limit = {
            supports_law = yes
        }
        save_temporary_scope_as = sup_ig_$rank$
    }
    ordered_activated_interest_group = {
        position = {
            value = $rank$
            subtract = 1
            min = 0
        }
        order_by = ig_clout
        check_range_bounds = no
        limit = {
            supports_law = yes
        }
        save_temporary_scope_as = sup_pred_ig_$rank$
    }
    if = {
        limit = {
            exists = scope:sup_ig_$rank$
            OR = {
                $rank$ = 0
                NOT = {scope:sup_pred_ig_$rank$ = scope:sup_ig_$rank$}
            }
        }
        set_variable = {
            name = bpm_sup_ig_$rank$
            value = scope:sup_ig_$rank$
        }
    }
}
