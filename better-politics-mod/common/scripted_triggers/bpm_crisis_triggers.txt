bpm_is_crisis_or_sticky = {
    has_variable = bpm_active_sticky_crisis
}

bpm_is_crisis = {
    hidden_trigger = {
        NAND = {
            has_law = law_type:law_advisory_body
            has_law = law_type:law_no_separate_hog
        }
    }
    OR = {
        radical_fraction = { value > 0.4 }
        any_political_movement = {
            count >= 2
            AND = { 
                political_movement_radicalism > 0.75
                political_movement_support > 0.15
            }
        }
        any_interest_group = {
            is_powerful = yes
            bpm_ig_is_very_radical_opposition = yes
        }
        any_scope_state = {
            bpm_auth_state_has_collapsed_auth = yes
        }
    }
}

bpm_institutional_appointment_is_hoshog = {
    owner = { 
        modifier:country_bpm_hog_inst = yes
        bpm_country_is_republic = yes
    }
}

bpm_institutional_appointment_is_hosapp = {
    owner = { 
        modifier:country_bpm_hog_inst = yes
        bpm_country_is_republic = no
    }
}

bpm_is_hoshog = {
    owner = { OR = {
        modifier:country_bpm_hog_hoshogsame = yes
        bpm_institutional_appointment_is_hoshog = yes
    } }
}

bpm_is_hosapp = {
    owner = { OR = {
        modifier:country_bpm_hog_top_hosapp = yes
        bpm_institutional_appointment_is_hosapp = yes
    } }
}


bpm_country_has_rising_extremist_during_crisis = {
    has_journal_entry = je_bpm_crises_government_scandal
    je:je_bpm_crises_government_scandal.var:bpm_government_scandal_progress >= 50
}

bpm_country_extremist_exist_or_would_spawn = {
    OR = {
        AND = {
            bpm_consciousness_liberal_percent > 0.75
            bpm_consciousness_labor_percent > 0.35
        }
        any_interest_group = {
            ig_clout > 0.1
            OR = {
                bpm_ig_is_very_radical_opposition = yes
            }        
        }
    }
}

bpm_country_has_devastating_war = {
    save_temporary_scope_value_as = {
        name = casualty_comp
        value = {
            value = total_population
            divide = 1000
        }
    }
    save_temporary_scope_as = target
    any_diplomatic_play = {
        OR = {
            initiator = {this = scope:target}
            target = {this = scope:target}
        }
        war = {
            has_war_support = {
                target = scope:target
                value < 20
            }
            num_country_casualties = {
                target = scope:target
                value > scope:casualty_comp
            }
        }
    }
}
