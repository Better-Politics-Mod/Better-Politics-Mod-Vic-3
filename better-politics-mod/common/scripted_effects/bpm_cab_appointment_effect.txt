bpm_appoint_mineco_effect = {
	owner = {
		if = {
			limit = {
				has_variable = bpm_mineco
			}
			var:bpm_mineco ?= {
				remove_character_role = bpm_character_role_minister
			}
		}
		set_variable = { name = bpm_mineco value = prev }
		set_variable = { name = bpm_minister_of_economy value = prev }
		set_variable = { name = bpm_mineco_last_ig value = prev.interest_group }
	}

	add_character_role = bpm_character_role_minister
	show_as_tooltip = {
		bpm_reload_institution_modifiers_loc_mineco = yes
	}
	hidden_effect = {
		owner = {
			bpm_reload_institution_modifiers = yes
			bpm_update_government_and_opposition = yes
		}
	}
}

bpm_appoint_minfor_effect = {
	owner = {
		if = {
			limit = {
				has_variable = bpm_minfor
			}
			var:bpm_minfor ?= {
				remove_character_role = bpm_character_role_minister
			}
		}
		set_variable = { name = bpm_minfor value = prev }

		set_variable = { name = bpm_minister_of_colonial_affairs value = prev }
		set_variable = { name = bpm_minister_of_diplomacy value = prev }
		set_variable = { name = bpm_minfor_last_ig value = prev.interest_group }
	}

	add_character_role = bpm_character_role_minister
	show_as_tooltip = {
		bpm_reload_institution_modifiers_loc_minfor = yes
	}
	hidden_effect = {
		owner = {
			bpm_reload_institution_modifiers = yes
			bpm_update_government_and_opposition = yes
		}
	}
	# Appointing a Minister of Foreign Affairs
    # This will affect lobbies and current forpol laws
    if = {
        limit = {
            owner = { has_institution = institution_diplomacy }
        }

        # Law Change (If permitted by laws)
        if = {
            limit = { 
				owner = {
					modifier:country_bpm_forpol_minister_appointment_switches_law = yes
					is_player = yes
				}
			}

            if = { # Fash
                limit = {
                    OR = {
                        bpm_leader_is_fascist = yes
                        interest_group = {
                            law_stance = {
                                law = law_type:law_ultramilitarism
                                value = strongly_approve
                            }
                        }
                    }
                }
                owner = { activate_law = law_type:law_ultramilitarism }
            } else_if = { # Hawk (2A)
                limit = {
                    OR = {
                        bpm_leader_is_hawkish = yes
                        AND = {
                            bpm_leader_is_doveish = no
                            interest_group = {
                                law_stance = {
                                    law = law_type:law_jingoism
                                    value >= approve
                                }
                            }
                        }
                    }
                } 
                owner = { activate_law = law_type:law_jingoism }
            } else_if = { # Dove
                limit = {
                    OR = {
                        bpm_leader_is_doveish = yes
                        AND = {
                            bpm_leader_is_hawkish = no
                            interest_group = {
                                law_stance = {
                                    law = law_type:law_dove_politics
                                    value >= approve
                                }
                            }
                        }
                    }
                }
                owner = { activate_law = law_type:law_dove_politics }
            } else = { # Everyone else makes policy flexible
                owner = { activate_law = law_type:law_flexible_foreign_policy }
            }
        }
    }
}

bpm_auto_cabinet_appoint = {
	bpm_auto_cabinet_appoint_pm = yes
	bpm_auto_cabinet_appoint_subp = { m = mineco }
	bpm_auto_cabinet_appoint_subp = { m = minint }
	bpm_auto_cabinet_appoint_subp = { m = minsoc }
	bpm_auto_cabinet_appoint_subp = { m = minmor }
	bpm_auto_cabinet_appoint_subp = { m = minfor }
}

bpm_auto_cabinet_appoint_subp = {
	remove_local_variable = bpm_$m$appt_candidate
	if = {
		limit = {
			bpm_has_gov_regency = yes
		}
		var:bpm_regent = {
			save_scope_as = bpm_new_ruler_scope
		}
	}
	else_if = {
		limit = {
			bpm_has_acting_hog = yes
		}
		var:bpm_acting_hog = {
			save_scope_as = bpm_new_ruler_scope
		}
	}
	else_if = {
		limit = {
			modifier:country_bpm_hog_veto = yes
		}
		random_scope_character = {
			limit = {
				has_role = bpm_character_role_hog
			}
			save_scope_as = bpm_new_ruler_scope
		}
	}
	else = {
		ruler = {
			save_scope_as = bpm_new_ruler_scope
		}
	}

	if = {
		limit = {
			OR = {
				NOT = { has_variable = bpm_$m$ }
				NOT = { exists = var:bpm_$m$ }
				var:bpm_$m$ ?= {
					is_character_alive = no
				}
			}
		}
		if = {
			limit = {
				has_variable = $m$_locked_to
			}
			ordered_scope_character = {
				limit = {
					interest_group = {
						owner.var:$m$_locked_to ?= this
					}
					bpm_char_can_be_minister = yes
				}
				order_by = prominence
				position = 0
				set_local_variable = {
					name = bpm_$m$appt_candidate
					value = this
				}
			}
			set_variable = bpm_used_locked_for_$m$
		}
		else_if = {
			limit = {
				has_variable = bpm_cabappt_strategy
				var:bpm_cabappt_strategy ?= flag:bpm_greedy_ruler
			}
			scope:bpm_new_ruler_scope.interest_group = {
				save_scope_as = minister_ig
			}
			bpm_get_ministerial_candidate = {
				m = $m$
			}
			set_variable = bpm_used_greedy_for_$m$
		}
		else_if = {
			limit = {
				has_variable = bpm_cabappt_strategy
				var:bpm_cabappt_strategy ?= flag:bpm_pick_prominent_idg
			}
			ordered_scope_character = {
				limit = {
					interest_group = {
						custom_tooltip = {
							text = "bpm_not_anti_establishment_tt"
							bpm_ig_is_radical_opposition = no
						}
						custom_tooltip = {
							text = "bpm_not_extremist_tt"
							bpm_ig_is_very_radical_opposition = no
						}
					}
					bpm_char_can_be_minister = yes
					bpm_ig_is_institutional = no
				}
				order_by = prominence
				position = 0
				set_local_variable = {
					name = bpm_$m$appt_candidate
					value = this
				}
			}
			set_variable = bpm_used_prom_$m$
		}
		else_if = {
			limit = {
				has_variable = bpm_cabappt_strategy
				var:bpm_cabappt_strategy ?= flag:bpm_pick_prominent_ing
			}
			ordered_scope_character = {
				limit = {
					interest_group = {
						custom_tooltip = {
							text = "bpm_not_anti_establishment_tt"
							bpm_ig_is_radical_opposition = no
						}
						custom_tooltip = {
							text = "bpm_not_extremist_tt"
							bpm_ig_is_very_radical_opposition = no
						}
					}
					bpm_char_can_be_minister = yes
					bpm_ig_is_institutional = yes
				}
				order_by = prominence
				position = 0
				set_local_variable = {
					name = bpm_$m$appt_candidate
					value = this
				}
			}
			set_variable = bpm_used_prom_$m$
		}
		else_if = {
			limit = {
				has_variable = bpm_$m$_last_ig
				any_scope_character = {
					bpm_char_can_be_minister = yes
					interest_group ?= scope:minister_ig
				}
			}
			var:bpm_$m$_last_ig = {
				save_scope_as = minister_ig
			}
			bpm_get_ministerial_candidate = {
				m = $m$
			}
			set_variable = bpm_used_last_$m$
		}
		else_if = {
			limit = {
				any_scope_character = {
					bpm_char_can_be_minister = yes
					interest_group = {
						is_in_government = yes
					}
				}
			}
			random_scope_character = {
				limit = {
					bpm_char_can_be_minister = yes
					interest_group = {
						is_in_government = yes
					}
				}
				set_local_variable = {
					name = bpm_$m$appt_candidate
					value = this
				}
			}
			set_variable = bpm_used_rand_$m$
		}
		else = {
			every_scope_character = {
				remove_variable = bpm_created_char_appt
			}
			create_character = {
				interest_group = scope:bpm_new_ruler_scope.interest_group
            	on_created = {
					set_variable = bpm_created_char_appt
				}
			}
			random_scope_character = {
				limit = {
					has_variable = bpm_created_char_appt
				}
				set_local_variable = {
					name = bpm_$m$appt_candidate
					value = this
				}
			}
			set_variable = { name = bpm_used_char_cre_$m$ value = local_var:bpm_$m$appt_candidate } 
		}

		set_local_variable = {
			name = bpm_minname
			value = flag:$m$
		}

		local_var:bpm_$m$appt_candidate ?= {
			bpm_appoint_$m$_effect = yes
			set_local_variable = {
				name = bpm_newmin
				value = this
			}
		}

		post_notification = bpm_minister_replaced
	}
}

bpm_get_ministerial_candidate = {
	random_scope_character = {
		limit = {
			bpm_char_can_be_minister = yes
			interest_group ?= scope:minister_ig
		}
		set_local_variable = {
			name = bpm_$m$appt_candidate
			value = this
		}
	}
}

bpm_auto_cabinet_appoint_pm = {
	clear_saved_scope = bpm_newmin

	if = {
		limit = {
			AND = {
				OR = {
					has_law_or_variant = law_type:law_appointed_by_hos
					has_law_or_variant = law_type:law_appointed_on_advice 
					# protege sucession already handled in bpm_do_hog_transition
				}
				OR = {
					NOT = { has_variable = bpm_headofgov }
					var:bpm_headofgov ?= {
						is_character_alive = no
					}
					bpm_ruler_is_acting_as_interim_head_of_government = yes
				}
			}
		}
		ordered_scope_character = {
			limit = {
				interest_group = {
					custom_tooltip = {
						text = "bpm_not_anti_establishment_tt"
						bpm_ig_is_radical_opposition = no
					}
					custom_tooltip = {
						text = "bpm_not_extremist_tt"
						bpm_ig_is_very_radical_opposition = no
					}
				}
				bpm_char_can_be_hog = yes
				bpm_ig_is_institutional = yes
			}
			order_by = prominence
			position = 0
			set_local_variable = {
				name = bpm_hogappt_candidate
				value = this
			}
		}

		set_local_variable = {
			name = bpm_minname
			value = flag:bpm_headofgov
		}

		if = {
			limit = {
				NOT = {
					has_local_variable = bpm_hogappt_candidate
				}
			}
			create_character = {
				interest_group = owner.ruler.interest_group
            	on_created = {
					save_scope_as = bpm_newmin
					bpm_appoint_as_hog = yes
				}
			}
		}

		local_var:bpm_hogappt_candidate ?= {
			save_scope_as = bpm_newmin
			bpm_appoint_as_hog = yes
		}
		#post_notification = bpm_minister_replaced
	}
}
