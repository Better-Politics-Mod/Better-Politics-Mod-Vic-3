# Accession of Ieyoshi
shogunate.1 = {
	type = country_event
	placement = root

	duration = 3

	title = shogunate.1.t
	desc = shogunate.1.d
	flavor = shogunate.1.f

    gui_window = event_window_1char_tabloid

	left_icon = scope:shogun_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	dlc = dlc018

	icon = "gfx/interface/icons/faction_icons/faction_japan_tokugawa.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		# Triggered by on_new_ruler
	}

	cancellation_trigger = {

	}

	immediate = {
		var:bpm_headofgov ?= {
			save_scope_as = shogun_scope
		}
		random_scope_character = {
			limit = {
				has_trait = special_character_japanese_emperor
			}
			save_scope_as = emperor_scope
		}
		ig:ig_landowners ?= {
			save_scope_as = daimyo_scope
		}
	}

	option = {
		name = shogunate.1.a
		default_option = yes
		scope:shogun_scope ?= {
			remove_trait = reserved
			add_trait = imperious
		}
	}
}

shogunate.2 = { # Debate about Iesada vs. Yoshinobu - Ieyoshi
	type = country_event
	placement = root

	duration = 3

	title = shogunate.2.t
	desc = shogunate.2.d
	flavor = shogunate.2.f

    event_image = {
		video = "ep2_important_japanese_men_negotiating"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/faction_icons/faction_japan_tokugawa.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		game_date >= 1850.1.1
		var:bpm_headofgov ?= {
			has_template = JAP_ieyoshi_tokugawa
		}
	}

	cancellation_trigger = {

	}

	immediate = {
		var:bpm_headofgov ?= {
			save_scope_as = shogun_scope
		}
		ig:ig_armed_forces ?= {
			leader = {
				save_scope_as = armed_forces_leader
			}
		}
		if = {
			limit = {
				NOT = {
					any_scope_character = {
						has_template = JAP_iesada_tokugawa
					}
				}
			}
			create_character = {
				template = JAP_iesada_tokugawa
			}
		}
		if = {
			limit = {
				NOT = {
					any_scope_character = {
						has_template = JAP_yoshinobu_tokugawa
					}
				}
			}
			create_character = {
				template = JAP_yoshinobu_tokugawa
			}
		}
		random_scope_character = {
			limit = {
				has_template = JAP_iesada_tokugawa
			}
			save_scope_as = iesada_scope
		}
		random_scope_character = {
			limit = {
				has_template = JAP_yoshinobu_tokugawa
			}
			save_scope_as = yoshinobu_scope
		}
	}

	option = {
		name = shogunate.2.a
		default_option = yes
		if = {
			limit = {
				var:bpm_hog_heir ?= scope:iesada_scope
			}
			custom_tooltip = iesada_stays_heir_tt
		}
		else = {
            every_scope_character = {
                limit = {has_role = bpm_character_role_protege}
                remove_character_role = bpm_character_role_protege
            }
            scope:iesada_scope ?= {
                bpm_appoint_as_hog_heir = yes
            }
		}
		ig:ig_landowners ?= {
			add_modifier = {
				name = modifier_japan_selected_shogun_candidate
				days = short_modifier_time
			}
			leader ?= {
				add_modifier = {
					name = modifier_power_behind_the_shogun
					days = normal_modifier_time
				}
			}
		}
		ig:ig_armed_forces ?= {
			add_modifier = {
				name = modifier_japan_capability_disregarded
				days = normal_modifier_time
			}
		}
        set_variable = bpm_rejected_yoshinobu
	}

	option = {
		name = shogunate.2.b
		trigger = {
			ig:ig_armed_forces ?= {
				is_in_government = yes
			}
			legitimacy >= 50
		}
		show_as_unavailable = {
			always = yes
		}
		highlighted_option = yes
		scope:yoshinobu_scope ?= {
            bpm_appoint_as_hog_heir = yes
        }
		if = {
			limit = {
				scope:yoshinobu_scope ?= {
					is_adult = no
				}
			}
			custom_tooltip = yoshinobu_is_the_guy_info_tt
		}
		ig:ig_armed_forces ?= {
			add_modifier = {
				name = modifier_japan_selected_shogun_candidate
				days = short_modifier_time
			}
			leader ?= {
				add_modifier = {
					name = modifier_power_behind_the_shogun
					days = normal_modifier_time
				}
			}
		}
		ig:ig_landowners ?= {
			add_modifier = {
				name = modifier_japan_lineage_disregarded
				days = normal_modifier_time
			}
		}
        set_variable = bpm_rejected_iesada
	}
}

shogunate.6 = { # Accession of Iemochi
	type = country_event
	placement = root

	duration = 3

	title = shogunate.6.t
	desc = shogunate.6.d
	flavor = shogunate.6.f

    gui_window = event_window_1char_tabloid

	left_icon = scope:shogun_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	dlc = dlc018

	icon = "gfx/interface/icons/faction_icons/faction_japan_tokugawa.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		# Triggered by on_new_ruler
		bpm_has_acting_hog = no
	}

	cancellation_trigger = {

	}

	immediate = {
		random_scope_character = {
			limit = {
				has_template = JAP_iemochi_tokugawa
			}
			save_scope_as = shogun_scope
		}
		ruler ?= {
			save_scope_as = emperor_scope
		}
		if = {
			limit = {
				scope:shogun_scope ?= {
					age < 16
				}
			}
			if = {
				limit = {
					NOT = {
						any_scope_character = {
							has_template = JAP_yoshinobu_tokugawa
						}
					}
				}
				create_character = {
					template = JAP_yoshinobu_tokugawa
				}
			}
			random_scope_character = {
				limit = {
					has_template = JAP_yoshinobu_tokugawa
				}
				save_scope_as = yoshinobu_scope
			}
		}
	}

	option = {
		name = shogunate.6.a
		default_option = yes
		trigger = {
			scope:shogun_scope ?= {
				is_adult = yes
			}
		}
		scope:shogun_scope ?= {
			set_first_name = Iemochi
		}
	}

	option = {
		name = shogunate.6.a_regency
		default_option = yes
		trigger = {
			scope:shogun_scope ?= {
				is_adult = no
			}
		}
		scope:shogun_scope ?= {
			set_first_name = Iemochi
		}
		if = {
			limit = {
				scope:shogun_scope ?= {
					is_adult = no
				}
			}
			create_character = {
				template = JAP_yoshiyori_tokugawa
				on_created = {
					bpm_set_character_as_acting_hog = yes
					set_variable = { name = regency_years value = 0 }
					save_scope_as = acting_hog_scope
				}
			}
			add_acting_hog_modifier = yes
		}
	}

	option = {
		name = shogunate.6.b_regency
		trigger = {
			scope:shogun_scope ?= {
				is_adult = no
			}
		}
		scope:shogun_scope ?= {
			set_first_name = Iemochi
		}
		if = {
			limit = {
				scope:shogun_scope ?= {
					is_adult = no
				}
			}
			scope:yoshinobu_scope ?= {
					bpm_set_character_as_acting_hog = yes
					set_variable = { name = regency_years value = 0 }
					save_scope_as = acting_hog_scope
			}
			add_acting_hog_modifier = yes
		}
	}

	after = {
		if = {
			limit = {
				scope:shogun_scope ?= {
					has_variable = daimyo_var
				}
			}
			scope:shogun_scope ?= {
				remove_variable = daimyo_var
				remove_modifier = modifier_shinpan_daimyo
			}
			create_character = {
				template = JAP_tokugawa_mochitsugu
			}
		}
	}
}

shogunate.8 = { # Death of Ieyoshi
	type = country_event
	placement = root

	duration = 3

	title = shogunate.8.t
	desc = shogunate.8.d
	flavor = shogunate.8.f

    event_image = {
		video = "unspecific_sick_in_hospital"
	}

	dlc = dlc018

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_dlc_feature = ep2_content
		var:bpm_headofgov ?= {
			has_template = JAP_ieyoshi_tokugawa
			age > define:NCharacters|OLD_AGE
		}
		OR = {
			month = 5
			month = 6
			month = 7
		}
	}

	cancellation_trigger = {

	}

	immediate = {
		var:bpm_headofgov ?= {
			save_scope_as = shogun_scope
		}
		random_scope_character = {
			limit = {
				has_trait = special_character_japanese_emperor
			}
			save_scope_as = emperor_scope
		}
	}

	option = {
		name = shogunate.8.a
		default_option = yes
		scope:shogun_scope ?= {
			kill_character = yes
		}
	}

	option = {
		name = shogunate.8.b
		highlighted_option = yes
		trigger = {
			has_technology_researched = pharmaceuticals
		}
		scope:shogun_scope ?= {
			random_list = {
				25 = {
					kill_character = yes
				}
				75 = {
					add_trait = scarred
				}
			}
		}
	}
