ig_socialists = {
	color = hsv{ 0 1 1 }
	texture = "gfx/interface/icons/political_parties/social_democrats.dds"
	
	ideologies = {
		ideology_proletarian
		ideology_egalitarian
		ideology_anti_slavery
		ideology_ig_united_socialist
	}
	
	enable = {
		always = yes
	}
	
	traits = {
		ig_trait_left_systematic_corruption
		ig_trait_left_bastion_of_freedom
		ig_trait_left_systematic_opposition
	}	
	
	on_enable = {}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
		}			
	}
	
	pop_weight = {
		value = 0
		
		add = {
			desc = "POP_EXCLUSIONS"
			if = {
				limit = { 
					OR = {
						is_pop_type = capitalists
						is_pop_type = peasants
					}
				}
				value = -50
			}
		}

		add = {
			desc = "URBAN_POP"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 50
			}
		}

		add = {
			desc = "URBAN_UNEMPLOYED"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					is_employed = no
				}
				value = 100
			}
		}		
	
		add = {
			desc = "POP_URBAN_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
					}
				}
				value = 50
			}
		}

		add = {
			desc = "POP_EDUCATED_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						literacy_rate >= 0.50
					}
					is_pop_type = machinists
				}
				value = 50
			}
		}	

		add = {
			desc = "POP_LEADERS"
			if = {
				limit = { 
					OR = {
						is_pop_type = academics
						is_pop_type = engineers
					}
				}
				value = 100
			}
		}

		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
					owner = {
						radical_fraction = {
							value >= 0.2
							pop_type = laborers
						}
					}
				}
				value = 10
			}
		}
		add = {
			desc = "POPS_ANGER_SYMPATHY"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
						is_pop_type = soldiers
						is_pop_type = farmers
						is_pop_type = academics
					}
					owner = {
						radical_fraction = {
							value >= 0.5
							pop_type = laborers
						}
					}
				}
				value = 100
			}
		}

		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 50
			}
		}	
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01
					add = 1
					max = 2.0
					min = 0.75
				}
			}
		}	

		multiply = {
			desc = "NO_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_marginal = yes
						}
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "ESTABLISHED_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_powerful = yes
						}
					}
				}
				value = 2.0
			}		
		}		

		multiply = {
			desc = "AT_WAR"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						is_at_war = yes
					}
				}
				value = 2.0
			}		
		}	

		multiply = {
			desc = "POLITICAL_AGITATION"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						has_technology_researched = political_agitation
					}
				}
				value = 2.0
			}		
		}	

		multiply = {
			desc = "CAN_GET_SOCIALISM"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						can_research = socialism
					}
				}
				value = 1.5
			}		
		}	

		multiply = {
			desc = "NO_SOCIALISM"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = socialism }
					}
				}
				value = 0.5
			}		
		}	

		multiply = {
			desc = "NO_LABOR_MOVEMENT"	
			value = 1.0
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = labor_movement }
					}
				}
				value = 0.5
			}		
		}		
		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = egalitarianism }
					}
				}
				value = 0.0
			}		
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}
	
	monarch_weight = {
		value = 0.0
	}	

	commander_weight = {
		value = 0.5
	}	
	
	noble_chance = {
		value = 0.0
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.1

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.4
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}

}
