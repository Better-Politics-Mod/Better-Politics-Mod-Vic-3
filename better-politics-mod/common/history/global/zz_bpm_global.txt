GLOBAL = {
	set_global_variable = bpm_2_3_save_var

	bpm_set_vanilla_ideologies_list = yes

	every_country = {	
		bpm_setup_country = yes
	}

	every_country = {
		trigger_event = { id = bpm_ideology_soft_compat_events.1 }
		trigger_event = { id = bpm_ideology_soft_compat_events.2 }
		trigger_event = { id = bpm_ideology_soft_compat_events.4 }
	}

	every_country = {
		limit = {
			OR = {
				has_law = law_type:law_universal_suffrage
				has_law = law_type:law_census_voting
				has_law = law_type:law_wealth_voting
			}
		}
		set_institution_investment_level = {
			institution = institution_suffrage
			level = 3
		}
	}

	every_country = {
		limit = { 
			has_law = law_type:law_total_separation 
		}
		if = {
			limit = { 
				OR = {
					this ?= c:USA # Allegedly they were actually secular. Fixing state religion for them
				}
			}
			set_state_religion = rel:atheist
		} else = {
			activate_law = law_type:law_freedom_of_conscience # Other countries will need to prove secularity to fit in.
		}
	}

	every_country = {
		activate_law = law_type:law_all_labor_unions_allowed
	}

	bpm_setup_global_laws_varlists = yes

	bpm_set_hidden_journal_entry_groups = yes
	bpm_setup_global_front_names = yes
	add_to_global_variable_list = {
		name = custom_button_list_flag
		target = flag:bpm_gui_sidebar_icon
	}

	every_country = {
        bpm_update_exe_ideology = yes
		bpm_update_consciousness = yes
		#bpm_setup_party_establishment = yes
    }    
}
