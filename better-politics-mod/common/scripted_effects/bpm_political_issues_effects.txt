# post hook
bpm_post_electoral_mandate_hook = {
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_sol
		}
		set_variable = {
			name = bpm_electoral_mandate_sol_baseline_sol
			value = average_sol
		}
	}
	else_if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_liberal
		}
		set_variable = {
			name = bpm_electoral_mandate_anti_liberal_baseline_liberal
			value = bpm_liberal_movements_support
		}
	}
	else_if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_socialist
		}
		set_variable = {
			name = bpm_electoral_mandate_anti_socialist_baseline_socialist
			value = bpm_liberal_movements_support
		}
	}
	else_if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_monarchist
		}
		set_variable = {
			name = bpm_electoral_mandate_anti_monarchist_baseline_monarchist
			value = bpm_labor_movements_support
		}
	}
	else_if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_trade_deficit
		}
		set_variable = {
			name = bpm_electoral_mandate_trade_deficit_baseline_trade_deficit
			value = total_trade_value
		}
	}
	else_if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_pro_slavery
		}
		set_variable = {
			name = bpm_electoral_mandate_pro_slavery_baseline_support
			value = bpm_antislavery_movements_support
		}
	}
	else_if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_slavery
		}
		if = {
			limit = {
				has_law = law_type:law_legacy_slavery
			}
			set_variable = bpm_electoral_mandate_anti_slavery_already_compromised
		}
		else = {
			remove_variable = bpm_electoral_mandate_anti_slavery_already_compromised
		}
	}
	else_if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_unemployment
		}
		set_variable = {
			name = bpm_electoral_mandate_unemployment_baseline_unemployment
			value = {
				value = 0
				every_scope_state = {
					add = state_unemployment_rate
				}
				divide = num_states
			}
		}
	}
	set_variable = {
		name = bpm_electoral_mandate_progress
		value = 0
	}
	set_variable = {
		name = bpm_electoral_mandate_time_progress
		value = 0
	}
	bpm_clear_macros_issues = yes
}

bpm_electoral_mandate_status_track_special = {
	# Increase SOL by 4 points.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_sol
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = average_sol
				subtract = var:bpm_electoral_mandate_sol_baseline_sol
				multiply = 4
				floor = yes
				min = 0
				max = 4
			}
		}	
	}
	# End the war.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_war
			is_at_war = no
		}
		# If not at war, full progress, the war issue is moot
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = 4
		}
	}
	# Suppress liberal movements by 50%.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_liberal
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = var:bpm_electoral_mandate_anti_liberal_baseline_liberal
				divide = bpm_liberal_movements_support
				multiply = 2
				floor = yes
				min = 0
				max = 4
			}
		}	
	}
	# Suppress socialist movements by 50%.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_socialist
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = var:bpm_electoral_mandate_anti_socialist_baseline_socialist
				divide = bpm_labor_movements_support
				multiply = 2
				floor = yes
				min = 0
				max = 4
			}
		}	
	}
	# Suppress reactionary movements by 50% or enact a republic.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_monarchist
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = var:bpm_electoral_mandate_anti_monarchist_baseline_monarchist
				divide = bpm_reactionary_movements_support
				multiply = 2
				floor = yes
				min = 0
				max = 4
			}
		}	
		if = {
			limit = {
				OR = {
					has_law = law_type:law_parliamentary_republic
					has_law = law_type:law_presidential_republic
				}
			}
			set_variable = {
				name = bpm_electoral_mandate_progress
				value = 4
			}
		}
	}
	# Reduce the deficit by 4x or to zero.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_trade_deficit
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = var:bpm_electoral_mandate_trade_deficit_baseline_trade_deficit
				divide = {
					value = total_trade_value
					max = -1
				}
				floor = yes
				min = 0
				max = 4
			}
		}
		if = {
			limit = {
				total_trade_value > 0
			}
			set_variable = {
				name = bpm_electoral_mandate_progress
				value = 4
			}
		}
	}

	# Suppress abolitionist movements by 50%.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_pro_slavery
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = var:bpm_electoral_mandate_pro_slavery_baseline_support
				divide = bpm_antislavery_movements_support
				multiply = 2
				floor = yes
				min = 0
				max = 4
			}
		}
	}
	
	# Ban the slave trade for 2 and ban slavery for 4
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_anti_slavery
		}
		if = {
			limit = {
				NOT = { has_variable = bpm_electoral_mandate_anti_slavery_already_compromised }
				has_law = law_type:law_legacy_slavery
			}
			set_variable = {
				name = bpm_electoral_mandate_progress
				value = 2
			}
		}
		if = {
			limit = {
				has_law = law_type:law_slavery_banned
			}
			set_variable = {
				name = bpm_electoral_mandate_progress
				value = 4
			}
		}
	}
	# Reduce unemployment by 25%.
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_unemployment
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = var:bpm_electoral_mandate_unemployment_baseline_unemployment
				divide = {
					value = 0
					every_scope_state = {
						add = state_unemployment_rate
					}
					divide = num_states
					min = 0.01
				}
				multiply = 4
				floor = yes
				min = 0
				max = 4
			}
		}	
	}

	# Land Reform (in on actions): Move to a more progressive or equally progressive land reform law.
	# Chauvinism : Passing four chauvinistic repeatables
}

bpm_add_electoral_mandate_progress = {
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_$issue$
		}
		change_variable = {
			name = bpm_electoral_mandate_progress
			add = $add$
		}
		clamp_variable = {
			name = bpm_electoral_mandate_progress
			min = 0
			max = 4
		}
		custom_tooltip = bpm_elecmandate_progress_tooltip
	}
}
