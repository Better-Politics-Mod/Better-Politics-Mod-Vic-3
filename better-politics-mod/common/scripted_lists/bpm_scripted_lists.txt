activated_interest_group = {
    base = interest_group
    conditions = {
        bpm_interest_group_is_activated = yes
    }
}

supportive_interest_group = {
    base = interest_group
    conditions = {
        var:bpm_ig_coalition_mov ?= scope:movement
    }
}

factional_interest_group = {
    base = interest_group
    conditions = {
        is_target_in_variable_list = {
            name = bpm_postrev_factions
            target = scope:faction
        }
        owner = {
            is_target_in_variable_list = {
                name = bpm_revolution_victory_igs
                target = prev
            }
        }
    }
}
