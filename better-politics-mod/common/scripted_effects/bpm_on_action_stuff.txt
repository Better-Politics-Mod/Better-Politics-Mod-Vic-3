bpm_on_election_campaign_end_effect = {
	every_active_party = {
		limit = {
			is_party_type = one_party_independents
		}

		add_momentum = -200
	}
	bpm_remove_political_stability_small = yes
	if = {
		limit = {
			this = c:USA
			NOT = { has_variable = bpm_john_brown }
		}
		trigger_event = { id = bpm_acw_events.24 days = 0 }
		set_variable = {
			name = bpm_last_president
			value = ruler
		}
	}
	bpm_reset_hog = yes
	bpm_initialize_government_mandate = yes
	bpm_reload_government_mandate = yes
	
	bpm_unsplat_issues = yes

	if = {
		limit = {
			bpm_country_has_executive_electoral_system = yes
		}
		ruler.party = {
			ordered_member = {
				limit = {
					has_variable = bpm_party_issue
				}
				order_by = ig_clout
				position = 0
				save_scope_as = leader_ig
			}
		}
		set_variable = {
			name = bpm_electoral_mandate
			value = scope:leader_ig.var:bpm_party_issue
		}
		bpm_post_electoral_mandate_hook = yes
	}

	if = {
		limit = { 
			has_variable = bpm_has_majority_consumable
		}
		remove_variable = bpm_has_majority_consumable
	}
	set_variable = bpm_organic_election
	set_variable = bpm_has_election_results
	set_variable = {
		name = polhook_main_reload
		days = 1
	}
}
