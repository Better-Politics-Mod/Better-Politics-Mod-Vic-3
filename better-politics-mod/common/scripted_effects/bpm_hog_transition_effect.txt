bpm_create_hog_heir_hook = {
    if = {
        limit = {
            THIS = c:JAP
            has_law = law_type:law_bakufu
        }
		if = {
			limit = {
				NOT = {
					has_variable = bpm_JAP_ieyoshi_tokugawa
				}
			}
			create_character = {
				template = JAP_ieyoshi_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
			set_variable = bpm_JAP_ieyoshi_tokugawa
		}
		else_if = {
			limit = {
				NOT = {
					has_variable = bpm_JAP_iesada_tokugawa
				}
			}
			create_character = {
				template = JAP_iesada_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
			set_variable = bpm_JAP_iesada_tokugawa
		}
		else_if = {
			limit = {
				NOT = {
					has_variable = bpm_JAP_iemochi_tokugawa
					game_date < 1846.7.16
				}
			}
			create_character = {
				template = JAP_iemochi_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
			set_variable = bpm_JAP_iemochi_tokugawa
		}
		else = {
			create_character = {
				last_name = Tokugawa
				age = 23
				interest_group = ruler.interest_group
				on_created = {
					owner = {
						set_variable = house_tokugawa
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
		}
    }
    else = {
        create_character = {
            age = 23
			interest_group = ruler.interest_group
			on_created = {
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}
    }
}
