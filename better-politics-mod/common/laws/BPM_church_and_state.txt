# icon = graphical icon shown in-game

INJECT:law_state_religion = {
	on_activate = { #PBF
		if = {
			limit = {
				country_has_state_religion = rel:atheist
			}
			set_state_religion = root.country_definition.religion
		}
	}
}

INJECT:law_freedom_of_conscience = {
	on_activate = { #PBF
		recalculate_pop_ig_support = yes
		if = {
			limit = {
				country_has_state_religion = rel:atheist
			}
			set_state_religion = root.country_definition.religion
		}
	}
}

INJECT:law_total_separation = {
	is_visible = {
		NOT = { has_law = law_type:law_total_separation_upgraded }
	}

	on_activate = { #PBF
		add_journal_entry = { type = je_bpm_total_separation_upgrade }	

		if = {
			limit = {
				country_has_state_religion = rel:atheist
			}
			set_state_religion = root.country_definition.religion
		}
	}
	
	modifier = {
		bpm_decorative_upgradable_law = yes
		country_acceptance_shared_religious_trait_add = -5
		country_acceptance_shared_religious_trait_group_add = -5
		country_acceptance_no_shared_religious_trait_add = -10
	}
}

law_total_separation_upgraded = {
	group = lawgroup_church_and_state
	parent = law_total_separation
	
	icon = "gfx/interface/icons/law_icons/total_separation.dds"
	
	progressiveness = 100

	unlocking_technologies = {
		empiricism
	}

	disallowing_laws = {
		law_theocracy
	}
	
	can_enact = {
		OR = {
			is_power_bloc_leader = no
			power_bloc ?= {
				NOT = { has_identity = identity:identity_religious }
			}
		}
	}

	is_visible = {
		has_law = law_type:law_total_separation_upgraded # Upgrade only visible when it is passed
	}

	on_activate = { #PBF
		recalculate_pop_ig_support = yes
		if = {
			limit = {
				NOT = { country_has_state_religion = rel:atheist }
			}
			set_state_religion = rel:atheist
		}
		if = {
			limit = {
				has_modifier = antireligious_policies
			}
			remove_modifier = antireligious_policies
		}
	}
	
	modifier = {
		state_pop_support_movement_religious_majority_mult = -0.5
		state_pop_support_movement_religious_minority_mult = -0.25
		state_conversion_mult = 0.25

		country_acceptance_state_religion_add = 15
		country_acceptance_shared_religious_trait_add = 15
		country_acceptance_shared_religious_trait_group_add = 15
		country_acceptance_no_shared_religious_trait_add = 15

		country_allow_conversion_violent_hostility_bool = yes
		country_allow_conversion_cultural_erasure_bool = yes
		country_allow_conversion_open_prejudice_bool = yes
		country_allow_conversion_second_rate_citizen_bool = yes
		country_allow_conversion_full_acceptance_bool = yes
	}
	
	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_church_and_state_in_power_bloc_bool = yes
					has_law_or_variant = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}	
	}
}
