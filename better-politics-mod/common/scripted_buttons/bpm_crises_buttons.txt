bpm_scandal_response_distract = {
	name = "bpm_scandal_response_distract"
	desc = "bpm_scandal_response_distract_desc"

	visible = {
		has_journal_entry = je_bpm_crises_government_scandal
	}

	possible = {
		custom_tooltip = {
			text = "bpm_scandal_response_distract_possible_tooltip"
			bpm_media_preparedness >= 20
		}
	}

	ai_chance = {
		value = 100
	}

	effect = {
		random_list = {
			1 = {
				custom_tooltip = bpm_end_the_crisis_tt 
				scope:journal_entry = {
					set_variable = {
						name = bpm_crisis_ended
						days = 180
					}
				}
				modifier = {
					if = {
						limit = {
							has_variable = bpm_headofgov
							var:bpm_headofgov = {
								bpm_character_skill_level >= 2
							}
						}
						add = 1
					}
					if = {
						limit = {
							has_variable = bpm_headofgov
							var:bpm_headofgov = {
								bpm_character_skill_level >= 4
							}
						}
						add = 1
					}
				 	if = {
						limit = {
							OR = {
								bpm_number_of_crises > 1
							}
						}
						# More crises means easier to distract
						add = bpm_number_of_crises
					}
				}
			}
			2 = {
				custom_tooltip = bpm_crisis_progresses_tt
				change_variable = {
					name = bpm_government_scandal_progress
					add = 10
				}
			}
		}
		change_variable = {
			name = bpm_media_preparedness
			subtract = 20
		}
		custom_tooltip = {
			text = "bpm_scandal_response_distract_effect_tooltip"
		}
	}

}

		has_journal_entry = je_bpm_crises_government_scandal
	}

	possible = {
		always = yes
	}

	ai_chance = {
		value = 50
	}

	effect = {
		every_interest_group = {
			limit = {
				is_in_government = yes
			}
			add_modifier = {
				name = bpm_ig_government_admission_of_failure_happiness
				days = 1825
				is_decaying = yes
			}
		}
		recalculate_pop_ig_support = yes
		if = {
			limit = {
				country_has_voting_franchise = yes
			}
			call_election = { months = 6 }
		}
		else = {
			ordered_scope_character = {
				limit = {
					NOT = {
						this = prev.ruler
					}
					interest_group = {
						is_in_government = no
					}
				}
				position = 0
				set_character_as_ruler = yes
			}
		}
		scope:journal_entry = {
			set_variable = {
				name = bpm_crisis_ended
				days = 180
			}
		}
	}

}
