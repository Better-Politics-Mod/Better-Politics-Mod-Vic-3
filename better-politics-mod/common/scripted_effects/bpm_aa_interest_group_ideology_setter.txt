bpm_ig_reset_ideologies = {
    bpm_remove_all_eco_ideologies = yes
    bpm_remove_all_gov_ideologies = yes
    bpm_remove_all_dop_ideologies = yes
    bpm_remove_all_rel_ideologies = yes
    bpm_remove_all_mil_ideologies = yes
    bpm_remove_all_fem_ideologies = yes
    bpm_remove_all_cit_ideologies = yes
    bpm_remove_all_wel_ideologies = yes
    bpm_remove_all_sta_ideologies = yes
    bpm_remove_all_exe_ideologies = yes
    bpm_remove_all_ban_ideologies = yes
}

bpm_ig_set_ideology_no_re = {
    bpm_remove_all_$type$_ideologies = yes
    add_ideology = ideology_$type$_$name$
}

bpm_update_ideology_all_igs = {
    bpm_update_ideology_armed_forces = yes
	bpm_update_ideology_landowners = yes
}

bpm_handle_ideological_ig = {
    if = {
        limit = {
            bpm_interest_group_is_activated = no
        }
        bpm_ig_reset_ideologies = yes
        add_ideology = ideology_nonexistent
    }
}

bpm_update_ideology_armed_forces = {
    ig:ig_armed_forces ?= {
        bpm_ig_reset_ideologies = yes

        ## DEFAULT
        bpm_ig_set_ideology_no_re = {
            type = mil
            name = institutional_militarist
        }
        bpm_ig_set_ideology_no_re = {
            type = sta
            name = institutional_state
        }
        bpm_ig_set_ideology_no_re = {
            type = dop
            name = oligarch
        }
    }

    # CONDITIONAL
	if = {
		limit = {
			NOT = { 
				OR = { 
					has_law = law_type:law_oligarchy
					has_law = law_type:law_military_junta
					has_law = law_type:law_autocracy
					is_country_type = unrecognized
				}
			}
		}
		ig:ig_armed_forces ?= {
			bpm_ig_set_ideology_no_re = {
				type = dop
				name = state_power
			}
		}
	}
    if = {
        limit = {
			bpm_country_has_artisanal_economy = yes
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = eco
                name = artisan
            }
        }
    }
    ## Technically you could have the red-national-guard
    if = {
        limit = {
            has_variable = red_army_var
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = mass_army
            }
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = party_socialist
            }
        }
    }
    else_if = {
        limit = {
            has_variable = bpm_samurai_ig_var
            bpm_country_has_artisanal_economy = no
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = traditional_power
            }
            bpm_ig_set_ideology_no_re = {
                type = gov
                name = radical_monarchist
            }
            bpm_ig_set_ideology_no_re = {
                type = eco
                name = trad_isolationist
            }
            bpm_ig_set_ideology_no_re = {
                type = cit
                name = samurai
            }
            bpm_ig_set_ideology_no_re = {
                type = sta
                name = decentralized_state
            }
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = aristocratic_militarist
            }
        }
    }
    else_if = {
        limit = {
            has_journal_entry = je_caudillo
            bpm_country_has_artisanal_economy = no
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = oligarch
            }
            bpm_ig_set_ideology_no_re = {
                type = gov
                name = liberal_republican
            }
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = institutional_militarist
            }
        }
    }

    ig:ig_armed_forces ?= { bpm_reorder_ideologies = yes }
}

            type = gov
            name = liberal_republican
        }
        bpm_ig_set_ideology_no_re = {
            type = dop
            name = moderate_democrat
        }
        bpm_ig_set_ideology_no_re = {
            type = sta
            name = weak_state
        }
        bpm_ig_set_ideology_no_re = {
            type = cit
            name = liberal_nationalist
        }
        bpm_ig_set_ideology_no_re = {
            type = rel
            name = secularist
        }
        bpm_ig_set_ideology_no_re = {
            type = mil
            name = patriotic_liberal
        }
        bpm_ig_set_ideology_no_re = {
            type = eco
            name = laissez_faire
        }
        bpm_ig_set_ideology_no_re = {
            type = wel
            name = market_welfare
        }
        bpm_ig_set_ideology_no_re = {
            type = fem
            name = benevolent_sexism
        }
        bpm_ig_set_ideology_no_re = {
            type = ban
            name = averse
        }
    }

    if = {
        limit = {
            c:USA ?= this
        }
        bpm_ig_set_ideology_no_re = {
            type = sta
            name = federated_state
        }
        bpm_ig_set_ideology_no_re = {
            type = rel
            name = moralist
        }
        bpm_ig_set_ideology_no_re = {
            type = cit
            name = immigrant_nationalist
        }
        bpm_ig_set_ideology_no_re = {
            type = dop
            name = radical_democrat
        }
        bpm_ig_set_ideology_no_re = {
            type = eco
            name = modernizer
        }
    }

    ig:ig_liberals ?= {
        if = {
            limit = { ROOT = { bpm_country_is_latin_american = yes } }
            bpm_ig_set_ideology_no_re = {
                type = wel
                name = public_welfare
            }
        }
        if = {
            limit = {
                NOR = {
                    ROOT = c:USA
                    ROOT = c:BEL
                    ROOT = c:GBR
                }
            }
            bpm_ig_set_ideology_no_re = {
                type = eco
                name = modernizer
            }
        }
    }


    ig:ig_liberals ?= {
        bpm_reorder_ideologies = yes
        bpm_handle_ideological_ig = yes
    }
}

bpm_update_ideology_market_liberals = {
    ig:ig_market_liberals ?= {
        bpm_ig_reset_ideologies = yes

        ## DEFAULT
        bpm_ig_set_ideology_no_re = {
            type = gov
            name = moderate_monarchist
        }
        bpm_ig_set_ideology_no_re = {
            type = dop
            name = authoritarian_democrat
        }
        bpm_ig_set_ideology_no_re = {
            type = sta
            name = powerful_state
        }
        bpm_ig_set_ideology_no_re = {
            type = cit
            name = liberal_nationalist
        }
        bpm_ig_set_ideology_no_re = {
            type = rel
            name = moralist
        }

        }
        bpm_ig_set_ideology_no_re = {
            type = wel
            name = market_welfare
        }
        bpm_ig_set_ideology_no_re = {
            type = fem
            name = benevolent_sexism
        }
        bpm_ig_set_ideology_no_re = {
            type = ban
            name = averse
        }
    }

    if = {
        limit = { bpm_country_is_republic = yes }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = gov
                name = liberal_republican
            }
        }
    }

    if = {
        limit = {
            OR = {
                bpm_country_is_latin_american = yes
                c:USA ?= this
            }
        }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = moderate_democrat
            }
        }
    }
    else_if = {
        limit = { c:BIC ?= this }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = authoritarian_democrat
            }
        }
    }
    else_if = {
        limit = {
            OR = {
                country_has_primary_culture = cu:north_german
                country_has_primary_culture = cu:south_german
            }
            NOT = {
                this ?= c:AUS
            }
        }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = radical_democrat
            }
        }
    }
    else_if = {
        limit = { c:BEL ?= this }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = doctrinaire_democrat
            }
        }
    }
    else_if = {
        limit = {
            has_law = law_type:law_supreme_legislature
        }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = authoritarian_parliamentarian
            }
        }
    }

    if = {
        limit = {
            OR = {
                has_law = law_type:law_federated
                c:BEL ?= this
                bpm_country_is_latin_american = yes
                c:USA ?= this
            }
        }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = sta
                name = federated_state
            }
        }
    }
    else_if = {
        limit = { c:RAJ ?= this }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = sta
                name = weak_state
            }
        }
    }
    else_if = {
        limit = {
            OR = {
                country_has_primary_culture = cu:north_german
                country_has_primary_culture = cu:south_german
            }
            NOT = {
                this ?= c:AUS
            }
        }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = sta
                name = weak_state
            }
        }
    }

    if = {
        limit = { bpm_country_is_in_new_world = yes }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {
                type = cit
                name = immigrant_nationalist
            }
        }
    }

    if = {
        limit = {
            OR = {
                c:USA ?= this

                country_has_primary_culture = cu:south_german
                c:BIC ?= this
            }
        }
        ig:ig_market_liberals ?= {
            bpm_ig_set_ideology_no_re = {

            }

    }
