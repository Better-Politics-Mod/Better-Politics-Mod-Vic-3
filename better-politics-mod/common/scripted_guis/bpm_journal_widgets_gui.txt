bpm_je_revolution_leader = {
	scope = country
	effect = {
		random_scope_character = {
			limit = {
				has_role = bpm_character_role_revolutionary_leader
			}
			save_scope_as = selected_character
		}
		set_variable = {
			name = bpm_selected_revolutionary_leader
			value = scope:selected_character
		}
	}
}
