ig_revolutionist_socialists = {
	color = hsv{ 0 1 1 }
	texture = "gfx/interface/icons/ig_icons/revolutionists.dds"
	
	ideologies = {
		ideology_nonexistent
	}
	
	enable = {
		has_global_variable = global_socialists_split_var
	}
	
	traits = {
		ig_trait_left_systematic_corruption
		ig_trait_left_bastion_of_freedom
		ig_trait_left_systematic_opposition
	}	
	
	on_enable = {}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
		}			
		owner = {
			has_variable = local_socialists_split_var
		}
	}
	
	pop_weight = {
		value = 20
		
		add = {
			desc = "POP_EXCLUSIONS"
			if = {
				limit = { 
					OR = {
						is_pop_type = capitalists
						is_pop_type = peasants
					}
				}
				value = -50
			}
		}

		add = {
			desc = "URBAN_POP"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 30
			}
		}

		add = {
			desc = "URBAN_UNEMPLOYED"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					is_employed = no
				}
				value = 100
			}
		}		
	
		add = {
			desc = "POP_URBAN_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
					}
				}
				value = 30
			}
		}

		add = {
			desc = "POP_EDUCATED_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						AND = {
							is_pop_type = laborers
							literacy_rate >= 0.60
						}
						is_pop_type = machinists
					}
				}
				value = 60
			}
		}	

		add = {
			desc = "POP_LEADERS"
			if = {
				limit = { 
					OR = {
						is_pop_type = academics
						is_pop_type = engineers
					}
				}
				value = 40
			}
		}

		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
					owner = {
						radical_fraction = {
							value >= 0.2
							pop_type = laborers
						}
					}
				}
				value = 50
			}
		}
		add = {
			desc = "POPS_ANGER_SYMPATHY"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
						is_pop_type = soldiers
						is_pop_type = farmers
						is_pop_type = academics
					}
					owner = {
						radical_fraction = {
							value >= 0.5
							pop_type = laborers
						}
					}
				}
				value = 150
			}
		}

		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 50
			}
		}	
		multiply = { # Revolutionists more popular in very unfree states
			desc = "NO_DEMOCRACY"
			value = 1.0
			if = {
				limit = {
					owner = {
						OR = { 
							has_law = law_type:law_autocracy
							has_law = law_type:law_oligarchy
							has_law = law_type:law_elder_council
							has_law = law_type:law_outlawed_dissent
							has_law = law_type:law_secret_police
						}
					}
				}
				value = 1.5
			}
		}
		multiply = {
			desc = "LIMITED_DEMOCRACY"
			value = 1.0
			if = {
				limit = {
					owner = {
						OR = {
							NOT = { has_law = law_type:law_universal_suffrage }
							has_law = law_type:law_censorship
							has_law = law_type:law_national_guard
						}
					}
				}
				value = 1.5
			}
		}

		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01
					add = 1
					max = 2.0
					min = 0.5
				}
			}
		}	

		multiply = {
			desc = "NO_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_marginal = yes
						}
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "WEAK_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							ig_clout < 5
						}
					}
				}
				value = 0.75
			}		
		}

		multiply = {
			desc = "ESTABLISHED_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_powerful = yes
						}
					}
				}
				value = 1.5
			}		
		}		
		multiply = {
			desc = "DEVASTATED"	
			value = 1.0

			if = {
				limit = { 
					state = {
						devastation > 5
					}
				}
				value = 1.5
			}		
		}	
		multiply = {
			desc = "AT_WAR"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						is_at_war = yes
					}
				}
				value = 2.0
			}		
		}	

		multiply = {
			desc = "REFORMISTS_RADICALS_SUPPRESSED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							NOT = { has_ideology = ideology:ideology_ig_revolutionist_socialist }
							OR = {
								has_ideology = ideology:ideology_proletarian
								has_ideology = ideology:ideology_republican
								has_ideology = ideology:ideology_egalitarian
								leader = {
									OR = {
										has_ideology = ideology:ideology_radical
										has_ideology = ideology:ideology_abolitionist
										has_ideology = ideology:ideology_feminist
									}
								}
							}
							is_being_suppressed = yes
						}
					}
				}
				value = 1.2
			}		
		}
		multiply = {
			desc = "CONSERVATIVES_BOLSTERED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							OR = {
								has_ideology = ideology:ideology_paternalistic
								has_ideology = ideology:ideology_reactionary
								has_ideology = ideology:ideology_pro_slavery
								has_ideology = ideology:ideology_stratocratic
								has_ideology = ideology:ideology_laissez_faire
							}
							is_being_bolstered = yes
						}
					}
				}
				value = 1.1
			}		
		}	
		multiply = {
			desc = "BEFORE_FIRST_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = first_intl_split_complete_var } }
				value = 0.8
			} 
		}
		multiply = {
			desc = "BEFORE_SECOND_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = second_intl_split_complete_var } }
				value = 0.9
			}
		}
		multiply = {
			desc = "BEFORE_HEROIC_PERIOD"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = global_socialist_heroic_period } }
				value = 0.5
			}
		}
		multiply = {
			desc = "INTERNATIONAL_EXCHANGE"	
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = international_exchange_standards } }
				value = 1.3
			}		
		}
		multiply = {
			desc = "POLITICAL_AGITATION"	
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = political_agitation } }
				value = 1.3
			}		
		}
		multiply = {
			desc = "MASS_PROPAGANDA"
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = mass_propaganda } }
				value = 1.3
			}		
		}	
		multiply = {
			desc = "NO_SOCIALISM"	
			value = 1.0
			if = {
				limit = { owner = { NOT = { has_technology_researched = socialism } } }
				value = 0.5
			}		
		}
		multiply = {
			desc = "NO_LABOR_MOVEMENT"	
			value = 1.0
			if = {
				limit = { owner = { NOT = { has_technology_researched = labor_movement } } }
				value = 0.5
			}		
		}		
		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			if = {
				limit = { owner = { NOT = { has_technology_researched = egalitarianism } } }
				value = 0.0
			}		
		}
		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}
	
	monarch_weight = {
		value = 0.0
	}	

	commander_weight = {
		value = 0.5
	}	
	
	noble_chance = {
		value = 0.0
	}

		value = 0.0

		value = 0.1

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property

			}
			add = {
				value = 0.2
			}
		}
	}

	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
