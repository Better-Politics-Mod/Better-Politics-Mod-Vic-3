on_character_death = {
    on_actions = {
        bpm_on_do_hog_transition
    }
}

on_monthly_pulse_country = {
    on_actions = {
        bpm_on_do_hog_transition
    }
}

bpm_on_do_hog_transition = {
    effect ={
        owner = {
            if = {
                limit = {
                    var:bpm_headofgov ?= {
                        NOT = {
                            has_variable = bpm_heir_proper_hog_marker
                        }
                    }
                    has_variable = bpm_hog_heir
                    has_law = law_type:law_advisory_body
                }
                var:bpm_hog_heir ?= {
                    # handle scripted heirs
                    set_character_immortal = no
                    bpm_set_headofgov_sticky = yes
                    set_variable = bpm_heir_proper_hog_marker
                }
                bpm_create_hog_heir_hook = yes
            }

            if = {
                limit = {
                    var:bpm_hog_heir ?= {
                        NOT = { is_alive = yes}
                    }
                }
                remove_variable = bpm_hog_heir
            }

            if = {
                limit = {
                    NOT = {
                        has_variable = bpm_hog_heir
                        has_law = law_type:law_answers_to_legislature
                        has_law = law_type:law_advisory_body
                    }
                }
                bpm_create_hog_heir_hook = yes

            }
        }
    }
}
