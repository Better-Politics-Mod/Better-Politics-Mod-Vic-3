bpm_reload_cabinet_max_level_from_skill_inst = {
    remove_modifier = bpm_cabinet_max_level_from_skill_$INST$
    var:bpm_minister_of_$INST$ ?= {
        save_scope_as = actor
        set_variable = {
            name = bpm_skill_level
            value = bpm_traits_skill_level
        }
        owner = {
            set_variable = {
                name = bpm_skill_level
                value = prev.var:bpm_skill_level
            }
            if = {
                limit = {
                    NOT = { scope:actor.var:bpm_skill_level = 0 }
                }
                switch = {  #few considerations - multiplier runs in scope root apparently, history root is null, explicit scope is required but still spams errors, this does no
                    trigger = scope:actor.var:bpm_skill_level
                    -1 = {
                        add_modifier = {
                            name = bpm_cabinet_max_level_from_skill_$INST$
                            multiplier = -1
                        }
                    }
                    0 = {
                        add_modifier = {
                            name = bpm_cabinet_max_level_from_skill_$INST$
                            multiplier = 0
                        }
                    }
                    1 = {
                        add_modifier = {
                            name = bpm_cabinet_max_level_from_skill_$INST$
                            multiplier = 1
                        }
                    }
                }
            }
        }
    }
