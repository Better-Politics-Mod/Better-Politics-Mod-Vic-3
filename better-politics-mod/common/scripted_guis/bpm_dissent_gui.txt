bpm_ig_is_revolutionary = {
	scope = interest_group
	is_valid = {
		bpm_ig_in_rev = yes
	}
}

bpm_nuke_revolution = {
	scope = journal_entry
	is_valid = {
		owner = { bpm_state_capacity > 100 }
		NOT = {
			has_variable = bpm_nuke_on_cd
		}
	}
	effect = {
		set_variable = {
			name = bpm_nuke_on_cd
			value = yes
			days = 180
		}
		save_scope_as = selected_je
		owner = {
			scope:selected_je.var:bpm_front_currently_planning = {
				save_scope_as = selected_front_name
			}
			bpm_update_front_organization = { v = -25 }
			bpm_save_front_organization_to_je = yes
			if = {
				limit = {
					scope:selected_je = { 
						has_variable = bpm_current_front_organization 
					}
					scope:selected_je.var:bpm_current_front_organization <= 0
				}
				scope:selected_je = {
					set_variable = {
						name = bpm_current_front_organization
						value = 0
					}
				}
			}
			custom_tooltip = bpm_reduce_organization
			bpm_auth_add_exhaustion_normal = yes
			add_radicals = {
				value = 0.01
			}
		}

	}
}

bpm_show_state_front = {
	scope = state

	saved_scopes = {
		selected_je
	}

	is_shown = {
		scope:selected_je.var:bpm_front_currently_planning = {
			save_temporary_scope_as = selected_front_name
		}
		bpm_trigger_front = {
			trigger = bpm_state_has_rev
		}
	}
}
