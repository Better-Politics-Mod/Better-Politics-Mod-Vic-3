bpm_ig_law_support_value = {
	value = 0
	add = {
		if = {
			limit = {
				law_enactment_stance = {
    			    law = scope:law.type
    			    value > approve
    			}
			}
			value = 2
		}
		else_if = {
			limit = {
				law_enactment_stance = {
    			    law = scope:law.type
    			    value < disapprove
    			}
			}
			value = -2
		}
		else_if = {
			limit = {
				law_enactment_stance = {
    			    law = scope:law.type
    			    value > neutral
    			}
			}
			value = 1
		}
		else_if = {
			limit = {
				law_enactment_stance = {
    			    law = scope:law.type
    			    value < neutral
    			}
			}
			value = -1
		}
		desc = bpm_ig_law_support_enacting_law
	}
}

				desc = "amenability_pressuring_supporting_movements"

bpm_law_enactment_ruler_effect_on_ig = {
	value = 0
	if = {
		limit = {
			owner.ruler ?= {
				has_variable = bpm_ig_enacting_law_change_steps
				interest_group = {
					has_variable = bpm_ig_enacting_law_change_steps
				}
			}
		}
		add = {
			if = {
				# If no ruler stance, use interest group stance
				limit = {
					owner.ruler.var:bpm_ig_enacting_law_change_steps = 0
				}
				add = owner.ruler.interest_group.var:bpm_ig_enacting_law_change_steps
			}
			else = {
				add = owner.ruler.var:bpm_ig_enacting_law_change_steps
			}

			if = {
				limit = { owner = { has_law = law_type:law_autocracy } }
				multiply = 10
			}
			else_if = {
				limit = { owner = { has_law = law_type:law_presidential_republic } }
				multiply = 2
			}
			else_if = {
				limit = { owner = { has_law = law_type:law_monarchy } }
				multiply = 2
			}
			else = {
				multiply = 0
			}
			multiply = {
				value = ig_approval
				divide = 5
				min = 0
				max = 1
			}
			desc = "bpm_ig_law_support_from_ruler"
		}
	}	
}

bpm_character_law_support = {
	value = 0

	if = {
		limit = {
			NOT = {exists = scope:law}
		}
		owner.currently_enacting_law ?= {
			save_temporary_scope_as = law
		}
	}
	add = {
		if = {
			# If no ruler stance, use interest group stance
			limit = {
				bpm_ig_law_support_value = 0
			}
			add = interest_group.bpm_ig_law_support_value
		}
		else = {
			add = bpm_ig_law_support_value
		}
		multiply = 8
		desc = "bpm_ig_law_support_from_ideology"
	}
	multiply = {
		value = 1
		add = modifier:character_bpm_veto_assertiveness
		desc = "bpm_ig_law_support_from_traits"
	}
	add = bpm_vetoer_support_from_vote
	add = bpm_vetoer_law_support_from_movements


	if = {
		limit = {
			OR = {
				is_ruler_of_own_country = yes
				has_role = bpm_character_role_regent
			}
		}
		add = {
			scope:law = { value = bpm_law_hos_power }
			scope:law.currently_active_law_in_group = { subtract = bpm_law_hos_power}
			multiply = 4
			desc = "bpm_hos_power_difference"
		}
	}
	else_if = {
		limit = {
			OR = {
				has_role = bpm_character_role_hog
				has_role = bpm_character_role_acting_hog
			}
		}
		add = {
			scope:law = { value = bpm_law_hog_power }
			scope:law.currently_active_law_in_group = { subtract = bpm_law_hog_power}
			multiply = 4
			desc = "bpm_hog_power_difference"
		}
	}

	if = {
		limit = {
			law_type:law_monarchy = {
				is_same_law_group_as = scope:law.type
			}
		}
		add = {
			value = -10
			desc = "bpm_changing_gov_principle"
		}
	}
}

					subtract = 8 #IG smaller than this actually seeks to get bigger.
					multiply = ig_clout_neg_multiplier
					multiply = 0.5
				}
			}
		}

		if = {
			limit = {
				owner = {
					has_variable = bpm_political_stability
					var:bpm_political_stability > 66
				}
			}
			add = {
				value = owner.var:bpm_political_stability
				subtract = 66
				# Weigh by clout difference with party whip
				multiply = -0.5
				desc = "bpm_ig_law_support_from_stability"
			}
		}

		##modifiers on IG
		add = {
			value = modifier:interest_group_amenability_add
			desc = "amenability_ig_additional_adds"
		}

		##modifiers on Nation
		add = {
			value = owner.modifier:country_amenability_add
			desc = "amenability_country_additional_adds"
		}

		### Enactment chance
		##if looking very likely will pass even without negotiating, scaled slightly based on lead. The IG are looking for a freebie, since law is gonna pass anyway.
		if = {
			limit = {
				owner = {
					bpm_country_yea_votes > bpm_law_enactment_vote_threshold
				}
			}
			add = {
				add = -25
				desc = "amenability_law_likely_to_pass"
			}
		}

		##Otherwise, extra amenability if sure their support will make no difference and it will fail anyway. Two levels.
		else_if = {
			limit = {
				string_law = yes
				ig_clout > {
					owner = {
						value = bpm_law_enactment_vote_threshold
						subtract = bpm_country_yea_votes
					}
				}
			}
			add = {
				add = 15
				desc = "amenability_law_worried_to_be_tie_breaker"
			}
		}
		else_if = {
			limit = {
				owner = {
					bpm_law_enactment_vote_threshold > {
						value = bpm_country_yea_votes
						multiply = 4
					}
				}
			}
			add = {
				add = -20
				desc = "amenability_law_makes_no_difference"
			}
		}
		##Otherwise, afraid of being the tiebreaker
		##Balance note, this is an easy way to make negotiation harder whenever it is most effective, while keeping it the same in other places. BEST LEVER HERE.
		
	if = {
		limit = {
			has_variable = bpm_times_ig_swayed
		}
		add = {
			value = var:bpm_times_ig_swayed
			multiply = -20
			desc = "amenability_times_swayed"
		}
	}

		round = 1
		min = define:NPolitics|MIN_AMENABILITY
		max = define:NPolitics|MAX_AMENABILITY
	}
	}
	else = {
		value = 100
	}
}
