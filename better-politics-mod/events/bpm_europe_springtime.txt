		}

			}

		}

			}

			every_country = {
				limit = {
					NOR = {

					}

					}
				}
				c:GER = {
					annex_with_incorporation = prev
				}
			}
		}

		activate_law = law_type:law_separated_powers
		activate_law = law_type:law_protected_speech
		activate_law = law_type:law_universal_suffrage
		activate_law = law_type:law_unitary

		hidden_effect = {
			every_interest_group = {
				limit = {
					OR = {
						has_modifier = bpm_stubborn_ig
						has_modifier = bpm_1848_stubborn_ig

				remove_modifier = bpm_1848_stubborn_ig
				remove_modifier = bpm_very_stubborn_ig
			}
		}
		
		default_option = yes
	}
