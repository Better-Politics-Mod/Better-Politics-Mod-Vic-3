ig_intelligentsia = {
	color = hsv{ 0 0 0.5 }
	texture = "gfx/interface/icons/ig_icons/intelligensia.dds"
	layer = "revolution_dynamic_intelligentsia"
	index = 3

	ideologies = {
		ideology_dop_moderate_democrat
		ideology_sta_powerful_state
		ideology_cit_liberal_nationalist
	}

	enable = {
		always = yes
	}
	
	on_enable = {
		ig:ig_intelligentsia = {
			set_ig_trait = ig_trait:ig_trait_urban_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_urban_well_oiled_machine
			set_ig_trait = ig_trait:ig_trait_urban_systematic_opposition	
		}
		# Hawaiian Democrats
		if = {
			limit = {
				country_has_primary_culture = cu:hawaiian
			}
			ig:ig_intelligentsia = {
				set_interest_group_name = ig_hawaiian_democrats
			}
		}
	}
	on_disable = {}	
	on_character_ig_membership = {}
	
	pop_potential = {
	}

	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_SHOPKEEPERS"
			if = {
				limit = { 
					is_pop_type = shopkeepers
				}
				value = 200
			}
		}	
		add = {
			desc = "POP_ACADEMICS"
			if = {
				limit = { 
					is_pop_type = academics
				}
				value = 100
			}
		}	
		add = {
			desc = "POP_CLERKS"
			if = {
				limit = { 
					is_pop_type = clerks
				}
				value = 200
			}
		}	
		add = {
			desc = "POP_BUREAUCRATS"
			if = {
				limit = { 
					is_pop_type = bureaucrats
				}
				value = 50
			}
		}
		add = {
			desc = "POP_ENGINEERS"
			if = {
				limit = { 
					is_pop_type = engineers
				}
				value = 100
			}
		}
		add = {
			desc = "NONPRIMARY_CULTURE"

			if = {
				limit = {
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					strata = middle
					pop_has_primary_culture = no
				}
				value = 150
			}
		}	
		add = {
			desc = "POLITICAL_MACHINES"
			if = {
				limit = {
					strata = middle
					owner = {
						OR = {
							has_law = law_type:law_landed_voting
							has_law = law_type:law_wealth_voting
							has_law = law_type:law_census_voting
							has_law = law_type:law_appointed_bureaucrats
						}
					}
				}
				value = 100
			}
		}	
		add = {
			desc = "MERCHANT_GUILDS"
			if = {
				limit = {
					is_pop_type = shopkeepers
					owner = {
						has_law = law_type:law_mercantilism
					}
				}
				value = 100
			}
		}	
		multiply = ig_attraction_to_interests_from_stability
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}		

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}

	monarch_weight = {
		value = 0.25
	}	

	agitator_weight = {
		value = 1.0
		multiply = bpm_base_agitator_weight
	}
	
	commander_weight = {
		value = 1.0
		multiply = bpm_commander_assignment_from_laws_factor
	}	
	
	noble_chance = {
		value = 0.1
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
