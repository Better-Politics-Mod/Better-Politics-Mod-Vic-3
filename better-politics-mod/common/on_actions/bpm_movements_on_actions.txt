on_monthly_pulse_country = {
    on_actions = {
        bpm_update_movements_oa
    }
}

bpm_update_movements_oa = {
    effect = {
        random_political_movement = {
            limit = {
                type = movement_type:movement_liberal
            }
            save_scope_as = movement
        }
        if = {
            limit = {
                exists = scope:movement
            }
            set_variable = {
                name = bpm_liberal_movement
                value = scope:movement
            }
        }
    }
}
