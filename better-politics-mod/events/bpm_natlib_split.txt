namespace = bpm_natlib_split

bpm_natlib_split.1 = { # French Bonapartists
	type = country_event
	placement = root

	trigger = {
		this = c:FRA
		NOT = { has_variable = local_natlib_split_var }
	}

	immediate = {
		set_variable = local_natlib_split_var

		ig:ig_national_liberals = {
			remove_ideology = ideology_nonexistent
			add_ideology = ideology_gov_moderate_monarchist
			add_ideology = ideology_dop_moderate_democrat
			add_ideology = ideology_sta_security_state
			add_ideology = ideology_cit_nationalist
			add_ideology = ideology_rel_moralist
			add_ideology = ideology_jingoist
			add_ideology = ideology_eco_laissez_faire
			add_ideology = ideology_wel_paternalistic_welfare
			add_ideology = ideology_fem_moderate_patriarchal
			set_interest_group_name = ig_bonapartists
		}
	}

	option = {
		name = bpm_natlib_split.1.a
		default_option = yes
	}
}

bpm_natlib_split.3 = { # Unifiers
	type = country_event
	placement = root

	trigger = {
		game_date > 1850.1.1
		is_unification_candidate = any
		OR = {
			is_at_war = yes
			is_diplomatic_play_committed_participant = yes
			has_revolution = yes
		}
		OR = {
			ig:ig_liberals = { is_marginal = no }
			ig:ig_conservatives = { is_marginal = no }
		}
		has_technology_researched = nationalism
	}

	weight_multiplier = {
		base = 0
		if = {
			limit = {
				is_diplomatic_play_committed_participant = yes
			}
			add = 0.5
		}
		if = {
			limit = {
				is_at_war = yes
			}
			add = 1.0
		}
		if = {
			limit = {
				has_revolution = yes
			}
			add = 1.5
		}
		if = {
			limit = {
				is_unification_candidate = any
			}
			add = 1.5
		}
		if = {
			limit = {
				has_technology_researched = pan-nationalism
			}
			add = 2.0
		}
		if = {
			OR = {
				ig:ig_liberals = { is_powerful = yes }
				ig:ig_conservatives = { is_powerful = yes }
			}
			multiply = 1.5
		}
	}

	immediate = {
		set_variable = local_natlib_split_var

		ig:ig_national_liberals = {
			remove_ideology = ideology_nonexistent
			add_ideology = ideology_gov_moderate_monarchist
			add_ideology = ideology_dop_moderate_democrat
			add_ideology = ideology_sta_security_state
			add_ideology = ideology_cit_nationalist
			add_ideology = ideology_rel_moralist
			add_ideology = ideology_jingoist
			add_ideology = ideology_eco_laissez_faire
			add_ideology = ideology_wel_paternalistic_welfare
			add_ideology = ideology_fem_moderate_patriarchal
		}
	}

	option = {
		name = bpm_natlib_split.3.a
		default_option = yes
	}
}
