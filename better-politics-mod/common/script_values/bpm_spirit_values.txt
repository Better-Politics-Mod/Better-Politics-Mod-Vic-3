bpm_spirit_value = {
	value = 0
	if = {
		limit = {
			has_variable = bpm_spirit_var
		}
		add = var:bpm_spirit_var
	}
}

bpm_spirit_value_change = {
	value = 10
}

bpm_spirit_max_value = {
	value = 500
}

bpm_spirit_surplus = {
	value = 0
	if = {
		limit = {
			bpm_spirit_value > 0
		}
		add = bpm_spirit_value
		divide = bpm_spirit_max_value
	}
}

bpm_spirit_deficit = {
	value = 0
	if = {
		limit = {
			bpm_spirit_value < 0
		}
		add = bpm_spirit_value
		divide = bpm_spirit_max_value
	}
}
