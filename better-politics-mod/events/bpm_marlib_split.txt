# Utilitarians (India)
bpm_marlib_split.2 = {
	type = country_event
	placement = root

	event_image = { video = "gfx/event_pictures/europenorthamerica_capitalists_meeting.bk2" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	title = bpm_marlib_split.2.t
	desc = bpm_marlib_split.2.d
	flavor = bpm_marlib_split.2.f
	duration = 3

	dlc = dlc_bpm

	option = {
		name = bpm_marlib_split.2.a
		default_option = yes
		show_as_tooltip = {
			create_character = {
				template = agitator_john_stuart_mill
			}
		}
		hidden_effect = {
			hidden_effect = {
				ig:ig_market_liberals ?= {
					if = {
						limit = {
							leader = {
								NOT = { has_variable = stuart_mill_var }
							}
						}
						leader = { kill_character = yes }
					}
				}
				if = {
					limit = {
						any_scope_character = {
							has_variable = stuart_mill_var
						}
					}
					random_scope_character =  {
						limit = {
							has_variable = stuart_mill_var
						}
						kill_character = {
							hidden = yes
						}
					}
				}
				else_if = {
					limit = {
						any_character_in_exile_pool = {
							has_variable = stuart_mill_var
						}
					}
					random_character_in_exile_pool =  {
						limit = {
							has_variable = stuart_mill_var
						}
						kill_character = {
							hidden = yes
						}
					}
				}
				else_if = {
					limit = {
						any_country = {
							NOT = {
								this = root
							}
							any_scope_character = {
								has_variable = stuart_mill_var
							}
						}
					}
					random_country = {
						limit = {
							NOT = {
								this = root
							}
							any_scope_character = {
								has_variable = stuart_mill_var
							}
						}
						random_scope_character = {
							limit = {
								has_variable = stuart_mill_var
							}
							kill_character = {
								hidden = yes
							}
						}
					}
				}
				create_character = {
					template = agitator_john_stuart_mill
					on_created = {		
						#set_variable = stuart_mill_var
						set_as_interest_group_leader = yes
						add_character_role = politician	
						remove_character_role = agitator
					}
				}
			}
		}
	}
}
