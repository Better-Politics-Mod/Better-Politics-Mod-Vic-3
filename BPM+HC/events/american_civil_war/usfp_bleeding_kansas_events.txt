			}

					any_scope_pop = {
						is_pop_type = slaves
					}
				}
				liberate_slaves = yes
			}
			if = {
				limit = { 
					is_incorporated = no
				}
				set_state_type = incorporated
			}
			if = {
				limit = { 
					cu:yankee = {
						NOT = {
							has_homeland = prev.state_region
						}
					}
				}
				state_region = {
					add_homeland = cu:yankee
				}
			}
		}
		hidden_effect = {
			usfp_cleanup_bleeding_kansas_variables = yes
		}
	}
}
