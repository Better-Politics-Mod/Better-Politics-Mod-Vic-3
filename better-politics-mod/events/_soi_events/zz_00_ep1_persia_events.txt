namespace = persia_events


persia_events.8 = {
	type = country_event
	placement = scope:herat_scope

	title = persia_events.8.t
	desc = persia_events.8.d
	flavor = persia_events.8.f

	event_image = {
		video = "ep1_redcoats"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	dlc = dlc010

	duration = 3

	trigger = {
		# triggered by event
	}

	immediate = {
		c:GBR ?= {
			save_scope_as = britain_scope
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_HERAT
			}
			save_scope_as = herat_scope
		}
		if = {
        	limit = {
	        	c:HER = {
	        		exists = no
	        	}
	        	c:GBR ?= {
	        		any_subject_or_below = {
						OR = {
							country_has_primary_culture = cu:pashtun
							country_has_primary_culture = cu:tajik
						}
	        		}
	        	}
	        }
	        c:GBR ?= {
		        random_subject_or_below = {
		        	limit = {
		        		OR = {
							country_has_primary_culture = cu:pashtun
							country_has_primary_culture = cu:tajik
						}
		        	}
		        	save_scope_as = british_afghan_subject
		        }
		    }
		}
	}

	option = {
		name = persia_events.8.a
        default_option = yes
        custom_tooltip = {
        	text = britain_wants_herat_tt
			scope:britain_scope = {
				set_secret_goal = {
					country = root
					secret_goal = conquer			
				}				
			}
			set_secret_goal = {
				country = scope:britain_scope
				secret_goal = antagonize	
			}
        }
		every_interest_group = {
			limit = {
				OR = {
					is_interest_group_type = ig_landowners
					is_interest_group_type = ig_devout
					is_interest_group_type = ig_armed_forces
				}
			}
			add_modifier = {
				name = did_not_bow_to_britain_modifier
				days = normal_modifier_time
			}
		}
	}
	option = {
		name = persia_events.8.b
        if = {
        	limit = {
        		c:HER ?= {
        			NOT = {
        				is_subject_of = root
        			}
        		}
        	}
			random_scope_state = {
				limit = {
					state_region = s:STATE_HERAT
				}
				if = {
					limit = {
						has_modifier = modifier_herat_inspiration
					}
					remove_modifier = modifier_herat_inspiration
				}
				set_state_owner = c:HER
			}
			c:HER ?= {
        		change_relations = {
					country = c:GBR
					value = 50
				}
			}
        }
        else_if = {
        	limit = {
	        	NOT = {
					exists = c:HER
				}
	        }
        	create_country = {
        		tag = HER
        		origin = root
        		state = scope:herat_scope
        		on_created = {
        			change_relations = {
						country = c:GBR
						value = 50
					}
        		}
        	}
			c:HER = {
				bpm_setup_country = yes
			}
        }
        else_if = {
        	limit = {
	        	NOT = {
					exists = c:HER
				}
	        	c:GBR ?= {
	        		any_subject_or_below = {
						OR = {
							country_has_primary_culture = cu:pashtun
							country_has_primary_culture = cu:tajik
						}
	        		}
	        	}
	        }
			random_scope_state = {
				limit = {
					state_region = s:STATE_HERAT
				}
				if = {
					limit = {
						has_modifier = modifier_herat_inspiration
					}
					remove_modifier = modifier_herat_inspiration
				}
				set_state_owner = scope:british_afghan_subject
			}
        }
        else = {
			create_dynamic_country = {
				origin = root
				country_type = unrecognized
				culture = cu:persian
				religion = rel:sunni
				tier = principality
				capital = scope:herat_scope
				cede_state_trigger = {
					state_region = s:STATE_HERAT
				}
				on_created = {
        			change_relations = {
						country = c:GBR
						value = 50
					}
				}
			}
        }
		change_relations = {
			country = c:GBR
			value = 10
		}
		every_interest_group = {
			limit = {
				OR = {
					is_interest_group_type = ig_landowners
					is_interest_group_type = ig_devout
					is_interest_group_type = ig_armed_forces
				}
			}
			add_modifier = {
				name = bowed_to_britain_modifier
				days = normal_modifier_time
			}
		}
	}
}
