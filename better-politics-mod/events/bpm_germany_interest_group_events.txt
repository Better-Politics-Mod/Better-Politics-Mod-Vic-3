namespace = germany_ig_events

germany_ig_events.1 = {
	type = country_event
	placement = root
	title = germany_ig_events.1.t
	desc = germany_ig_events.1.d
	flavor = germany_ig_events.1.f
	duration = 3

	event_image = {
		video = "unspecific_factory_closed"
	}

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	trigger = {
		OR = {
			c:PRU ?= this
			c:NGF ?= this
			c:GER ?= this
		}
		OR = {
			ig:ig_socialists = { ig_clout >= 0.05 }
			ig:ig_reformist_socialists = { ig_clout >= 0.05 }
			ig:ig_revolutionist_socialists = { ig_clout >= 0.05 }
			ig:ig_anarchists = { ig_clout >= 0.05 }
		}
		has_global_variable = first_intl_chain_complete_var
		NOT = { has_variable = bpm_ger_strong_socialists }
	}

	immediate = {
		set_variable = bpm_ger_strong_socialists
	}

	option = {
		name = germany_ig_events.1.a
		default_option = yes

		custom_tooltip = {
			text = bpm_movement_splits_fascists_tt
			bpm_add_movement_interaction = { int = fascist_split }
		}
	}
}
