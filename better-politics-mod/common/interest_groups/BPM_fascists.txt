ig_fascists = {
	color = hsv{ 0.10 1 0.5 }
	texture = "gfx/interface/icons/ig_icons/fascists.dds"
	layer = "revolution_dynamic_petty_bourgeoisie"
	index = 11
	
	ideologies = {
		ideology_patriotic
		ideology_ig_fascist
	}
	
	enable = {
		game_date > 1910.1.1
	}
	
	on_enable = {
		ig:ig_fascists = {
			set_ig_trait = ig_trait:ig_trait_right_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_right_bastion_of_freedom
			set_ig_trait = ig_trait:ig_trait_right_systematic_opposition
		}
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
	}
	
	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 100
			}
		}	

		add = {
			desc = "MIDDLE_CLASS"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					strata = middle
				}
				value = 50
			}
		}
		add = {
			desc = "URBAN_POP"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 50
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01 # Fascists are heavily dependent on their leader
					add = 1
					max = 2
					min = 0.1
				}
			}
		}	

		multiply = {
			desc = "UNEDUCATED"
			value = 1.0

			if = {
				limit = {
					literacy_rate <= 0.67
				}
				value = 0.5
			}
		}
		multiply = {
			desc = "VERY_UNEDUCATED"
			value = 1.0

			if = {
				limit = {
					literacy_rate <= 0.33
				}
				value = 0.5
			}
		}

		multiply = {
			desc = "NO_POLITICAL_AGITATION"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = political_agitation }
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}
	
	monarch_weight = {
		value = 0.5
	}	
	
	commander_weight = {
		value = 0.5
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.0

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
