
### RADICAL IG ###

bpm_front_attraction_u1 = {
	value = 0
	if = {
		limit = {
			ig_approval <= unhappy
		}
		add = 1
	}
}
bpm_front_attraction_u2 = {
	value = 0
	if = {
		limit = {
			ig_approval <= unhappy
		}
		add = 2
	}
}
bpm_front_attraction_u3 = {
	value = 0
	if = {
		limit = {
			ig_approval <= unhappy
		}
		add = 3
	}
}
bpm_front_attraction_n1 = {
	value = 1
}
bpm_front_attraction_n2 = {
	value = 2
}
bpm_front_attraction_n3 = {
	value = 3
}

bpm_front_urban_multiplier = {
	value = 1
	if = {
		limit = {
			owner = {
				has_variable_list = bpm_front_urban_members
				is_target_in_variable_list = {
					name = bpm_front_urban_members
					target = scope:selected_ig
				}
			}
		}
		add = 1
	}
}

bpm_front_popular_multiplier = {
	value = 1
	if = {
		limit = {
			owner = {
				has_variable_list = bpm_front_popular_members
				is_target_in_variable_list = {
					name = bpm_front_popular_members
					target = scope:selected_ig
				}
			}
		}
		add = 1
	}
}

bpm_front_liberal_multiplier = {
	value = 1
	if = {
		limit = {
			owner = {
				has_variable_list = bpm_front_liberal_members
				is_target_in_variable_list = {
					name = bpm_front_liberal_members
					target = scope:selected_ig
				}
			}
		}
		add = 1
	}
}

bpm_front_republican_multiplier = {
	value = 1
	if = {
		limit = {
			owner = {
				has_variable_list = bpm_front_republican_members
				is_target_in_variable_list = {
					name = bpm_front_republican_members
					target = scope:selected_ig
				}
			}
		}
		add = 1
	}
}

bpm_front_attraction_dispatch_radicals = {
	scope:selected_ig = {
		if = {
			limit = {
				prev = flag:bpm_front_urban
			}
			value = bpm_front_attraction_u2
			multiply = bpm_front_urban_multiplier
		}
		else_if = {
			limit = {
				prev = flag:bpm_front_popular
			}
			value = bpm_front_attraction_u2
			multiply = bpm_front_popular_multiplier
		}
		else_if = {
			limit = {
				prev = flag:bpm_front_liberal
			}
			value = bpm_front_attraction_n2
			multiply = bpm_front_liberal_multiplier
		}
		else_if = {
			limit = {
				prev = flag:bpm_front_republican
			}
			value = bpm_front_attraction_n3
			multiply = bpm_front_republican_multiplier
		}
		else = {
			value = 0
		}
	}
}
