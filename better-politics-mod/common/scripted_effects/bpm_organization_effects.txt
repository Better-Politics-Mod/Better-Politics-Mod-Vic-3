bpm_organization_do_trend = {
	every_scope_character = {
		bpm_character_skill_level_init = yes
	}
	every_interest_group = {
		change_variable = {
			name = bpm_ig_organization
			add = bpm_organization_trend
		}
		clamp_variable = {
			name = bpm_ig_organization
			min = 0
			max = 100
		}
	}
}
