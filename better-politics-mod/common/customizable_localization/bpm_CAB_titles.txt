bpm_hog_title = {
	type = country
	random_valid = no
	text = {
		trigger = {
			OR = {
				modifier:country_bpm_hog_top_parlelec = yes
				AND = {
					OR = {
						modifier:country_bpm_hog_top_advice = yes
						bpm_is_hosapp = yes
					}
					NOT = { var:bpm_headofgov ?= ruler }
				}
			}
		}
		localization_key = bpm_hog_title_default_contextless
	}
	text = {
		trigger = {
			OR = {
				bpm_is_hosapp = yes
				modifier:country_bpm_hog_top_advice = yes	
			}
		}
		localization_key = bpm_hog_title_ruler_interim
	}
	text = {
		trigger = {
			always = no
		}
		fallback = yes
		localization_key = bpm_hog_title_ruler
	}
}

bpm_hog_title_country_spec = {
	type = country
	random_valid = no

	text = {
		trigger = {
			bpm_shogunate_trigger = yes
		}
		localization_key = bpm_hog_title_default_shogunate
	}
	text = {
		trigger = {
			OR = {
				has_law = law_type:law_council_republic
			}
		}
		localization_key = bpm_hog_title_default_premier
	}
	text = {
		trigger = {
			c:BEL ?= owner
		}
		localization_key = bpm_hog_title_default_moi
	}
	text = {
		trigger = {
			c:ETH ?= owner
			owner = {
				has_variable = bpm_gondar_protectorate_var
				OR = {
					has_law = law_type:law_autocracy
					has_law = law_type:law_oligarchy
				}
			}
		}
		localization_key = bpm_hog_title_default_gondar
	}
	text = {
		trigger = {
			OR = {
				c:AUS ?= owner
				c:HUN ?= owner
			}
		}
		localization_key = bpm_hog_title_default_minister_president
	}
	text = {
		trigger = {
			OR = {
				country_has_primary_culture = cu:north_german
				country_has_primary_culture = cu:south_german
			}
			NOT = { c:AUS ?= owner }
		}
		localization_key = bpm_hog_title_default_chancellor
	}
	text = {
		trigger = {
			OR = {
				c:TUR ?= owner
				c:SOK ?= owner
				AND = {
					c:MUG ?= owner
					owner = {
						has_government_type = gov_mughal_empire
					}
				}
				AND = {
					c:PER ?= owner
					owner = {
						country_has_voting_franchise = no
					}
				}
			}
			owner = {
				has_law = law_type:law_monarchy
			}
		}
		localization_key = bpm_hog_title_default_grand_vizier
	}
	text = {
		trigger = {
			#c:MUG ?= owner
			owner = {
				has_government_type = gov_princely_state_sultan
				#NOT = { has_government_type = gov_mughal_empire }
				#has_law = law_type:law_monarchy
			}
		}
		localization_key = bpm_hog_title_default_mukhtar
	}
	text = {
		trigger = {
			owner = {
				any_active_law = {
					has_amendment = amendment_type:amendment_taikunate
				}
			}
		}
		localization_key = bpm_hog_title_default_taikun
	}
	text = {
		trigger = {
			c:USA ?= owner
			owner = { NOT = { has_law = law_type:law_answers_to_legislature } }
		}
		localization_key = bpm_hog_title_default_secretary_of_state
	}
	text = {
		trigger = {
			c:USA ?= owner
			owner = {
				has_law = law_type:law_answers_to_legislature
			}
		}
		localization_key = bpm_hog_title_default_speaker_of_the_house
	}
	text = {
		trigger = {
			always = no
		}
		fallback = yes
		localization_key = bpm_hog_title_default_default
	}
}
