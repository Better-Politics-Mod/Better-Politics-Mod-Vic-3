bpm_mov_wants_to_revolution = {
	political_movement_radicalism > 0.7
	political_movement_support > 0.1
}

bpm_ig_wants_to_revolution = {
	var:bpm_ig_organization > 20
	ig_approval <= angry
	bpm_interest_group_is_activated = yes
}

bpm_movement_in_front_check = {
	exists = scope:selected_front_name
	save_temporary_scope_as = selected_mov
	owner = {
		bpm_movement_in_front = yes
	}
}

bpm_ig_in_front_check = {
	exists = scope:selected_front_name
	save_temporary_scope_as = selected_ig
	owner = {
		bpm_ig_in_front = yes
	}
}

bpm_ig_in_rev = {
	owner.je:je_bpm_dissent_revolution = {
		var:bpm_front_currently_planning = {
			save_temporary_scope_as = selected_front_name
		}
	}
	bpm_ig_in_front_check = yes
}
