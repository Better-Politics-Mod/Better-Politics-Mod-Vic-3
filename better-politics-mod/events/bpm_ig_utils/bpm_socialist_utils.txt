namespace = bpm_socialist_utils

bpm_socialist_utils.1 = {
	type = country_event
	placement = root
	hidden = yes

	trigger = {
		owner = {
			NOT = { has_law = law_type:law_council_republic }
			any_interest_group = {
				has_ideology = ideology:ideology_ig_united_socialist
				NOT = { has_ideology = ideology:ideology_ig_social_democrat }
				NOT = { has_modifier = bpm_stubborn_ig }
			}
		}
	}

	immediate = {
		owner = {
			every_interest_group = {
				limit = {
					has_ideology = ideology:ideology_ig_united_socialist
					NOT = { has_ideology = ideology:ideology_ig_social_democrat }
				}
				add_modifier = { name = bpm_stubborn_ig }
			}
		}
	}
}
