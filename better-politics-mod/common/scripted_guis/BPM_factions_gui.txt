bpm_movement_factions_sg = {
    scope = interest_group
    saved_scopes = {
        polmov
        result
    }
    effect = {
        bpm_movement_setter = yes
        save_scope_as = selected_ig
        if = {
            limit = {
                scope:polmov = { any_influenced_interest_group = { type = scope:selected_ig.type } }
            }
            change_variable = {
                name = bpm_strongly_influenced_clout
                add = scope:result
            }
        }
        change_variable = {
            name = bpm_influenced_clout
            add = scope:result
        }
        set_variable = faction_hooked
    }
}

bpm_overall_revolutionary_zeal_level_show_sgui = {
    scope = country

    saved_scopes = {
        level
    }

    is_shown = {
        scope:faction = {
            has_variable = bpm_revolutionary_zeal
        }
        trigger_if = {
            limit = {
                scope:level ?= flag:lv1
            }
            bpm_overall_revolutionary_zeal < 25
        }
        trigger_else_if = {
            limit = {
                scope:level ?= flag:lv2
            }
            bpm_overall_revolutionary_zeal < 50
            bpm_overall_revolutionary_zeal >= 25
        }
        trigger_else_if = {
            limit = {
                scope:level ?= flag:lv3
            }
            bpm_overall_revolutionary_zeal < 75
            bpm_overall_revolutionary_zeal >= 50
        }
        trigger_else_if = {
            limit = {
                scope:level ?= flag:lv4
            }
            bpm_overall_revolutionary_zeal >= 75
        }

    }
}
