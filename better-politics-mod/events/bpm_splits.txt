namespace = bpm_splits

bpm_splits.1 = { # Split reformist/revolutionist socialists globally
	type = country_event
	placement = root
	hidden = yes

	trigger = {
		country_rank > rank_value:minor_power
		NOT = {
			has_global_variable = global_socialists_split_var
		}
		OR = {
			AND = {
				any_interest_group = {
					is_interest_group_type = ig_socialists
					is_marginal = no
					ig_clout > 0.03
				}
				game_date > 1850.1.1
			}
			has_global_variable = second_intl_first_split_var
		}
	}

	immediate = {
		set_global_variable = global_socialists_split_var
	}
}

bpm_splits.2 = { # Activate reformist/revolutionist socialists locally
	type = country_event
	placement = root
	hidden = yes

	trigger = {
		has_global_variable = global_socialists_split_var
		exists = ig:ig_reformist_socialists
		NOT = {
			has_variable = local_socialists_split_var
		}
		OR = {
			AND = {
				any_interest_group = {
					is_interest_group_type = ig_socialists
					is_marginal = no
					ig_clout > 0.03
				}
				game_date > 1836.1.1
			}
			has_global_variable = second_intl_first_split_var
		}
	}

	immediate = {
		set_variable = local_socialists_split_var

		ig:ig_reformist_socialists = {
			remove_ideology = ideology_nonexistent
			add_ideology = ideology_proletarian
			add_ideology = ideology_egalitarian
			add_ideology = ideology_anti_slavery
			add_ideology = ideology_ig_united_socialist
			add_ideology = ideology_ig_reformist_socialist
			every_scope_character = {
				set_ideology = ideology_socialist_leader
			}
			if = {
				limit = {
					owner = c:FRA
					NOT = { has_global_variable = global_first_major_rev_disunited_var }
				}
				set_interest_group_name = ig_possibilists
			}
			if = {
				limit = {
					owner = {
						OR = {
							country_has_primary_culture = cu:north_german
							country_has_primary_culture = cu:south_german
						}	
					}
					NOT = { has_global_variable = global_first_major_rev_disunited_var }
				}
				set_interest_group_name = ig_reformist_social_democrats
			}
		}

		ig:ig_revolutionist_socialists = {
			remove_ideology = ideology_nonexistent
			add_ideology = ideology_proletarian
			add_ideology = ideology_egalitarian
			add_ideology = ideology_anti_slavery
			add_ideology = ideology_ig_united_socialist
			add_ideology = ideology_ig_revolutionist_socialist
			every_scope_character = {
				set_ideology = ideology_socialist_leader
			}
			if = {
				limit = {
					owner = {
						OR = {
							country_has_primary_culture = cu:north_german
							country_has_primary_culture = cu:south_german
						}	
					}
					NOT = { has_global_variable = global_first_major_rev_disunited_var }
				}
				set_interest_group_name = ig_revolutionist_social_democrats
			}
		}

		trigger_event = { id = bpm_splits.3 days = 180 } 
		trigger_event = { id = bpm_2nd_intl.201 }
	}
}
