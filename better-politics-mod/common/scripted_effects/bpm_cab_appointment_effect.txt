bpm_appoint_mineco_effect = {
	owner = {
		if = {
			limit = {
				has_variable = bpm_mineco
			}
			var:bpm_mineco ?= {
				remove_character_role = bpm_character_role_minister
			}
		}
		set_variable = { name = bpm_mineco value = prev }
		set_variable = { name = bpm_minister_of_economy value = prev }
		set_variable = { name = bpm_mineco_last_ig value = prev.interest_group }
	}

	add_character_role = bpm_character_role_minister
	show_as_tooltip = {
		bpm_reload_institution_modifiers_loc_mineco = yes
	}
	hidden_effect = {
		owner = {
			bpm_reload_institution_modifiers = yes
		}
	}
}

bpm_auto_cabinet_appoint = {
	bpm_auto_cabinet_appoint_pm = yes
	bpm_auto_cabinet_appoint_subp = { m = mineco }
	bpm_auto_cabinet_appoint_subp = { m = minint }
	bpm_auto_cabinet_appoint_subp = { m = minsoc }
	bpm_auto_cabinet_appoint_subp = { m = minmor }
	bpm_auto_cabinet_appoint_subp = { m = minfor }
}

bpm_auto_cabinet_appoint_subp = {
	clear_saved_scope = bpm_newmin
	if = {
		limit = {
			has_law = law_type:law_answers_to_legislature
		}
		random_scope_character = {
			limit = {
				has_role = bpm_character_role_hog
			}
			save_scope_as = bpm_new_ruler_scope
		}
	}
	else = {
		ruler = {
			save_scope_as = bpm_new_ruler_scope
		}
	}

	if = {
		limit = {
			OR = {
				NOT = { has_variable = bpm_$m$ }
				NOT = { exists = var:bpm_$m$ }
				var:bpm_$m$ ?= {
					is_character_alive = no
				}
			}
		}
		if = {
			limit = {
				has_variable = bpm_cabappt_strategy
				var:bpm_cabappt_strategy ?= flag:bpm_greedy_ruler
			}
			scope:bpm_new_ruler_scope.interest_group = {
				save_scope_as = minister_ig
			}
			bpm_get_ministerial_candidate = yes
		}
		else_if = {
			limit = {
				has_variable = bpm_cabappt_strategy
				var:bpm_cabappt_strategy ?= flag:bpm_pick_prominent_idg
			}
			ordered_scope_character = {
				limit = {
					interest_group = {
						custom_tooltip = {
							text = "bpm_not_anti_establishment_tt"
							bpm_ig_is_radical_opposition = no
						}
						custom_tooltip = {
							text = "bpm_not_extremist_tt"
							bpm_ig_is_very_radical_opposition = no
						}
					}
					bpm_char_can_be_minister = yes
					bpm_ig_is_institutional = no
				}
				order_by = prominence
				position = 0
				set_local_variable = {
					name = bpm_cabappt_candidate
					value = this
				}
			}
		}
		else_if = {
			limit = {
				has_variable = bpm_cabappt_strategy
				var:bpm_cabappt_strategy ?= flag:bpm_pick_prominent_ing
			}
			ordered_scope_character = {
				limit = {
					interest_group = {
						custom_tooltip = {
							text = "bpm_not_anti_establishment_tt"
							bpm_ig_is_radical_opposition = no
						}
						custom_tooltip = {
							text = "bpm_not_extremist_tt"
							bpm_ig_is_very_radical_opposition = no
						}
					}
					bpm_char_can_be_minister = yes
					bpm_ig_is_institutional = yes
				}
				order_by = prominence
				position = 0
				set_local_variable = {
					name = bpm_cabappt_candidate
					value = this
				}
			}
		}
		else_if = {
			limit = {
				has_variable = bpm_$m$_last_ig
			}
			var:bpm_$m$_last_ig = {
				save_scope_as = minister_ig
			}
			bpm_get_ministerial_candidate = yes
		}
		else = {
			random_scope_character = {
				limit = {
					bpm_char_can_be_minister = yes
					interest_group = {
						is_in_government = yes
					}
				}
				set_local_variable = {
					name = bpm_cabappt_candidate
					value = this
				}
			}
		}
		if = {
			limit = {
				NOT = {
					has_local_variable = bpm_cabappt_candidate
				}
			}
			create_character = {
				interest_group = owner.ruler.interest_group
            	save_scope_as = bpm_selchar
			}
			scope:bpm_selchar = {
				set_local_variable = {
					name = bpm_cabappt_candidate
					value = this
				}
			}
		}
		set_local_variable = {
			name = bpm_minname
			value = flag:$m$
		}
		local_var:bpm_cabappt_candidate = {
			save_scope_as = bpm_newmin
			bpm_appoint_$m$_effect = yes
		}
		post_notification = bpm_minister_replaced
	}
}

bpm_get_ministerial_candidate = {
	random_scope_character = {
		limit = {
			bpm_char_can_be_minister = yes
			interest_group ?= scope:minister_ig
		}
		set_local_variable = {
			name = bpm_cabappt_candidate
			value = this
		}
	}
}

bpm_auto_cabinet_appoint_pm = {
	clear_saved_scope = bpm_newmin
	if = {
		limit = {
			has_law_or_variant = law_type:law_answers_to_legislature
		}
		random_scope_character = {
			limit = {
				has_role = bpm_character_role_hog
			}
			save_scope_as = bpm_new_ruler_scope
		}
	}
	else = {
		ruler = {
			save_scope_as = bpm_new_ruler_scope
		}
	}

	if = {
		limit = {
			AND = {
				NOT = {
					has_law_or_variant = law_type:law_answers_to_legislature
					has_law_or_variant = law_type:law_advisory_body 
					# protege sucession already handled in bpm_do_hog_transition
				}
				OR = {
					NOT = { has_variable = bpm_headofgov }
					var:bpm_headofgov ?= {
						is_character_alive = no
					}
				}
			}
		}
		ordered_scope_character = {
			limit = {
				interest_group = {
					custom_tooltip = {
						text = "bpm_not_anti_establishment_tt"
						bpm_ig_is_radical_opposition = no
					}
					custom_tooltip = {
						text = "bpm_not_extremist_tt"
						bpm_ig_is_very_radical_opposition = no
					}
				}
				bpm_char_can_be_hog = yes
				bpm_ig_is_institutional = yes
			}
			order_by = prominence
			position = 0
			set_local_variable = {
				name = bpm_cabappt_candidate
				value = this
			}
		}
	
		if = {
			limit = {
				NOT = {
					has_local_variable = bpm_cabappt_candidate
				}
			}
			create_character = {
				interest_group = owner.ruler.interest_group
            	save_scope_as = bpm_selchar
			}
			scope:bpm_selchar = {
				set_local_variable = {
					name = bpm_cabappt_candidate
					value = this
				}
			}
		}
		set_local_variable = {
			name = bpm_minname
			value = flag:bpm_headofgov
		}
		local_var:bpm_cabappt_candidate = {
			save_scope_as = bpm_newmin
			bpm_appoint_as_hog = yes
		}
		#post_notification = bpm_minister_replaced
	}
}
