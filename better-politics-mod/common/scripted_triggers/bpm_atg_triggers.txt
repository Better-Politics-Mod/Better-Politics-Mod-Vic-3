bpm_country_has_parliamentary_electoral_system = {
	OR = {
		has_law = law_type:law_appointed_on_advice
		has_law_or_variant = law_type:law_answers_to_legislature
		AND = {
			has_law = law_type:law_appointed_by_hos
			has_law = law_type:law_monarchy
		}
	}
	NOT = { has_law = law_type:law_organic_democracy }
	country_has_voting_franchise = yes
	NOT = {
		has_law = law_type:law_provisional_government
	}
}

bpm_country_has_parliamentary_oligarchic_system = {
	OR = {
		has_law = law_type:law_appointed_on_advice
		has_law_or_variant = law_type:law_answers_to_legislature
		AND = {
			has_law = law_type:law_appointed_by_hos
			has_law = law_type:law_monarchy
		}
		AND = {
			has_law = law_type:law_monarchy
			has_law = law_type:law_no_separate_hog
			country_has_voting_franchise = yes
		}
	}
	NOT = { has_law = law_type:law_organic_democracy }
	OR = {
		country_has_voting_franchise = no
		AND = {
			country_has_voting_franchise = yes
			has_law = law_type:law_monarchy
			has_law = law_type:law_no_separate_hog
		}
	}
	NOT = {
		has_law = law_type:law_provisional_government
	}
}

bpm_country_has_executive_electoral_system = {
	OR = {
		has_law = law_type:law_no_separate_hog
		has_law = law_type:law_appointed_by_hos
	}
	NOT = {
		has_law = law_type:law_organic_democracy
		has_law = law_type:law_monarchy
	}
	country_has_voting_franchise = yes
	NOT = {
		has_law = law_type:law_provisional_government
	}
}

bpm_country_has_executive_dictatorial_system = {
	OR = {
		has_law = law_type:law_no_separate_hog
		AND = {		
			has_law = law_type:law_appointed_by_hos
			NOT = { has_law = law_type:law_monarchy }
		}			
	}
	NOT = { has_law = law_type:law_organic_democracy }
	country_has_voting_franchise = no
	NOT = {
		has_law = law_type:law_provisional_government
	}
}

bpm_country_has_provisional_government_system = {
	has_law = law_type:law_provisional_government
}

bpm_country_has_organic_electoral_system = {
	has_law = law_type:law_organic_democracy
	NOT = {
		has_law = law_type:law_provisional_government
	}
}
