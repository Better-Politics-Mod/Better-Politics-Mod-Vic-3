american_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		OR = {
			home_country = {
				exists = c:USA
				c:USA = this
				is_subject = no
			}
			home_country = {
				exists = c:FSA
				c:FSA = this
				is_subject = no
			}
			home_country = {
				exists = c:CSA
				c:CSA = this
				is_subject = no
			}
		}
	}
}
