bpm_no_cabinet_picks_government_igs_count = {
    value = 0

    every_interest_group = {
        limit = {
            is_in_government = yes
            has_variable = bpm_number_of_cabpicks
            var:bpm_number_of_cabpicks ?= 0
        }
        add = 1
    }
}

bpm_pick_new_char_weight = {
    value = var:bpm_cost_ig_position_var
    add = {
        value = bpm_expected_cabpos
        subtract = bpm_actual_cabpos
        multiply = 5
    }
    add = {
        value = ig_clout
        max = 0.5
        multiply = 10
    }
    add = var:bpm_randomval
}

bpm_political_preparedness_change = {
    value = {
        value = 0.6
        desc = "BPM_BASE_POLITICAL_PREPAREDNESS_CHANGE"
    }
    add = {
        value =  bpm_focusing_cabinet
        multiply = 0.2
        if = {
            limit = {
                is_target_in_variable_list = {
                    name = bpm_focused_institutions
                    target = institution:institution_culture
                }
            }
            subtract = 0.2
        }
        desc = "BPM_FROM_NUMBER_OF_FOCUSED_INSTITUTIONS"
    }
}
