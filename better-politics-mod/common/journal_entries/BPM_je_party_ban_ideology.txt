je_bpm_ban_fascists = {
	icon = "gfx/interface/icons/law_icons/reactionary_parties_banned.dds"

	group = bpm_je_group_party_laws

	is_shown_when_inactive = {
		always = yes
	}

	can_deactivate = yes
	possible = {
		hidden_trigger = {
			has_technology_researched = mass_propaganda
		}
		OR = {
			any_interest_group = {
				bpm_leader_is_fascist = yes
				OR = {
					is_in_government = yes
					is_insurrectionary = yes
				}
			}
			bpm_banned_reactionary_parties = yes
			bpm_banned_antiliberal_parties = yes
		}
	}

	immediate = {
		bpm_ig_update_ig_ban_support = yes
	}

	complete = {
		hidden_trigger = {
			always = no
		}
		NOR = {
			any_interest_group = {
				OR = {
					bpm_leader_is_fascist = yes
					bpm_ig_is_fascist = yes
				}
				OR = {
					is_in_government = yes
					is_insurrectionary = yes
				}
			}
			bpm_banned_reactionary_parties = yes
			bpm_banned_antiliberal_parties = yes
		}
	}

	weight = 25

	should_be_pinned_by_default = no
}

je_bpm_ban_communists = {
	icon = "gfx/interface/icons/law_icons/socialist_parties_banned.dds"

	group = bpm_je_group_party_laws

	is_shown_when_inactive = {
		NOT = { has_law = law_type:law_council_republic }
	}

	can_deactivate = yes
	possible = {
		OR = {
			ig:ig_socialists ?= { is_marginal = no }
			ig:ig_revolutionist_socialists ?= { is_marginal = no }
			ig:ig_reformist_socialists ?= { is_marginal = no }
			ig:ig_anarchists ?= { is_marginal = no }
			ig:ig_trade_unions ?= {
				is_marginal = no
				OR = {
					bpm_ig_is_socialist = yes
					bpm_leader_is_socialist = yes
				}
			}
			bpm_banned_socialist_parties = yes
			bpm_banned_antiliberal_parties = yes
		}
		hidden_trigger = {
			NOT = { has_law = law_type:law_council_republic }
		}
	}

	immediate = {
		bpm_ig_update_ig_ban_support = yes
	}

	complete = {
		hidden_trigger = {
			always = no
		}
		OR = {
			NOR = {
				any_interest_group = {
					OR = {
						bpm_leader_is_socialist = yes
						bpm_ig_is_socialist = yes
					}
					is_marginal = no
				}
				bpm_banned_socialist_parties = yes
				bpm_banned_antiliberal_parties = yes
			}
			has_law = law_type:law_council_republic
		}
	}

	weight = 25

	should_be_pinned_by_default = no
}

je_bpm_ban_anti_communists = {
	icon = "gfx/interface/icons/law_icons/socialist_parties_only.dds"

	is_shown_when_inactive = {

	}

	can_deactivate = yes
	possible = {
		hidden_trigger = {

		}
	}

	immediate = {
		bpm_ig_update_ig_ban_support = yes
	}

	complete = {
		hidden_trigger = {
			always = no
		}
		OR = {
			NOR = {
				any_interest_group = {
					bpm_leader_is_socialist = no
					bpm_ig_is_socialist = no
					is_insurrectionary = yes
				}
				has_law = law_type:law_party_socialist_only
			}
			NOT = { has_law = law_type:law_council_republic }
		}
	}

	weight = 25

	should_be_pinned_by_default = no
}
