bpm_acw_events.24 = {
	type = country_event
	placement = ROOT
	
	title = bpm_acw_events.24.t
	desc = bpm_acw_events.24.d
	flavor = bpm_acw_events.24.f
	
	event_image = {
	}

	on_created_soundeffect = ""

	icon = ""
	
	duration = 3
	
	trigger = {
		exists = c:USA
		c:USA = ROOT
	}

	immediate = {	
		if = {
			limit = {
				ruler.interest_group = {
					OR = { 
  						is_interest_group_type = ig_rural_folk
						is_interest_group_type = ig_petty_bourgeoisie
						is_interest_group_type = ig_landowners
						is_interest_group_type = ig_reactionaries
						is_interest_group_type = ig_market_liberals
						is_interest_group_type = ig_conservatives
						AND = {
							is_interest_group_type = ig_liberals
							is_member_of_party = owner.py:conservative_party
						}
					}
				}
			}	
			change_variable = {
				name = bpm_slave_power_prog_var
				add = 2
			}
			change_variable = {
				name = bpm_secession_tension_prog_var
				add = 2
			}
		}
		if = {
			limit = {
				ruler.interest_group = {
					OR = { 
  						is_interest_group_type = ig_trade_unions
						is_interest_group_type = ig_intelligentsia
						is_interest_group_type = ig_industrialists
						is_interest_group_type = ig_devout
						is_interest_group_type = ig_national_liberals
						is_interest_group_type = ig_socialists
						is_interest_group_type = ig_agrarian_populists
						is_interest_group_type = ig_radicals
						AND = {
							is_interest_group_type = ig_liberals
							NOT = { is_member_of_party = owner.py:conservative_party }
						}
					}
				}
			}	
			change_variable = {
				name = bpm_states_rights_prog_var
				add = 2
			}
			change_variable = {
				name = bpm_secession_tension_prog_var
				add = 2
			}
		}
	}

	option = {
		name = bpm_acw_events.24.a
		default_option = yes
		custom_tooltip = bpm_acw_president.2.tt
	}
}

bpm_acw_events.26 = {
	type = country_event
	placement = ROOT
	
	title = bpm_acw_events.26.t
	desc = bpm_acw_events.26.d
	flavor = bpm_acw_events.26.f
	
	event_image = {
	}

	on_created_soundeffect = ""

	icon = ""
	
	duration = 3
	
	trigger = {
		exists = c:USA
		c:USA = ROOT
		any_scope_state = {
			owner = c:USA
			is_incorporated = yes
			NOT = { 
				has_modifier = bpm_voting_state 
			}
		}
	}

	immediate = {	
		random_state = {
			limit = {
				is_incorporated = yes
				NOT = { has_modifier = bpm_voting_state }
				owner = { has_modifier = missouri_compromise }
			}
			save_scope_as = bpm_new_state
		}							
		scope:bpm_new_state = {
			add_modifier = {
				name = bpm_voting_state
				is_decaying = no
			}
		}
	}

	option = {
		name = bpm_acw_events.26.b
		bpm_acw_states_rights.1.tt

		change_variable = {
			name = bpm_states_rights_prog_var
			add = 1
		}
		change_variable = {
			name = bpm_secession_tension_prog_var
			add = 1
		}
		hidden_effect = {
			scope:bpm_new_state = {
				liberate_slaves = yes
			}
		}
	}

	option = {
		name = bpm_acw_events.26.a
		bpm_acw_slave_power.1.tt
		
		change_variable = {
			name = bpm_slave_power_prog_var
			add = 1
		}
		change_variable = {
			name = bpm_secession_tension_prog_var
			add = 1
		}
		hidden_effect = {
			scope:bpm_new_state = {
				create_pop = {
					pop_type = slaves
					culture = afro_american
					size = 1000			
				}
			}
		}
	}
}
