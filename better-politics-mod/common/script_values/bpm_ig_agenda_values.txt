# IG AGENDA Slots
bpm_number_of_agendas = {
    value = 0

    if = {
        limit = {
            has_variable = bpm_agendas_ig_agenda_slot_1
        }
        add = 1
    }
    if = {
        limit = {
            has_variable = bpm_agendas_ig_agenda_slot_2
        }
        add = 1
    }
    if = {
        limit = {
            has_variable = bpm_agendas_ig_agenda_slot_3
        }
        add = 1
    }
}

bpm_maximum_number_of_agenda_slots = {
    # Todo! This shouldn't be a constant
    value = 3
}

bpm_number_of_free_agenda_slots = {
    value = bpm_maximum_agenda_slots

    subtract = bpm_number_of_agendas
}
