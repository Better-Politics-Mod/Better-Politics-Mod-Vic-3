REPLACE:two_spains_liberal_pivot = {
	if = {
		limit = {
			has_journal_entry = je_the_two_spains
		}
		add = {
			add = ig:ig_intelligentsia.ig_clout
			desc = TWO_SPAINS_INTELLIGENTSIA_CLOUT_TT
		}
		if = {
			limit = {
				ig:ig_intelligentsia ?= {
					leader = {
						OR = {
							bpm_leader_is_conservative = yes
							bpm_leader_is_reactionary = yes
						}
					}
				}
			}
			subtract = {
				add = ig:ig_intelligentsia.ig_clout
				divide = 2
				desc = TWO_SPAINS_CONSERVATIVE_INTELLIGENTSIA_TT
			}
		}
		add = {
			add = ig:ig_trade_unions.ig_clout
			desc = TWO_SPAINS_TRADE_UNIONS_CLOUT_TT
		}
		if = {
			limit = {
				ig:ig_trade_unions ?= {
					leader = {
						OR = {
							bpm_leader_is_conservative = yes
							bpm_leader_is_reactionary = yes
						}
					}
				}
			}
			subtract = {
				add = ig:ig_trade_unions.ig_clout
				divide = 2
				desc = TWO_SPAINS_CONSERVATIVE_TRADE_UNIONS_TT
			}
		}
		if = {
			limit = {
				any_interest_group = {
					OR = {
						leader = {
							OR = {
								bpm_leader_is_radical_liberal = yes
								bpm_leader_is_progressive_liberal = yes
							}
						}
						bpm_ig_is_liberal = yes
					}

					NOR = {
						is_interest_group_type = ig_intelligentsia
						is_interest_group_type = ig_trade_unions
						bpm_ig_is_traditionalist = yes
					}
				}
			}
			add = {
				value = 0
				desc = TWO_SPAINS_LIBERAL_IGS_CLOUT_TT
				every_interest_group = {
					limit = {
						OR = {
							leader = {
								OR = {
									bpm_leader_is_radical_liberal = yes
									bpm_leader_is_progressive_liberal = yes
								}
							}
							bpm_ig_is_liberal = yes
						}

						NOR = {
							is_interest_group_type = ig_intelligentsia
							is_interest_group_type = ig_trade_unions
							bpm_ig_is_traditionalist = yes
						}
					}
					add = this.ig_clout
				}
			}
		}
		if = {
			limit = {
				any_scope_character = {
					has_role = general
					interest_group = {
						OR = {
							is_interest_group_type = ig_trade_unions
							leader = {
								OR = {
									bpm_leader_is_radical_liberal = yes
									bpm_leader_is_progressive_liberal = yes
								}
							}
							bpm_ig_is_liberal = yes
						}
					}
				}
			}
			add = {
				desc = TWO_SPAINS_ARMY_UNITS_LIBERAL_COMMANDERS_TT
				every_scope_character = {
					limit = {
						has_role = general
						interest_group = {
							OR = {
								is_interest_group_type = ig_trade_unions
								leader = {
									OR = {
										bpm_leader_is_radical_liberal = yes
										bpm_leader_is_progressive_liberal = yes
									}
								}
								bpm_ig_is_liberal = yes
							}
						}
					}
					add = this.num_units_share
				}
				divide = 100
			}
		}
		if = {
			limit = {
				any_scope_character = {
					has_role = admiral
					interest_group = {
						OR = {
							is_interest_group_type = ig_trade_unions
							leader = {
								OR = {
									bpm_leader_is_radical_liberal = yes
									bpm_leader_is_progressive_liberal = yes
								}
							}
							bpm_ig_is_liberal = yes
						}
					}
				}
			}
			add = {
				desc = TWO_SPAINS_NAVY_UNITS_LIBERAL_COMMANDERS_TT
				every_scope_character = {
					limit = {
						has_role = admiral
						interest_group = {
							OR = {
								is_interest_group_type = ig_trade_unions
								leader = {
									OR = {
										bpm_leader_is_radical_liberal = yes
										bpm_leader_is_progressive_liberal = yes
									}
								}
								bpm_ig_is_liberal = yes
							}
						}
					}
					add = this.num_units_share
				}
				divide = 100
			}
		}
		if = {
			limit = {
				any_political_movement = {
					bpm_movement_is_liberal_bourgeois = yes
				}
			}
			add = {
				desc = TWO_SPAINS_LIBERAL_MOVEMENTS_TT
				every_political_movement = {
					limit = {
						bpm_movement_is_liberal_bourgeois = yes
					}
					add = {
						value = political_movement_support
					}
				}
			}
		}
		if = {
			limit = {
				modifier:country_two_spains_liberal_drift_add != 0
			}
			add = {
				desc = TIMED_MODIFIERS
				value = modifier:country_two_spains_liberal_drift_add
			}
		}
		min = 0
	}
	else = {
		value = 0
	}
}

REPLACE:two_spains_tension_pivot = {
	if = {
		limit = {
			has_journal_entry = je_the_two_spains
		}
		add = {
			desc = TWO_SPAINS_BASE_VALUE_TT
			value = 0.25
		}
		add = {
			desc = TWO_SPAINS_LEGITIMACY_TT
			value = 75
			subtract = legitimacy # Increases tension if legitimacy is below 75
			divide = 250
		}
		add = {
			desc = TWO_SPAINS_RADICAL_FRACTION_TT
			add = radical_fraction
		}
		subtract = {
			desc = TWO_SPAINS_LOYALIST_FRACTION_TT
			add = loyalist_fraction
		}
		subtract = {
			desc = TWO_SPAINS_CONSERVATIVE_IG_APPROVAL_TT
			every_interest_group = {
				limit = {
					OR = {
						is_interest_group_type = ig_landowners
						is_interest_group_type = ig_devout
						AND = {
							NOR = {
								is_interest_group_type = ig_intelligentsia
								is_interest_group_type = ig_trade_unions
							}
							OR = {
								leader = {
									OR = {
										bpm_leader_is_conservative = yes
										bpm_leader_is_reactionary = yes
									}
								}
								bpm_ig_is_moderate_conservative = yes
								bpm_ig_is_traditionalist = yes
							}
						}
					}
				}
				add = ig_approval
			}
			divide = 100
		}
		subtract = {
			desc = TWO_SPAINS_LIBERAL_IG_APPROVAL_TT
			every_interest_group = {
				limit = {
					OR = {
						is_interest_group_type = ig_intelligentsia
						is_interest_group_type = ig_trade_unions
						AND = {
							NOR = {
								is_interest_group_type = ig_landowners
								is_interest_group_type = ig_devout
							}
							OR = {
								leader = {
									OR = {
										bpm_leader_is_radical_liberal = yes
										bpm_leader_is_progressive_liberal = yes
									}
								}
								bpm_ig_is_liberal = yes
							}
						}
					}
				}
				add = ig_approval
				max = 10 # Having some very happy IGs and some very angry IGs is the definition of tension
			}
			divide = 100
		}
		min = 0
		max = 1
	}
	else = {
		value = 0
	}
}
