bpm_journal_progress_complete = {
	custom_tooltip = {
		text = "bpm_je_progress_complete"
		var:$VAR$ >= 100
	}
}

bpm_journal_progress_zero = {
	custom_tooltip = {
		text = "bpm_je_progress_zero"
		has_variable = $VAR$
		var:$VAR$ <= 0
	}
}
