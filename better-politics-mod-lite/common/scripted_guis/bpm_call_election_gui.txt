bpm_call_election_gui = {
	scope = country

	is_valid = {
		country_has_voting_franchise = yes
		in_election_campaign = no
		NOT = { has_variable = bpm_recently_called_election }
	}

	effect = {
		set_variable = {
			name = bpm_recently_called_election
			months = 36
		}
		call_election = {
			months = 3
		}
	}
}
