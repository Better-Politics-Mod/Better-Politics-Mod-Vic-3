on_monthly_pulse_state = {
	on_actions = {
		bpmcs_on_monthly
	}
}

on_half_yearly_pulse_state = {
	on_actions = {
		bpmcs_on_halfyearly
	}
}

bpmcs_on_monthly = {
	effect = {
		if = {
			limit = {
				bpm_state_has_civil_society = yes
			}
			bpmcs_civsoc_process_monthly = yes
		}
	}
}

bpmcs_on_halfyearly = {
	effect = {
		if = {
			limit = {
				bpm_state_has_civil_society = yes
			}
			bpmcs_civsoc_process_halfyearly = yes
		}
	}
}
