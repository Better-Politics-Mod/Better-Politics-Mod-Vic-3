on_half_yearly_pulse_country = {
    on_actions = {
        bpm_reload_cabinet
    }
}

on_investment_decreased = {
    on_actions = {
        bpm_inv_reload_cabinet
    }
}

on_investment_increased = {
    on_actions = {
        bpm_inv_reload_cabinet
    }
}

on_monthly_pulse_country = {
    on_actions = {
        bpm_reload_hog
        bpm_reload_synergy
    }
}

on_government_reformed = {
	on_actions = {
		bpm_reload_hog
	}
}

on_election_campaign_start = {
	on_actions = {
		bpm_reload_hog
	}
}

on_election_campaign_end = {
	on_actions = {
		bpm_reload_hog
        bpm_cab_fill_positions_oa
	}
}

on_character_death = {
    on_actions = {
        bpm_reload_hog
        bpm_reload_list_chr
    }
}


on_yearly_pulse_country = {
    on_actions = {
        bpm_reload_cabinet_pool
    }
}

bpm_cab_fill_positions_oa = {
    effect = {
        if = {
            limit = {
                NOT = { is_player = yes}
                has_variable = bpm_has_cabinet
            }
            bpm_cab_fill_positions = yes
        }
    }
}

bpm_reload_cabinet_pool = {
    effect = {
        if = {
            limit = {
                has_variable = bpm_has_cabinet
            }
            clear_variable_list = bpm_CAB_char_select_pool
            bpm_fill_cabinet_pool = yes
        }
    }
}

bpm_reload_hog = {
    effect = {
        bpm_choose_hog = yes
    }
}

bpm_reload_cabinet = {
    effect = {
        if = {
            limit = {
                has_variable = bpm_has_cabinet
            }
            bpm_reload_institution_modifiers = yes
        }
    }
}

bpm_inv_reload_cabinet = {
    effect = {
        owner = {
            if = {
                limit = {
                    has_variable = bpm_has_cabinet
                }
                bpm_reload_institution_modifiers = yes
            }
        }
    }
}

bpm_reload_synergy = {
    effect = {
        if = {
            limit = {
                has_variable = bpm_has_cabinet
            }
            bpm_cab_update_synergy = yes
        }
    }
}
