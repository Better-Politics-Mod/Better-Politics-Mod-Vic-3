bpm_china.1 = {
	type = country_event
	placement = root
	title = bpm_china.1.t
	desc = bpm_china.1.d
	flavor = bpm_china.1.f

	dlc = dlc_bpm

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = {
		video = "votp_gunboat_diplomacy"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
        # exists = c:CHI
		# this = c:CHI
		# scope:target_state = {
		#     state_region = s:STATE_SHAOZHOU
        #     is_treaty_port = yes
		# }
        NOT = { has_variable = bpm_nanking_signed }
	}

	option = {
		name = bpm_china.1.a
        set_variable = {
            name = bpm_nanking_signed
        }
		add_modifier = {
			name = bpm_opium_war_lost
			months = very_long_modifier_time
			is_decaying = yes
		}
		add_treasury = -100000
		c:GBR = {
			add_treasury = 100000
		}
		remove_modifier = bpm_canton_system
		set_variable = lost_opium_wars

		hidden_effect = {
			trigger_event = { 
				id = bpm_china.2 
				days = 120
			}
		}
	}
}
