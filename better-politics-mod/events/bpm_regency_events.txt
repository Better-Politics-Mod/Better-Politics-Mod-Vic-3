namespace = regency

# Start of the regency
regency.1 = {
	type = country_event
	placement = root

	duration = 3

	title = regency.1.t
	desc = regency.1.d
	flavor = regency.1.f

    event_image = {
		video = "votp_monarch_holding_court"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	### No cooldown

	trigger = {
		# Is a monarchy, ruler is under age of majority
		has_law = law_type:law_monarchy
		NOT = {has_law_or_variant = law_type:law_answers_to_legislature}
		bpm_has_gov_regency = no
		has_game_started = yes
	}

	cancellation_trigger = {
		OR = {
			NOT = {
				has_law = law_type:law_monarchy
			}
			scope:monarch_scope ?= {
				is_adult = yes
			}
			has_law_or_variant = law_type:law_answers_to_legislature
		}
	}

	immediate = {
		bpm_start_regency = yes
		ruler = {
			save_scope_as = monarch_scope
		}
		var:bpm_regent = {
			save_scope_as = regent_scope
		}
	}
	option = {
		name = regency.1.a
		default_option = yes
	}
}

# End of the regency
regency.3 = {
	type = country_event
	placement = root

	duration = 3

	title = regency.3.t
	desc = regency.3.d
	flavor = regency.3.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:monarch_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/2Characters"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	cooldown = { days = 100 }

	trigger = {
		bpm_has_gov_regency = yes
		ruler ?= {
			is_adult = yes
		}
	}

	cancellation_trigger = {

	}

	immediate = {
		var:bpm_regent ?= {
			save_scope_as = regent_scope
		}
		ruler ?= {
			save_scope_as = monarch_scope
		}
        bpm_end_regency = yes
	}

	option = {
		name = regency.3.a
		default_option = yes
	}
}
