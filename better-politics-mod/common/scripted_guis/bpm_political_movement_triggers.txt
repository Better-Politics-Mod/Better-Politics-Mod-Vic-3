bpm_movement_is_reactionary = {
	scope = political_movement
	is_shown = {
		NOR = {
			movement_is_socialist = yes
			movement_is_liberal = yes
		}
	}
}

bpm_any_movement_is_reactionary = {
	scope = country
	is_shown = {
		any_political_movement = {
			NOR = {
				movement_is_socialist = yes
				movement_is_liberal = yes
			}
		}
	}
}

bpm_movement_is_minority = {
	scope = political_movement
	is_shown = {
		OR = {
			is_political_movement_type = movement_cultural_minority
			is_political_movement_type = movement_religious_minority
		}
	}
}

bpm_minority_faction_sgui = {
    scope = political_movement
    effect = {
		if = {
			limit = {is_political_movement_type = movement_cultural_minority}
			culture = {
				save_scope_as = cult
			}
			custom_tooltip_no_bullet = "bpm_cultural_minority_movement_faction_default_name"
		}
		else_if = {
			limit = {is_political_movement_type = movement_religious_minority}
			religion = {
				save_scope_as = relig
			}
			custom_tooltip_no_bullet = "bpm_religious_minority_movement_faction_default_name"
		}
    }
}
