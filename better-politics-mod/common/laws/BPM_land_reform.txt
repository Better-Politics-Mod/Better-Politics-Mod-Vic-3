INJECT:law_manorialism = {
	modifier = {
		country_rigidity_baseline_add = 15
	}
	can_enact = {
		OR = {
			c:AUS ?= this # For the Austrian Empire and its subjects
			AND = {
				exists = c:AUS
				is_subject_of = c:AUS
				OR = {
					is_subject_type = subject_type_crown_land
					is_subject_type = subject_type_personal_union
				}
				c:AUS ?= { has_law = law_type:law_manorialism }
			}
			country_has_primary_culture = cu:romanian # Also for the Danubian principalities
		}
	}
}
