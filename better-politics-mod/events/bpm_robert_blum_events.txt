namespace = robert_blum
robert_blum.1 = { # Blum appears in Prussia
	type = country_event
	placement = root
	title = robert_blum.1.t
	desc = robert_blum.1.d
	flavor = robert_blum.1.f


	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
        # Triggered by JE
	}

	immediate = {
        create_character = {
			template = pru_robert_blum_template
		}
		random_scope_character = {
			limit = { has_template = pru_robert_blum_template }
			save_scope_as = robert_blum_scope
		}
	}

	option = {
		name = robert_blum.1.a
		default_option = yes
        
		scope:robert_blum_scope = {
			add_modifier = {
				name = 1848_popular_radical
				days = short_modifier_time
			}
		}
    }

	option = {
		name = robert_blum.1.b
        
		scope:robert_blum_scope = {
			add_modifier = {
				name = treasonous_traitor
				days = short_modifier_time
			}
		}

    }
}

robert_blum.2 = { # The Fate of Blum
	type = country_event
	placement = root
	title = robert_blum.2.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_global_variable = bpm_frankfurt_complete 
				} 
				desc = robert_blum.2.d1
			}
			triggered_desc = {
				trigger = {
					has_global_variable = frankfurt_dissolved
				} 
				desc = robert_blum.2.d2
			}
		}
	}
	 
	 
	flavor = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_global_variable = bpm_frankfurt_complete 
				} 
				desc = robert_blum.2.f1
			}
			triggered_desc = {
				trigger = {
					has_global_variable = frankfurt_dissolved
				} 
				desc = robert_blum.2.f2
			}
		}
	}
	

	duration = 3

	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_election.dds"

	trigger = {
		OR = {
			has_global_variable = frankfurt_dissolved
			has_global_variable = bpm_frankfurt_complete
		}
		any_character = { has_variable = is_robert_blum }
		NOT = { has_variable = fate_of_robert_blum }
		this = c:PRU
	}

	immediate = {
		hidden_effect = {
			set_variable = fate_of_robert_blum
			random_scope_character = {
				limit = { has_variable = is_robert_blum }
				save_scope_as = robert_blum
			}
		}
	}

	option = {
		name = robert_blum.2.a # Blum survives and becomes leader of Demokraten
		
		trigger = {
			has_global_variable = bpm_frankfurt_complete
		}

		random_scope_character = {
			limit = {
				has_variable = is_robert_blum
			}
			remove_character_role = agitator
			add_character_role = politician
			set_as_interest_group_leader = yes
		}

    }

	option = {
		name = robert_blum.2.b # Blum gets shafted
		
		trigger = {
			has_global_variable = frankfurt_dissolved
		}

		random_scope_character = {
			limit = {
				has_variable = is_robert_blum
			}
			kill_character = yes
		}
    }

}
