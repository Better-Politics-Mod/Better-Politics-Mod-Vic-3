	c:JAP ?= {
		create_character = { # Ruler
			template = JAP_ienari_tokugawa
			on_created = {
				bpm_set_headofgov_sticky = yes
			}
		}

		create_character = { # Protege
			template = JAP_ieyoshi_tokugawa
			heir = no
			on_created = {
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}

		create_character = { # The Emperor
			template = JAP_ninko_yamato
			ruler = yes
			on_created = {
				c:JAP ?= {
					set_variable = { name = emperor_var value = PREV }
				}
			}
		}

		create_character = { # Yamato Shunnin - Devout
			template = JAP_yamato_shunnin
		}

		create_character = { # Hotta Masyoshi - Intelligentsia
			template = JAP_masayoshi_hotta
			on_created = { 
				bpm_appoint_minint_effect = yes
				add_character_role = character_role_magnate
			}
		}

		create_character = { # Narioki Shimazu - Industrialists
			template = JAP_narioki_shimazu
		}

		create_character = { # Yukizane Konoike - Petit-Bourgeoisie
			template = JAP_yukizane_konoike
		}

		create_character = { # Tadakuni Mizuno - Landowners
			template = JAP_tadakuni_mizuno
			on_created = {
				bpm_appoint_mineco_effect = yes
			}
		}

		create_character = { # Ninomiya Sontoku - Rural Folk
			template = JAP_sontoku_ninomiya
		}

		create_character = { # Tokugawa Nariaki - Armed Forces
			template = JAP_nariaki_tokugawa
		}

		create_character = { # JAP, general
			template = JAP_hidetatsu_egawa
		}

		create_character = { # JAP, general
			template = JAP_yozo_torii
		}

		# Floating Daimyo

		create_character = { # Daimyo of Chugoku
			template = JAP_takachika_mori
		}

		create_character = { # Daimyo of Shikoku
			template = JAP_hachisuka_narimasa
		}

		create_character = { # Daimyo of Kansai
			template = JAP_tokugawa_nariyuki
		}

		create_character = { # Daimyo of Tokai

		create_character = { # Daimyo of Kyoto
			template = JAP_ii_naoaki

		}

		create_character = { # Daimyo of Tohoku
			template = JAP_date_narikuni
		}

		create_character = { # Daimyo of Ezogashima
			template = JAP_matsumae_yoshihiro
