bpm_setup_issues_varlist = {
	save_scope_value_as = {
		name = threshold
		value = 0
	}
	clear_variable_list = bpm_country_issues
	bpm_setup_issues_varlist_subp = { issue = sol }
	bpm_setup_issues_varlist_subp = { issue = anti_war }
	bpm_setup_issues_varlist_subp = { issue = anti_liberal }
	bpm_setup_issues_varlist_subp = { issue = trade_deficit }
	bpm_unsplat_issues = yes
}

bpm_clear_macros_issues = {
	remove_variable = bpm_macro_issue_sol
	remove_variable = bpm_macro_issue_anti_war
	remove_variable = bpm_macro_issue_anti_liberal
	remove_variable = bpm_macro_issue_trade_deficit
}

bpm_add_macro_issue = {
	bpm_add_macro_issue_subp = { issue = sol }
	bpm_add_macro_issue_subp = { issue = anti_war }
	bpm_add_macro_issue_subp = { issue = anti_liberal }
	bpm_add_macro_issue_subp = { issue = trade_deficit }
}

# post hook
bpm_post_electoral_mandate_hook = {
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_sol
		}
		set_variable = {
			name = bpm_electoral_mandate_sol_baseline_sol
			value = average_sol
		}
	}
	set_variable = {
		name = bpm_electoral_mandate_progress
		value = 0
	}
	set_variable = {
		name = bpm_electoral_mandate_time_progress
		value = 0
	}
	bpm_clear_macros_issues = yes
}

bpm_electoral_mandate_status_track = {
	if = {
		limit = {
			var:bpm_electoral_mandate = flag:bpm_issue_sol
		}
		set_variable = {
			name = bpm_electoral_mandate_progress
			value = {
				value = average_sol
				subtract = var:bpm_electoral_mandate_sol_baseline_sol
				multiply = 4
				floor = yes
				min = 0
				max = 4
			}
		}	
	}
	change_variable = {
		name = bpm_electoral_mandate_time_progress
		add = 1
	}
	save_scope_value_as = {
		name = bpm_electoral_mandate_time_progress_years
		value = {
			value = var:bpm_electoral_mandate_time_progress
			divide = 12
			floor = yes
		}
	}
	if = {
		limit = {
			scope:bpm_electoral_mandate_time_progress_years > var:bpm_electoral_mandate_progress
		}
		add_radicals = {
			value = 0.01
		}
		bpm_auth_add_exhaustion_small = yes
	}
	else_if = {
		limit = {
			scope:bpm_electoral_mandate_time_progress_years < var:bpm_electoral_mandate_progress
		}
		add_loyalists = {
			value = 0.01
		}
	}
}
