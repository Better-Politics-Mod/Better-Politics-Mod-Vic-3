		law_local_police
		law_dedicated_police

		law_elected_police
	}	
	
	pop_support = {
		value = 0
	}

	can_impose = { 
		is_subject_of = scope:initiator	
	} 

	# AI shouldn't try to abolish institutions
	ai_will_do = {
		always = no
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

			}
			add = 750
		}
	}
}
