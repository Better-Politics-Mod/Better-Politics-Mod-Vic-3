je_bpm_fascism = {
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	
	group = je_group_internal_affairs

	is_shown_when_inactive = {
		NOT = { has_law = law_type:law_council_republic }
		NOT = { has_variable = completed_fascism_1_var }
		has_technology_researched = political_agitation
	}

	possible = {
		NOT = { has_law = law_type:law_council_republic }
		any_interest_group = {
			OR = {
				is_being_bolstered = yes
				is_in_government = yes
			}
			leader = {
				bpm_leader_is_fascist = yes
            }
		}
	}

	immediate = {
		ig:ig_fascists = {
			save_scope_as = fascist_ig
		}
		random_interest_group = {
			limit = {
				OR = {
					is_being_bolstered = yes
					is_in_government = yes
				}
				leader = {
					bpm_leader_is_fascist = yes
            	}
			}
			save_scope_as = fascist_ig
			set_variable = og_fascist_ig
		}
	}

	complete = {
		any_interest_group = {
            is_in_government = yes
            leader = {
				bpm_leader_is_fascist = yes
            }
        }
		is_at_war = no
        institution_investment_level = {
            institution = institution_police
            value >= 4
        }
        OR = {
            has_law = law_type:law_dedicated_police
            has_law = law_type:law_militarized_police
        }
        any_scope_character = {
            has_role = general
            is_busy = no
            is_ruler = no
            count >= 2
			bpm_leader_is_fascist = no
        }
    }

	on_complete = {
		set_variable = {
			name = completed_fascism_1_var
		}
		trigger_event = { id = bpm_fascism_events.1 }
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = fascism_events.2
		}
	}

	invalid = {
		has_law = law_type:law_council_republic
	}

	weight = 10000
	can_deactivate = no
}
