bpm_bureau_reform_effect_centralization = {
	if = {
		limit = {
			has_variable = bpm_bureau_reform_result_centralization
		}
		bpm_bureau_reform_effect_centralization_success = yes
	}
	else = {
		bpm_bureau_reform_effect_centralization_failure = yes
	}
}
