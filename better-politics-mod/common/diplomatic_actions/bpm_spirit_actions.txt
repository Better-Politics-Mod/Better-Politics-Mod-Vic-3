bpm_spawn_pro_lobby = {
	groups= {
		general
	}
	
	requires_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = yes
	
	selectable = {
		NOT = { is_country_type = decentralized }
		always = no # For some reason I can't pick proper IGs that fit so we are hiding it off for now
	}
	
	potential = {
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		has_diplomatic_pact = { who = scope:target_country type = increase_relations }
	}
	
	possible = {
		bpm_spirit_can_spend_large = yes
		has_diplomatic_relevance = scope:target_country
		NOT = {
			any_political_lobby = {
				OR = {
					is_political_lobby_type = lobby_pro_country
					is_political_lobby_type = lobby_pro_overlord
				}
				target = scope:target_country
			}
		}
		num_political_lobbies <= 3
	}
	
	accept_effect = {
		random_interest_group = {
			limit = {
				scope:target_country = {
					any_interest_group = {
						is_interest_group_type = prev.prev.ig_type
						is_in_government = yes
					}
				}
				# is_member_of_any_lobby = no
				bpm_ig_is_marginal = no
			}
			save_scope_as = relevant_ig
		}
		if = {
			limit = {
				is_subject_of = scope:target_country
			}
			create_political_lobby = {
				type = lobby_pro_overlord
				target = scope:target_country
				add_interest_group = scope:relevant_ig
			}
		} 
		else = {
			create_political_lobby = {
				type = lobby_pro_country
				target = scope:target_country
				add_interest_group = scope:relevant_ig
			}
		}
		bpm_spirit_spend_large = yes
	}
}
