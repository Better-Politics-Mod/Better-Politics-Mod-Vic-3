bpm_update_ideology_armed_forces = {
    ig:ig_armed_forces ?= {
        bpm_ig_reset_ideologies = yes

        ## DEFAULT
        bpm_ig_set_ideology = {
            type = mil
            name = institutional_militarist
        }
        bpm_ig_set_ideology = {
            type = sta
            name = institutional_state
        }
        bpm_ig_set_ideology = {
            type = dop
            name = oligarch
        }
    }

    # CONDITIONAL
	if = {
		limit = {
			NOT = { 
				OR = { 
					has_law = law_type:law_oligarchy
					has_law = law_type:law_military_junta
					has_law = law_type:law_autocracy
					is_country_type = unrecognized
				}
			}

		ig:ig_armed_forces ?= {
			bpm_ig_set_ideology = {
				type = dop
				name = state_power
			}
		}
	}
    ig:ig_armed_forces ?= { bpm_trait_eco_pro_artisan_in_artisan_economy = yes }

    if = {
        limit = { has_law = law_type:law_colonial_administration }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology = {
                type = dop
                name = metropole_loyalist
            }

        limit = {
            has_variable = red_army_var

            bpm_ig_set_ideology = {
                type = mil
                name = mass_army
            }
            bpm_ig_set_ideology = {
                type = dop
                name = party_socialist
            }
        }

                type = cit
                name = samurai

                type = sta
                name = decentralized_state

                type = dop
                name = oligarch

            bpm_ig_set_ideology = {
                type = gov
                name = liberal_republican
            }
            bpm_ig_set_ideology = {
                type = mil
                name = institutional_militarist

        }
    }
    else_if = {

                c:POR ?= this

        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology = {
                type = cit
                name = nationalist
            }

                name = moderate
            }
            bpm_ig_set_ideology = {
                type = gov

    else_if = {
