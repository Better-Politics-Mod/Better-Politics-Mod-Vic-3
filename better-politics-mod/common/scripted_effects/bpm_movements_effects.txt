bpm_add_movement_interaction = {
    add_to_variable_list = {
		name = bpm_movement_interactions
		target = flag:$int$
	}
}

 	# do an event here
	bpm_add_movement_interaction = { int = liberal_split }
	trigger_event = { id = bpm_movement_splits.1 days = 31 }
}
