bpm_china_revolts.1 = {
	type = country_event
	placement = root
	title = bpm_china_revolts.1.t
	desc = bpm_china_revolts.1.d
	flavor = bpm_china_revolts.1.f

	dlc = dlc_bpm

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	event_image = {
		video = "votp_prison"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		game_date >= 1855.1.1
		NOT = { has_variable = bpm_panthay_var }
	}

	immediate = {
		set_variable = {
			name = bpm_panthay_var
		}
	}

	option = {
		name = bpm_china_revolts.1.a
		s:STATE_YUNNAN.region_state:CHI = {
			kill_population_in_state = {
				religion = sunni
				value = 3000
			}
		}
		trigger_event = { 
			id = bpm_china_revolts.2 
			days = 120
			popup = yes
		}
	}
}

bpm_china_revolts.2 = {
	type = country_event
	placement = root
	title = bpm_china_revolts.2.t
	desc = bpm_china_revolts.2.d
	flavor = bpm_china_revolts.2.f

	dlc = dlc_bpm

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	gui_window = event_window_1char_tabloid
	left_icon = scope:bpm_du_wenxiu

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {

	}

	immediate = {
		create_country = {
			tag = YUN
			origin = c:CHI
			state = s:STATE_YUNNAN.region_state:CHI
		}
		c:YUN = {
			set_state_religion = rel:sunni
			activate_law = law_type:law_oligarchy
			activate_law = law_type:law_tenant_farmers
			set_variable = {
				name = bpm_pingnan_guo_var
			}
			add_modifier = {
				name = bpm_mass_revolter
				years = 10
				is_decaying = yes
			}
			create_character = {
				first_name = Du
				last_name = Wenxiu
				historical = yes
				birth_date = 1823.1.1
				culture = cu:han
				religion = rel:sunni
				interest_group = ig_intelligentsia
				ideology = ideology_moderate
				is_politician = yes
				traits = {
					inspirational_orator
					literary
				}

				save_scope_as = bpm_du_wenxiu
				on_created = {
					set_character_as_ruler = yes
				}
			}

			create_character = {
				first_name = Ma
				last_name = Dexin
				historical = yes
				birth_date = 1794.1.1
				culture = cu:han
				religion = rel:sunni
				interest_group = ig_devout
				ideology = ideology_pacifist
				is_politician = yes
				traits = {
					pious
					erudite
					direct
				}
				on_created = {
					set_as_interest_group_leader = yes
				}
			}
		}
	}

	option = {
		name = bpm_china_revolts.2.a
		custom_tooltip = {
			text = bpm_panthay_war_tt
			trigger_event = { 
				id = bpm_china_revolts.3
				days = 90
				popup = yes
			}
		}
	}
	option = {
		name = bpm_china_revolts.2.b
		trigger = { is_player = yes }
		c:CHI ?= {
			trigger_event = { 
				id = bpm_china_revolts.3
				days = 90
				popup = yes
			}
		}
		
		play_as = c:YUN
	}
}
