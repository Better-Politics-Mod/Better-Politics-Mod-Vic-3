ig_socialists = {
	color = hsv{ 0 1 1 }
	texture = "gfx/interface/icons/ig_icons/socialists.dds"
	layer = "revolution_dynamic_trade_unions"
	index = 16
	
	ideologies = {
		ideology_gov_moderate_proletarian
		ideology_dop_radical_democrat
		ideology_sta_decentralized_state
		ideology_cit_internationalist
		ideology_rel_secularist
		ideology_mil_antimilitarist
		ideology_eco_proletarian_economy
		ideology_wel_expanded_public_welfare
		ideology_fem_gender_egalitarian
	}
	
	enable = {
		always = yes
	}
	
	on_enable = {
		ig:ig_socialists = {
			set_ig_trait = ig_trait:ig_trait_left_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_left_bastion_of_freedom
			set_ig_trait = ig_trait:ig_trait_left_systematic_opposition
		}
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
		}			
	}
	
	pop_weight = {
		value = 20

		add = {
			desc = "POP_EXCLUSIONS"
			if = {
				limit = { 
					OR = {
						is_pop_type = capitalists
						is_pop_type = peasants
					}
				}
				value = -50
			}
		}

		add = {
			desc = "URBAN_POP"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 40
			}
		}

		add = {
			desc = "URBAN_UNEMPLOYED"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					is_employed = no
				}
				value = 100
			}
		}		
	
		add = {
			desc = "POP_URBAN_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
					}
				}
				value = 30
			}
		}

		add = {
			desc = "POP_EDUCATED_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						AND = {
							is_pop_type = laborers
							literacy_rate >= 0.60
						}
						is_pop_type = machinists
					}
				}
				value = 50
			}
		}	

		add = {
			desc = "POP_LEADERS"
			if = {
				limit = { 
					OR = {
						is_pop_type = academics
						is_pop_type = engineers
					}
				}
				value = 60
			}
		}

		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
					owner = {
						radical_fraction = {
							value >= 0.2
							pop_type = laborers
						}
					}
				}
				value = 10
			}
		}
		add = {
			desc = "POPS_ANGER_SYMPATHY"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
						is_pop_type = soldiers
						is_pop_type = farmers
						is_pop_type = academics
					}
					owner = {
						radical_fraction = {
							value >= 0.5
							pop_type = laborers
						}
					}
				}
				value = 100
			}
		}

		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 50
			}
		}	
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01
					add = 1
					max = 2.0
					min = 0.5
				}
			}
		}	

		multiply = {
			desc = "NO_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_marginal = yes
						}
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "WEAK_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							ig_clout < 5
						}
					}
				}
				value = 0.75
			}		
		}

		multiply = {
			desc = "ESTABLISHED_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_powerful = yes
						}
					}
				}
				value = 1.5
			}		
		}		

		multiply = {
			desc = "DEVASTATED"	
			value = 1.0

			if = {
				limit = { 
					state = {
						devastation > 5
					}
				}
				value = 1.5
			}		
		}	
		multiply = {
			desc = "AT_WAR"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						is_at_war = yes
					}
				}
				value = 2.0
			}		
		}	


		multiply = {
			desc = "BEFORE_FIRST_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = first_intl_chain_complete_var } }
				value = 0.8
			} 
		}
		multiply = {
			desc = "BEFORE_SECOND_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = second_intl_split_complete_var } }
				value = 0.7
			}
		}
		multiply = {
			desc = "BEFORE_HEROIC_PERIOD"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = global_socialist_heroic_period } }
				value = 0.7
			}
		}
		multiply = {
			desc = "INTERNATIONAL_EXCHANGE"	
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = international_exchange_standards } }
				value = 1.3
			}		
		}
		multiply = {
			desc = "POLITICAL_AGITATION"	
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = political_agitation } }
				value = 1.3
			}		
		}
		multiply = {
			desc = "MASS_PROPAGANDA"
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = mass_propaganda } }
				value = 1.3
			}		
		}	
		multiply = {
			desc = "NO_SOCIALISM"	
			value = 1.0
			if = {
				limit = { owner = { NOT = { has_technology_researched = socialism } } }
				value = 0.5
			}		
		}
		multiply = {
			desc = "NO_LABOR_MOVEMENT"	
			value = 1.0
			if = {
				limit = { owner = { NOT = { has_technology_researched = labor_movement } } }
				value = 0.5
			}		
		}		
		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			if = {
				limit = { owner = { NOT = { has_technology_researched = egalitarianism } } }
				value = 0.0
			}		
		}

		multiply = {
			desc = "SPLIT_SOCIALISTS"
			value = 1.0

			if = {
				limit = {
					owner = {
						has_variable = socialists_split_complete_var
					}
				}
				value = -0.1
			}
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}
		multiply = ig_attraction_from_radicals_and_loyalists
	}
	
	monarch_weight = {
		value = 0.0
	}	

	commander_weight = {
		value = 0.5
	}	
	
	noble_chance = {
		value = 0.0
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.1

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.2
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}

}
