namespace = bpm_japan

bpm_japan.1 = { # For if Japan somehow gets elections without abolishing the bakufu first
	type = country_event
	placement = root

	duration = 3

	title = bpm_japan.1.t
	desc = bpm_japan.1.d
	flavor = bpm_japan.1.f

    gui_window = event_window_1char_tabloid

	left_icon = scope:shogun_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	dlc = dlc_bpm

	icon = "gfx/interface/icons/faction_icons/faction_japan_tokugawa.dds"

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		has_law = law_type:law_bakufu
        country_has_voting_franchise = yes
	}

	cancellation_trigger = {

	}

	immediate = {
		random_scope_character = {
            limit = {has_role = bpm_character_role_hog}
            save_scope_as = shogun_scope
        }
        random_political_movement = {
			limit = {
				is_political_movement_type = movement_tokugawa_loyalist
			}
			save_scope_as = loyalist_movement_scope
		}
        add_radicals = {
            value = very_large_radicals
            interest_group = ig:ig_landowners
        }
        add_radicals = {
            value = huge_radicals
            interest_group = ig:ig_armed_forces
        }
	}

	option = {
        highlighted_option = yes
		name = bpm_japan.1.a
        trigger = {
            custom_tooltip = {
                text = shogun_is_democrat_or_loyalists_marginalized
                OR = {
                    scope:shogun_scope = {
                        OR = {
                            bpm_character_is_radical_democrat = yes
		                    bpm_character_is_moderate_democrat = yes
                        }
                    }
                    scope:loyalist_movement_scope ?= {
                        political_movement_support < 0.05
                    }
                    NOT = {exists = scope:loyalist_movement_scope}
                }
            }
		}
		show_as_unavailable = {
			always = yes
		}
        scope:shogun_scope = {
            add_modifier = modifier_former_shogun
        }
	}

    option = {
        highlighted_option = yes
        name = bpm_japan.1.b
        default_option = yes
        trigger = {
            custom_tooltip = {
                text = loyalists_not_marginalized
                scope:loyalist_movement_scope ?= {
                    political_movement_support >= 0.05
                }
            }
        }
        show_as_unavailable = {
			always = yes
		}
        add_modifier = {
			name = modifier_meiji_interregnum
			days = short_modifier_time
		}
        scope:loyalist_movement_scope = {
            add_modifier = {
				name = modifier_retreat_to_osaka
				days = short_modifier_time
			}
        }
        every_interest_group = {
            limit = {
                bpm_interest_group_is_activated = yes
                has_ideology = ideology_dop_traditional_power
            }
            add_modifier = {
                name = modifier_preparing_for_revolt
                days = short_modifier_time
            }
        }
    }

    after = {
        activate_law = law_type:law_answers_to_legislature
        if = {
            limit = {has_law = law_type:law_advisory_body}
            activate_law = law_type:law_separated_powers
        }
    }
}
