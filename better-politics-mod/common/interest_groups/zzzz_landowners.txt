REPLACE:ig_landowners = {
	color = hsv{ 0.78 0.1 0.5 }
	texture = "gfx/interface/icons/ig_icons/landowners.dds"
	layer = "revolution_dynamic_landowners"
	index = 5
	
	ideologies = {
		ideology_sta_decentralized_state
		ideology_mil_aristocratic_militarist
		ideology_eco_traditional_agrarian
		ideology_wel_no_welfare
		# Vanilla
		ideology_paternalistic
		ideology_hierarchic
		ideology_patriarchal
	}
	
	character_ideologies = {
		ideology_moderate
		ideology_traditionalist
		ideology_jingoist_leader
		# ideology_legitimist
		ideology_authoritarian
		# ideology_theocrat
		# ideology_luddite
	}

	enable = {
		always = yes
	}
	
	on_enable = {
		bpm_update_ideology_landowners = yes
		bpm_update_name_landowners = yes
		# Caudillo Ideology
		if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:platinean
					country_has_primary_culture = cu:south_andean
					country_has_primary_culture = cu:north_andean
					country_has_primary_culture = cu:central_american
					country_has_primary_culture = cu:mexican
				}
			}
			ig:ig_landowners = {
				set_variable = bpm_caudillismo_ig_var
			}
		}

		ig:ig_landowners = { bpm_reorder_ideologies = yes }
	}
	on_disable = {}

	priority_cultures = { #Country scope
		rule = {
			trigger = {
				c:BIC ?= this
				any_primary_culture = { 
					cu:british = this
				}
			}
			cultures = {
				assamese avadhi baluchi bengali bihari gujarati kannada malayalam marathi oriya panjabi pathan rajput sindi tamil telegu hindustani bundeli pahari gondi bageli chhattisgarhi deccani
			}
		}
		rule = {
			trigger = {
				c:ZAN ?= this
				any_primary_culture = { cu:bedouin = this }
				NOT = { any_primary_culture = { cu:swahili = this } }
			}
			cultures = {
				bedouin
			}
		}
	}

	on_character_ig_membership = {}
	
	pop_potential = {
		custom_tooltip = {
			text = BPM_VALID_POP_FOR_IG
			owner ?= {
				NOR = {
					has_law_or_variant  = law_type:law_collectivized_agriculture
					has_law_or_variant  = law_type:law_commercialized_agriculture
				}
			}
		}
		OR = {
			is_pop_type = aristocrats
			is_pop_type = clergymen
			is_pop_type = officers
			is_pop_type = farmers
		}		
	}
	
	pop_weight = {
		value = 0

		add = {
			desc = "SLAVERY_DETRACTORS"
			if = {
				limit = {
					owner = {
						NOR = {
							has_law_or_variant = law_type:law_slavery_banned
							has_law_or_variant = law_type:law_colonial_slavery
						} 
							NOR = {
								has_law_or_variant = law_type:law_slavery_banned
								has_law_or_variant = law_type:law_colonial_slavery
							} 
					}
					NOT = { is_pop_type = aristocrats }
					OR = {
						#pop_has_primary_culture = no
						pop_acceptance <= acceptance_status_3
						state = { is_slave_state = no }
					}
				}
				value = -300
			}
			else_if = {
				limit = {
					owner = {
						has_law = law_type:law_colonial_slavery
					}
					OR = {
						pop_acceptance <= acceptance_status_3
						state = { is_incorporated = no }
					}
					NOT = { is_pop_type = aristocrats }
				}
				value = -300
			}
			else_if = {
				limit = {
					owner = {
						NOR = { has_law_or_variant = law_type:law_slavery_banned has_law_or_variant = law_type:law_colonial_slavery } 
					}
					NOT = { is_pop_type = aristocrats }
					OR = {
						pop_acceptance <= acceptance_status_3
						state = { is_slave_state = no }
					}
				}
				value = -300
			}
			
		}

		if = {
			limit = {
				is_pop_type = aristocrats
				NOT = { owner = { country_has_state_religion = rel:confucian } }
			}
			add = {
				desc = "POP_ARISTOCRATS"
				value = 250
			}
			if = {
				limit = { state = { is_slave_state = yes } }
				add = {
					desc = "SLAVEHOLDERS"
					value = 400
				}
			}
			else_if = {
				limit = { state = { is_incorporated = yes } owner = { has_law_or_variant= law_type:law_colonial_slavery } }
				add = {
					desc = "SLAVEHOLDERS"
					value = 200
				}
			}
		}
		if = {
			limit = {
				is_pop_type = aristocrats
				owner = { country_has_state_religion = rel:confucian }
			}
			add = {
				# desc = "RURAL_UPPER_CLASS" # Gotta find a way to fix this post update
				desc = "POP_ARISTOCRATS_CON"
				value = 75
			}
		}
		else_if = {
			limit = { is_pop_type = clergymen }
			add = {
				desc = "POP_CLERGYMEN"
				value = 50
			}
		}
		else_if = {
			limit = { is_pop_type = officers }
			add = {
				desc = "POP_OFFICERS"
				value = 50
			}
		}
		else_if = {
			limit = { is_pop_type = farmers }
			add = {
				desc = "POP_FARMERS"
				value = 50
			}
		}

		if = {
			limit = {
				c:BIC ?= owner
				culture = {
					has_discrimination_trait_group = heritage_group_european
				}
			}
			multiply = {
				desc = "POP_INDIA_COLONIAL_RULE"
				value = 0.1
			}
		}

		multiply = ig_attraction_to_interests_from_stability	
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}

	}
	
	monarch_weight = {
		owner.ig:ig_landowners = {
			value = bpm_ig_monarch_weight
		}
	}

	agitator_weight = {
		owner.ig:ig_landowners = {
			value = bpm_base_agitator_weight
		}
	}
	
	commander_weight = {
		owner.ig:ig_landowners = {
			value = bpm_commander_weight
		}
	}	

	executive_weight = {
		value = 0.1
	}
	
	noble_chance = {
		value = 1.0
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}

	female_executive_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.5
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
