REPLACE:ig_devout = {
	color = hsv{ 0.47 0.1 0.5 }
	texture = "gfx/interface/icons/ig_icons/devout.dds"
	layer = "revolution_dynamic_devout"
	index = 1
	
	ideologies = {
		ideology_rel_radical_moralist
		ideology_wel_religious_welfare
		ideology_fem_moderate_patriarchal
		# Vanilla
		ideology_pious
		ideology_moralist
		ideology_patriarchal
	}

	character_ideologies = {
		ideology_moderate
		ideology_traditionalist
		# ideology_reformer
		# ideology_land_reformer
		# ideology_legitimist
		# ideology_authoritarian
		ideology_theocrat
	}
	
	enable = {
		always = yes
	}
	
	on_enable = {
		bpm_update_name_devout = yes
		bpm_update_ideology_devout = yes
		

		ig:ig_devout = { bpm_reorder_ideologies = yes }
	}
	on_disable = {}

	priority_cultures = { #Country scope
		rule = {
			trigger = {
				c:BIC ?= this
				any_primary_culture = { 
					cu:british = this
				}
			}
			cultures = {
				assamese avadhi baluchi bengali bihari gujarati kannada malayalam marathi oriya panjabi pathan rajput sindi tamil telegu hindustani bundeli pahari gondi bageli chhattisgarhi deccani
			}
		}
		rule = {
			trigger = {
				c:ZAN ?= this
				any_primary_culture = {
					cu:bedouin = this
				}
			}
			cultures = {
				swahili
			}
		}
		rule = {
			trigger = {
				c:HAW ?= this
				NOT = { has_variable = deported_americans }
			}
			cultures = {
				yankee dixie british hawaiian
			}
		}
	}

	on_character_ig_membership = {
		if = {
			limit = {
				owner = {
					c:BIC ?= this
				}
			}
			change_character_religion = rel:hindu
		}
		else_if = {
			limit = {
				owner = {
					c:ZAN ?= this
				}
			}
			change_character_religion = rel:sunni
		}
		else_if = {
			limit = {
				owner = {
					NOR = {
						has_law_or_variant = law_type:law_total_separation
						has_law = law_type:law_state_atheism
					}
				}
			}
			change_character_religion = scope:interest_group.owner.religion
		}
		else = {
			change_character_religion = scope:interest_group.owner.country_definition.religion
		}
	}
		
	pop_potential = {
		OR = {
			has_state_religion = yes
			AND = {
				exists = c:BIC
				owner ?= c:BIC
				has_pop_religion = hindu
			}
			AND = {
				exists = c:ZAN
				owner ?= c:ZAN
				has_pop_religion = sunni
			}
		}
		NOT = { 
			has_pop_religion = atheist
		}
	}
	
	pop_weight = {
		value = 25

		if = {
			limit = { is_pop_type = peasants }
			add = {
				desc = "POP_PEASANTS"
				value = 100
			}
		}
		else_if = {
			limit = { is_pop_type = clergymen }
			add = {
				desc = "POP_CLERGYMEN"
				value = 500
			}
		}
		
		if = {
			limit = {
				owner = { has_law_or_variant = law_type:law_state_religion }
			}
			multiply = {
				desc = "STATE_RELIGION_LAW"
				value = 2.0
			}
		}
		else_if = {
			limit = {
				owner = { has_law_or_variant = law_type:law_total_separation }
			}
			multiply = {
				desc = "TOTAL_SEPARATION_LAW"
				value = 0.8
			}
		}
		else_if = {
			limit = {
				owner = { has_law_or_variant  = law_type:law_state_atheism }
			}
			multiply = {
				desc = "STATE_ATHEISM_LAW"
				value = 0.1
			}
		}

		add = {
			desc = "RELIGIOUS_INSTITUTIONS"
			if = {
				limit = {
					owner = {
						OR = {
							has_law_or_variant  = law_type:law_religious_schools
							has_law_or_variant  = law_type:law_charitable_health_system
						}
					}
				}
				value = 50
			}
		}

		add = {
			desc = "DEVOUT_LITERACY"
			if = {
				limit = { 
					NOT = { 
						is_pop_type = clergymen
					}
				}
				value = literacy_rate
				multiply = -100
				add = 50
				max = 0
			}
		}

		add = {
			desc = "THEOCRACY"
			if = {
				limit = {
					owner = { has_law_or_variant  = law_type:law_theocracy }
				}
				value = 100
			}
		}

		multiply = ig_attraction_to_interests_from_stability	
		
		multiply = {
			desc = "RELIGION_SUPPRESSED"
			value = 1.0
			if = {
				limit = {
					is_pop_type = clergymen
					owner = {
						has_law_or_variant = law_type:law_total_separation
					}
				}
				value = 1.5
			}
		}		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
	}
	
	monarch_weight = {
		owner.ig:ig_devout = {
			value = bpm_ig_monarch_weight
		}
	}	

	agitator_weight = {
		owner.ig:ig_devout = {
			value = bpm_base_agitator_weight
		}
	}
	
	commander_weight = {
		owner.ig:ig_devout = {
			value = bpm_commander_weight
		}
	}	

	executive_weight = {
		value = 0.1
	}
	
	noble_chance = {
		value = 0.5
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}

	female_executive_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
