je_bpm_weekly_pulsor = { # TODO: MAKE IT HIDDEN
	icon = "gfx/interface/icons/event_icons/event_election.dds"
	group = je_group_internal_affairs
	is_shown_when_inactive = {
		always = yes
	}
	possible = {
		always = yes
	}
	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					is_player = yes
				}
			}
	
			# Updating Spirit
			if = { # Using an if check because spirit can theoretically be overstocked by direct addition of it and it shouldn't be clamped because of that
				limit = {
					var:bpm_spirit_var <= bpm_spirit_max_value
				}
				change_variable = {
					name = bpm_spirit_var 
					add = bpm_spirit_value_change
				}
				clamp_variable = {
					name = bpm_spirit_var 
					min = 0
					max = bpm_spirit_max_value
				}
			}

			# Updating Chaos
			change_variable = {
				name = bpm_chaos_var 
				add = bpm_chaos_value_change
			}
			clamp_variable = {
				name = bpm_chaos_var 
				min = bpm_chaos_min
				max = bpm_chaos_max
			}
		}
	}
	should_be_pinned_by_default = no
}
