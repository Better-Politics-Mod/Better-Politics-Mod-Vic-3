bpm_subjugate_tribes = {
	groups= {
		general
	}
	
	requires_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = yes
	
	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
		capital = { state_in_africa = yes }
		is_country_type = unrecognized
		is_player = yes # we ain't letting AI use it lmao

		has_variable = bpm_st_debug_allow
	}
	
	potential = {
		scope:target_country = {
			is_country_type = decentralized
		}
	}
	
	possible = {	
		NOT = { has_war_with = scope:target_country }	
		has_strategic_land_adjacency = scope:target_country
		bpm_auth_has_no_exhaustion = yes
	}
	
	accept_effect = {
		change_infamy = 5
		scope:target_country = {
			every_scope_state = {
				add_radicals_in_state = {
					value = 0.75
				}
			}
		}
		annex = scope:target_country
		bpm_auth_add_exhaustion_huge = yes
	}
	
	ai = {
		evaluation_chance = {
			value = 0
		}	
	
		propose_score = {
			value = 0
		}
	
		will_propose = {
			always = no
		}
	}
}
