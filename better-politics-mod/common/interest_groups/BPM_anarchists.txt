ig_anarchists = {
	color = hsv{ 0 0 0 }
	texture = "gfx/interface/icons/ig_icons/anarchists.dds"
	layer = "revolution_dynamic_trade_unions"
	index = 9

	ideologies = {
		ideology_nonexistent
	}
	
	enable = {
		has_global_variable = global_anarchists_split_var
	}	
	
	on_enable = {
		ig:ig_anarchists = {
			# set_ig_trait = ig_trait:ig_trait_left_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_left_bastion_of_freedom
			set_ig_trait = ig_trait:ig_trait_left_systematic_opposition

			if = {
				limit = { country_has_primary_culture = cu:russian }
				set_interest_group_name = ig_narodniks_anarchist
			}
		}
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
		}
		owner = {
			has_variable = local_anarchists_split_var
		}
	}
	
	pop_weight = {
		value = 15

		add = ig_attraction_agrarian_socialism

		add = {
			desc = "URBAN_UNEMPLOYED"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
					is_employed = no
				}
				value = 100
			}
		}
	
		add = {
			desc = "POP_URBAN_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
					}
				}
				value = 30
			}
		}

		add = {
			desc = "POP_EDUCATED_PROLETARIAT"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
					}
				}
				value = literacy_rate
				multiply = 100
			}
		}
		add = {
			desc = "POP_EDUCATED_FARMERS"
			if = {
				limit = { 
					OR = {
						is_pop_type = peasants
						is_pop_type = farmers
					}
				}
				value = literacy_rate
				multiply = 100
			}
		}	

		add = {
			desc = "POP_LEADERS"
			if = {
				limit = { 
					OR = {
						is_pop_type = academics
						is_pop_type = engineers
					}
				}
				value = 40
			}
		}

		add = {
			desc = "POP_SOLDIERS"
			if = {
				limit = { 
					is_pop_type = soldiers
					owner = {
						radical_fraction = {
							value >= 0.2
							pop_type = laborers
						}
					}
				}
				value = 10
			}
		}
		add = {
			desc = "POPS_ANGER_SYMPATHY"
			if = {
				limit = { 
					OR = {
						is_pop_type = laborers
						is_pop_type = machinists
						is_pop_type = soldiers
						is_pop_type = farmers
						is_pop_type = academics
					}
					owner = {
						radical_fraction = {
							value >= 0.2
							pop_type = laborers
						}
					}
				}
				value = 100
			}
		}

		add = ig_attraction_tmp_bpm_non_rural_socialist

		add = {
			desc = "POP_DISCRIMINATED"
			if = {
				limit = { 
					pop_is_discriminated = yes
				}
				value = 60
			}
		}	
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01
					add = 1
					max = 2.0
					min = 0.5
				}
			}
		}	

		multiply = bpm_ig_agitators_popularity

		multiply = {
			desc = "NO_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_marginal = yes
						}
						NOT = { has_law = law_type:law_council_republic }
					}
				}
				value = 0.2
			}		
		}

		multiply = {
			desc = "WEAK_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							ig_clout < 5
						}
						NOT = { has_law = law_type:law_council_republic }
					}
				}
				value = 0.7
			}		
		}

		multiply = {
			desc = "ESTABLISHED_UNIONS"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						ig:ig_trade_unions = {
							is_powerful = yes
						}
					}
				}
				value = 1.5
			}		
		}		

		multiply = {
			desc = "AT_WAR"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						is_at_war = yes
					}
				}
				value = 1.8
			}		
		}	
		multiply = {
			desc = "BEFORE_FIRST_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = first_intl_chain_complete_var } }
				value = 0.6
			}
		}
		multiply = {
			desc = "BEFORE_SECOND_INTERNATIONAL"
			value = 1.0
			if = {
				limit = { NOT = { has_global_variable = second_intl_chain_complete_var } }
				value = 0.8
			}
		}
		multiply = {
			desc = "POLITICAL_AGITATION"	
			value = 1.0
			if = {
				limit = { owner = { has_technology_researched = political_agitation } }
				value = 1.3
			}		
		}
		multiply = {
			desc = "NO_ANARCHISM"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = anarchism }
					}
				}
				value = 0.7
			}		
		}	

		multiply = {
			desc = "NO_LABOR_MOVEMENT"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = labor_movement }
					}
				}
				value = 0.7
			}		
		}		

		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = egalitarianism }
					}
				}
				value = 0.5
			}		
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}	
		multiply = ig_attraction_from_radicals_and_loyalists
		#multiply = ig_attraction_to_revolutionists_from_stability
	}
	
	monarch_weight = {
		value = 0.0
	}	

	agitator_weight = {
		value = 1.0
		multiply = bpm_base_agitator_weight
	}

	commander_weight = {
		value = 1.0
		multiply = bpm_commander_assignment_from_laws_factor
	}	
	
	noble_chance = {
		value = 0.0
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}

}
