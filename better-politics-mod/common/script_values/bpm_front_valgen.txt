
# agrarian_populists
bpm_front_attraction_dispatch_agrarian_populists = {
	scope:selected_ig = {

        if = {
			limit = {
				prev = flag:bpm_front_popular
			}
			value = bpm_front_attraction_u2
			multiply = bpm_front_popular_multiplier
		}

        else_if = {
			limit = {
				prev = flag:bpm_front_peasant
			}
			value = bpm_front_attraction_n3
			multiply = bpm_front_peasant_multiplier
		}

        else = {
            value = 0
        }
    }
}
