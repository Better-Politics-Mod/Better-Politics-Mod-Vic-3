REPLACE:law_bakufu = {
	group = lawgroup_executive_principles
	parent = law_answers_to_legislature

	icon = "gfx/interface/icons/law_icons/bakufu.dds"

	progressiveness = -100

	unlocking_laws = {
		law_monarchy
		law_social_monarchy
	}

	is_visible = {
		country_has_primary_culture = cu:japanese
		OR = {
			has_law = law_type:law_bakufu
			NOT = {has_variable = japan_restoration_complete}
		}
	}

	on_activate = {
		hidden_effect = {
			country_assess_daimyo_alignments = yes
		}
		if = {
			limit = {
				has_game_started = yes
			}
			hidden_effect = {
				var:bpm_headofgov ?= {
					bpm_unappoint_as_hog = yes
				}
				remove_variable = bpm_hog_heir
				every_scope_character = {
					limit = {has_role = bpm_character_role_protege}
					remove_character_role = bpm_character_role_protege
				}
			}
		}
		bpm_on_new_hog_dictatorship = yes
		if = {
			limit = {has_game_started = yes}
			bpm_end_regency = yes
		}
	}

	on_deactivate = {
		if = {
			limit = {
				country_has_monarchy_law = yes
				country_has_non_monarchy_law = no # To handle an execution order issue
			}
			hidden_effect = {
				country_assess_daimyo_alignments = yes
				random_scope_character = {
        		    limit = {
        		        has_role = bpm_character_role_hog
        		    }
        		    set_variable = former_shogun_var
        		}
				var:bpm_headofgov ?= {
					bpm_unappoint_as_hog = yes
				}
				remove_variable = bpm_hog_heir
				every_scope_character = {
					limit = {has_role = bpm_character_role_protege}
					remove_character_role = bpm_character_role_protege
				}
			}
		}
		hidden_effect = {
			if = {
				limit = {NOT = {has_law_or_variant = law_type:law_answers_to_legislature}}
				bpm_end_acting_hog = yes
			}
		}
		if = {
			limit = {
				has_law = law_type:law_monarchy
				ruler = {
					is_adult = no
					NOT = { has_variable = regency_complete }
				}
			}
			trigger_event = {
				id = regency.1
				popup = yes
			}
		}
	}

	modifier = {
		country_legitimacy_govt_size_add = 1
		country_aristocrats_pol_str_mult = 0.25
		country_officers_pol_str_mult = 0.5
		country_soldiers_pol_str_mult = 0.25
		country_bpm_hog_veto = yes
		country_legitimacy_govt_total_votes_add = 20
		country_bpm_hog_top_parlelec = yes
		country_bpm_cabinet_government_bonus = yes

		country_synergy_from_ideological_compatibility = 80
	}

	can_impose = {
		always = no
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
	}

	ai_impose_chance = {
		value = 0
	}
}
