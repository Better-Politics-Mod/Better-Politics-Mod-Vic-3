bpm_ig_reset_ideologies = {
    bpm_remove_all_eco_ideologies = yes
    bpm_remove_all_gov_ideologies = yes
    bpm_remove_all_dop_ideologies = yes
    bpm_remove_all_rel_ideologies = yes
    bpm_remove_all_mil_ideologies = yes
    bpm_remove_all_fem_ideologies = yes
    bpm_remove_all_cit_ideologies = yes
    bpm_remove_all_wel_ideologies = yes
    bpm_remove_all_sta_ideologies = yes
    bpm_remove_all_exe_ideologies = yes
    bpm_remove_all_ban_ideologies = yes
}

bpm_ig_set_ideology_no_re = {
    bpm_remove_all_$type$_ideologies = yes
    add_ideology = ideology_$type$_$name$
}

bpm_update_ideology_all_igs = {
    bpm_update_ideology_armed_forces = yes
	bpm_update_ideology_landowners = yes
}

bpm_update_ideology_landowners = {
    ig:ig_landowners ?= {
        bpm_ig_reset_ideologies = yes

        bpm_ig_set_ideology_no_re = {
            type = eco
            name = traditional_agrarian
        }
        bpm_ig_set_ideology_no_re = {
            type = wel
            name = no_welfare
        }
        bpm_ig_set_ideology_no_re = {
            type = mil
            name = aristocratic_militarist
        }
    }

		limit = {
			c:PRU ?= this
		}

            bpm_ig_set_ideology_no_re = {
                type = gov
                name = radical_monarchist
            }
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = patriotic
            }
        }

	else_if = {
        limit = {
            c:BRZ ?= this
        }
		ig:ig_landowners ?= {
			bpm_ig_set_ideology_no_re = {
				type = gov
				name = moderate_monarchist
			}
			bpm_ig_set_ideology_no_re = {
				type = cit
				name = segregationist
			}
			bpm_ig_set_ideology_no_re = {
				type = fem
				name = patriarchal
			}
		}
    }
	else_if = {
        limit = {
            c:PRG ?= this
        }
		ig:ig_landowners ?= {
			bpm_ig_set_ideology_no_re = {
				type = dop
				name = oligarch
			}
		}
    }
	else_if = {
        limit = {
            c:ARG ?= this
        }
		ig:ig_landowners ?= {
			bpm_ig_set_ideology_no_re = {
				type = dop
				name = oligarch
			}
			bpm_ig_set_ideology_no_re = {
				type = ban

            c:VNZ ?= this
        }
		ig:ig_landowners ?= {
			bpm_remove_all_dop_ideologies = yes
		}
    }

        limit = {
            c:PAP ?= this
        }
		ig:ig_landowners ?= {
			bpm_ig_set_ideology_no_re = {
				type = gov
				name = theocrat
			}
		}
    }
	else_if = {
		limit = {
			OR = {
				c:SPA ?= this
				c:SPC ?= this
			}
		}
		ig:ig_landowners ?= {
			bpm_ig_set_ideology_no_re = {
				type = rel
				name = radical_moralist
			}
		}
	}
	else_if = {
		limit = {
			c:USA ?= this
		}
		ig:ig_landowners ?= {
			bpm_ig_set_ideology_no_re = {
				type = gov
				name = liberal_republican
			}
			bpm_ig_set_ideology_no_re = {
				type = dop
				name = moderate_democrat #authoritarian_democrat ???
			}
			bpm_ig_set_ideology_no_re = {
				type = dop
				name = antebellum_economy
			}
			bpm_ig_set_ideology_no_re = {
				type = eco
				name = antebellum_economy
			}

	}

				type = eco

			}

		limit = {
			country_has_state_religion = rel:confucian

			NOT = { has_journal_entry = je_confucian_state }
		}

			bpm_ig_set_ideology_no_re = {
				type = dop
				name = traditional_power
			}
		}
	}
	if = {
		limit = {
			has_law = law_type:law_purchased_officers
			NOT = { has_law = law_type:law_peasant_levies }
		}

			bpm_ig_set_ideology_no_re = {

				name = oligarch

				type = mil
				name = institutional_militarist
			}
		}

		limit = {
			has_law = law_type:law_colonial_administration
		}

			bpm_ig_set_ideology_no_re = {
				type = dop
				name = metropole_loyalist
			}
		}
	}

bpm_update_ideology_armed_forces = {
    ig:ig_armed_forces ?= {
        bpm_ig_reset_ideologies = yes

        ## DEFAULT
        bpm_ig_set_ideology_no_re = {
            type = mil
            name = institutional_militarist
        }
        bpm_ig_set_ideology_no_re = {
            type = sta
            name = institutional_state
        }
        bpm_ig_set_ideology_no_re = {
            type = dop
            name = oligarch
        }
    }

    # CONDITIONAL
	if = {
		limit = {
			NOT = { 
				OR = { 
					has_law = law_type:law_oligarchy
					has_law = law_type:law_military_junta
					has_law = law_type:law_autocracy
					is_country_type = unrecognized
				}
			}
		}
		ig:ig_armed_forces ?= {
			bpm_ig_set_ideology_no_re = {
				type = dop
				name = state_power
			}
		}
	}
    if = {
        limit = {
			bpm_country_has_artisanal_economy = yes
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = eco
                name = artisan
            }
        }
    }
    ## Technically you could have the red-national-guard
    if = {
        limit = {
            has_variable = red_army_var
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = mass_army
            }
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = party_socialist
            }
        }
    }
    else_if = {
        limit = {
            has_variable = bpm_samurai_ig_var
            bpm_country_has_artisanal_economy = no
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = traditional_power
            }
            bpm_ig_set_ideology_no_re = {
                type = gov
                name = radical_monarchist
            }
            bpm_ig_set_ideology_no_re = {
                type = eco
                name = trad_isolationist
            }
            bpm_ig_set_ideology_no_re = {
                type = cit
                name = samurai
            }
            bpm_ig_set_ideology_no_re = {
                type = sta
                name = decentralized_state
            }
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = aristocratic_militarist
            }
        }
    }
    else_if = {
        limit = {
            has_journal_entry = je_caudillo
            bpm_country_has_artisanal_economy = no
        }
        ig:ig_armed_forces ?= {
            bpm_ig_set_ideology_no_re = {
                type = dop
                name = oligarch
            }
            bpm_ig_set_ideology_no_re = {
                type = gov
                name = liberal_republican
            }
            bpm_ig_set_ideology_no_re = {
                type = mil
                name = institutional_militarist
            }
        }
    }

    ig:ig_armed_forces ?= { bpm_reorder_ideologies = yes }
}
