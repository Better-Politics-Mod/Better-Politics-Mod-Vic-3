GLOBAL = {
	set_global_variable = bpm_2_4_save_var

	set_global_variable = {
		name = bpm_je_march_to_radicalism_progress
		value = 0
	}

	bpm_set_vanilla_ideologies_list = yes

	every_country = {
		limit = {
			is_country_type = colonial
			OR = {
				has_law = law_type:law_oligarchy
				has_law = law_type:law_autocracy
			}
		}
		bpm_set_colony_law = yes
	}
	every_country = {
		limit = {
			has_law = law_type:law_colonial_administration
		}
		activate_law = law_type:law_presidential_republic
	}

	every_country = {	
		bpm_setup_country = yes
	}

	every_country = {
		trigger_event = { id = bpm_ideology_soft_compat_events.1 }
		trigger_event = { id = bpm_ideology_soft_compat_events.2 }
		trigger_event = { id = bpm_ideology_soft_compat_events.4 }
	}

	every_country = {
		limit = {
			OR = {
				has_law = law_type:law_universal_suffrage
				has_law = law_type:law_census_voting
				has_law = law_type:law_wealth_voting
			}
		}
		set_institution_investment_level = {
			institution = institution_suffrage
			level = 3
		}
	}

	every_country = {
		if = {
			limit = {
				bpm_country_is_in_europe = no
			}
			activate_law = law_type:law_right_to_associate
		}
		else = {
			activate_law = law_type:law_guild_system
		}
	}

	bpm_setup_global_laws_varlists = yes

	every_country = {
        bpm_update_exe_ideology = yes
		bpm_update_consciousness = yes
		#bpm_setup_party_establishment = yes
    }    
}
