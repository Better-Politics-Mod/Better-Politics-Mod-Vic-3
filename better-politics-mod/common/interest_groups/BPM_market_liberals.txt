ig_market_liberals = {
	color = hsv{ 0.45 0.7 0.9 }
	texture = "gfx/interface/icons/ig_icons/market_liberals.dds"
	layer = "revolution_dynamic_industrialists"
	index = 12
	
	ideologies = {
		ideology_nonexistent
	}
	
	enable = {
		owner ={ 
			has_variable = local_marlib_split_var
			NOT = {
				is_country_type = decentralized
			}
		} 
	}
	
	on_enable = {
		ig:ig_market_liberals = {
			# set_ig_trait = ig_trait:ig_trait_urban_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_urban_well_oiled_machine
			set_ig_trait = ig_trait:ig_trait_urban_systematic_opposition
		}
		ig:ig_market_liberals = {
			remove_ideology = ideology_nonexistent
			if = {
				limit = { ROOT = { bpm_country_is_republic = yes } }
				add_ideology = ideology_gov_liberal_republican
			}
			else = { add_ideology = ideology_gov_moderate_monarchist }
			# America Marlib
			if = {
				limit = { 
					OR = {
						ROOT = { bpm_country_is_latin_american = yes }
						ROOT = c:USA
					}
				}
				add_ideology = ideology_dop_moderate_democrat
				add_ideology = ideology_sta_federated_state
			}
			else_if = {
				limit = {
					ROOT = { has_law = law_type:law_federated }
				}
				add_ideology = ideology_dop_technocrat
				add_ideology = ideology_sta_federated_state
			}
			else = { 
				add_ideology = ideology_dop_technocrat
				add_ideology = ideology_sta_powerful_state
			}
			if = {
				limit = { bpm_country_is_in_new_world = yes }
				add_ideology = ideology_cit_immigrant_nationalist
			}
			else = {
				add_ideology = ideology_cit_liberal_nationalist
			}
			if = {
				limit = { ROOT = c:USA }
				add_ideology = ideology_rel_secularist
			}
			else = {
				add_ideology = ideology_rel_moralist
			}
			add_ideology = ideology_mil_plutocratic_militarist
			add_ideology = ideology_eco_laissez_faire
			add_ideology = ideology_wel_market_welfare
			add_ideology = ideology_fem_benevolent_sexism
			add_ideology = ideology_ban_averse

			# Vanilla
			add_ideology = ideology_liberal
			add_ideology = ideology_plutocratic
			add_ideology = ideology_laissez_faire
			add_ideology = ideology_individualist

			every_scope_character = {
				limit = {
					has_ideology = ideology:ideology_placeholder_leader
				}
				set_ideology = ideology:ideology_center_leader
			}
			bpm_reorder_ideologies = yes
		}

		ig:ig_liberals = {
			if = {
				limit = { 
					ROOT = { bpm_country_is_latin_american = yes }
				}
				bpm_remove_all_wel_ideologies = yes
				add_ideology = ideology_wel_public_welfare
			}
			bpm_remove_all_eco_ideologies = yes
			add_ideology = ideology_eco_modernizer
			bpm_reorder_ideologies = yes
			set_interest_group_name = ig_moderate_liberals
		}

		bpm_rename_igs = yes
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		custom_tooltip = {
			text = BPM_VALID_POP_FOR_IG
			owner = {
				NOT = {
					is_country_type = decentralized
				}
			}
		}
	}
	
	pop_weight = {
		value = 40

		if = {
			limit = { is_pop_type = peasants }
			add = {
				desc = "POP_EXCLUSIONS"
				value = -50
			}
		}
		else_if = {
			limit = { is_pop_type = aristocrats }
			add = {
				desc = "POP_ARISTOCRATS"
				value = 60
			}
		}
		else_if = {
			limit = { is_pop_type = capitalists }
			add = {
				desc = "POP_CAPITALISTS"
				value = 195
			}
		}
		else_if = {
			limit = {
				strata = middle
				NOT = { is_pop_type = clergymen }
			}
			add = {
				desc = "MIDDLE_CLASS"
				value = 20
			}
		}

		# URBAN EMPLOYMENT
		if = {
			limit = {
				bpm_pop_employment_is_rural = no
			}
			add = {
				value = 30
				desc = "URBAN"
			}
			if = {
				limit = {
					strata = middle
					NOT = { is_pop_type = clergymen }
				}
				add = {
					value = 80
					desc = "URBAN_MIDDLE_CLASS"
				}
			}
			else_if = {
				limit = { strata = rich }
				add = {
					value = 60
					desc = "URBAN_UPPER_CLASS"
				}
			}
		}

		if = {
			limit = { owner = { bpm_performance_significant_country = yes } }
			add = {
				desc = "SLAVERY_DETRACTORS"
				if = {
					limit = {
						owner = { NOT = { has_law = law_type:law_slavery_banned } }
						OR = {
							pop_has_primary_culture = no
							state = { is_slave_state = no }
						}
					}
					value = 100
				}
			}
			multiply = {
				desc = "LEADER_POPULARITY"
				scope:interest_group = {
					leader = {
						value = popularity
						multiply = 0.005
						add = 1
						max = 1.5
						min = 0.67
					}
				}
			}
			multiply = {
				desc = "TRADITIONAL_MONARCHY"	
				value = 1.0
				
				if = {
					limit = { 
						owner = {
							has_law = law_type:law_monarchy
							OR = {
								has_law = law_type:law_oligarchy
								has_law = law_type:law_autocracy
							}
						}
					}
					value = 0.75
				}		
			}
			multiply = bpm_ig_agitators_popularity
			multiply = bpm_ig_ruler_popularity
			multiply = ig_attraction_from_radicals_and_loyalists
		}
		multiply = {
			desc = "UNEDUCATED"
			value = literacy_rate
			multiply = 1.5
			max = 1.0
		}
		multiply = ig_attraction_to_ruler_ideology_from_stability
	}
	
	monarch_weight = {
		value = 0.2
	}	
	
	commander_weight = {
		value = 1.0
		multiply = bpm_commander_assignment_from_laws_factor
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
