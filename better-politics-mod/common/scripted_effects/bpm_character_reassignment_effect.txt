bpm_map_ideology_to_ig = {
	if = {
		limit = {
			interest_group ?= {
				bpm_ig_is_institutional = yes
			}
			has_ideology = ideology:ideology_$ideology$
		}
		set_interest_group = owner.ig:ig_$ig$
		set_ideology = ideology:ideology_center_leader
	}
	else_if = {
		limit = {
			has_ideology = ideology:ideology_$ideology$
		}
		set_ideology = ideology:ideology_center_leader
	}
}

bpm_character_remap_ideology_effect = {
	bpm_map_ideology_to_ig = { ig = radicals ideology = radical }
	bpm_map_ideology_to_ig = { ig = radicals ideology = republican_leader }
	bpm_map_ideology_to_ig = { ig = liberals ideology = liberal_leader }
	bpm_map_ideology_to_ig = { ig = conservatives ideology = reformer }
	bpm_map_ideology_to_ig = { ig = fascists ideology = fascist }
	bpm_map_ideology_to_ig = { ig = anarchists ideology = anarchist }
	bpm_map_ideology_to_ig = { ig = market_liberals ideology = market_liberal }
	bpm_map_ideology_to_ig = { ig = national_liberals ideology = positivist }
	
	if = {
		limit = {
			owner = {bpm_has_movement_interaction = { int = refsoc_split }}
		}
		bpm_map_ideology_to_ig = { ig = reformist_socialists ideology = social_democrat }
	}
	else = {
		bpm_map_ideology_to_ig = { ig = socialists ideology = social_democrat }
	}

	if = {
		limit = {
			owner = {bpm_has_movement_interaction = { int = revsoc_split }}
		}
		bpm_map_ideology_to_ig = { ig = revolutionist_socialists ideology = vanguardist }
		bpm_map_ideology_to_ig = { ig = revolutionist_socialists ideology = communist }
	}
	else = {
		bpm_map_ideology_to_ig = { ig = socialists ideology = vanguardist }
		bpm_map_ideology_to_ig = { ig = socialists ideology = communist }
	}

	if = {
		limit = {
			OR = {
				has_ideology = ideology:ideology_authoritarian
				has_ideology = ideology:ideology_royalist
				has_ideology = ideology:ideology_theocrat
			}
		}
		if = {
			limit = {
				interest_group ?= {
					bpm_ig_is_institutional = yes
				}
			}
			set_ideology = ideology:ideology_moderate
		}
		else = {
			set_ideology = ideology:ideology_center_leader
		}
	}
}
