    # the variable for each party is bpm_party_$PARTY$_voting_power
    # example: bpm_party_liberal_voting_power, bpm_party_radical_voting_power
    if = {
        limit = {
            has_variable = bpm_organic_election
        }
        remove_variable = bpm_organic_election
        if = {
            limit = {
                bpm_country_has_parliamentary_electoral_system = yes 
                NOT = {
                    has_variable = bpm_has_majority_consumable
                } 
            }
            trigger_event = { 
                id = bpm_form_government_events.1
                popup = yes
            }
        }
        else_if = {
            limit = {
                has_variable = bpm_has_majority_consumable
                bpm_country_has_parliamentary_electoral_system = yes 
            }
            remove_variable = bpm_has_majority_consumable
            bpm_update_government_and_opposition = yes
            trigger_event = {
                id = bpm_form_government_events.2
                popup = yes
            }
        }
        else = {
            bpm_update_government_and_opposition = yes
        }
    }
}
