REPLACE:exile_character = {
	clicksound = "event:/SFX/DLC/1.3_ip1/UI/exile_character"

	should_ai_evaluate = {
		country_rank >= rank_value:major_power
	}

	potential = {
		is_in_exile_pool = no
		is_in_void = no
		exists = owner
		# Cannot exile pure commanders, we have the retire interaction for that
		OR = {
			has_role_of_type = politician
			has_role_of_type = agitator
		}
		hidden_trigger = { # BPM addition: Making AI incapable of exiling non-institutional politicians unless they are a great power
			OR = {
				NOT = { has_variable = bpm_character_in_institutional_ig_var }
				has_role_of_type = agitator
				owner ?= {
					OR = {
						country_rank >= rank_value:great_power	
						is_player = yes
					}
				}
			}
		}
		is_ruler_of_own_country = no
		is_heir_of_own_country = no
		NOT = { has_variable = bpm_untouchable_char }
		NOT = {
			has_role = bpm_character_role_hog
			has_role = bpm_character_role_protege
			has_role = bpm_character_role_regent
            has_role = bpm_character_role_acting_hog
		}
	}

	show_requirements = yes

	possible = {
		OR = {
			has_role_of_type = agitator
			interest_group = {
				is_in_government = no
				is_insurrectionary = no
			}
		}
		interest_group = {
			bpm_ig_is_marginal = no
		}
		owner ?= {
			NOR = {
				has_law_or_variant = law_type:law_protected_speech
				has_law_or_variant = law_type:law_guaranteed_liberties
				has_law = law_type:law_traditional_ostracism
			}
		}
		is_busy = no
		trigger_if = {
			limit = {owner = {is_player = yes}}
			custom_tooltip = {
				text = ideology_not_too_close_tt
				bpm_synergy_with_executive < 0.7
			}
		}
		trigger_if = {
			limit = {
				c:JAP ?= owner
			}
			trigger_if = {
				limit = {
					owner ?= {
						has_law = law_type:law_bakufu
					}
				}
				custom_tooltip = {
					text = house_tokugawa_not_tt
					NOT = {
						has_variable = house_tokugawa
					}
				}
			}
			trigger_if = {
				limit = {
					owner ?= {
						NOR = {
							has_law_or_variant = law_type:law_council_republic
							has_law_or_variant = law_type:law_corporate_state
						}
					}
				}
				custom_tooltip = {
					text = house_yamato_not_tt
					NOT = {
						has_variable = house_yamato
					}
				}
			}
		}
	}

	effect = {
		this.interest_group = {
			save_scope_as = relevant_ig
		}
		this = {
			save_scope_as = exile_scope
		}
		owner ?= {
			save_scope_as = exiling_country_scope
		}
		if = {
			limit = {
				owner ?= { has_journal_entry = je_caudillo }
				interest_group ?= {
					is_interest_group_type = ig_armed_forces
				}
			}
			if = {
				limit = {
					interest_group ?= {
						leader = {
							this = ROOT
						}
					}
				}
				owner ?= {
					custom_tooltip = {
						text = lose_large_caudillo_points
						je:je_caudillo ?= {
							add_progress = { value = -200 name = je_caudillo_progress_bar }
						}
					}
				}
			}
			else = {
				owner ?= {
					custom_tooltip = {
						text = lose_caudillo_points
						je:je_caudillo ?= {
							add_progress = { value = -100 name = je_caudillo_progress_bar }
						}
					}
				}
			}
		}
		if = {
			limit = {
				owner ?= {
					is_player = yes
					NOT = { has_variable = achievement_agitate_elsewhere_var }
				}
				popularity >= 100
			}
			owner ?= { set_variable = achievement_agitate_elsewhere_var }
		}
		if = {
			limit = {
				has_role_of_type = politician
			}
			owner ?= {
				scope:relevant_ig = {
					add_modifier = {
						name = modifier_exiled_leader
						days = normal_modifier_time
						is_decaying = yes
					}
				}
			}
		}
		owner ?= {
			custom_tooltip = exiling_dissident_radicals_factors
		}
		# Extra radical from exiling a popular character
		if = {
			limit = {
				popularity >= 50
			}
			owner ?= {
				add_radicals = {
					value = small_radicals
					interest_group = scope:relevant_ig
				}
			}
		}
		# Standard radicals from having Right of Assembly
		if = {
			limit = {
				owner ?= { has_law = law_type:law_right_of_assembly }
			}
			owner ?= {
				add_radicals = {
					value = medium_radicals
					interest_group = scope:relevant_ig
				}
			}
		}
		# Fewer radicals from Censorship
		else_if = {
			limit = {
				owner ?= { has_law = law_type:law_censorship }
			}
			owner ?= {
				add_radicals = {
					value = small_radicals
					interest_group = scope:relevant_ig
				}
			}
		}
		# Boring characters should be made interesting when being exiled
		if = {
			limit = {
				has_ideology = ideology:ideology_moderate
			}
			adopt_random_antigovernment_ideology = yes
		}
		hidden_effect = {
			if = {
				limit = {
					popularity > 15
				}
				add_modifier = {
					name = in_exile_modifier
				}
				every_country = {
					limit = {
						has_diplomatic_relevance = scope:exiling_country_scope
					}
					post_notification = exile_notification
				}
			}
			else = {
				add_modifier = {
					name = in_exile_modifier
					multiplier = 3
				}
			}
		}
		custom_tooltip = {
			text = exile_dissident_tt
			exile_character_with_martyrdom = yes
		}
	}

	cooldown = { days = normal_modifier_time }
	show_confirmation_box = yes

	ai_chance = {
		value = -50
		if = {
			limit = {
				interest_group = {
					ig_approval <= -12
				}		
			}
			add = 55
		}
		else_if = {
			limit = {
				interest_group = {
					ig_approval <= -8
				}		
			}
			add = 25
		}
		# only exile if IG approval < 0
		if = {
			limit = {
				interest_group = {
					ig_approval < 0
				}		
			}
			if = {
				limit = {
					scope:actor ?= {
						NOR = {
							has_law = law_type:law_autocracy
							has_law = law_type:law_oligarchy
							has_law = law_type:law_single_party_state
						}
					}			
				}
				add = -25
			}
			if = {
				limit = {
					scope:actor ?= {
						has_law = law_type:law_national_guard
					}			
				}
				add = 15
			}
			else_if = {
				limit = {
					scope:actor ?= {
						has_law = law_type:law_guaranteed_liberties
					}			
				}
				add = -100
			}
			else_if = {
				limit = {
					scope:actor ?= {
						has_law = law_type:law_secret_police
					}			
				}
				add = 30
			}
			if = {
				limit = {
					scope:actor ?= {
						has_law = law_type:law_censorship
					}			
				}
				add = 10
			}
			else_if = {
				limit = {
					scope:actor ?= {
						has_law = law_type:law_right_of_assembly
					}			
				}
				add = -10
			}
			else_if = {
				limit = {
					scope:actor ?= {
						has_law = law_type:law_outlawed_dissent
					}			
				}
				add = 30
			}
			else_if = {
				limit = {
					scope:actor ?= {
						has_law = law_type:law_protected_speech
					}			
				}
				add = -20
			}
		}
		multiply = 0.3
	}
}

			limit = { is_interest_group_type = ig_anarchists }

			limit = { is_interest_group_type = ig_reformist_socialists }

			limit = { is_interest_group_type = ig_revolutionist_socialists }

			limit = { is_interest_group_type = ig_market_liberals }
			scope:actor ?= { exists = ig:ig_market_liberals }
		}
		trigger_else_if = {
			limit = { is_interest_group_type = ig_national_liberals }
			scope:actor ?= { exists = ig:ig_national_liberals }
		}
		trigger_else_if = {
			limit = { is_interest_group_type = ig_fascists }
			scope:actor ?= { exists = ig:ig_fascists }
		}

		# cannot invite culturally discriminated exiles
		character_acceptance = { target = scope:actor value >= acceptance_status_3 }

		# Agitator must be able to start or join a movement
		can_agitate = scope:actor
	}

REPLACE:arrange_accident = {
	clicksound = "event:/SFX/DLC/1.3_ip1/UI/exile_character"

	potential = {
		is_in_exile_pool = no
		is_in_void = no
		is_ruler_of_own_country = no
		is_heir_of_own_country = no
		is_immortal = no
		OR = {
			has_role_of_type = general
			has_role_of_type = admiral
			has_role_of_type = politician
			has_role_of_type = agitator
		}
		NOT = { has_variable = bpm_untouchable_char }
	}

	show_requirements = yes

	possible = {
		owner ?= {
			has_law_or_variant = law_type:law_secret_police
		}
		interest_group = {
			bpm_ig_is_marginal = no
		}
		trigger_if = {
			limit = {
				c:JAP ?= owner
			}
			trigger_if = {
				limit = {
					owner ?= {
						has_law = law_type:law_bakufu
					}
				}
				custom_tooltip = {
					text = house_tokugawa_not_tt
					NOT = {
						has_variable = house_tokugawa
					}
				}
			}
			trigger_if = {
				limit = {
					owner ?= {
						NOR = {
							has_law_or_variant = law_type:law_council_republic
							has_law_or_variant = law_type:law_corporate_state
						}
					}
				}
				custom_tooltip = {
					text = house_yamato_not_tt
					NOT = {
						has_variable = house_yamato
					}
				}
			}
		}
	}

	effect = {
		this.interest_group ?= {
			save_scope_as = relevant_ig
		}
		this = {
			save_scope_as = assassinated_scope
		}
		if = {
			limit = {
				exists = owner.institution:institution_home_affairs
			}
			random_list = {
				40 = {
					modifier = {
						if = {
							limit = {
								owner.institution:institution_home_affairs.investment >= 2
							}
							add = 25
						}
						if = {
							limit = {
								owner.institution:institution_home_affairs.investment >= 3
							}
							add = 25
						}
						if = {
							limit = {
								owner.institution:institution_home_affairs.investment >= 4
							}
							add = 25
						}
						if = {
							limit = {
								owner.institution:institution_home_affairs.investment >= 5
							}
							add = 25
						}
					}
					kill_character = yes
				}
				40 = {
					owner ?= {
						post_notification = arranged_accident_failed
					}
					custom_tooltip = assassination_attempt_failed_tt
				}
				20 = {
					modifier = {
						if = {
							limit = {
								owner.institution:institution_home_affairs.investment >= 2
							}
							subtract = 5
						}
