update_variable_map_value = {
    if = {
        limit = {
            is_key_in_variable_map = {
                name = $name$
                target = $key$
            }
        }
        remove_from_variable_map = {
            name = $name$
            key = $key$
        }
    }
    add_to_variable_map = {
        name = $name$
        key = $key$
        value = $name$
    }
}
