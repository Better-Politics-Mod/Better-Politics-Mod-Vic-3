
bpm_electoral_weight = {
	if = {
		limit = {
			owner = { has_law = law_type:law_census_voting }
		}
		value = literacy_rate
	}
	else_if = {
		limit = {
			owner = { has_law = law_type:law_wealth_voting }
		}
		value = standard_of_living
		subtract = state.modifier:state_standard_of_living_add
		if = {
			limit = {
				standard_of_living <= 20
			}
			value = 0
		}
	}
	else_if = {
		limit = {
			owner = { has_law = law_type:law_universal_suffrage }
		}
		value = 1
	}
	else_if = {
		limit = {
			owner = { has_law = law_type:law_landed_voting }
		}
		value = 0
		if = {
			limit = {
				OR = {
					is_pop_type = clergymen
					is_pop_type = capitalists
				}
			}
			value = 1
		}
		else_if = {
			limit = {
				is_pop_type = aristocrats
			}
			value = 3
		}
	}
	else_if = {
		limit = {
			owner = { 
				has_law = law_type:law_autocracy
				has_law = law_type:law_hereditary_bureaucrats
			}
		}
		if = {
			limit = {
				is_pop_type = aristocrats
			}
			value = 1
		}
	}
	else_if = {
		limit = {
			owner = { 
				has_law = law_type:law_autocracy
			}
		}
		if = {
			limit = {
				is_pop_type = officers
			}
			value = 1
		}

	}
}

bpm_total_electorate = {
	value = 0
	every_scope_pop = {
		add = {
			value = total_size
			multiply = bpm_electoral_weight
		}
	}
}

bpm_issue_interest = {
	if = {
		limit = {
			this = flag:bpm_issue_sol
		}
		scope:country = {
			value = bpm_issue_sol_interest
		}
	}
	else_if = {
		limit = {
			this = flag:bpm_issue_anti_war
		}
		scope:country = {
			value = bpm_issue_anti_war_interest
		}
	}
	else_if = {
		limit = {
			this = flag:bpm_issue_anti_liberal
		}
		scope:country = {
			value = bpm_issue_anti_liberal_interest
		}
	}
}

bpm_momentum_impact = {
	if = {
		limit = {
			has_variable = bpm_party_issue_impact
		}
		value = var:bpm_party_issue_impact
	}
	else = {
		value = 0
	}
}

		limit = {

		}
		if = {
			limit = {
				standard_of_living < state.average_expected_sol
			}
			add = total_size
		}
	}

		value = 0
		every_scope_pop = {
			limit = {
				is_pop_type = peasants
			}
			add = total_size
		}
		min = 1
	}
}

bpm_issue_anti_war_interest = {
	value = 0
	add = {
		value = bpm_issue_anti_war_interest_peasantry
		multiply = bpm_issue_anti_war_interest_peasantry_weight
	}
	add = {
		value = bpm_issue_anti_war_interest_workers
		multiply = bpm_issue_anti_war_interest_workers_weight
	}
	add = {
		value = bpm_issue_anti_war_interest_professionals
		multiply = bpm_issue_anti_war_interest_professionals_weight
	}
	add = {
		value = bpm_issue_anti_war_interest_soldiers
		multiply = bpm_issue_anti_war_interest_soldiers_weight
	}
	if = {
		limit = {
			has_variable = bpm_macro_issue_anti_war
		}
		add = var:bpm_macro_issue_anti_war
	}
	if = {
		limit = {
			is_at_war = no
		}
		multiply = 0
	}
}
