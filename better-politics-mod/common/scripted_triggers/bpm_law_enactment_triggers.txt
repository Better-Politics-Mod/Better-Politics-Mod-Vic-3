bpm_law_enactment_ruler_must_accept = {
	enacting_any_law = yes
	OR = {
		trigger_if = {
			limit = {
				modifier:country_bpm_ruler_complete_veto_enabled = yes
			}
			always = no
		}
		trigger_else_if = {
			limit = {
				modifier:country_bpm_ruler_partial_veto_enabled = yes
			}
			bpm_country_yea_votes > 0.6666
		}
		trigger_else = {
			always = yes
		}
		has_modifier = bpm_enacting_no_confidence
	}
	NAND = {
		bpm_country_yea_votes < 0.66
		bpm_is_changing_constitutional_law = yes
	}
}

bpm_law_enactment_must_accept = {
	enacting_any_law = yes
	trigger_if = {
		limit = { modifier:country_law_enactment_ruler_veto_bypass = yes }
		always = yes 
	}
	trigger_else_if = {
		limit = { 
			OR = {
				has_law = law_type:law_autocracy 
				has_law = law_type:law_colonial_administration
			}
		}
		always = no
	}
	trigger_else_if = {
		limit = {
			OR = {
				has_law = law_type:law_monarchy
				has_law = law_type:law_presidential_republic
			}
		}
		var:bpm_law_enactment_yea_votes_cached > 0.6666
	}
	trigger_else = {
		always = yes
	}
}

bpm_valid_sway_enacting_amendment_type = {
	save_temporary_scope_as = possible_amendment
	scope:ig = {
		any_in_list = {
			variable = bpm_sways
			count = 0
			has_variable = amendment
			var:amendment = scope:possible_amendment
		}
	}
	scope:ig = {would_sponsor_amendment = scope:possible_amendment}
	amendment_banned_from_enactment = no
	NOT = {
		prev = {
			any_scope_amendment = {
				type ?= scope:possible_amendment
			}
		}
	}
}
