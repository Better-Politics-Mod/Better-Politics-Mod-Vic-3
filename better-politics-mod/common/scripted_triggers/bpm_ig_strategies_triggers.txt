	NOT = {
		has_variable = bpm_ig_strategy
	}
}

	ig_approval <= unhappy
}

	is_in_government = false
	bpm_ig_front_clout >= 0.3
}

	ig_approval <= angry
	bpm_ig_front_organizational_clout >= 0.2
}
