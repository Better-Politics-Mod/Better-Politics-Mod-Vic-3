bpm_CAB_toggle_char_select = {
	scope = institution
	is_shown = {
	}
	is_valid = {
	}
	effect = {
        owner = { 
            set_variable = {
                name = bpm_cabinet_char_select
                value = prev
            }
            every_in_list = {
                variable = bpm_CAB_char_select_pool
                bpm_cab_character_calculate_hyp_synergy = yes
            }
            bpm_fix_cabinet_pool = yes
        }
	}
}

bpm_CAB_toggle_char_select_off = {
    scope = institution
    is_shown = {
    }
    is_valid = {
    }
    effect = {
        remove_variable = bpm_cabinet_char_select
    }
}

bpm_CAB_is_antagonistic = {
    scope = character
    is_shown = {
        trigger_if = {
            limit = {
                NOT = { has_variable = bpm_cabinet_minister }
            }
            always = no
        }
        trigger_else_if = {
            limit = {
                var:bpm_cabinet_minister.type = owner.institution:institution_colonial_affairs.type
            }
            bpm_is_antagonistic_colonial_affairs = yes
        }
        trigger_else = {
            always = no
        }
    }
    is_valid = {
    }
    effect = {
    }
}
