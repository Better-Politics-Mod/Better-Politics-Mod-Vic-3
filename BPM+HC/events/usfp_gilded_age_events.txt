usfp_gilded_age.2 = {
	type = country_event
	placement = root

	title = usfp_gilded_age.2.t
	desc = usfp_gilded_age.2.d
	flavor = usfp_gilded_age.2.f
	dlc = dlc_usfp
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		ruler = {
			interest_group = {
				has_party = yes
			}
		}
		in_election_campaign = yes
		NOT = {
			has_variable = had_credit_mobilier
		}
	}

	immediate = {
		set_variable = had_credit_mobilier
		ruler = {
			interest_group = {
				party = {
					save_scope_as = scandal_party
				}
			}
		}
	}
	
	#investigation I guess
	option = {
		name = usfp_gilded_age.2.a
		default_option = yes
		add_treasury = -1000000
		scope:scandal_party = {
			add_momentum = momentum_large_decrease
		}
		add_modifier = {
			name = usfp_scandal_outrage_modifier
			years = 10
			is_decaying = yes
		}
		ai_chance = {
			base = 100
		}
	}
	#lets root out corruption!
	option = {
		name = usfp_gilded_age.2.b
		add_treasury = -1000000
		every_interest_group = {
			limit = {
				has_party = yes
				party = scope:scandal_party
			}
			add_modifier = {
				name = usfp_investigated_scandal_modifier
				years = 5
			}
		}
		add_modifier = {
			name = usfp_scandal_outrage_modifier
			years = 10
			is_decaying = yes
			multiplier = 0.5
		}
		ai_chance = {
			base = 0
		}
	}
	#coverup
	option = {
		name = usfp_gilded_age.2.c
		add_treasury = -1000000
		random_list = {
			30 = {
				scope:scandal_party = {
					add_momentum = momentum_very_large_decrease
				}
				add_modifier = {
					name = usfp_scandal_outrage_modifier
					years = 10
					is_decaying = yes
				}
				add_radicals = {
					value = medium_radicals
					strata = poor
				}
			}
			30 = {
				scope:scandal_party = {
					add_momentum = momentum_large_decrease
				}
				add_modifier = {
					name = usfp_scandal_outrage_modifier
					years = 10
					is_decaying = yes
				}
			}
			30 = {
				scope:scandal_party = {
					add_momentum = momentum_medium_decrease
				}
				add_modifier = {
					name = usfp_scandal_outrage_modifier
					years = 10
					is_decaying = yes
					multiplier = 0.5
				}
			}
		}
		ai_chance = {
			base = 10
		}
	}
	#censor
	option = {
		name = usfp_gilded_age.2.e
		add_treasury = -1000000
		trigger = {
			OR = {
				has_law = law_type:law_censorship
				has_law = law_type:law_outlawed_dissent
			}
		}
		highlighted_option = yes
		scope:scandal_party = {
			add_momentum = momentum_small_decrease
		}
		ai_chance = {
			base = 1000
		}
	}
}
