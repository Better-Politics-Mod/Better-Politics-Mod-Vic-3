ig_fascists = {
	color = hsv{ 0.10 1 0.5 }
	texture = "gfx/interface/icons/ig_icons/fascists.dds"
	layer = "revolution_dynamic_petty_bourgeoisie"
	index = 11
	
	ideologies = {
		ideology_nonexistent

		ideology_fem_benevolent_sexism
		ideology_ban_conservative_supremacist
		# Vanilla
		ideology_reactionary
		ideology_patriotic
	}
	
	enable = {
		has_variable = local_fascist_split_var
		owner = {
			NOT = { is_country_type = decentralized }
		}
	}
	
	on_enable = {
		ig:ig_fascists = {
			# set_ig_trait = ig_trait:ig_trait_right_systematic_corruption
			set_ig_trait = ig_trait:ig_trait_right_bastion_of_freedom
			set_ig_trait = ig_trait:ig_trait_right_election_interference
		}
		if = {
			limit = { 
				ig:ig_fascists = { 
					has_ideology = ideology:ideology_nonexistent
				}
			}
			ig:ig_fascists = {
				remove_ideology = ideology_nonexistent
				if = {
					limit = {
						ROOT = { 
							bpm_country_is_republic = yes
							country_has_primary_culture = cu:brazilian
						}		
					}
					add_ideology = ideology_gov_liberal_republican
				}
				else_if = {
					limit = {
						ROOT = { country_has_primary_culture = cu:brazilian }
					}
					add_ideology = ideology_gov_radical_monarchist
				}
				if = {
					limit = {
						ROOT = { country_has_primary_culture = cu:brazilian	}	
					}
					add_ideology = ideology_dop_organic_power
				}
				else_if = {
					limit = {
						ROOT = c:CLM
					}
					add_ideology = ideology_dop_authoritarian_democrat
				}
				else = { add_ideology = ideology_dop_party_state }
				if = {
					limit = {
						OR = {
							ROOT = { country_has_primary_culture = cu:brazilian	}	
							ROOT = c:CLM
						}
					}
					add_ideology = ideology_sta_organic_state
					add_ideology = ideology_cit_multiethnic_nationalist
				}
				else = { 
					add_ideology = ideology_sta_security_state
					add_ideology = ideology_cit_ethnonationalist
				}
				add_ideology = ideology_jingoist
				add_ideology = ideology_eco_autarky
				if = {
					limit = {
						ROOT = { country_has_primary_culture = cu:brazilian	}		
					}
					add_ideology = ideology_rel_moralist
					add_ideology = ideology_wel_distributist_welfare
				}
				else = { 
					add_ideology = ideology_wel_paternalistic_welfare
				}
	
				every_scope_character = {
					limit = {
						has_ideology = ideology:ideology_placeholder_leader
					}
					set_ideology = ideology:ideology_center_leader
				}
				bpm_reorder_ideologies = yes
			}
			ig:ig_reactionaries = {
				if = {
					limit = {
						ROOT = { country_has_state_religion = rel:catholic }
					}
					bpm_remove_all_dop_ideologies = yes
					add_ideology = ideology_dop_organic_power
					bpm_remove_all_sta_ideologies = yes
					add_ideology = ideology_sta_organic_state
					bpm_remove_all_cit_ideologies = yes
					add_ideology = ideology_cit_multiethnic_nationalist
					bpm_remove_all_wel_ideologies = yes
					add_ideology = ideology_wel_distributist_welfare
					bpm_reorder_ideologies = yes
				}
			}
			bpm_rename_igs = yes
		}
		if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:brazilian
					c:CLM ?= this
				}		
			}
			ig:ig_fascists = {
				set_interest_group_name = ig_integralists
			}
		}
	}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
		custom_tooltip = {
			text = BPM_VALID_POP_FOR_IG
			owner = {
				NOT = { is_country_type = decentralized }
			}
		}
	}
	
	pop_weight = {
		value = 0
	
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 100
			}
		}	

		add = {
			desc = "MIDDLE_CLASS"
			if = {
				limit = { 
					strata = middle
				}
				value = 50
			}
		}
		add = {
			desc = "URBAN_POP"
			if = {
				limit = { 
					bpm_pop_employment_is_rural = no
				}
				value = 50
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.01 # Fascists are heavily dependent on their leader
					add = 1
					max = 2
					min = 0.1
				}
			}
		}	

		multiply = bpm_ig_agitators_popularity
		multiply = bpm_ig_ruler_popularity
		multiply = ig_attraction_to_ruler_ideology_from_stability

		multiply = {
			desc = "UNEDUCATED"
			value = literacy_rate
			multiply = 1.5
			max = 1.0
		}

		multiply = {
			desc = "NO_POLITICAL_AGITATION"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = political_agitation }
					}
				}
				value = 0.5
			}		
		}

		multiply = ig_attraction_from_radicals_and_loyalists	
	}
	
	monarch_weight = {
		value = 0.2
	}	

	agitator_weight = {
		value = 1.0
		multiply = bpm_base_agitator_weight
	}
	
	commander_weight = {
		value = 2.5
		multiply = bpm_commander_assignment_from_laws_factor
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 1.0
		multiply = bpm_female_commander_spawn_chance
	}
	
	female_politician_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance
	}	

	female_agitator_chance = {
		value = 1.0
		multiply = bpm_female_character_spawn_chance		
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
