je_addressing_grunderkrach = {
    icon = "gfx/interface/icons/event_icons/event_trade.dds"

    group = je_group_historical_content

    immediate = {
        ig:ig_landowners.leader.ideology = { save_scope_as = landowners_ideology_scope }
        ig:ig_landowners.leader = { set_ideology = ideology:ideology_protectionist }

        ig:ig_industrialists.leader.ideology = { save_scope_as = industrialists_ideology_scope }
        ig:ig_industrialists.leader = { set_ideology = ideology:ideology_protectionist }

        ig:ig_national_liberals.leader.ideology = { save_scope_as = national_liberals_ideology_scope }
        ig:ig_national_liberals.leader = { set_ideology = ideology:ideology_protectionist }

        set_variable = {
            name = bpm_schutzzoll_progress_var
            value = 0
        }
        trigger_event = { id = schutzzoll.1 popup = yes}
    }

    current_value = {
		value = root.var:bpm_schutzzoll_progress_var
	}

	goal_add_value = {
		value = 36
	}

    should_be_pinned_by_default = yes
    
    progressbar = yes

    complete = {
        custom_tooltip = {
			text = bpm_schutzzoll_timer_tt
			var:bpm_schutzzoll_progress_var >= 36
		}
        has_law_or_variant = law_type:law_protectionism
        capital = {
            custom_tooltip = {
                text = bpm_schutzzoll_grain
                    sg:grain = {
				    or = {
					    import_tariff_level = low_tariffs
					    import_tariff_level = high_tariffs
                        import_tariff_level = max_tariffs
				    }
			    }
            }
           custom_tooltip = {
                text = bpm_schutzzoll_meat
                    sg:meat = {
				    or = {
					    import_tariff_level = low_tariffs
					    import_tariff_level = high_tariffs
                        import_tariff_level = max_tariffs
				    }
			    }
            }
            custom_tooltip = {
                text = bpm_schutzzoll_wood
                    sg:wood = {
				    or = {
					    import_tariff_level = low_tariffs
					    import_tariff_level = high_tariffs
                        import_tariff_level = max_tariffs
				    }
			    }
            }
            custom_tooltip = {
                text = bpm_schutzzoll_iron
                    sg:iron = {
				    or = {
					    import_tariff_level = low_tariffs
					    import_tariff_level = high_tariffs
                        import_tariff_level = max_tariffs
				    }
			    }
            }
            custom_tooltip = {
                text = bpm_schutzzoll_tea
                    sg:tea = {
				    or = {
					    import_tariff_level = low_tariffs
					    import_tariff_level = high_tariffs
                        import_tariff_level = max_tariffs
				    }
			    }
            }
            custom_tooltip = {
                text = bpm_schutzzoll_coffee
                    sg:coffee = {
				    or = {
					    import_tariff_level = low_tariffs
					    import_tariff_level = high_tariffs
                        import_tariff_level = max_tariffs
				    }
			    }
            }
            custom_tooltip = {
                text = bpm_schutzzoll_tobacco
                    sg:tobacco = {
				    or = {
					    import_tariff_level = low_tariffs
					    import_tariff_level = high_tariffs
                        import_tariff_level = max_tariffs
				    }
			    }
            }
		}
    }

    on_monthly_pulse = {
        effect = {
            if = {
                limit = {
                    has_law_or_variant = law_type:law_protectionism
                    capital = {
                        sg:grain = {
				            or = {
				    	        import_tariff_level = low_tariffs
			    		        import_tariff_level = high_tariffs
                                import_tariff_level = max_tariffs
	    			        }
    			        }
                        sg:meat = {
				            or = {
				    	        import_tariff_level = low_tariffs
			    		        import_tariff_level = high_tariffs
                                import_tariff_level = max_tariffs
	    			        }
    			        }
                        sg:wood = {
				            or = {
				    	        import_tariff_level = low_tariffs
    		    			    import_tariff_level = high_tariffs
                                import_tariff_level = max_tariffs
	    	    		    }
    			        }

	    			        or = {
		    	    		    import_tariff_level = low_tariffs
			        		    import_tariff_level = high_tariffs
                                import_tariff_level = max_tariffs
	    			        }

			    	    	    import_tariff_level = low_tariffs
				        	    import_tariff_level = high_tariffs
                                import_tariff_level = max_tariffs

					            import_tariff_level = high_tariffs

    				        }
	    		        }

    					        import_tariff_level = high_tariffs
                                import_tariff_level = max_tariffs
		    		        }
			            }
                    }
                }
                change_variable = { name = bpm_schutzzoll_progress_var add = 1 }
            }
            else = {
                set_variable = { name = bpm_schutzzoll_progress_var value = 0 }
            }
        }
    }

    on_complete = {
        #trigger_event = { id = }
        ig:ig_landowners.leader = { set_ideology = scope:landowners_ideology_scope }
        ig:ig_industrialists.leader = { set_ideology = scope:industrialists_ideology_scope }

            remove_variable = bpm_schutzzoll_progress_var
