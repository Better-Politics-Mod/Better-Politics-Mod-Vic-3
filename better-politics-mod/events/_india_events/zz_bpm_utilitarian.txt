namespace = utilitarian

# Good Governance For All, on JE completion
utilitarian.1 = {
	type = country_event
	placement = root
	title = utilitarian.1.t
	desc = utilitarian.1.d
	flavor = {
		first_valid = {
			triggered_desc = {
				desc = utilitarian.1.f # Loc if a Utilitarian character exists
				trigger = {
					any_scope_character = {
						has_ideology = ideology:ideology_utilitarian_leader	
					}
				}
			}
			triggered_desc = { 
				desc = utilitarian.1.f2 # Loc if no Utilitarian character exists
				trigger = {
					always = yes
				}
			}
		}
	}

	dlc = dlc011

	event_image = {
		video = "ip2_india_factory_floor"
	}

	dlc = dlc011

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	cooldown = { months = normal_modifier_time }

	trigger = {
		# Triggered by Imperialism of Promise JE
	}

	immediate = {
		random_interest_group = {
			limit = {
				any_scope_character = {
					has_ideology = ideology:ideology_utilitarian_leader	
				}
			}
			save_scope_as = relevant_ig
			random_scope_character = {
				limit = {
					has_ideology = ideology:ideology_utilitarian_leader	
				}
				save_scope_as = relevant_character
			}
		}
	}

	cancellation_trigger = {
		OR = {
			NOT = {
				exists = c:BIC
			}
			NOT = {
				country_has_primary_culture = cu:british
			}
		}
	}

	option = {
		name = utilitarian.1.a
		default_option = yes
		add_modifier = {
			name = utilitarian_bonus
			months = long_modifier_time
		}
		ig:ig_industrialists = {
			if = {
				limit = {
					OR = {
						has_ideology = ideology:ideology_malthusian
						has_ideology = ideology:ideology_wel_market_welfare
					}
				}
				remove_ideology = ideology_malthusian
				add_ideology = ideology_utilitarian_millian
				bpm_remove_all_wel_ideologies = yes
				add_ideology = ideology_wel_public_welfare
				bpm_reorder_ideologies = yes
			}
		}
		every_interest_group = {
			limit = {
				any_scope_character = {
					OR = {
						has_role = agitator
						has_role = politician
					}
					has_ideology = ideology:ideology_utilitarian_leader	
				}
			}
			add_modifier = {
				name = util_colonial_admin_success
				months = normal_modifier_time
			}
		}
	}
}

# A Project Without Progress, on JE time out
utilitarian.3 = {
	type = country_event
	placement = root
	title = utilitarian.3.t
	desc = utilitarian.3.d
	flavor = {
		first_valid = {
			triggered_desc = {
				desc = utilitarian.3.f # Loc if a Utilitarian character exists
				trigger = {
					any_scope_character = {
						has_ideology = ideology:ideology_utilitarian_leader	
					}
				}
			}
			triggered_desc = { 
				desc = utilitarian.3.f2 # Loc if no Utilitarian character exists
				trigger = {
					always = yes
				}
			}
		}
	}

	event_image = {
		video = "asia_politician_parliament_motion"
	}

	dlc = dlc011

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	cooldown = { months = normal_modifier_time }

	trigger = {
		# Triggered by Imperialism of Promise JE
	}

	immediate = {
		random_interest_group = {
			limit = {
				any_scope_character = {
					has_ideology = ideology:ideology_utilitarian_leader	
				}
			}
			save_scope_as = relevant_ig
			random_scope_character = {
				limit = {
					has_ideology = ideology:ideology_utilitarian_leader	
				}
				save_scope_as = relevant_character
			}
		}
		set_variable = {
			name = util_end_var
			value = 0
		}
		if = {
			limit = {
				root.average_sol >= 10
			}
			change_variable = {
				name = util_end_var
				add = 1
			}
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_proportional_taxation
					has_law = law_type:law_graduated_taxation
				}
			}
			change_variable = {
				name = util_end_var
				add = 1
			}
		}
		if = {
			limit = {
				has_law = law_type:law_compulsory_primary_school
			}
			change_variable = {
				name = util_end_var
				add = 1
			}
		}
		if = {
			limit = {
				has_law = law_type:law_public_schools
			}
			change_variable = {
				name = util_end_var
				add = 1
			}
		}
		if = {
			limit = {
				NOT = {
					has_law = law_type:law_no_womens_rights
				}
			}
			change_variable = {
				name = util_end_var
				add = 1
			}
		}
	}

	cancellation_trigger = {
		OR = {
			NOT = {
				exists = c:BIC
			}
			NOT = {
				country_has_primary_culture = cu:british
			}
		}
	}

	option = {
		name = utilitarian.3.a
		default_option = yes
		add_modifier = {
			name = utilitarian_partial_malus
			months = long_modifier_time
		}
		if = {
			limit = {
				var:util_end_var >= 1
			}
			every_interest_group = {
				limit = {
					any_scope_character = {
						OR = {
							has_role = agitator
							has_role = politician
						}
						has_ideology = ideology:ideology_utilitarian_leader	
					}
				}
				add_modifier = {
					name = util_colonial_admin_partial_success
					months = normal_modifier_time
				}
			}
		}
		if = {
			limit = {
				var:util_end_var > 2
			}
			ig:ig_industrialists = {
				if = {
					limit = {
						OR = {
							has_ideology = ideology:ideology_malthusian
							has_ideology = ideology:ideology_wel_market_welfare
						}
					}
					remove_ideology = ideology_malthusian
					add_ideology = ideology_utilitarian_millian
					bpm_remove_all_wel_ideologies = yes
					add_ideology = ideology_wel_public_welfare
					bpm_reorder_ideologies = yes
				}
			}
		}
	}
}
