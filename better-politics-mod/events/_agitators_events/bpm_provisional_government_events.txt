bpm_provisional_government_events.2 = {
	type = country_event
	placement = root
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = bpm_provisional_government_events.2.t
	desc = bpm_provisional_government_events.2.d
	flavor = bpm_provisional_government_events.2.f

	duration = 3

	trigger = {
		#Triggered by JE
	}

	immediate = {
		random_interest_group = {
			limit = {
				has_variable = bpm_law_sway_promise_proposed
			}
			save_scope_as = bpm_government_promise_ig
			ordered_preferred_law = {
				limit = {
					bpm_ig_law_can_be_promised = yes
				}
				min = 0
				max = 2
				check_range_bounds = no
				save_scope_as = bpm_government_promise_law_one
			}

			ordered_preferred_law = {
				limit = {
					bpm_ig_law_can_be_promised = yes
					NOT = { scope:bpm_government_promise_law_one ?= THIS }
				}
				min = 1
				max = 3
				check_range_bounds = no
				save_scope_as = bpm_government_promise_law_two
			}

        }
		currently_enacting_law = {
			save_scope_as = bpm_currently_enacting_law
		}
	}

	cancellation_trigger = {
		OR = {
			NOT = { exists = scope:bpm_government_promise_ig }
			NOT = { exists = scope:bpm_government_promise_law_one }
		}
	}

	option = {
		name = bpm_provisional_government_events.2.a
		trigger = {
			exists = scope:bpm_government_promise_law_one
		}
		set_variable = { name = bpm_promise_desired_law value = scope:bpm_government_promise_law_one }
		custom_tooltip = bpm_government_promise_law_tt
		scope:bpm_government_promise_ig = {
			set_variable = bpm_law_sway_promise_made
			bpm_subtract_from_law_sways = yes

		}
		add_journal_entry = {
			type = je_bpm_government_promise_law
		}
		hidden_effect = { bpm_update_law_enactment_variables_modifiers = yes }
	}

	option = {
		name = bpm_provisional_government_events.2.b
		trigger = {
			exists = scope:bpm_government_promise_law_two
		}
		set_variable = { name = bpm_promise_desired_law value = scope:bpm_government_promise_law_two }
		custom_tooltip = bpm_government_promise_law_tt_two
		scope:bpm_government_promise_ig = {
			set_variable = bpm_law_sway_promise_made
			bpm_subtract_from_law_sways = yes
		}
		add_journal_entry = {
			type = je_bpm_government_promise_law
		}
		hidden_effect = { bpm_update_law_enactment_variables_modifiers = yes }
	}

	option = {
		name = bpm_provisional_government_events.2.c
		default_option = yes
		scope:bpm_government_promise_ig = { 
			remove_variable = bpm_law_sway_promise_proposed
		}
		custom_tooltip = bpm_government_promise_law_tt_cancel
		set_variable = bpm_cannot_promise_a_law
		set_variable = {
			name = bpm_cpal_removal_flag
			value = 0
		}
		hidden_effect = { bpm_update_law_enactment_variables_modifiers = yes }
	}
	

}

			limit = {has_variable = bpm_cabinet_promised_ig}

		}
	}
	

}


bpm_provisional_government_events.4 = {
	type = country_event
	placement = root
	event_image = {
		video = "votp_monarch_holding_court"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	title = bpm_provisional_government_events.4.t
	desc = bpm_provisional_government_events.4.d
	#flavor = bpm_provisional_government_events.4.f

	duration = 3

	trigger = {
	}

	after = {
		scope:ig = {
			remove_variable = bpm_negotiation_selected_ig
		}
		remove_variable = bpm_negotiating_amendment
		remove_variable = bpm_temp_amendment1
		remove_variable = bpm_temp_amendment2
		remove_variable = bpm_temp_amendment3
	}

	immediate = {
		random_interest_group = {
			limit = {
				has_variable = bpm_negotiation_selected_ig
			}
			save_scope_as = ig
		}
		currently_enacting_law = {
			save_scope_as = law
		}
		scope:law = {
			random_possible_amendment_type = {
				limit = {
					bpm_amendment_banned_from_enactment = no
					scope:ig = { would_sponsor_amendment = prev }
				}
				save_scope_as = amendment_type1
			}
			random_possible_amendment_type = {
				limit = {
					bpm_amendment_banned_from_enactment = no
					amendment_is_generic = no
					scope:ig = { would_sponsor_amendment = prev }
					NOT = { 
						this = scope:amendment_type1
					}
				}
				save_scope_as = amendment_type2
			}
			random_possible_amendment_type = {
				limit = {
					bpm_amendment_banned_from_enactment = no
					amendment_is_generic = no
					scope:ig = { would_sponsor_amendment = prev }
					NOT = { 
						this = scope:amendment_type1
					}
					NOT = { 
						this = scope:amendment_type2
					}
				}
				save_scope_as = amendment_type3
			}
		}
		set_variable = {
			name = bpm_temp_amendment1
			value = scope:amendment_type1
		}
		set_variable = {
			name = bpm_temp_amendment2
			value = scope:amendment_type2
		}
		set_variable = {
			name = bpm_temp_amendment3
			value = scope:amendment_type3
		}
	}

	option = {
		name = bpm_provisional_government_events.4.a
		trigger = {
			exists = scope:amendment_type1
		}
		scope:ig = {
			set_variable = bpm_negotiated_amendment_sway_used
			bpm_subtract_from_law_sways = yes
		}
		set_variable = { name = bpm_negotiate_amendment_cd days = 90 }
		scope:law = {
			add_amendment = {
				type = scope:amendment_type1
				sponsor = scope:ig
				cooldown = 120
			}
		}
	}
	option = {
		name = bpm_provisional_government_events.4.b
		trigger = {
			exists = scope:amendment_type2
		}
		scope:ig = {
			set_variable = bpm_negotiated_amendment_sway_used
			bpm_subtract_from_law_sways = yes
		}
		set_variable = { name = bpm_negotiate_amendment_cd days = 90 }
		scope:law = {
			add_amendment = {
				type = scope:amendment_type2
				sponsor = scope:ig
				cooldown = 120
			}
		}
	}
		option = {
		name = bpm_provisional_government_events.4.c
		trigger = {
			exists = scope:amendment_type3
		}
		scope:ig = {
			set_variable = bpm_negotiated_amendment_sway_used
			bpm_subtract_from_law_sways = yes
		}
		set_variable = { name = bpm_negotiate_amendment_cd days = 90 }
		scope:law = {
			add_amendment = {
				type = scope:amendment_type3
				sponsor = scope:ig
				cooldown = 120
			}
		}
	}

	option = {
		name = bpm_provisional_government_events.4.cancel
		default_option = yes
		# cancel
	}
}
