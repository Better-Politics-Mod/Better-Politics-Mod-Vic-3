# Formalize the German Empire
bpm_frankfurt_parliament_events.12 = {
	type = country_event
	placement = root
	title = bpm_frankfurt_parliament_events.12.t
	flavor = bpm_frankfurt_parliament_events.12.f
	desc = bpm_frankfurt_parliament_events.12.d
	
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}
	duration = 3

	trigger = {
		# event
	}

	immediate = {

	}

	# Kleindeutschland
	option = {
		name = bpm_frankfurt_parliament_events.12.a
		default_option = yes

		change_tag = GER

		# Enacts selected laws to the constitution
		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_speech_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_centralization_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_voting_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_hog_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = law_type:law_separated_powers
			level = constitutional
		}

		# Transfers leaders of minor states to the German Empire, integrates non-Austrian states
		every_character = {
			limit = {
				owner = {
					country_is_in_german_confederation = yes
					NOT = { has_variable = bpm_frankfurt_monarch_var }
					NOR = {
						c:AUS ?= this
						is_subject_of = c:AUS
					}
				}
				has_role_of_type = ruler
			}
			transfer_character = c:GER
			remove_character_role = politician
			add_character_role = magnate
		}

		every_country = {
			limit = {
				country_is_in_german_confederation = yes
				NOT = { has_variable = bpm_frankfurt_monarch_var }
				NOR = {
					c:AUS ?= this
					is_subject_of = c:AUS
				}
			}

			c:GER ?= {
				annex_with_incorporation = prev
			}
		}
	}

	# Großdeutschland
	option = {
		name = bpm_frankfurt_parliament_events.12.b

	}
}

			}
			add_target_backers = { ROOT }
		}
		if = {
			limit = { c:PRU ?= THIS }
			random_diplomatic_play = {
				limit = {
					target = {
						scope:frankfurt_opposer ?= this
					}
				}

					holder = scope:frankfurt_crown_candidate 
					type = humiliation
					target_country = root

				}
			}
		}
	}
}
