	type = institution
	random_valid = no
	
	text = {
		trigger = {
			type = owner.institution:institution_schools.type
		}
		localization_key = bpm_is_institution_schools
	}

    text = {
		trigger = {
			type = owner.institution:institution_centralization.type
		}
		localization_key = bpm_is_institution_centralization
	}

    text = {
		trigger = {
			type = owner.institution:institution_police.type
		}
		localization_key = bpm_is_institution_police
	}

    text = {
        trigger = {
            always = no
        }
        fallback = yes
        localization_key = bpm_is_institution_unknown
    }
}
