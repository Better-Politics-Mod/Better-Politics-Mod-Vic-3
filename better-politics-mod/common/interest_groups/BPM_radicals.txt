ig_radicals = {
	color = hsv{ 0.11 1 1 }
	texture = "gfx/interface/icons/ig_icons/radicals.dds"
	
	ideologies = {
		ideology_republican
		ideology_radical_liberal
		ideology_egalitarian
		ideology_anti_slavery
	}
	
	enable = {
		always = yes
	}
	
	traits = {
		ig_trait_left_systematic_corruption
		ig_trait_left_bastion_of_freedom
		ig_trait_left_systematic_opposition
	}	
	
	on_enable = {}
	on_disable = {}
	on_character_ig_membership = {}
	
	pop_potential = {
	}
	
	pop_weight = {
		value = 20
		add = {
			desc = "POP_EXCLUSIONS"
			if = {
				limit = { 
					is_pop_type = peasants
				}
				value = -10
			}
		}
		add = {
			desc = "POP_CAPITALISTS"
			if = {
				limit = { 
					is_pop_type = capitalists
				}
				value = 100
			}
		}	

		add = {
			desc = "MIDDLE_CLASS"
			if = {
				limit = { 
					strata = middle
				}
				value = 50
			}
		}
		add = {
			desc = "URBAN_POP"
			if = {
				limit = { 
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 40
			}
		}
		
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader = {
					value = popularity
					multiply = 0.005
					add = 1
					max = 1.5
					min = 0.67
				}
			}
		}	

		multiply = {
			desc = "LIBERALS_SUPPRESSED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							NOT = { is_interest_group_type = ig_radicals }
							OR = {
								has_ideology = ideology_liberal
								has_ideology = ideology_anti_slavery
								leader = {
									OR = {
										has_ideology = ideology_radical
										has_ideology = ideology_reformer
										has_ideology = ideology_abolitionist
										has_ideology = ideology_feminist
									}
								}
							}
							is_being_suppressed = yes
						}
					}
				}
				value = 1.1
			}		
		}	
		multiply = {
			desc = "CONSERVATIVES_BOLSTERED"	
			value = 1.0

			if = {
				limit = { 
					owner = {
						any_interest_group = {
							OR = {
								has_ideology = ideology_paternalistic
								has_ideology = ideology_reactionary
								has_ideology = ideology_pro_slavery
								has_ideology = ideology_stratocratic
							}
							is_being_bolstered = yes
						}
					}
				}
				value = 1.1
			}		
		}	

		multiply = {
			desc = "UNEDUCATED"
			value = 1.0

			if = {
				limit = {
					literacy_rate <= 0.67
				}
				value = 0.5
			}
		}
		multiply = {
			desc = "VERY_UNEDUCATED"
			value = 1.0

			if = {
				limit = {
					literacy_rate <= 0.33
				}
				value = 0.5
			}
		}

		multiply = {
			desc = "NO_EGALITARIANISM"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = democracy }
					}
				}
				value = 0.5
			}		
		}
		multiply = {
			desc = "NO_DEMOCRACY"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						NOT = { has_technology_researched = democracy }
					}
				}
				value = 0.0
			}		
		}

		multiply = {
			desc = "decentralized_power"	
			value = 1.0
			
			if = {
				limit = { 
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}		
		}		
	}
	
	monarch_weight = {
		value = 0.5
	}	
	
	commander_weight = {
		value = 0.5
	}	
	
	noble_chance = {
		value = 0.3
	}
	
	female_commander_chance = {	
		value = 0.0
	}
	
	female_politician_chance = {
		value = 0.1

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}
				
			}
			add = {
				value = 0.2
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}
				
			}
			add = {
				value = 0.1
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
				
			}
			add = {
				value = 0.1
			}
		}
	}
	
	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
