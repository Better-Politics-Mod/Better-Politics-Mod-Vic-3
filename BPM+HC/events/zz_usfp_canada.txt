namespace = can_aus
# Canada complete event for Canada
can_aus.5 = {
	type = country_event
	placement = ROOT

	title = can_aus.5.t
	desc = can_aus.5.d
	flavor = can_aus.5.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		top_overlord = {
			save_scope_as = canada_overlord_country
		}
	}

	option = {
		name = can_aus.5.a
		add_modifier = {
			name = can_unified_canada
			months = normal_modifier_time
		}
		if = {
			limit = {
				NOT = { is_subject_type = subject_type_dominion }
			}
			change_subject_type = subject_type_dominion
		}
		if = {
			limit = { 
				any_scope_state = {
					state_region = s:STATE_BRITISH_COLUMBIA
				}
			}
			s:STATE_BRITISH_COLUMBIA = {
				if = {
					limit = {
						exists = region_state:SYX
					}
					region_state:SYX = {
						set_state_owner = root
					}
				}
				if = {
					limit = {
						exists = region_state:SAL
					}
					region_state:SAL = {
						set_state_owner = root
					}
				}
				if = {
					limit = {
						exists = region_state:LLT
					}
					region_state:LLT = {
						set_state_owner = root
					}
				}
				if = {
					limit = {
						exists = region_state:SWP
					}
					region_state:SWP = {
						set_state_owner = root
					}
				}
			}
		}
		if = {
			limit = {
				any_scope_state = {
					state_region = s:STATE_ONTARIO
				}
			 }
			 s:STATE_ONTARIO = {
				if = {
					limit = {
						exists = region_state:OJI
					 }
					 region_state:OJI = {
						set_state_owner = root
					 }
				 }
			 }

		}
	}
}
