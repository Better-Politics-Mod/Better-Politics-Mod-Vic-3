        owner = { 
            set_variable = {
                name = bpm_cabinet_char_select
                value = prev
            }
            every_in_list = {
                variable = bpm_CAB_char_select_pool
                bpm_cab_character_calculate_hyp_synergy = yes

	}

bpm_cabinet_shuffle_gui = {
    scope = country
    is_shown = {
    }
    is_valid = {
        NOT = { has_variable = bpm_cab_cannot_shuffle }
    }
    effect = {
        set_variable = {
            name = bpm_cab_cannot_shuffle
            days = 730
        }
        bpm_cab_shuffle = yes
    }
}
