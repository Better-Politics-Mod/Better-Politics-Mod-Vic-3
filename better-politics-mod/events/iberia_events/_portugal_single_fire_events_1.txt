portugal_single_fire_events.2 = { # Asylums of [state]
	type = country_event
	placement = scope:azores_state_scope

	duration = 3

	dlc = dlc016

	title = portugal_single_fire_events.2.t
	desc = portugal_single_fire_events.2.d
	flavor = portugal_single_fire_events.2.f

	event_image = {
		video = "southamerica_christianity"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	immediate = {
		p:x90a080.state = {
			save_scope_as = azores_state_scope
		}
		ig:ig_intelligentsia ?= {
			save_scope_as = intelligentsia_scope
		}
		ig:ig_devout = {
			save_scope_as = devout_scope
		}
	}

	trigger = {
		c:POR ?= this
		p:x90a080.state = {
			owner = ROOT
		}
		NOT = {
			has_variable = asylums_of_azores_happened
		}
	}

	cancellation_trigger = {
		p:x90a080.state = {
			NOT = {
				owner = ROOT
			}
		}
	}

	option = {
		name = portugal_single_fire_events.2.a
		default_option = yes
		add_modifier = {
			name = portugal_single_fire_events_azores_support_mod
			days = normal_modifier_time
		}
		scope:azores_state_scope = {
			add_modifier = {
				name = portugal_single_fire_events_azores_state_devout_mod
				days = normal_modifier_time
			}
		}
	}
	option = {
		name = portugal_single_fire_events.2.b
		add_modifier = {
			name = portugal_single_fire_events_azores_not_support_mod
			days = normal_modifier_time
		}
		scope:azores_state_scope = {
			add_modifier = {
				name = portugal_single_fire_events_azores_state_intelligentsia_mod
				days = normal_modifier_time
			}
		}
	}
	option = {
		name = portugal_single_fire_events.2.c
		add_modifier = {
			name = portugal_single_fire_events_azores_ignore_mod
			days = normal_modifier_time
		}
	}

	after = {
		set_variable = asylums_of_azores_happened
	}
}
