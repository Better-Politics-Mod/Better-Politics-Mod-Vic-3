namespace = hog_regency

# Start of the regency
hog_regency.1 = {
	type = country_event
	placement = root

	duration = 3

    cooldown = { days = 1}
	title = hog_regency.1.t
	desc = hog_regency.1.d
	flavor = hog_regency.1.f

    event_image = {
		video = "votp_monarch_holding_court"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	### No cooldown

	trigger = {
		has_law_or_variant = law_type:law_answers_to_legislature
        has_law = law_type:law_advisory_body
        NOT = {has_modifier = modifier_hog_regency}
	}

	cancellation_trigger = {
		OR = {
			NOT = {
		        has_law_or_variant = law_type:law_answers_to_legislature
                has_law = law_type:law_advisory_body
			}
			scope:hog_scope ?= {
				is_adult = yes
			}
            has_modifier = modifier_hog_regency
		}
	}

	immediate = {
		ordered_interest_group = { # Just the most notable IG in government
			limit = {
				is_in_government = yes
			}
			order_by = ig_clout
			position = 0
			check_range_bounds = no
			save_scope_as = ig_candidate_1
		}
		ordered_interest_group = {
			limit = {
				NOT = {
					scope:ig_candidate_1 ?= this
				}
				law_stance = {
					law = law_type:law_answers_to_legislature
					value >= approve
				}
				ig_counts_as_marginal = no
			}
			order_by = ig_clout
			position = 0
			check_range_bounds = no
			save_scope_as = ig_candidate_2
		}
		ordered_interest_group = {
			limit = {
				NOR = {
					scope:ig_candidate_1 ?= this
					scope:ig_candidate_2 ?= this
				}
				law_stance = {
					law = law_type:law_answers_to_legislature
					value >= neutral
				}
				ig_counts_as_marginal = no
				is_insurrectionary = no
			}
			order_by = ig_clout
			position = 0
			check_range_bounds = no
			save_scope_as = ig_candidate_3
		}
		ordered_interest_group = {
			limit = {
				NOR = {
					scope:ig_candidate_1 ?= this
					scope:ig_candidate_2 ?= this
					scope:ig_candidate_3 ?= this
				}
				law_stance = {
					law = law_type:law_answers_to_legislature
					value >= neutral
				}
				ig_counts_as_marginal = no
				is_insurrectionary = no
			}
			order_by = this.leader.popularity
			position = 0
			check_range_bounds = no
			save_scope_as = ig_candidate_4
		}
		var:bpm_headofgov ?= {
			save_scope_as = hog_scope
		}
		set_variable = hog_regency1_active
	}

	option = {
		name = hog_regency.1.a
		default_option = yes
		add_modifier = modifier_hog_regency
        custom_tooltip = {
            text = hog_regency_c1
		    scope:ig_candidate_1.leader = {
		    	bpm_appoint_as_hog = yes
		    	hidden_effect = {
		    		if = {
		    			limit = {
		    				is_immortal = no
		    			}
		    			set_character_immortal = yes
		    		}
		    	}
		    }
		    scope:hog_scope = {
                owner = {
		    		set_variable = {
		    			name = bpm_hog_heir
		    			value = prev
		    		}
		    	}
		    	add_character_role = bpm_character_role_protege
                remove_character_role = bpm_character_role_hog
            }
            every_scope_character = {
                limit = {
                    NOT = {scope:hog_scope = this}
                    has_role = bpm_character_role_protege
                }
                remove_character_role = bpm_character_role_protege
            }
        }
	}

	option = {
		name = hog_regency.1.b
		trigger = {
			exists = scope:ig_candidate_2
		}
		add_modifier = modifier_hog_regency
        custom_tooltip = {
            text = hog_regency_c2
		    scope:ig_candidate_2.leader = {
		    	bpm_appoint_as_hog = yes
		    	hidden_effect = {
		    		if = {
		    			limit = {
		    				is_immortal = no
		    			}
		    			set_character_immortal = yes
		    		}
		    	}
		    }
		    scope:hog_scope = {
                owner = {
		    		set_variable = {
		    			name = bpm_hog_heir
		    			value = prev
		    		}
		    	}
		    	add_character_role = bpm_character_role_protege
                remove_character_role = bpm_character_role_hog
            }
            every_scope_character = {
                limit = {
                    NOT = {scope:hog_scope = this}
                    has_role = bpm_character_role_protege
                }
                remove_character_role = bpm_character_role_protege
            }
        }
	}

	option = {
		name = hog_regency.1.c
		trigger = {
			exists = scope:ig_candidate_3
		}
		add_modifier = modifier_hog_regency
        custom_tooltip = {
            text = hog_regency_c3
		    scope:ig_candidate_3.leader = {
		    	bpm_appoint_as_hog = yes
		    	hidden_effect = {
		    		if = {
		    			limit = {
		    				is_immortal = no
		    			}
		    			set_character_immortal = yes
		    		}
		    	}
		    }
		    scope:hog_scope = {
                owner = {
		    		set_variable = {
		    			name = bpm_hog_heir
		    			value = prev
		    		}
		    	}
		    	add_character_role = bpm_character_role_protege
                remove_character_role = bpm_character_role_hog
            }
            every_scope_character = {
                limit = {
                    NOT = {scope:hog_scope = this}
                    has_role = bpm_character_role_protege
                }
                remove_character_role = bpm_character_role_protege
            }
        }
	}

	option = {
		name = hog_regency.1.e
		trigger = {
			exists = scope:ig_candidate_4
		}
		add_modifier = modifier_hog_regency
        custom_tooltip = {
            text = hog_regency_c4
		    scope:ig_candidate_4.leader = {
		    	bpm_appoint_as_hog = yes
		    	hidden_effect = {
		    		if = {
		    			limit = {
		    				is_immortal = no
		    			}
		    			set_character_immortal = yes
		    		}
		    	}
		    }
		    scope:hog_scope = {
                owner = {
		    		set_variable = {
		    			name = bpm_hog_heir
		    			value = prev
		    		}
		    	}
		    	add_character_role = bpm_character_role_protege
                remove_character_role = bpm_character_role_hog
            }
            every_scope_character = {
                limit = {
                    NOT = {scope:hog_scope = this}
                    has_role = bpm_character_role_protege
                }
                remove_character_role = bpm_character_role_protege
            }
        }
	}

	after = {
		remove_variable = hog_regency1_active
	}
}

# End of the regency
hog_regency.3 = {
	type = country_event
	placement = root

	duration = 3

	title = hog_regency.3.t
	desc = hog_regency.3.d
	flavor = hog_regency.3.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:hog_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/2Characters"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	cooldown = { days = 100 }

	trigger = {
		has_modifier = modifier_hog_regency
		var:bpm_hog_heir ?= {
			is_adult = yes
		}
	}

	cancellation_trigger = {

	}

	immediate = {
		var:bpm_headofgov ?= {
			save_scope_as = regent_hog_scope
		}
		var:bpm_hog_heir ?= {
			save_scope_as = hog_scope
		}
	}

	option = {
		name = hog_regency.3.a
		default_option = yes
		remove_modifier = modifier_hog_regency
		scope:regent_hog_scope ?= {
			hidden_effect = {
				if = {
					limit = {
						is_immortal = yes
					}
					set_character_immortal = no
				}
			}
		}
		scope:hog_scope ?= {
			bpm_appoint_as_hog = yes
            remove_character_role = bpm_character_role_protege
		}
        remove_variable = bpm_hog_heir
        bpm_do_hog_transition = yes
	}
}
