on_law_activated = {
    on_actions = {
        bpm_codification_set_new_law_level
    }
}

bpm_codification_set_new_law_level = {
    effect = {
        owner = { save_scope_as = associated_country }
        if = {
            limit = {
                owner = { has_variable = bpm_codification_law_passed_by_enactment_consumable }
            }
            owner = { remove_variable = bpm_codification_law_passed_by_enactment_consumable }
        } 
        else_if = {
            limit = { 
                game_date > 1836.1.1 
                type = { bpm_codification_can_be_decree = yes }
            }
            type = {
                bpm_codification_set_decree = yes
            }
        }
    }
}
