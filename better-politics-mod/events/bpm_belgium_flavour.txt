bpm_belgian_law_events.1 = { # Municipal and Pronvicial Act (March 30th & April 30th, 1836)

	type = country_event

	dlc = dlc_bpm

	event_image = {
		video = "europenorthamerica_london_center"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	title = bpm_belgian_law_events.1.t
	desc = {
		triggered_desc = {
			desc = bpm_belgian_law_events.1.federated
			trigger = {
				has_law = law_type:law_federated
			}
		}
		triggered_desc = {
			desc = bpm_belgian_law_events.1.devolved
			trigger = {
				has_law = law_type:law_devolved
			}
		}
	}

	flavor = bpm_belgian_law_events.1.f

	duration = 3

	cooldown = {
		years = 100
	}

	trigger = {
		OR = {
			has_law = law_type:law_federated
			has_law = law_type:law_devolved
		}
	}

	immediate = {
		change_variable = {
			name = bpm_belgian_je_organic_laws_introduced
			add = 1
		}
	}

	option = {
		name = bpm_belgian_law_events.1.a

		default_option = yes

		custom_tooltip = {
			text = bpm_codifying_elected_bureaucrats
			bpm_codify_law = { # Codifying the electable city councils
				law = law_type:law_elected_bureaucrats
				level = constitutional
			}
		}

		add_modifier = {
			name = bpm_organizing_municipalities
			years = 2
			is_decaying = yes
		}

		if = {
			limit = {
				has_law = law_type:law_devolved
			}
			set_variable = reactionary_law_introduced_1842

			bpm_belgium_make_libs_mad_clericalism = yes
		}

		else_if = {
			limit = {
				has_law = law_type:law_federated
			}
			set_variable = liberal_law_introduced_1836

			bpm_belgium_make_libs_mad_lonelyness = yes
		}

		set_variable = bpm_je_municipal_provincial_laws_passed

		bpm_belgium_make_consolidation_progress_bigger_10 = yes
	}
}

	title = bpm_belgian_law_events.2.t
	desc = {
		triggered_desc = {
			desc = bpm_belgian_law_events.2.religious
			trigger = {
				has_law = law_type:law_religious_schools
			}
		}
		triggered_desc = {
			desc = bpm_belgian_law_events.2.public
			trigger = {
				has_law = law_type:law_public_schools
			}
		}
	}

	flavor = bpm_belgian_law_events.2.f

	duration = 3

	cooldown = {
		years = 100
	}

	trigger = {
		OR = {
			has_law = law_type:law_religious_schools
			has_law = law_type:law_public_schools
		}
	}

	immediate = {
		change_variable = {
			name = bpm_belgian_je_organic_laws_introduced
			add = 1
		}
	}

	option = {
		name = bpm_belgian_law_events.2.a

		default_option = yes		

		if = { ### Every secularist IG will be particularly unhappy
			limit = {
				has_law = law_type:law_religious_schools
			}

			every_interest_group = {
				limit = {
					OR = {
						has_ideology = ideology:ideology_rel_secularist
						has_ideology = ideology:ideology_rel_radical_secularist
					}
				}
				add_modifier = {
					name = bpm_stubborn_ig
					days = normal_modifier_time
				}
			}
			set_variable = religious_schooling_introduced_1842
		}

		else_if = { # Religion and morality are subjects present in schooling (Article 6-7)
			limit = {
				has_law = law_type:law_public_schools
			}

			active_law:lawgroup_education_system ?= {
				add_amendment = {
					type = amendment_church_organised_schools
					sponsor = PREV.ig:ig_devout
				}
			}
			set_variable = concessioned_schooling_introduced_1842
		}

		set_variable = bpm_je_education_law_passed

		add_modifier = {
			name = bpm_funding_and_organizing_primary_schooling
			years = 5
			is_decaying = yes
		}

		bpm_belgium_make_consolidation_progress_bigger_10 = yes

		bpm_belgium_make_libs_mad_clericalism = yes
	}
}
