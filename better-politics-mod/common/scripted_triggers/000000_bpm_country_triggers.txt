bpm_belgium_government_is_mixed = {
	OR = {
		bpm_right_unionist_ig_government = bpm_left_unionist_ig_government
		AND = {
			bpm_minimum_for_dominance_government >= bpm_left_unionist_ig_government
			bpm_minimum_for_dominance_government >= bpm_right_unionist_ig_government
		}
	}
}

bpm_springtime_democrat_radical_victory_trigger = {
	custom_tooltip = {
		text = bpm_republic_is_constitutional_law
		owner = {
			OR = {
				has_law = law_type:law_parliamentary_republic
				has_law = law_type:law_presidential_republic
			}
			}
		active_law:lawgroup_governance_principles = {
			bpm_codification_is_constitutional_law = yes
		}
	}

	custom_tooltip = {
		text = bpm_seppow_supleg_is_constitutional_law
		owner = {
			OR = {
				has_law = law_type:law_separated_powers
				has_law = law_type:law_supreme_legislature
			}
		}
		active_law:lawgroup_legislative_principles = {
			OR = {
				bpm_codification_is_constitutional_law = yes
				bpm_codification_is_legal_law = yes
			}
		}
	}
		
	custom_tooltip = {
		text = bpm_has_modern_bureaucrats_law
		NOT = {
			has_law_or_variant = law_type:law_hereditary_bureaucrats
		}
	}
	custom_tooltip = {
		text = bpm_has_regulatory_or_wage_law
		OR = {
			has_law = law_type:law_regulatory_bodies
			has_law = law_type:law_wage_subsidies
		}
	}
		
	custom_tooltip = {
		text = bpm_has_enacted_universal_suffrage
		has_law = law_type:law_universal_suffrage
	}
	custom_tooltip = {
		text = bpm_has_enacted_no_speech
		has_law = law_type:law_protected_speech
	}

	ig:ig_radicals = {
		is_in_government = yes
	}

	custom_tooltip = {
		text = bpm_has_abolished_at_least_4_privileges
		var:bpm_springtime_abolished_privileges >= 4
	}
}

bpm_springtime_reform_moderate_victory_trigger = {
	custom_tooltip = {
		text = bpm_seppow_supleg_is_constitutional_law
		owner = {
			OR = {
				has_law = law_type:law_separated_powers
				has_law = law_type:law_supreme_legislature
			}
		}
		active_law:lawgroup_legislative_principles = {
			OR = {
				bpm_codification_is_constitutional_law = yes
				bpm_codification_is_legal_law = yes
			}
		}
	}

	custom_tooltip = {
		text = bpm_has_modern_bureaucrats_law
		NOT = {
			has_law_or_variant = law_type:law_hereditary_bureaucrats
		}
	}

	custom_tooltip = {
		text = bpm_has_enacted_no_speech
		has_law = law_type:law_protected_speech
	}
		
	OR = {
		custom_tooltip = {
			text = bpm_seppow_libsuf_is_enacted
			OR = {
				has_law = law_type:law_wealth_voting
				has_law = law_type:law_census_voting
			}
		}

		custom_tooltip = {
			text = bpm_seppow_unisuf_and_average_is_enacted
			has_law = law_type:law_universal_suffrage
			bpm_springtime_average_revolutionary_drift_value >= 30
		}
	}

	custom_tooltip = {
		text = bpm_has_abolished_at_least_4_privileges
		var:bpm_springtime_abolished_privileges >= 4
	}
}
