bpm_state_about_to_collapse_sgui = {
    scope = country
    effect = {
		every_scope_state = {
			limit = {
				AND = {
					turmoil >= {
						value = bpm_auth_radicals_in_state_collapse_threshold
						subtract = 0.05
					}
					bpm_auth_can_collapse = yes
				}
				NOT = {
					has_decree = decree_violent_suppression
				}
			}
			save_scope_as = bpm_tt_sgui_state
			custom_tooltip = "bpm_state_is_about_to_collapse"
		}
	}
}

bpm_state_collapsed_sgui = {
    scope = country
    effect = {
		every_scope_state = {
			limit = {
				bpm_auth_state_has_collapsed_auth = yes
				NOT = {
					OR = {
						modifier:state_bpm_auth_restores_order = yes
						owner = { 
							modifier:country_bpm_auth_restores_order = yes 
							is_incorporated = no
						}
					}
				}
			}
			save_scope_as = bpm_tt_sgui_state
			custom_tooltip = "bpm_state_is_about_to_collapse"
		}
	}
}

bpm_movement_endorses_law = {
	scope = political_movement
	saved_scopes = {
		law
	}
	is_shown = {
		law_enactment_stance = {
			law = scope:law.type
			value > neutral
		}
	}
}

bpm_movement_opposes_law = {
	scope = political_movement
	saved_scopes = {
		law
	}
	is_shown = {
		law_enactment_stance = {
			law = scope:law.type
			value < neutral
		}
	}
}
