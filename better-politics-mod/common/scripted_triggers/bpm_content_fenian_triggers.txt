bpm_country_is_involved_with_fenians = {
    OR = {
        any_state_in_bpm_ireland = {
            count > 0
        }
        has_variable = bpm_has_fenian
        has_variable = bpm_fenian_target
    }
    bpm_is_ireland_free = no
}

bpm_is_ireland_free = {
    any_country = {
        has_primary_culture = cu:irish
        is_in_geographic_region = bpm_geographic_region_ireland
        is_subject = no
    }
}

bpm_is_irish_occupier = {
    any_scope_state = {
        is_in_geographic_region = bpm_geographic_region_ireland
        
    }
}

bpm_is_fenian_attacker_candidate = {
    any_country = {
        is_adjacent_to_country = prev
        bpm_is_fenian_target_candidate = yes
    }
}

bpm_is_fenian_target_candidate = {
    army_power_projection <= 1000
    
}
