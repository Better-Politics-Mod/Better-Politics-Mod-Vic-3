# Taxation is theft
# Stall event for any tax law
tax.1 = {
	type = country_event
	placement = ROOT
	
	title = tax.1.t
	desc = tax.1.d
	flavor = tax.1.f

	category = enactment
	
	event_image = {
		video = "middleeast_courtroom_upheaval"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"
	
	duration = 3
	
	trigger = {
		OR = {
			is_enacting_law = law_type:law_proportional_taxation
			is_enacting_law = law_type:law_graduated_taxation
			is_enacting_law = law_type:law_land_based_taxation
			is_enacting_law = law_type:law_per_capita_based_taxation
		}
		any_political_movement = {
			bpm_is_political_movement_to_preserve = yes
		}
		NOT = { has_modifier = taxation_is_theft }
	}

	immediate = {
		random_political_movement = {
			limit = {
				bpm_is_political_movement_to_preserve = yes
			}
			save_scope_as = antitax_rebels
		}
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = {
		name = tax.1.a
		default_option = yes
		# lose success chance
		if = { 
			limit = { 
				is_player = yes
			}
			random_interest_group = { 
				limit = { 
					bpm_ig_neutral_enactment = yes
					or = { 
						bpm_ig_is_lower_institutional = yes
						bpm_ig_is_ideological = yes
					}
				}
				bpm_ig_decrease_law_enactment_approval_medium = yes
			}
			random_interest_group = { 
				limit = { 
					bpm_ig_neutral_enactment = yes
					or = { 
						bpm_ig_is_upper_institutional = yes
						bpm_ig_is_middle_institutional = yes
					}
				}
				bpm_ig_decrease_law_enactment_approval_medium = yes
			}
		}
		else = { 
			add_enactment_modifier = {
				name = taxation_is_theft
				multiplier = 2
			}
		}
		# movement gets angrier
		scope:antitax_rebels = {
			add_modifier = {
				name = antitax_anger
				days = long_modifier_time
			}
		}
	}
	option = { # promise to keep taxes low
		name = tax.1.b
		# lose less success chance

		if = { 
			limit = { 
				is_player = yes
			}
			random_interest_group = { 
				limit = { 
					OR = {
						bpm_ig_is_upper_institutional = yes
						bpm_ig_is_middle_institutional = yes
					}
					NOT = { owner = { has_law = law_type:law_consumption_based_taxation } }
					NOT = { has_modifier = bpm_law_sway_tax_break_middle_mod }
				}
				bpm_ig_decrease_law_enactment_approval_medium = yes
				bpm_subtract_from_law_sways = yes
			}
		}
		else = { 
			add_enactment_modifier = {
				name = taxation_is_theft
			}
		}
		add_modifier = {
			name = bpm_law_sway_tax_break_middle_used
			years = 5
		}
	}
}
