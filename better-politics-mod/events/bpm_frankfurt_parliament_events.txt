# Formalize the German Empire
bpm_frankfurt_parliament_events.12 = {
	type = country_event
	placement = root
	title = bpm_frankfurt_parliament_events.12.t
	flavor = bpm_frankfurt_parliament_events.12.f
	desc = bpm_frankfurt_parliament_events.12.d
	
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}
	duration = 3

	trigger = {
		# event
	}

	immediate = {

	}

	# Kleindeutschland
	option = {
		name = bpm_frankfurt_parliament_events.12.a
		trigger = {
			has_global_variable = bpm_frankfurt_parliament_lesser_germany
		}

		set_variable = german_48_unifier
		set_global_variable = bpm_je_frankfurt_parliament_complete
		change_tag = GER

		# Enacts selected laws to the constitution
		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_speech_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_centralization_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_voting_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_hog_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = law_type:law_separated_powers
			level = constitutional
		}

		# Transfers leaders of minor states to the German Empire, integrates non-Austrian states
		every_character = {
			limit = {
				owner = {
					country_is_in_german_confederation = yes
					NOT = { has_variable = bpm_frankfurt_monarch_var }
					NOR = {
						c:AUS ?= this
						is_subject_of = c:AUS
					}
				}
				has_role_of_type = ruler
			}
			transfer_character = c:GER
			set_is_noble = yes
			remove_character_role = politician
			add_character_role = magnate
		}

		every_country_in_german_confederation = {
			limit = {
				country_is_in_german_confederation = yes
				NOT = { has_variable = bpm_frankfurt_monarch_var }
				NOR = {
					c:AUS ?= this
					is_subject_of = c:AUS
				}
			}

			trigger_event = { id = bpm_frankfurt_parliament_events.20 days = 7 popup = yes }

			# c:GER ?= {
			# 	annex_with_incorporation = prev
			# }
		}
	}

	# Großdeutschland
	option = {
		name = bpm_frankfurt_parliament_events.12.b
		trigger = {
			has_global_variable = bpm_frankfurt_parliament_greater_germany
		}

		set_variable = german_48_unifier
		set_global_variable = bpm_je_frankfurt_parliament_complete
		change_tag = GER

		# Enacts selected laws to the constitution
		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_speech_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_centralization_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_voting_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = global_var:bpm_frankfurt_parliament_hog_law_chosen
			level = constitutional
		}

		bpm_enact_law = {
			law = law_type:law_separated_powers
			level = constitutional
		}

		# Transfers leaders of minor states to the German Empire, integrates German states
		every_character = {
			limit = {
				owner = {
					country_is_in_german_confederation = yes
					NOT = { has_variable = bpm_frankfurt_monarch_var }
				}
				has_role_of_type = ruler
			}
			transfer_character = c:GER
			set_is_noble = yes
			remove_character_role = politician
			add_character_role = magnate
		}

		every_country = {
			limit = {
				country_is_in_german_confederation = yes
				NOT = { has_variable = bpm_frankfurt_monarch_var }
			}
			trigger_event = { id = bpm_frankfurt_parliament_events.20 days = 7 popup = yes }

		}

		c:GER = { trigger_event = { id = bpm_frankfurt_parliament_events.13 days = 14 popup = yes } }
	}
}

# Splitting the Austrian realms in Großdeutschland
bpm_frankfurt_parliament_events.13 = {
	type = country_event
	placement = root
	title = bpm_frankfurt_parliament_events.13.t
	flavor = bpm_frankfurt_parliament_events.13.f
	desc = bpm_frankfurt_parliament_events.13.d
	
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}
	duration = 3

	trigger = {
		# event
	}

	immediate = {

	}

	# Assuming Austrian led empire
	option = {
		name = bpm_frankfurt_parliament_events.13.a
		default_option = yes
		
		# Hungary (if it has not successfully split away)
		if = {
			limit = {
				exists = c:HUN
				c:HUN = {
					is_subject_of = c:GER
				}
			}
			c:HUN = {
				make_independent = yes
			}
		}

		# Galicia-Lodomeria
		if = {
			limit = {
				exists = c:GAL
				c:GAL = {
					is_subject_of = c:GER
				}
			}
			c:GAL = {
				make_independent = yes

				# Karl Joeseph of Austria-Este is elected to rule Galicia-Lodomeria seperatly
				create_character = {
					template = aus_ferdinand_karl_joseph_of_austria-este
					on_created = {
						set_character_as_ruler = yes
					}
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = c:GAL }
				c:GER = { owns_entire_state_region = STATE_EAST_GALICIA }
			}
			create_country = {
				tag = GAL
				origin = c:GER
				state = s:STATE_EAST_GALICIA.region_state:GER
			}
			s:STATE_WEST_GALICIA.region_state:GER = { set_state_owner = c:GAL set_state_type = incorporated }
			s:STATE_BUKOVINA.region_state:GER = { set_state_owner = c:GAL set_state_type = incorporated }
			c:GAL = {
				# Karl Joeseph of Austria-Este is elected to rule Galicia-Lodomeria seperatly
				create_character = {
					template = aus_ferdinand_karl_joseph_of_austria-este
					on_created = {
						set_character_as_ruler = yes
					}
				}

				bpm_enact_law = {
					law = law_type:law_monarchy
					level = constitutional
				}

				bpm_enact_law = {
					law = law_type:law_census_voting
					level = legal
				}

				bpm_enact_law = {
					law = law_type:law_unitary
					level = constitutional
				}

				bpm_enact_law = {
					law = law_type:law_freedom_of_conscience
					level = constitutional
				}
			}
		}

		# Lombardy–Venetia
		if = {
			limit = {
				c:GER = { owns_entire_state_region = s:STATE_LOMBARDY }
			}

			create_country = {
				tag = LOM
				origin = c:GER
				state = s:STATE_LOMBARDY.region_state:GER
			}
			s:STATE_VENETIA.region_state:GER = { set_state_owner = c:LOM set_state_type = incorporated }

			c:LOM = {
				# Maximillian is elected to rule Lombardy seperatly (tfw Cancun trip cancelled), needs a regent
				create_character = {
					template = aus_maximilian_I_habsburg_template
					on_created = {
						set_character_as_ruler = yes
					}
				}

				bpm_enact_law = {
					law = law_type:law_monarchy
					level = constitutional
				}

				bpm_enact_law = {
					law = law_type:law_census_voting
					level = legal
				}

				bpm_enact_law = {
					law = law_type:law_devolved
					level = constitutional
				}

				bpm_enact_law = {
					law = law_type:law_freedom_of_conscience
					level = constitutional
				}
			}
		}
	}
}
