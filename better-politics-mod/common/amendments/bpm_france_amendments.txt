bpm_amendment_artisanal_economy = {

	parent = law_industry_banned

	allowed_laws = {
		law_laissez_faire
		law_interventionism
		law_industry_banned
		law_agrarianism
	}

	modifier = {
		bpm_france_artisanal_lowest_level_pm = yes
		country_shopkeepers_pol_str_mult = 0.25
		interest_group_ig_intelligentsia_pol_str_mult = 0.5
		country_capitalists_pol_str_mult = -0.75
		building_group_bg_manufacturing_allowed_collectivization_add = 0.9
		building_group_bg_extraction_allowed_collectivization_add = 0.9
		building_group_bg_private_infrastructure_allowed_collectivization_add = 0.9
	}

	possible = {
		c:FRA ?= this
	}

	on_activate = {
		ig:ig_intelligentsia = {
			bpm_remove_all_eco_ideologies = yes
			add_ideology = ideology_eco_artisan
			bpm_reorder_ideologies = yes
		}
		ig:ig_petty_bourgeoisie = {
			bpm_remove_all_eco_ideologies = yes
			add_ideology = ideology_eco_artisan
			bpm_reorder_ideologies = yes
		}
		ig:ig_armed_forces = {
			bpm_remove_all_eco_ideologies = yes
			add_ideology = ideology_eco_artisan
			bpm_reorder_ideologies = yes
		}
		ig:ig_rural_folk = {
			bpm_remove_all_eco_ideologies = yes
			add_ideology = ideology_eco_artisan
			bpm_reorder_ideologies = yes
		}
	}

	on_deactivate = {
		ig:ig_intelligentsia = {
			if = {
				limit = { has_ideology = ideology:ideology_eco_artisan }
				bpm_remove_all_eco_ideologies = yes
				bpm_reorder_ideologies = yes
			}
		}
		ig:ig_petty_bourgeoisie = {
			if = {
				limit = { has_ideology = ideology:ideology_eco_artisan }
				bpm_remove_all_eco_ideologies = yes
				bpm_reorder_ideologies = yes
			}
		}
		ig:ig_armed_forces = {
			if = {
				limit = { has_ideology = ideology:ideology_eco_artisan }
				bpm_remove_all_eco_ideologies = yes
				bpm_reorder_ideologies = yes
			}
		}
		ig:ig_rural_folk = {
			if = {
				limit = { has_ideology = ideology:ideology_eco_artisan }
				bpm_remove_all_eco_ideologies = yes
				bpm_reorder_ideologies = yes
				add_ideology = ideology_eco_modern_agrarian
			}
		}
	}

	would_sponsor = {
		is_interest_group_type = ig_intelligentsia
	}

	ai_will_revoke = {
		always = yes
	}

	can_repeal = {
		bpm_cab_synergy_value >= legitimacy_to_repeal_amendment
	}
}
