bpm_coalition_and_actions_init = {
	add_to_global_variable_list = {
		name = bpm_ig_strategies
		target = flag:bpm_strategy_default
	}
	add_to_global_variable_list = {
		name = bpm_coalition_actions
		target = flag:bpm_action_revolt
	}
}

bpm_coalition_and_actions = {
	save_scope_as = country
	# == Coalition and Actions
	# First each IG should decide on a strategy & what coalition (movement) to join
	every_activated_interest_group = {
		save_scope_as = ig
		bpm_ig_decide_strategy = yes #v
		bpm_ig_decide_coalition = yes
		clear_saved_scope = ig
	}

	# Then, movements select which action they wanna do
	every_political_movement = {
		save_scope_as = movement
		bpm_movement_select_action = yes
		clear_saved_scope = movement
	}

	every_in_list = {
		variable = bpm_active_coalition_actions
		save_scope_as = action
		bpm_action_on_monthly = yes
		clear_saved_scope = action
	}

	clear_saved_scope = country
}

bpm_ig_decide_strategy = {
	ordered_in_global_list = {
		variable = bpm_ig_strategies
		position = 0
		order_by = bpm_strategy_attraction

		save_scope_as = selected_strategy
	}

	set_variable = {
		name = bpm_ig_strategy
		value = scope:selected_strategy
	}

	clear_saved_scope = selected_strategy
}


bpm_ig_decide_coalition = {
	remove_variable = bpm_ig_coalition_mov

	OWNER = { ordered_political_movement = {
		limit = {
			bpm_ig_movement_attraction > 0
		}
		position = 0
		order_by = bpm_ig_movement_attraction

		save_scope_as = selected_movement
	} }

	set_variable = {
		name = bpm_ig_coalition_mov
		value = scope:selected_movement
	}

	clear_saved_scope = selected_movement
}

bpm_movement_select_action = {
	if = {
		limit = {
			scope:country = { NOT = { any_in_list = {
				variable = bpm_active_coalition_actions
				var:bpm_initiator ?= scope:movement
			} } }
		}
		ordered_in_global_list = {
			limit = {
				bpm_movement_action_attraction > 0
			}
			variable = bpm_coalition_actions
			position = 0
			order_by = bpm_movement_action_attraction

			save_scope_as = selected_action_type
		}
		if = {
			limit = {
				exists = scope:selected_action_type
			}
			scope:country = {
				create_struct = {
					struct_scope = created_action
					struct_type = bpm_coalition_action_num
				}
			}

			scope:created_action = {
				set_variable = {
					name = bpm_initiator
					value = scope:movement
				}
				set_variable = {
					name = bpm_action_type
					value = scope:selected_action_type
				}
			}

			scope:country = {
				add_to_variable_list = {
					name = bpm_active_coalition_actions
					target = scope:created_action
				}
			}

			clear_saved_scope = created_action
			clear_saved_scope = selected_action_type
		}
	}
	else = {
		random_in_list = {
			variable = bpm_active_coalition_actions
			var:bpm_initiator ?= scope:movement
			save_scope_as = selected_action
		}
		if = {
			limit = {
				NOT = {
					scope:selected_action.var:bpm_action_type = {
						bpm_movement_action_attraction > 0
					}
				}
			}
			remove_list_variable = {
				name = bpm_active_coalition_actions
				target = scope:selected_action
			}
			destroy_struct = {
				struct = scope:selected_action
			}
		}

	}
}

bpm_action_on_monthly = {
	if = {
		limit = {
			NOT = {
				exists = scope:action.var:bpm_initiator
			}
		}
		remove_list_variable = {
			name = bpm_active_coalition_actions
			target = scope:action
		}
		destroy_struct = {
			struct = scope:action
		}
	}
	else = {
		scope:action.var:bpm_initiator = { save_scope_as = movement }
		scope:action = { 
			if = {
				limit = {
					var:bpm_action_type = flag:bpm_action_revolt
				}
				bpm_action_revolt_on_monthly = yes
			}
			else = {
				scope:country = {
					remove_from_variable_map = {
						name = bpm_revolt_progress
						key = scope:movement
					}
					remove_from_variable_map = {
						name = bpm_revcell_recently_subsided
						key = scope:movement
					}
					every_scope_state = {
						remove_from_variable_map = {
							name = bpm_revcells
							key = scope:movement
						}
						remove_from_variable_map = {
							name = bpm_revcell_recently_subsided
							key = scope:movement
						}
					}
				}
			}
		}
		clear_saved_scope = movement
	}

}
