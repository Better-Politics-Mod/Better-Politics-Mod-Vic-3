POLITICAL_MOVEMENTS = {
	every_country = {
		limit = {
			NOT = { is_country_type = decentralized }
		}
		bpm_create_necessary_movements = yes
	}
	c:JAP ?= {
		create_political_movement = {
			type = movement_kaikoku
		}
	}
	c:USA ?= {
		create_political_movement = { type = movement_socialist }
		create_political_movement = { type = movement_market_liberal }
	}

	every_country = {
		every_political_movement = {
			remove_modifier = initial_movement_enthusiasm
		}
	}
	# brewing rebellions of 1837
	c:ONT ?= {
		create_political_movement = { type = movement_national_liberal }
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_national_liberal
			}
			add_modifier = {
				name = can_reformer_movement_fury
				days = very_long_modifier_time
			}
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_national_liberal
			}
			add_modifier = {
				name = can_reformer_movement_fury
				days = very_long_modifier_time
			}
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_radical
			}
			add_movement_enthusiasm_modifier = yes
		}
	}
	c:QUE ?= {
		create_political_movement = { type = movement_national_liberal }
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_national_liberal
			}
			add_modifier = {
				name = can_reformer_movement_fury
				days = very_long_modifier_time
			}
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_radical
			}
			add_movement_enthusiasm_modifier = yes
		}
	}
}
