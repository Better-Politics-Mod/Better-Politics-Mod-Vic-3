on_monthly_pulse_country = {
	on_actions = {
		on_bpm_monthly_pulse_country_christdem
		#on_bpm_monthly_pulse_country_je_christians
	}
}

on_bpm_monthly_pulse_country_christdem = {
	effect = {
		if = {
			limit = { 
				has_global_variable = first_intl_chain_complete_var 
				owner = { global_country_ranking = 1 } # Making trigger happer once per month because paradox apparently has no on_action for global events??? what? how??
			}
			if = {
				limit = { NOT = { has_global_variable = je_christian_democracy_progress } }
				set_global_variable = {
					name = je_christian_democracy_progress
					value = 0
				}
			}
			bpm_je_christian_democrat_progress_rate_update = yes 
			bpm_handle_steady_global_var_growth = {
				VARIABLE = je_christian_democracy_progress
			}
		}
	}
}

on_bpm_monthly_pulse_country_je_christians = {
	effect = {
		if = {
			limit = {
				ig:ig_christian_democrats ?= {
					has_ideology = ideology:ideology_nonexistent 
				}
				OR = {
					any_interest_group = {
						is_powerful = yes
						OR = {
							has_ideology = ideology_rel_moralist
							has_ideology = ideology_rel_radical_moralist
							has_ideology = ideology_rel_secularist
						}
					}
					this = c:GER
				}
				has_technology_researched = egalitarianism
				has_technology_researched = corporatism
			}
			if = {
				limit = {
					OR = {
						AND = {
							exists = c:GER
							this = c:GER
						}
						AND = {
							exists = c:ITA
							this = c:ITA
						}
						AND = {
							exists = c:USA
							this = c:USA
						}
						AND = {
							exists = c:FRA
							this = c:FRA
						}
					}
				}
				trigger_event = { id = bpm_christdem_split.1 }
			}
		}
	}
}
