bpm_ig_is_actively_revolting = {
	var:bpm_ig_coalition_mov ?= {
		bpm_movement_is_revolting = yes
	}
}


bpm_movement_is_revolting = {
	save_temporary_scope_as = movement
	owner = { save_temporary_scope_as = country }
	any_country = {
		has_war_with = scope:country
		var:bpm_civil_war_origin ?= scope:country
		var:bpm_country_revolutionary_movement_type ?= scope:movement.type
	}
}

bpm_coalition_crossroad_takes_over = {
	bpm_coalition_has_supportive_general_or_minister = yes
	bpm_coalition_controls_capital = yes
	bpm_country_has_strong_center = yes
}

bpm_coalition_crossroad_bloodless = {
	bpm_very_low_state_capacity = yes
}

# whether a front has a strong supportive general in the country
bpm_coalition_has_supportive_general_or_minister = {
	any_scope_character = {
		interest_group = { 
			var:bpm_ig_coalition_mov ?= scope:movement
		}
		OR = {
			commander_rank >= 2
			has_character_role = bpm_character_role_minister
		}
        loyalty < define:NCharacters|LOYALTY_UNCERTAIN_THRESHOLD
	}
}

# whether the front controls the capital of the country
bpm_coalition_controls_capital = {
	capital = {
		is_key_in_variable_map = {
			name = bpm_revcells
			target = scope:movement
		}
		"variable_map(bpm_revcells|scope:movement)" >= 3
	}
}

# whether the country has a strong center
# determines if it will simply fall after
# the capital is taken or if it will put up
# a fight and potentially cause a civil war
bpm_country_has_strong_center = {
	NOR = {
		has_law = law_type:law_ad_hoc_centralization
		has_law = law_type:law_confederated
	}
}
