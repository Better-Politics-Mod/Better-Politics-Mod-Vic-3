## Reorganize Mark

bpm_government_formation_result = {
    type = country
    random_valid = no

    text = {
        trigger = {
            always = no
        }
        fallback = yes
        localization_key = bpm_form_government_events_minority
    }
    text = {
        trigger = {
            OR = {
                NOT = { has_variable_list = bpm_in_gov_party_sel }
                variable_list_size = {
                    name = bpm_in_gov_party_sel
                    value = 0
                }
            }
        }
        localization_key = bpm_form_government_events_election
    }
    text = {
        trigger = {
            has_variable = bpm_in_gov_party_sel_votes
            var:bpm_in_gov_party_sel_votes >= 0.5
        }
        localization_key = bpm_form_government_events_coalition
    }
}

bpm_government_hostility_level = {
    type = country
    random_valid = no

    text = {
        trigger = {
            bpm_selected_government_interparty_hostility_value >= 30
        }
        localization_key = bpm_government_hostility_level_completely_hostile
    }
    text = {
        trigger = {
            bpm_selected_government_interparty_hostility_value >= 20
            bpm_selected_government_interparty_hostility_value < 30
        }
        localization_key = bpm_government_hostility_level_hostile
    }
    text = {
        trigger = {
            bpm_selected_government_interparty_hostility_value >= 10
            bpm_selected_government_interparty_hostility_value < 20
        }
        localization_key = bpm_government_hostility_level_tense
    }
    text = {
        trigger = {
            bpm_selected_government_interparty_hostility_value >= 1
            bpm_selected_government_interparty_hostility_value < 10
        }
        localization_key = bpm_government_hostility_level_neutral
    }
    text = {
        trigger = {
            always = yes
        }
        localization_key = bpm_government_hostility_level_united
    }
}
