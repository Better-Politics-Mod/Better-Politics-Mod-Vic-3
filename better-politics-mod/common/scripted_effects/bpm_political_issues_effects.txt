bpm_setup_issues_varlist = {
	save_scope_value_as = {
		name = threshold
		value = 0
	}
	clear_variable_list = bpm_country_issues
	if = {
		limit = {
			bpm_issue_sol_interest > scope:threshold
		}
		add_to_variable_list = {
			name = bpm_country_issues
			target = flag:bpm_issue_sol
		}
	}
	if = {
		limit = {
			bpm_issue_anti_war_interest > scope:threshold
		}
		add_to_variable_list = {
			name = bpm_country_issues
			target = flag:bpm_issue_anti_war
		}
	}
	if = {
		limit = {
			bpm_issue_anti_liberal_interest > scope:threshold
		}
		add_to_variable_list = {
			name = bpm_country_issues
			target = flag:bpm_issue_anti_liberal
		}
	}
	bpm_unsplat_issues = yes
}

bpm_setup_party_issues_campaign = {
	bpm_setup_issues_varlist = yes
	clear_variable_list = bpm_box_of_issues
	save_scope_as = country
	every_in_list = {
		variable = bpm_country_issues
		scope:country = {
			add_to_variable_list = {
				name = bpm_box_of_issues
				target = prev
			}
		}
	}
	every_active_party = {
		save_scope_as = checking_party
		scope:country = {
			ordered_in_list = {
				variable = bpm_box_of_issues
				order_by = bpm_issue_attraction
				position = 0
				save_scope_as = selected_issue
			}
			# remove_list_variable = {
			# 	name = bpm_box_of_issues
			# 	target = scope:selected_issue
			# }
			bpm_splat_issue = yes
		}
	}
	bpm_unsplat_issues = yes
}

bpm_unsplat_issues = {
	save_scope_as = country
	every_interest_group = {
		remove_variable = bpm_party_issue
	}
	every_active_party = {
		save_scope_as = checking_party
		ordered_member = {
			order_by = ig_clout
			position = 0
			save_scope_as = leader_ig
		}
		scope:country = {
			bpm_unsplat_issue = yes
			# calculate momentum impact
			save_scope_value_as = {
				name = bpm_momentum_impact
				value = {
					value = scope:sel_issue.bpm_issue_interest
					subtract = 0.15
					multiply = {
						value = scope:sel_issue.bpm_issue_interest
						subtract = 0.15
					}
					if = {
						limit = {
							scope:sel_issue.bpm_issue_interest < 0.15
						}
						multiply = -1
					}
				}
			}
		}
		scope:leader_ig = {
			set_variable = {
				name = bpm_party_issue
				value = scope:sel_issue
			}
			set_variable = {
				name = bpm_party_issue_impact
				value = scope:bpm_momentum_impact
			}
		}
	}
}
