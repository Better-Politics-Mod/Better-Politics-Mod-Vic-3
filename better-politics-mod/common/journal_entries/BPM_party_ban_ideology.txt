je_bpm_ban_fascists = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		always = yes
	}

	can_deactivate = yes
	possible = {
		OR = {
			any_interest_group = {
				OR = {
					bpm_leader_is_fascist = yes
					bpm_ig_is_fascist = yes
				}
				OR = {
					is_in_government = yes
					is_marginal = no
				}
			}
			bpm_banned_reactionary_parties = yes
		}
	}

	complete = {
		hidden_trigger = {
			always = no
		}
		NOR = {
			any_interest_group = {
				OR = {
					bpm_leader_is_fascist = yes
					bpm_ig_is_fascist = yes
				}
				OR = {
					is_in_government = yes
					is_marginal = no
				}
			}
			bpm_banned_reactionary_parties = yes
		}
	}

	weight = 25

	should_be_pinned_by_default = no
}

je_bpm_ban_communists = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		NOT = { has_law = law_type:law_council_republic }
	}

	can_deactivate = yes

		OR = {
			any_interest_group = {

				}
				is_insurrectionary = yes
			}
			bpm_banned_socialist_parties = yes
		}
		hidden_trigger = {
			NOT = { has_law = law_type:law_council_republic }
		}
	}

	complete = {
		hidden_trigger = {
			always = no
		}
		OR = {
			NOR = {
				any_interest_group = {
					OR = {
						bpm_leader_is_socialist = yes
						bpm_ig_is_socialist = yes
					}
					is_insurrectionary = yes
				}
				bpm_banned_socialist_parties = yes
			}
			has_law = law_type:law_council_republic
		}
	}

	weight = 25

	should_be_pinned_by_default = no
}

je_bpm_ban_anti_communists = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		has_law = law_type:law_council_republic
	}

	can_deactivate = yes
	possible = {
		OR = {
			any_interest_group = {
				bpm_leader_is_socialist = no
				bpm_ig_is_socialist = no
				is_insurrectionary = yes
			}
			has_law = law_type:law_party_socialist_only
		}
		hidden_trigger = {
			has_law = law_type:law_council_republic
		}
	}

	complete = {
		hidden_trigger = {
			always = no
		}
		OR = {
			NOR = {
				any_interest_group = {
					bpm_leader_is_socialist = no
					bpm_ig_is_socialist = no
					is_insurrectionary = yes
				}
				has_law = law_type:law_party_socialist_only
			}
			NOT = { has_law = law_type:law_council_republic }
		}
	}

	weight = 25

	should_be_pinned_by_default = no
}
