	custom_tooltip = {
		text = socialist_revolution_ongoing_tt

				any_influenced_interest_group = {
					law_stance = {

					}
				}
				any_supporting_character = {
					has_role = agitator
					law_stance = {
						law = law_type:law_council_republic
						value > neutral
					}
				}
			}		
		}	
	}		
}
