ep2_meiji.51 = { # Restoration - Resignation accepted
	type = country_event
	placement = root

	duration = 3

	title = ep2_meiji.51.t
	desc = ep2_meiji.51.d
	flavor = ep2_meiji.51.f

    event_image = {
		video = "ep2_important_japanese_men_negotiating"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	cooldown = { days = long_modifier_time }

	cancellation_trigger = {

	}

	immediate = {
		random_scope_character = {
			limit = {
				has_role = bpm_character_role_hog
			}
			save_scope_as = shogun_scope
		}
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_meiji_restorationist
			}
			save_scope_as = restorationist_movement_scope
		}
		ordered_scope_character = {
			limit = {
				has_variable = daimyo_var
			}
			order_by = loyalty
			position = 0
			set_character_immortal = yes
			save_scope_as = loyal_daimyo_scope
		}
		ordered_scope_character = {
			limit = {
				has_variable = daimyo_var
			}
			order_by = {
				value = loyalty
				multiply = -1
			}
			position = 0
			set_character_immortal = yes
			save_scope_as = disloyal_daimyo_scope
		}
	}

	option = {
		name = ep2_meiji.51.a
		default_option = yes
		random_scope_character = {
			limit = {has_role = bpm_character_role_hog}
			set_as_interest_group_leader = yes
		}
		je:je_meiji_restoration ?= {
			add_modifier = modifier_meiji_interregnum
		}
		activate_law = law_type:law_oligarchy
		if = {
			limit = {
				OR = {
					AND = {
						scope:shogun_scope ?= { # If one is pursuing Unification of Court and Shogunate and somehow get to this point, Traditionalist shoguns will fail automatically
							has_ideology = ideology:ideology_traditionalist
						}
						has_variable = kobu_gattai
					}
					government_legitimacy <= 50
				}
			}
			trigger_event = {
				id = ep2_meiji.6
				days = 7
			}
		} 
		else = {
			if = { # If one is pursuing Unification of Court and Shogunate and somehow get to this point, non-Traditionalist shoguns will automatically switch to Parliamentary Government
				limit = {
					has_variable = kobu_gattai
				}
				custom_tooltip = {
					text = kobu_gattai_no_longer_viable_tt
					remove_variable = kobu_gattai
					set_variable = kogi_yoron
				}
			}
			trigger_event = {
				id = ep2_meiji.7
				days = 7
			}
		}
		hidden_effect = {
			scope:loyal_daimyo_scope ?= {
				set_character_immortal = no
			}
			scope:disloyal_daimyo_scope ?= {
				set_character_immortal = no
			}
		}
	}
}

	on_opened_soundeffect = "event:/SFX/Events/misc/1Character_Banner"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	cooldown = { days = long_modifier_time }

	cancellation_trigger = {

	}

	immediate = {
		ruler = {
			save_scope_as = emperor_scope
		}
		random_scope_character = {
			limit = {
				has_role = bpm_character_role_hog
			}
			save_scope_as = taikun_scope
		}
		add_journal_entry = {
			type = je_meiji_main
		}
	}

	option = {
		name = ep2_meiji.9.a
		default_option = yes
		add_modifier = {
			name = modifier_continuity_and_enlightenment
			days = very_long_modifier_time
			is_decaying = yes
		}
	}

	option = {
		name = ep2_meiji.9.b
		add_modifier = {
			name = modifier_taikun_monarchy
			days = very_long_modifier_time
			is_decaying = yes
		}
		every_scope_character = {
			limit = {
				has_variable = house_tokugawa
			}
			add_modifier = {
				name = modifier_taikun_monarchy_character
				days = stupidly_long_modifier_time
			}
		}
	}

	after = {
		if = {
			limit = {
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_modernizer
					}
				}
			}
			create_political_movement = { type = movement_modernizer }
		}
		if = {
			limit = {
				NOT = { has_variable = completed_je_meiji_economy }
				in_default = no
				any_scope_state = {
					filter = {
						is_incorporated = yes
					}
					any_scope_building = {
						is_building_type = building_urban_center
						level >= 5
					}
					has_building = building_railway
					percent > 0.7
				}
			}
			change_variable = {
				name = meiji_var
				add = 1
			}

					is_in_government = no
				}

				NOT = {
					any_scope_building = {
						is_building_type = building_barrack
						OR = {
							has_active_production_method = pm_no_organization
							# has_active_production_method = pm_no_artillery
							# has_active_production_method = pm_cannon_artillery
						}
					}
				}
			}
			change_variable = {
				name = meiji_var
				add = 1
			}
			trigger_event = {
				id = meiji.3
			}
		}
		else = {
			add_journal_entry = {
				type = je_meiji_army
			}
		}
		if = {
			limit = {
				NOT = { has_dlc_feature = ep2_content }
				NOT = { has_variable = completed_je_meiji_diplomacy }
				NOT = { is_country_type = unrecognized }
				NOT = { has_law_or_variant = law_type:law_isolationism }
			}

				name = meiji_var
				add = 1
			}
			set_variable = completed_je_meiji_diplomacy # If the player manages to skip je_meiji_diplomacy we still need to set the variable
		}
		else_if = {
			limit = {
				NOT = { has_dlc_feature = ep2_content }
				OR = {
					has_law_or_variant = law_type:law_isolationism
					has_law_or_variant = law_type:law_traditionalism
					has_law_or_variant = law_type:law_closed_borders

				}
