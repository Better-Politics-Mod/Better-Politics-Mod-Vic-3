corporatism = {
	era = era_3
	texture = "gfx/interface/icons/invention_icons/corporatism.dds"
	category = society

	on_researched = {
		if = {
			limit = {
				NOT = { has_variable = corporatism_researched }
			}
			set_variable = corporatism_researched

			if = {
				limit = { 
					NOT = { this = c:BIC }
				}
				ig:ig_industrialists = {
					bpm_remove_all_wel_ideologies = yes
					add_ideology = ideology_wel_paternalistic_welfare
					bpm_reorder_ideologies = yes
				}
			}
			
			ig:ig_trade_unions ?= {
				if = {
					limit = {
						NOT = { has_ideology = ideology_sta_progressive_state }
					}
					bpm_remove_all_sta_ideologies = yes
					add_ideology = ideology_sta_progressive_state
					bpm_reorder_ideologies = yes
				}
			}

			ig:ig_christian_democrats ?= {
				if = {
					limit = {
						NOT = { has_ideology = ideology:ideology_nonexistent }
					}
					bpm_remove_all_wel_ideologies = yes
					if = {
						limit = {
							has_ideology = ideology:ideology_pragmatic_moralist
						}
						add_ideology = ideology_wel_pragmatic_distributist
					}
					else = {
						add_ideology = ideology_wel_distributist_welfare
					}
					bpm_reorder_ideologies = yes
				}
			}
		}
	}
	
	modifier = {
		state_radicals_from_political_movements_mult = -0.10
	}
	
	unlocking_technologies = {
		labor_movement
		nationalism
	}
	
	ai_weight = {
		value = 1
		
		if = {
			limit = { 
				has_strategy = ai_strategy_egalitarian_agenda
			}
			add = -0.5
		}	
		if = {
			limit = { 
				OR = {
					has_strategy = ai_strategy_conservative_agenda
					has_strategy = ai_strategy_reactionary_agenda
				}			
			}
			add = 1.5
		}
	}
}
