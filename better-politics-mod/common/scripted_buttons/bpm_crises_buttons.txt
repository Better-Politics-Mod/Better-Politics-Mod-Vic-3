bpm_scandal_response_distract = {
	name = "bpm_scandal_response_distract"
	desc = "bpm_scandal_response_distract_desc"

	visible = {
		has_journal_entry = je_bpm_crises_government_scandal
	}

	possible = {
		custom_tooltip = {
			text = "bpm_scandal_response_distract_possible_tooltip"
			bpm_media_preparedness >= 20
		}
	}

	ai_chance = {
		value = 100
	}

	effect = {
		random_list = {
			1 = {
				custom_tooltip = bpm_end_the_crisis_tt 
				set_variable = {
					name = bpm_crisis_ended
					days = 180
				}
				modifier = {
					if = {
						limit = {
							has_variable = bpm_headofgov
							var:bpm_headofgov = {
								bpm_character_skill_level >= 2
							}
						}
						add = 1
					}
					if = {
						limit = {
							has_variable = bpm_headofgov
							var:bpm_headofgov = {
								bpm_character_skill_level >= 4
							}
						}
						add = 2
					}
				}
			}
			1 = {
				custom_tooltip = bpm_crisis_progresses_tt
				change_variable = {
					name = bpm_government_scandal_progress
					add = 10
				}
			}
		}
		change_variable = {
			name = bpm_media_preparedness
			subtract = 20
		}
		custom_tooltip = {
			text = "bpm_scandal_response_distract_effect_tooltip"
		}
	}

}
