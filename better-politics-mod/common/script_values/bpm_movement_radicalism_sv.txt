bpm_additional_radicalism_factors = {
	value = 0
	add = bpm_additional_radicalism_factors_regime
	add = bpm_additional_radicalism_factors_institutions
	add = bpm_additional_radicalism_factors_diplomacy
	add = bpm_additional_radicalism_factors_law_enactment
	#multiply = bpm_additional_radicalism_factor_popular_support_factors
}

bpm_additional_radicalism_factors_regime = {
	if = {
		limit = {
			OR = {
				bpm_movement_is_nationalist = yes
				bpm_movement_is_modernizer = yes
				bpm_movement_is_socialist = yes
			}
			owner = { 
				bpm_country_laws_are_colonial = yes
			}
		}
		add = {
			desc = "HAS_COLONIAL_LAWS"
			add = 0.25
		}

	}
	else_if = {
		limit = {
			OR = {
				bpm_movement_is_socialist = yes
				bpm_movement_is_liberal_bourgeois = yes
			}
			owner = { OR = {
				bpm_country_is_pre_constitutional = yes
				bpm_country_is_pre_modern_absolutist = yes
			} }
		}
		add = {
			desc = "IS_PRE_MODERN_ABSOLUTIST"
			value = 0.25
		}
	}
	else_if = {
		limit = {
			owner = {
				bpm_country_laws_are_very_undemocratic = yes
			}
			bpm_movement_is_liberal_bourgeois = yes
		}
		add = {
			desc = "VERY_UNDEMOCRATIC"
			value = 0.25
		}
	}
	else_if = {
		limit = {
			owner = { 
				bpm_country_laws_are_moderately_undemocratic = yes
			}
			bpm_movement_is_liberal_bourgeois = yes
		}
		add = {
			desc = "MODERATLY_UNDEMOCRATIC"
			value = 0.2
			if = {
				limit = { bpm_movement_is_progressive_liberal = no }
				add = -0.1
			}
		}
	}
	else_if = {
		limit = {
			owner = { bpm_country_laws_unfair_voting = yes }
			bpm_movement_is_progressive_liberal = yes
		}
		add = {
			desc = "UNFAIR_VOTING"
			value = 0.15
		}
	}
	else_if = {
		limit = {
			owner = { 
				bpm_country_laws_are_liberal_democratic = no
			}
			bpm_movement_is_progressive_liberal = yes
		}
		add = {
			desc = "NOT_LIBERAL_DEMOCRATIC"
			value = 0.1
		}
	}
	
	if = {
		limit = {
			is_political_movement_type = movement_communist
			has_global_variable = first_intl_chain_complete_var
			owner = { NOT = { has_law_or_variant = law_type:law_council_republic } }
		}
		add = {
			desc = "COMMUNISTS_FIST_INTERNATIONAL"
			value = 0.2
		}
	}
	if = {
		limit = {
			is_political_movement_type = movement_communist
			has_global_variable = second_intl_chain_complete_var
			owner = { NOT = { has_law_or_variant = law_type:law_council_republic } }
		}
		add = {
			desc = "COMMUNISTS_SECOND_INTERNATIONAL"
			value = 0.2 
		}
	}
	if = {
		limit = {
			bpm_movement_is_modernizer = yes
			owner = { country_has_laws_in_need_of_modernization = yes }
		}
		add = {
			desc = "WE_MUST_MODERNIZE"
			value = 0.25
		}
	}
	else_if = {
		limit = {
			OR = {
				bpm_movement_is_liberal_nationalist = yes
				bpm_movement_is_laissez_faire = yes
				bpm_movement_is_modernizer = yes
			}
			owner = { 
				bpm_country_laws_have_traditional_economics = yes
			}
		}
		add = {
			desc = "TRADITIONAL_ECONOMIC_SYSTEM"
			add = 0.2
		}
	}
	else_if = {
		limit = {
			bpm_movement_is_laissez_faire = yes
		}
		if = {
			limit = { owner = { bpm_country_laws_are_very_capitalist = no } }
			add = {
				desc = "NOT_FREE_MARKET" 
				value = 0.25
			}
		}
		# if = {
		# 	limit = { owner = { bpm_country_laws_have_max_welfare = yes } }
		# 	add = {
		# 		desc = "HATES_WELFARE"
		# 		value = 0.1
		# 	}
		# }
	}
	else_if = {
		limit = {
			bpm_movement_is_abolitionist = yes
			owner = { bpm_country_has_unfree_labor = yes }
		}
		add = {
			desc = "OPPOSES_SLAVERY"
			value = 0.2 
			if = {
				limit = { owner = { has_law_or_variant = law_type:law_slave_trade } }
				add = 0.1
			}
		}
	}
	else_if = {
		limit = {
			bpm_movement_is_slaver = yes
			owner = { 
				OR = {
					has_law_or_variant = law_type:law_legacy_slavery
					has_law_or_variant = law_type:law_slavery_banned
				}
			}
		}
		add = {
			desc = "SUPPORTS_SLAVERY"
			value = 0.2 
			if = {
				limit = { owner = { has_law_or_variant = law_type:law_slavery_banned } }
				add = 0.1
			}
		}
	}
	if = {
		limit = {
			OR = {
				bpm_movement_is_absolutist_monarchist = yes
				bpm_movement_is_liberal_bourgeois = yes
				is_political_movement_type = movement_fascist
				is_political_movement_type = movement_reactionary
			}
			owner = { 
				bpm_country_laws_has_socialist_economics = yes
			}
		}
		add = {
			desc = "OPPOSES_SOCIALISM"
			value = 0.2
		}
	}
	if = {
		limit = {
			OR = {
				bpm_movement_is_absolutist_monarchist = yes
				is_political_movement_type = movement_fascist
				is_political_movement_type = movement_reactionary
			}
			owner = { OR = {
				bpm_country_laws_are_liberal_democratic = yes
				bpm_country_laws_are_moderately_liberal_democratic = yes
			} }
		}
		add = {
			desc = "ANTI_DEMOCRATIC"
			value = 0.2
		}
	}
	if = {
		limit = {
			bpm_movement_is_restorationist = yes
			owner = { bpm_country_is_republic = yes }
		}
		add = {
			desc = "MONARCHIST_MOVEMENT"
			value = 0.2
		}
	}
	else_if = {
		limit = {
			bpm_movement_is_absolutist_monarchist = yes
			owner = { bpm_country_is_constitutional_monarchy = yes }
		}
		add = {
			desc = "ABSOLUTE_MONARCHIST_MOVEMENT"
			value = 0.2
		}
	}
	if = {
		limit = {
			bpm_movement_is_anti_monarchy = yes
			owner = { bpm_country_is_republic = no }
		}
		add = {
			desc = "REPUBLICAN_MOVEMENT"
			value = 0.2
		}
	}

	if = {
		limit = {
			is_political_movement_type = movement_feminist
			bpm_country_laws_are_patriarchal = yes
		}
		add = {
			desc = "Patriachal laws"
			value = 0.2
		}
	}

	if = {
		limit = {
			bpm_movement_is_nationalist = yes
			bpm_movement_is_minority = no
			owner = {
				OR = {
					bpm_country_laws_are_racist = no
				}
			}
		}
		add = {
			desc = "NOT_RACISTS_LAWS"
			value = 0.2
		}
	}
	
	if = {
		limit = {
			OR = {
				is_political_movement_type = movement_religious_minority
				is_political_movement_type = movement_hindu_nationalist
				is_political_movement_type = movement_muslim_nationalist
			}
			bpm_movement_is_minority = yes
			owner = { bpm_country_laws_are_moralist = yes }
		}
		add = {
			desc = "RELIGIOUS_MINORITY"
			value = 0.2
		}
	}
}

bpm_additional_radicalism_factors_institutions = {
	if = {
		limit = {
			OR = {
				bpm_movement_is_liberal_bourgeois = yes
				#bpm_movement_is_moderate_socialist = yes
				is_political_movement_type = movement_utilitarian
			}
		}
		owner = {
				if = {
					limit = { has_institution = institution_schools }
					
					add = {
						value = 3
						subtract = "institution_investment_level(institution_schools)"
						multiply = 0.05
						desc = "INSTITUTION_FUNDING_LEVEL_schools"
					}
				}
				else = {
					add = {
						value = 0.1
						desc = "NO_INSTITUTION_schools"
					}		
				}
			}
	}
	if = {
		limit = {
			bpm_movement_is_moderate_socialist = yes
		}
		owner = {
			if = {
				limit = { has_institution = institution_social_security }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_social_security)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_social_security"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_social_security"
				}		
			}				
			
			if = {
				limit = { has_institution = institution_workplace_safety }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_workplace_safety)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_workplace_safety"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_workplace_safety"
				}		
			}
		}
	}
	if = {
		limit = {
			bpm_movement_is_modernizer = no
			bpm_movement_is_laissez_faire = yes
		}
		owner = {
			if = {
				limit = { 
					has_institution = institution_social_security 
					NOT = { has_law_or_variant = law_type:law_poor_laws }
				}
				
				add = {
					value = "institution_investment_level(institution_social_security)"
					multiply = 0.02
					desc = "INSTITUTION_FUNDING_LEVEL_LF_MOVEMENT_social_security"
				}
			}
			else_if = {
				limit = {
					has_law_or_variant = law_type:law_no_welfare
				}
				add = {
					value = 0.1
					desc = "NO_POOR_LAWS"
				}		
			}
			if = {
				limit = { 
					has_institution = institution_heathcare 
					NOT = { has_law_or_variant = law_type:law_private_health_insurance }
				}
				
				add = {
					value = "institution_investment_level(institution_health_system)"
					multiply = 0.02
					desc = "INSTITUTION_FUNDING_LEVEL_LF_MOVEMENT_health_system"
				}
			}
			else_if = {
				limit = {
					has_law_or_variant = law_type:law_no_health_system
				}
				add = {
					value = 0.1
					desc = "NO_PRIVATE_HEALTHCARE"
				}		
			}
			if = {
				limit = { 
					has_institution = institution_workplace_safety 
				}
				
				add = {
					value = "institution_investment_level(institution_workplace_safety)"
					multiply = 0.02
					desc = "INSTITUTION_FUNDING_LEVEL_workplace_safety"
				}
			}
		}
	}
	if = {
		limit = {
			owner = { 
				is_ai = no
				has_institution = institution_suffrage 
			}
		}
		owner = {
			if = {
				limit = {
					ROOT = {
						OR = {
							bpm_movement_is_socialist = yes
							bpm_movement_is_progressive_liberal = yes
						}
					}
				}
				add = {
					value = 2
					subtract = "institution_investment_level(institution_suffrage)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_suffragy"
				}
			}
			else_if = {
				limit = {
					ROOT = { bpm_movement_is_regressive_liberal = yes }
					bpm_country_laws_ok_voting = yes
				}
				add = {
					value = -3
					add = "institution_investment_level(institution_suffrage)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_suffragy"
				}
			}
			else_if = {
				limit = {
					ROOT = { 
						OR = {
							bpm_movement_is_absolutist_monarchist = yes
							is_political_movement_type = movement_reactionary
							is_political_movement_type = movement_fascist
						}
					}
					OR = {
						bpm_country_laws_unfair_voting = yes
						bpm_country_laws_ok_voting = yes
					}
				}
				add = {
					add = "institution_investment_level(institution_suffrage)"
					multiply = 0.03
					desc = "INSTITUTION_FUNDING_LEVEL_suffragy"
				}
			}
		}
		
	}
	if = {
		limit = {
			bpm_movement_is_liberal_nationalist = yes
		}
		owner = {
			if = {
				limit = { 
					has_institution = institution_centralization
				}
				
				add = {
					value = -3
					add = "institution_investment_level(institution_centralization)"
					if = {
						limit = { is_ai = yes }
						min = 0
					}
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_centralization"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_centralization"
				}		
			}		
			if = {
				limit = { 
					NOT = { has_law_or_variant = law_type:law_council_republic }
					bpm_country_laws_are_colonial = no
				}
				if = {
					limit = { 
						has_institution = institution_police
					}
					
					add = {
						value = -3
						add = "institution_investment_level(institution_police)"
						multiply = 0.05
						desc = "INSTITUTION_FUNDING_LEVEL_police"
					}
				}
				else = {	
					add = {
						value = 0.1
						desc = "NO_INSTITUTION_police"
					}
				}
				if = {
					limit = { 
						has_institution = institution_culture
						OR = {
							NOT = { has_law_or_variant = law_type:law_outlawed_dissent }
							bpm_movement_is_regressive = yes # Fascists effectively
						}
					}
					
					add = {
						value = -3
						add = "institution_investment_level(institution_culture)"
						multiply = 0.05
						desc = "INSTITUTION_FUNDING_LEVEL_culture"
					}
				}
				else = {	
					add = {
						value = 0.1
						desc = "NO_INSTITUTION_culture"
					}
				}
				
			}
			
		}
	}	

}

bpm_additional_radicalism_factors_diplomacy = {
	# Diplomacy
	if = {
		limit = {
			OR = {
				religion = rel:catholic
				AND = {
					owner = { 
						OR = {
							country_has_state_religion = rel:catholic
							country_definition.religion = rel:catholic
						}
					}
					OR = {
						bpm_movement_is_absolutist_monarchist = yes
						is_political_movement_type = movement_reactionary
					}
				}
			}
			
			exists = c:PAP
			c:PAP ?= {
				country_has_state_religion = rel:catholic
				has_law_or_variant = law_type:law_theocracy
			}
			"owner.relations(c:PAP)" < 0
		}
		add = {
			desc = "POOR_RELATIONS_WITH_PAPACY"
			value = "owner.relations(c:PAP)"
			divide = 100
			multiply = -0.25
		}
	}
}

bpm_additional_radicalism_factors_law_enactment = {
	if = {
		limit = {
			bpm_movement_is_restorationist = yes
			trigger_if = {
				limit = { is_political_movement_type = movement_orleanist }
				owner = { ruler = { has_variable = house_orleans } }
			}
			trigger_else_if = {
				limit = { is_political_movement_type = movement_bonapartist }
				owner = { ruler = { has_variable = house_bonaparte } }
			}
			owner ?= {
				if = {
					limit = {		
						has_law = law_type:law_monarchy
					}
					if = {
						limit = {
							OR = {
								is_enacting_law = law_type:law_presidential_republic
								is_enacting_law = law_type:law_parliamentary_republic
								is_enacting_law = law_type:law_theocracy
								is_enacting_law = law_type:law_council_republic	
							}					
						}
						add = {
							if = {
								limit = {
									ROOT = { bpm_movement_is_absolutist_monarchist = yes }
								}
								value = 0.5
							}
							else = { value = 0.25 }
							desc = "MONARCHY_BEING_ABOLISHED"
						}
											
					}
				}		
			}
		}
	}

	if = {
		limit = {
			OR = {
				bpm_movement_is_radical_socialist = yes
				is_political_movement_type = movement_socialist
			}
		}
		owner = {
			if = {
				limit = {
					has_law_or_variant = law_type:law_factory_councils
				}
				if = {
					limit = {
						OR = {
							country_is_abolishing_factory_councils = yes
							AND = {
								ROOT = { is_political_movement_type = movement_anarchist }
								is_enacting_law = law_type:law_socialist_labor_unions
							}
						}
					}
					add = {
						value = 0.25
						desc = "FACTORY_COUNCILS_BEING_ABOLISHED"
					}
				}
			}
		}
	}
}
