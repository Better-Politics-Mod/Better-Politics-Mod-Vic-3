bpm_electoral_weight = {
	if = {
		limit = {
			owner = { has_law = law_type:law_census_voting }
		}
		value = literacy_rate
	}
	else_if = {
		limit = {
			owner = { has_law = law_type:law_wealth_voting }
		}
		value = standard_of_living
		subtract = state.modifier:state_standard_of_living_add
		if = {
			limit = {
				standard_of_living <= 20
			}
			value = 0
		}
	}
	else_if = {
		limit = {
			owner = { has_law = law_type:law_universal_suffrage }
		}
		value = 1
	}
	else_if = {
		limit = {
			owner = { has_law = law_type:law_landed_voting }
		}
		value = 0
		if = {
			limit = {
				OR = {
					is_pop_type = clergymen
					is_pop_type = capitalists
				}
			}
			value = 1
		}
		else_if = {
			limit = {
				is_pop_type = aristocrats
			}
			value = 3
		}
	}
	else_if = {
		limit = {
			owner = { 
				has_law = law_type:law_autocracy
				has_law = law_type:law_hereditary_bureaucrats
			}
		}
		if = {
			limit = {
				is_pop_type = aristocrats
			}
			value = 1
		}
	}
	else_if = {
		limit = {
			owner = { 
				has_law = law_type:law_autocracy
			}
		}
		if = {
			limit = {
				is_pop_type = officers
			}
			value = 1
		}

	}
}

bpm_total_electorate = {
	value = 0
	every_scope_pop = {
		add = {
			value = total_size
			multiply = bpm_electoral_weight
		}
	}
}

bpm_issue_trade_deficit_interest_peasantry = { 
	value = 0
	market = {
		if = {
			limit = {
				# It makes no sense to care about a trade deficit if you down't own the market
				owner = root
				total_export_value < total_import_value
			}
			# Ratio of exports to imports; since 0 cannot be less than zero, imports will always be nonzero here, and the value will always be between zero and one
			add = {
				value = total_import_value
			}
			divide = {
				value = total_export_value
			}
		}
	}
}
