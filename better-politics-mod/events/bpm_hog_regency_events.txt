namespace = hog_regency

# Start of the regency
hog_regency.1 = {
	type = country_event
	placement = root

	duration = 3

    cooldown = { days = 1}
	title = hog_regency.1.t
	desc = hog_regency.1.d
	flavor = hog_regency.1.f

    event_image = {
		video = "votp_monarch_holding_court"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	### No cooldown

	trigger = {
		has_law_or_variant = law_type:law_answers_to_legislature
        has_law = law_type:law_advisory_body
        bpm_has_acting_hog = no
	}

	cancellation_trigger = {
		OR = {
			NOT = {
		        has_law_or_variant = law_type:law_answers_to_legislature
                has_law = law_type:law_advisory_body
			}
			scope:hog_scope ?= {
				is_adult = yes
			}
            bpm_has_acting_hog = yes
		}
	}

	immediate = {
		bpm_start_acting_hog = yes
		var:bpm_headofgov = {
			save_scope_as = hog_scope
		}
		var:bpm_acting_hog = {
			save_scope_as = acting_hog_scope
		}
	}

	option = {
		name = hog_regency.1.a
		default_option = yes
	}
}

# End of the regency
hog_regency.3 = {
	type = country_event
	placement = root

	duration = 3

	title = hog_regency.3.t
	desc = hog_regency.3.d
	flavor = hog_regency.3.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:hog_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/misc/2Characters"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	cooldown = { days = 100 }

	trigger = {
		bpm_has_acting_hog = yes
		var:bpm_headofgov ?= {
			is_adult = yes
		}
	}

	cancellation_trigger = {

	}

	immediate = {
		var:bpm_acting_hog ?= {
			save_scope_as = acting_hog_scope
		}
		var:bpm_headofgov ?= {
			save_scope_as = hog_scope
		}
        bpm_end_acting_hog = yes
	}

	option = {
		name = hog_regency.3.a
		default_option = yes
	}
}
