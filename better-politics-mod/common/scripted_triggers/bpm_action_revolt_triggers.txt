bpm_has_revcell = {
	is_key_in_variable_map = {
		name = bpm_revcells
		target = scope:movement
	}
}

bpm_has_revcell_level = {
	is_key_in_variable_map = {
		name = bpm_revcells
		target = scope:movement
	}
	"variable_map(bpm_revcells|scope:movement)" = $lvl$
}

bpm_revcell_can_be_spread_to = {
	turmoil > 0.25
	NOT = {
		bpm_has_revcell_level = { lvl = 3 }
	}
}

bpm_revcell_recently_subsided = {
	OR = {
		is_key_in_variable_map = {
			name = bpm_revcell_recently_subsided
			target = scope:movement
		}
		owner = {
			is_key_in_variable_map = {
				name = bpm_revcell_recently_subsided
				target = scope:movement
			}					
		}
	}
}

bpm_movement_is_broadly_modernizing = {
	OR = {
		law_stance = {
			law = law_type:law_separated_powers
			value > neutral
		}
		law_stance = {
			law = law_type:law_universal_suffrage
			value > neutral
		}
		law_stance = {
			law = law_type:law_census_voting
			value > neutral
		}
		law_stance = {
			law = law_type:law_presidential_republic
			value > neutral
		}
		AND = {
			owner = {
				has_law = law_type:law_serfdom
			}
			law_stance = {
				law = law_type:law_peasant_proprietorship
				value > neutral
			}
		}
	}
}

bpm_movement_is_broadly_proletarian = {
	OR = {
		law_stance = {
			law = law_type:law_factory_councils
			value > neutral
		}
		law_stance = {
			law = law_type:law_council_republic
			value > neutral
		}
	}
}

bpm_movement_is_broadly_reactionary = {
	NOR = {
		bpm_movement_is_broadly_modernizing = yes
		bpm_movement_is_broadly_proletarian = yes
	}
}
