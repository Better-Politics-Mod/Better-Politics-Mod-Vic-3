bpm_re_add_ideology = {
	if = {
		limit = { has_ideology = ideology:$IDEOLOGY$ }
		remove_ideology = ideology_$TYPE$_placeholder
		remove_ideology = $IDEOLOGY$
		add_ideology = $IDEOLOGY$
	}
}
