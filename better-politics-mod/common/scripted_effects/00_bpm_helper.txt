update_variable_map_value = {
    if = {
        limit = {
            is_key_in_variable_map = {
                name = $name$
                target = $key$
            }
        }
        remove_from_variable_map = {
            name = $name$
            key = $key$
        }
    }
    add_to_variable_map = {
        name = $name$
        key = $key$
        value = $value$
    }
}

add_variable_map_value = {
    set_local_variable = {
        name = temp
        value = {
            value = "variable_map($name$|$key$)"
            add = $add$
        }
    }
    update_variable_map_value = {
        name = $name$
        key = $key$
        value = local_var:temp
    }
    remove_local_variable = temp
}

sgui_execute_abuse = {
    if = {
        limit = {
            $trigger$
        }
        custom_tooltip = $text$_checked
    }
    else = {
        custom_tooltip = $text$_unchecked
    }
}
