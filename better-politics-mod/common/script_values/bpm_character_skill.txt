bpm_character_skill_level = {
	if = {
		limit = {
			has_trait = bpm_political_skill_zero
		}
		value = 0
	}
	if = {
		limit = {
			has_trait = bpm_political_skill_one
		}
		value = 1
	}
	if = {
		limit = {
			has_trait = bpm_political_skill_two
		}
		value = 2
	}
	if = {
		limit = {
			has_trait = bpm_political_skill_three
		}
		value = 3
	}
	if = {
		limit = {
			has_trait = bpm_political_skill_four
		}
		value = 4
	}
	if = {
		limit = {
			has_trait = bpm_political_skill_five
		}
		value = 5
	}
}

bpm_character_agenda_level = {
	if = {
		limit = {
			bpm_character_skill_level < 2
		}
		value = 1
	}
	else_if = {
		limit = {
			bpm_character_skill_level < 4
		}
		value = 2
	}
	else = {
		value = 3
	}
	owner = { subtract = bpm_focusing_cabinet }
}

bpm_focusing_cabinet = {
	value = 0
	every_in_list = {
		variable = bpm_focused_institutions
		add = 1
	}
}
