bpm_start_civil_war = {
	save_scope_as = origin_country
	if = {
		limit = {
			NOT = {
				any_scope_state = {
					NOT = { bpm_state_has_rev = { front = $front$ } }
				}
			}
		}
		set_local_variable = bpm_civil_war_has_all
	}
	else = {
		remove_local_variable = bpm_civil_war_has_all
	}
	every_state = {
		remove_variable = bpm_state_to_cede
	}
	every_scope_state = {
		limit = {
			bpm_state_has_rev = { front = $front$ }
			OR = {
				NOT = { is_capital = yes }
				NOT = { has_local_variable = bpm_civil_war_has_all }
			}
		}
		set_variable = bpm_state_to_cede
	}
	every_country = {
		remove_variable = bpm_coniq
	}
	create_dynamic_country = {
		origin = scope:origin_country
		cede_state_trigger = {
			has_variable = bpm_state_to_cede
		}
		on_created = {
			set_variable = bpm_coniq
		}
	}
	random_country = {
		limit = {
			has_variable = bpm_coniq
		}
		save_scope_as = new_country
		set_variable = {
			name = bpm_country_revolutionary_front
			value = flag:bpm_front_$front$
		}
		set_variable = {
			name = bpm_civil_war_origin
			value = scope:origin_country
		}
	}
	create_diplomatic_play = { 
		target_country = scope:new_country
		type = dp_revolution
		escalation = 80
		annex_as_civil_war = yes
	}  
	if = {
		limit = {
			exists = scope:origin_country
		}
		every_country = {
			limit = {
				has_variable = bpm_civil_war_origin
				var:bpm_civil_war_origin = scope:origin_country
				NOT = {
					this = scope:new_country
				}
			}
			save_scope_as = target_country
			scope:new_country = {
				create_diplomatic_play = { 
					target_country = scope:target_country
					type = dp_revolution
					escalation = 80
					annex_as_civil_war = yes
				}  
			}
		}
	}
}
