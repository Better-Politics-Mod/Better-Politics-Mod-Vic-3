
on_monthly_pulse_country = {
    on_actions = {
        bpm_update_front_membership
    }
}

on_half_yearly_pulse_country = {
	on_actions = {
		bpm_update_front_membership_ai
	}
}

bpm_update_front_membership = {
	effect = {
		if = {
			limit = {
				NOT = { is_country_type = decentralized }
				is_player = yes
			}
			bpm_choose_fronts = yes
		}
		if = {
			limit = {
				NOT = { is_country_type = decentralized }
			}
			bpm_dissent_inactive_je_effects = yes
		}
	}
}

# identical to bpm_update_front_membership, but for AI countries
bpm_update_front_membership_ai = {
	effect = {
		if = {
			limit = {
				NOT = { is_country_type = decentralized }
				is_player = no
			}
			bpm_choose_fronts = yes
		}
	}
}
