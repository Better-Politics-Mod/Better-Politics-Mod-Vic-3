bpm_create_hog_heir_hook = {
    if = {
        limit = {
            THIS = c:JAP
            OR = {
				has_law = law_type:law_bakufu
				any_active_law = {
					has_amendment = amendment_type:amendment_taikunate
				}
			}
        }

		if = {
			limit = {
				any_scope_character = {
					has_template = JAP_ienari_tokugawa
					NOT = {has_role = bpm_character_role_hog}
					is_character_alive = yes
				}
			}
			random_scope_character = {
				limit = {has_template = JAP_ienari_tokugawa}
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}

		else_if = {
			limit = {
				any_scope_character = {
					has_template = JAP_ieyoshi_tokugawa
					NOT = {has_role = bpm_character_role_hog}
					is_character_alive = yes
				}
			}
			random_scope_character = {
				limit = {has_template = JAP_ieyoshi_tokugawa}
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}
		else_if = {
			limit = {
				NOT = {
					is_template_used = JAP_ieyoshi_tokugawa
				}
			}
			create_character = {
				heir = no
				template = JAP_ieyoshi_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
		}

		else_if = {
			limit = {
				NOT = {has_variable = bpm_rejected_iesada}
				any_scope_character = {
					has_template = JAP_iesada_tokugawa
					NOT = {has_role = bpm_character_role_hog}
					is_character_alive = yes
				}
			}
			random_scope_character = {
				limit = {has_template = JAP_iesada_tokugawa}
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}
		else_if = {
			limit = {
				NOT = {
					has_variable = bpm_rejected_iesada
					is_template_used = JAP_iesada_tokugawa
				}
			}
			create_character = {
				template = JAP_iesada_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
		}

		else_if = {
			limit = {
				NOT = {has_variable = bpm_rejected_yoshinobu}
				any_scope_character = {
					has_template = JAP_yoshinobu_tokugawa
					NOT = {has_role = bpm_character_role_hog}
					is_character_alive = yes
				}
			}
			random_scope_character = {
				limit = {has_template = JAP_yoshinobu_tokugawa}
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}
		else_if = {
			limit = {
				NOT = {
					has_variable = bpm_rejected_yoshinobu
					is_template_used = JAP_yoshinobu_tokugawa
					game_date < 1837.10.28
				}
			}
			create_character = {
				template = JAP_yoshinobu_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
		}

		else_if = {
			limit = {
				NOT = {has_variable = bpm_rejected_iemochi}
				any_scope_character = {
					has_template = JAP_iemochi_tokugawa
					NOT = {has_role = bpm_character_role_hog}
					is_character_alive = yes
				}
			}
			random_scope_character = {
				limit = {has_template = JAP_iemochi_tokugawa}
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}
		else_if = {
			limit = {
				NOT = {
					has_variable = bpm_rejected_iemochi
					is_template_used = JAP_iemochi_tokugawa
					game_date < 1846.7.16
				}
			}
			create_character = {
				template = JAP_iemochi_tokugawa
				on_created = {
					set_variable = house_tokugawa
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
		}

		else = {
			create_character = {
				template = JAP_fallback_tokugawa
				on_created = {
					owner = {
						set_variable = {
							name = bpm_hog_heir
							value = prev
						}
					}
					add_character_role = bpm_character_role_protege
				}
			}
		}
    }
    else = {
		if = {
			limit = {
				has_variable = bpm_headofgov
				exists = var:bpm_headofgov.interest_group
			}
			var:bpm_headofgov.interest_group = {
				save_scope_as = heir_ig
			}
		}
		else_if = {
			limit = {
				bpm_has_acting_hog = yes
				exists = var:bpm_acting_hog.interest_group
			}
			var:bpm_acting_hog.interest_group = {
				save_scope_as = heir_ig
			}
		}
		else = {
			ordered_interest_group = {
				limit = {is_in_government = yes}
				alternative_limit = {always = yes}
				order_by = ig_clout
				save_scope_as = heir_ig
			}
		}
        create_character = {
            age = 23
			interest_group = scope:heir_ig
			on_created = {
				owner = {
					set_variable = {
						name = bpm_hog_heir
						value = prev
					}
				}
				add_character_role = bpm_character_role_protege
			}
		}
    }
}
