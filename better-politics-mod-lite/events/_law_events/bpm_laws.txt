namespace = laws

laws.1 = {
	type = country_event
	placement = root
	title = laws.1.t
	desc = laws.1.d
	flavor = laws.1.f

	category = enactment

	duration = 3

	event_image = {
		video = "southamerica_public_figure_assassination"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	cooldown = { months = normal_modifier_time }

	trigger = {
		is_enacting_law = law_type:law_autocracy
		any_interest_group = {
			is_in_government = no
			is_marginal = no
			law_stance = {
				law = owner.currently_enacting_law.type
				value < neutral
			}
		}
	}

	immediate = {
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
		random_interest_group = {
			limit = {
				is_in_government = no
				is_marginal = no
				law_stance = {
					law = owner.currently_enacting_law.type
					value < neutral
				}
			}
			save_scope_as = jedi_scope
		}
		ruler = {
			save_scope_as = ruler_scope
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = {
		name = laws.1.a
		default_option = yes
		add_enactment_phase = 1
		scope:ruler_scope = {
			add_modifier = {
				name = scarred_and_deformed
				months = normal_modifier_time
			}
		}
	}
	option = {
		name = laws.1.b
		if ={ 
			limit ={ 
				is_player = yes
			}
			random_interest_group ={ 
				limit ={ 
					bpm_ig_is_monarchist = yes
				}
				bpm_ig_increase_law_enactment_approval_medium = yes
			}
		}
		else ={ 
			add_enactment_modifier = {
				name = thunderous_applause
			}
		}
		scope:jedi_scope = {
			add_modifier = {
				name = hunted_down_and_defeated
				months = short_modifier_time
			}
		}
	}
}
