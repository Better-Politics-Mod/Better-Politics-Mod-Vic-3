american_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		OR = {
			home_country = {
				exists = c:USA
				c:USA = this
				is_subject = no
			}
			home_country = {
				exists = c:FSA
				c:FSA = this
				is_subject = no
			}
			home_country = {
				exists = c:CSA
				c:CSA = this
				is_subject = no
			}
		}
	}
}

american_clothes_pop_trigger = {

}

carlist_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		home_country = {
			is_subject = no
			country_has_primary_culture = cu:spanish
		}
		OR = {
			character_has_carlist_ideology = yes
			AND = {
				is_interest_group_type = ig_reactionaries
				owner = {
					NOT = { has_variable = monarchy_stamped_out_var }
				}
			}
			home_country = {
				c:SPC ?= this
			}
		}
	}
}

carlist_clothes_pop_trigger = {
	exists = scope:pop
	scope:pop = {
		exists = home_country
		home_country = {
			is_subject = no
			country_has_primary_culture = cu:spanish
			OR = {
				c:SPC ?= this
				ruler = {
					OR = {
						character_has_carlist_ideology = yes
						AND = {
							is_interest_group_type = ig_reactionaries
							owner = {
								NOT = { has_variable = monarchy_stamped_out_var }
							}
						}
					}
				}
			}
		}
	}
}
