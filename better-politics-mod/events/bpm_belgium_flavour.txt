bpm_belgian_law_events.2 = { # Organic Law on Primary Education of September 23rd, 1842

	type = country_event

	dlc = dlc_bpm

	event_image = {
		video = "southamerica_aristocrats"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = bpm_belgian_law_events.2.t
	desc = {
		triggered_desc = {
			desc = bpm_belgian_law_events.2.religious
			trigger = {
				has_law = law_type:law_religious_schools
			}
		}
		triggered_desc = {
			desc = bpm_belgian_law_events.2.public
			trigger = {
				has_law = law_type:law_public_schools
			}
		}
	}

	flavor = bpm_belgian_law_events.2.f

	duration = 3

	cooldown = {
		years = 100
	}

	trigger = {
		OR = {
			has_law = law_type:law_religious_schools
			has_law = law_type:law_public_schools
		}
	}

	immediate = {

	}

	option = {
		name = bpm_belgian_law_events.2.a

		default_option = yes		

		if = { ### Every secularist IG will be particularly unhappy
			limit = {
				has_law = law_type:law_religious_schools
			}

			every_interest_group = {
				limit = {
					OR = {
						has_ideology = ideology:ideology_rel_secularist
						has_ideology = ideology:ideology_rel_radical_secularist
					}
				}
				add_modifier = {
					name = bpm_stubborn_ig
					days = normal_modifier_time
				}
			}
		}

		else_if = { # Religion and morality are subjects present in schooling (Article 6-7)
			limit = {
				has_law = law_type:law_public_schools
			}

			active_law:lawgroup_education_system ?= {
				add_amendment = {
					type = amendment_church_organised_schools
					sponsor = PREV.ig:ig_devout
				}
			}
		}

		add_modifier = {
			name = bpm_funding_and_organizing_primary_schooling
			years = 5
			is_decaying = yes
		}

		#je:je_bpm_unions_des_oppositions = {
		#	add_progress = { value = 1 name = bpm_consolidation_of_the_state_progress_bar }
		#}
	}
}
