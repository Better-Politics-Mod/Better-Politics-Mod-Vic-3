movement_royalist_constitutional = {
	category = movement_category_ideological
	
	ideology = ideology_gov_constitutional_royalist
	character_ideologies = {
		ideology_royalist
		ideology_humanitarian_royalist
	}
	
	disband_trigger = {
		has_variable = monarchy_stamped_out_var # Allowing voting and then abolishing it won't suddenly make absolute monarchism chic again
	}

	creation_trigger = {
		has_law = law_type:law_monarchy
		country_has_voting_franchise = yes
		NOT = { c:FRA ?= this }
	}	
	creation_weight = {
		value = 100
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}

	can_pressure_interest_group = {
		bpm_ig_is_radical_left = no
	}
	
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_royalist
			has_ideology = ideology:ideology_humanitarian_royalist
			AND = {
				NOT = {
					law_stance = {
						law = law_type:law_monarchy
						value < approve
					}
				}
				law_stance = {
					law = law_type:law_separated_powers
					value > neutral
				}
			}
		}
	}
	character_support_weight = {
		value = 300
		if = {
			limit = {
				has_ideology = ideology:ideology_royalist
			}
			multiply = {
				value = 5
			}
		}
	}
	
	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}
	
	pop_support_factors = {
		movement_support_ruler_popularity
		movement_support_low_urbanization
		movement_support_high_literacy
		movement_support_aristocrats
		movement_support_farmers
		movement_support_peasants
		movement_support_clergymen
		movement_support_officers
		movement_support_shopkeepers
		movement_support_clerks
	}

	pop_support_weight = {	
		add = {
			desc = "URBAN_STATE" # Increases support in rural states, decreases in urban
			value = 1
			subtract = state.state_urbanization_rate
			multiply = 15
		}
	
		if = {
			limit = {
				is_pop_type = aristocrats
			}
			add = {
				value = 30
				desc = "POP_ARISTOCRATS"
			}
		}	
		else_if = {
			limit = {
				is_pop_type = farmers
			}
			add = {
				value = 12
				desc = "POP_FARMERS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = peasants
			}
			add = {
				value = 6
				desc = "POP_PEASANTS"
			}
		}			
		else_if = {
			limit = {
				is_pop_type = clergymen
			}
			add = {
				value = 12
				desc = "POP_CLERGYMEN"
			}
		}	
		else_if = {
			limit = {
				is_pop_type = officers
			}
			add = {
				value = 20
				desc = "POP_OFFICERS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = shopkeepers
			}
			add = {
				value = 6
				desc = "POP_SHOPKEEPERS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = clerks
			}
			add = {
				value = 6
				desc = "POP_CLERKS"
			}
		}

		if = {
			limit = {
				strata = upper
			}
			multiply = 1.25
		}
		else_if = {
			limit = {
				strata = middle
			}
			multiply = 1.25
		}
		else = {
			multiply = 0.1
		}

		if = {
			limit = {
				owner ?= {
					has_law = law_type:law_monarchy
				}
				owner.ruler ?= {
					NOT = { has_trait = trait_child }
				}
			}
			multiply = { # Constitutionalists are less reliant on the monarchs popularity than absolutists
				add = owner.ruler.popularity
				multiply = 0.01
				add = 1
				min = 0.5
				max = 1.5
				desc = "RULER_POPULARITY"
			}
		}
		else_if = {
			limit = {
				owner ?= {
					has_law = law_type:law_monarchy
				}
				owner.ruler ?= {
					has_trait = trait_child
				}
			}
			multiply = {
				value = 0.75
				desc = "RULER_IS_CHILD"
			}			
		}

		min = 0

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = { value = BPM_LIBERAL_STRENGTH desc = BPM_LIBERAL_AND_CONS } 
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.15
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = 1

			subtract = {
				value = state_urbanization_rate
				multiply = 0.5
			}
			
			if = {
				limit = { is_capital = yes }
				if = {
					limit = { owner ?= { has_law = law_type:law_monarchy } }
					multiply = 10.0
				}
				else = {
					multiply = 0.1
				}
			}			
		}
	}
	
	secession = {
		possible = {
			always = no
		}
		
		weight = {
			value = 1000
		}	

		state_weight = {
			value = 1

			subtract = {
				value = state_urbanization_rate
				multiply = 0.5
			}
			
			if = {
				limit = { is_capital = yes }
				if = {
					limit = { owner ?= { has_law = law_type:law_monarchy } }
					multiply = 10.0
				}
				else = {
					multiply = 0.1
				}
			}			
		}		
		
		forced_tags = {
			BHT = {
				trigger = {
					owner ?= { c:BIC ?= this }
				}
				weight = 1000
			}
		}
	}	
	
	additional_radicalism_factors = {
		owner ?= {
			if = {
				limit = {		
					has_law = law_type:law_monarchy
				}
				
				if = {
					limit = {
						OR = {
							is_enacting_law = law_type:law_presidential_republic
							is_enacting_law = law_type:law_parliamentary_republic
							is_enacting_law = law_type:law_corporate_state
							is_enacting_law = law_type:law_theocracy
							is_enacting_law = law_type:law_council_republic	
						}					
					}
					
					add = {
						value = 0.25
						desc = "MONARCHY_BEING_ABOLISHED"
					}					
				}
				else = {
					add = { # Only care about legitimacy in the context of a monarchy that isn't under direct threat
						value = 50
						subtract = legitimacy
						multiply = 0.005
						desc = "concept_legitimacy"
					}				
				}
			}		
		}
	}	
}

	}

			limit = {
				is_in_exile_pool = yes

			OR = {
				has_ideology = ideology:ideology_market_liberal

			OR = {
				has_ideology = ideology:ideology_market_liberal
				has_ideology = ideology:ideology_liberal_leader
				has_ideology = ideology:ideology_republican_leader

				has_ideology = ideology:ideology_humanitarian

					bpm_ig_is_liberal = yes
				}
			}
		}
	}

		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_republican_leader
					has_ideology = ideology:ideology_liberal_leader
					has_ideology = ideology:ideology_market_liberal
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_radical
				NOT = {
					owner ?= {
						any_political_movement = {
							is_political_movement_type = movement_radical
						}
					}
				}
			}
			multiply = {
				value = 5
			}
		}
		else_if = {
			limit = {
				has_ideology = ideology:ideology_humanitarian
				NOT = {
					interest_group = { is_interest_group_type = ig_trade_unions }
				}
			}
			multiply = {
				value = 5
			}
		}
		if = { # Liberal agitators just inherently love liberalism
			limit = {
				interest_group ?= {
					OR = {
						has_ideology = ideology:ideology_liberal
						has_ideology = ideology:ideology_liberal_modern
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}

		bpm_ig_is_radical_right = no
	}

			is_pop_type = slaves
		}

					is_pop_type = bureaucrats
				}
				add = {
					value = 24
					desc = "POP_BUREAUCRATS"
				}
			}
			else = {
				add = {
					value = 9
					desc = "MIDDLE_NO_ICON"
				}
			}
			if = {
				limit = {
					owner = {
						NOT = {
							exists = ig:ig_liberals
						}
					}
				}
				multiply = 6
			}
			if = {
				limit = {
					owner.ruler ?= {
						interest_group = { NOR = { 
							is_interest_group_type = ig_industrialists
							is_interest_group_type = ig_liberals
							is_interest_group_type = ig_market_liberals
							is_interest_group_type = ig_national_liberals
						} }
					}
				}
				multiply = 2
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 6
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants
				}
				add = {
					value = 0
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		min = 0
		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}
		min = 0
		# Hub-movement
		# If the country is not liberal democratic,
		# then other liberal movements are weaker
		# and the main liberal movement is stronger
		# this is because liberals will only diversify
		# their support if their basic demands are met
		#
		# Under an illiberal regime, liberals are 
		# primarily considered about regime change
		# rather than national or market liberal flavors
		if = {
			limit = {
				owner = {
					bpm_country_laws_are_moderately_liberal_democratic = yes
				}
			}
			multiply = 2
		}
		if = {
			limit = {
				exists = ig:ig_liberals
				ig:ig_liberals ?= {
					ig_clout > 0.15
					bpm_ig_is_establishment = yes
				}
			}
			multiply = {
				value = 2
				subtract = {
					value = ig:ig_liberals.ig_approval
					divide = 20
				}
				divide = 2
				desc = "BPM_INSTITUTIONAL_ALTERNATIVE"
			}
		}
		multiply = { value = BPM_LIBERAL_STRENGTH desc = BPM_LIBERAL_AND_CONS } 
	}

		possible = {
			political_movement_support >= 0.15
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = state_urbanization_rate	
		}
	}
	
	secession = {
		possible = {
			always = no
		}
		
		weight = {
			value = 1000
		}

		state_weight = {
			value = state_urbanization_rate	
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_schools }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_schools)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_schools"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_schools"
				}		
			}

				}
				add = {
					value = 0.1
					desc = "IS_PRE_MODERN_ABSOLUTIST"
				}
			}			
		}
	}

					owner = {

					}

				multiply = 6

							is_interest_group_type = ig_liberals
							is_interest_group_type = ig_market_liberals
							is_interest_group_type = ig_national_liberals

				value = 3

		}
